"""Learning event schema and storage."""
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import List, Optional, Dict, Any
import json
import hashlib
from pathlib import Path


class EventType(Enum):
    """Type of learning event."""
    CONFIRMED = "confirmed"      # True positive confirmed
    REJECTED = "rejected"        # False positive rejected
    ADJUSTED = "adjusted"        # Manual confidence adjustment


@dataclass
class LearningEvent:
    """A single learning event capturing verdict feedback."""

    # Identity
    event_id: str
    event_type: EventType
    timestamp: datetime

    # What triggered learning
    pattern_id: str
    finding_id: str

    # Context for similarity matching
    function_signature: str       # e.g., "withdraw(uint256)"
    modifier_hash: str            # Hash of modifiers present
    context_hash: str             # Hash of surrounding code context

    # Feedback
    reason: str                   # Why this verdict
    confidence_adjustment: float  # Applied adjustment (after decay)

    # Metadata
    auditor_id: Optional[str] = None
    bead_id: Optional[str] = None

    @classmethod
    def create(
        cls,
        event_type: EventType,
        pattern_id: str,
        finding_id: str,
        function_signature: str,
        modifiers: List[str],
        context_code: str,
        reason: str,
        auditor_id: Optional[str] = None,
    ) -> "LearningEvent":
        """Factory method to create an event."""
        event_id = cls._generate_id(pattern_id, finding_id)
        modifier_hash = hashlib.md5("|".join(sorted(modifiers)).encode()).hexdigest()[:8]
        context_hash = hashlib.md5(context_code.encode()).hexdigest()[:16]

        # Default adjustments
        if event_type == EventType.CONFIRMED:
            adjustment = 0.02
        elif event_type == EventType.REJECTED:
            adjustment = -0.05
        else:
            adjustment = 0.0

        return cls(
            event_id=event_id,
            event_type=event_type,
            timestamp=datetime.now(),
            pattern_id=pattern_id,
            finding_id=finding_id,
            function_signature=function_signature,
            modifier_hash=modifier_hash,
            context_hash=context_hash,
            reason=reason,
            confidence_adjustment=adjustment,
            auditor_id=auditor_id,
        )

    @staticmethod
    def _generate_id(pattern_id: str, finding_id: str) -> str:
        timestamp = datetime.now().strftime("%Y%m%d%H%M%S")
        return f"evt-{pattern_id[:8]}-{timestamp}"

    def to_dict(self) -> Dict[str, Any]:
        return {
            "event_id": self.event_id,
            "event_type": self.event_type.value,
            "timestamp": self.timestamp.isoformat(),
            "pattern_id": self.pattern_id,
            "finding_id": self.finding_id,
            "function_signature": self.function_signature,
            "modifier_hash": self.modifier_hash,
            "context_hash": self.context_hash,
            "reason": self.reason,
            "confidence_adjustment": self.confidence_adjustment,
            "auditor_id": self.auditor_id,
            "bead_id": self.bead_id,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "LearningEvent":
        data["event_type"] = EventType(data["event_type"])
        data["timestamp"] = datetime.fromisoformat(data["timestamp"])
        return cls(**data)


class EventStore:
    """Persistent storage for learning events."""

    def __init__(self, path: Path):
        self.path = path
        self.path.mkdir(parents=True, exist_ok=True)
        self._events_file = self.path / "events.jsonl"

    def record(self, event: LearningEvent) -> None:
        """Append an event to storage."""
        with open(self._events_file, "a") as f:
            f.write(json.dumps(event.to_dict()) + "\n")

    def get_events_for_pattern(self, pattern_id: str) -> List[LearningEvent]:
        """Get all events for a pattern."""
        events = []
        if not self._events_file.exists():
            return events

        with open(self._events_file, "r") as f:
            for line in f:
                data = json.loads(line)
                if data["pattern_id"] == pattern_id:
                    events.append(LearningEvent.from_dict(data))

        return events

    def get_recent_events(self, days: int = 30) -> List[LearningEvent]:
        """Get events from the last N days."""
        cutoff = datetime.now().timestamp() - (days * 86400)
        events = []

        if not self._events_file.exists():
            return events

        with open(self._events_file, "r") as f:
            for line in f:
                data = json.loads(line)
                event_time = datetime.fromisoformat(data["timestamp"]).timestamp()
                if event_time > cutoff:
                    events.append(LearningEvent.from_dict(data))

        return events

    def count_by_type(self, pattern_id: str) -> Dict[str, int]:
        """Count events by type for a pattern."""
        counts = {"confirmed": 0, "rejected": 0, "adjusted": 0}
        for event in self.get_events_for_pattern(pattern_id):
            counts[event.event_type.value] += 1
        return counts

"""Tests for learning events."""
import pytest
from datetime import datetime
from true_vkg.learning.events import LearningEvent, EventType, EventStore

class TestLearningEvent:
    def test_create_confirmed_event(self):
        event = LearningEvent.create(
            event_type=EventType.CONFIRMED,
            pattern_id="vm-001",
            finding_id="VKG-001",
            function_signature="withdraw(uint256)",
            modifiers=["nonReentrant"],
            context_code="function withdraw() ...",
            reason="Verified via PoC",
        )
        assert event.event_type == EventType.CONFIRMED
        assert event.confidence_adjustment > 0

    def test_create_rejected_event(self):
        event = LearningEvent.create(
            event_type=EventType.REJECTED,
            pattern_id="vm-001",
            finding_id="VKG-002",
            function_signature="withdraw(uint256)",
            modifiers=["nonReentrant"],
            context_code="function withdraw() ...",
            reason="Has reentrancy guard",
        )
        assert event.event_type == EventType.REJECTED
        assert event.confidence_adjustment < 0

    def test_round_trip(self):
        event = LearningEvent.create(
            event_type=EventType.CONFIRMED,
            pattern_id="vm-001",
            finding_id="VKG-001",
            function_signature="withdraw(uint256)",
            modifiers=[],
            context_code="code",
            reason="test",
        )
        data = event.to_dict()
        restored = LearningEvent.from_dict(data)
        assert restored.event_id == event.event_id
