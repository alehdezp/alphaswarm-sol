## Key Design Decisions

### What Makes Two Findings "Similar"?

**Pattern ID alone is too broad:**
- `vm-001` matches all reentrancy - but some have guards, some don't

**Function signature alone is too narrow:**
- `withdraw(uint256)` might have many different implementations

**Solution: Context-based similarity**
```python
# Similarity key = pattern_id + modifier_hash + context features
similar_if:
    - Same pattern_id
    - Same modifiers present (e.g., nonReentrant)
    - Similar code structure (context_hash bucket)
```

```python
"""False positive recording and warning system."""
from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Dict, Optional, Set
from collections import defaultdict
import json
from pathlib import Path

from .events import LearningEvent, EventType, EventStore


@dataclass
class FPPattern:
    """A false positive pattern identified from rejections."""
    pattern_id: str
    modifier_signature: str       # Modifiers that indicate FP
    occurrence_count: int
    last_seen: datetime
    reasons: List[str]           # Why these were FPs

    def matches(self, finding) -> bool:
        """Check if a finding matches this FP pattern."""
        # Match if same pattern and similar modifiers
        if finding.pattern_id != self.pattern_id:
            return False

        # Check modifier overlap
        finding_mods = set(getattr(finding, 'modifiers', []))
        pattern_mods = set(self.modifier_signature.split('|'))

        # FP pattern indicates these modifiers make it safe
        if pattern_mods.issubset(finding_mods):
            return True

        return False


class FalsePositiveRecorder:
    """Track and warn about false positive patterns."""

    # Minimum occurrences before generating warning
    MIN_OCCURRENCES_WARNING = 2

    # Strong warning threshold
    STRONG_WARNING_THRESHOLD = 5

    def __init__(self, storage_path: Path):
        self.storage_path = storage_path
        self.storage_path.mkdir(parents=True, exist_ok=True)
        self._fp_patterns: Dict[str, FPPattern] = {}
        self._load()

    def record(self, finding, reason: str) -> None:
        """Record a false positive."""
        key = self._make_key(finding)

        if key not in self._fp_patterns:
            self._fp_patterns[key] = FPPattern(
                pattern_id=finding.pattern_id,
                modifier_signature=self._get_modifier_signature(finding),
                occurrence_count=0,
                last_seen=datetime.now(),
                reasons=[],
            )

        pattern = self._fp_patterns[key]
        pattern.occurrence_count += 1
        pattern.last_seen = datetime.now()
        if reason not in pattern.reasons:
            pattern.reasons.append(reason)

        self._save()

    def get_warnings(self, finding) -> List[str]:
        """Get FP warnings for a finding.

        Returns list of warning strings. Empty if no warnings.
        """
        warnings = []

        for fp in self._fp_patterns.values():
            if fp.pattern_id != finding.pattern_id:
                continue

            if not fp.matches(finding):
                continue

            if fp.occurrence_count >= self.STRONG_WARNING_THRESHOLD:
                warnings.append(
                    f"LIKELY FALSE POSITIVE: {fp.occurrence_count} similar "
                    f"findings were rejected. Common reasons: {', '.join(fp.reasons[:3])}"
                )
            elif fp.occurrence_count >= self.MIN_OCCURRENCES_WARNING:
                warnings.append(
                    f"Possible FP: {fp.occurrence_count} similar rejections. "
                    f"Check: {fp.reasons[0] if fp.reasons else 'unknown'}"
                )

        return warnings

    def get_fp_rate(self, pattern_id: str) -> float:
        """Get estimated FP rate for a pattern based on recorded data."""
        relevant = [fp for fp in self._fp_patterns.values()
                   if fp.pattern_id == pattern_id]
        if not relevant:
            return 0.0

        total_fps = sum(fp.occurrence_count for fp in relevant)
        # This is just FP count, not rate - would need TP count for rate
        return total_fps

    def _make_key(self, finding) -> str:
        """Make a unique key for FP pattern matching."""
        pattern_id = finding.pattern_id
        mod_sig = self._get_modifier_signature(finding)
        return f"{pattern_id}:{mod_sig}"

    def _get_modifier_signature(self, finding) -> str:
        """Get modifier signature for similarity matching."""
        modifiers = getattr(finding, 'modifiers', [])
        # Sort for consistency
        return "|".join(sorted(modifiers))

    def _load(self) -> None:
        """Load FP patterns from storage."""
        fp_file = self.storage_path / "fp_patterns.json"
        if not fp_file.exists():
            return

        with open(fp_file, "r") as f:
            data = json.load(f)

        self._fp_patterns = {}
        for key, item in data.items():
            self._fp_patterns[key] = FPPattern(
                pattern_id=item["pattern_id"],
                modifier_signature=item["modifier_signature"],
                occurrence_count=item["occurrence_count"],
                last_seen=datetime.fromisoformat(item["last_seen"]),
                reasons=item["reasons"],
            )

    def _save(self) -> None:
        """Save FP patterns to storage."""
        fp_file = self.storage_path / "fp_patterns.json"
        data = {}
        for key, fp in self._fp_patterns.items():
            data[key] = {
                "pattern_id": fp.pattern_id,
                "modifier_signature": fp.modifier_signature,
                "occurrence_count": fp.occurrence_count,
                "last_seen": fp.last_seen.isoformat(),
                "reasons": fp.reasons,
            }

        with open(fp_file, "w") as f:
            json.dump(data, f, indent=2)

    def summary(self) -> str:
        """Generate summary of FP patterns."""
        lines = ["# False Positive Patterns", ""]

        by_pattern = defaultdict(list)
        for fp in self._fp_patterns.values():
            by_pattern[fp.pattern_id].append(fp)

        for pattern_id, fps in sorted(by_pattern.items()):
            total = sum(fp.occurrence_count for fp in fps)
            lines.append(f"## {pattern_id}: {total} total FPs")
            for fp in fps:
                lines.append(f"  - {fp.modifier_signature}: {fp.occurrence_count} occurrences")
                for reason in fp.reasons[:2]:
                    lines.append(f"    Reason: {reason}")
            lines.append("")

        return "\n".join(lines)
```

```python
"""Tests for FP recorder."""
import pytest
from pathlib import Path
from unittest.mock import Mock
from true_vkg.learning.fp_recorder import FalsePositiveRecorder

class TestFPRecorder:
    def test_record_and_warn(self, tmp_path):
        recorder = FalsePositiveRecorder(tmp_path)

        finding = Mock()
        finding.pattern_id = "vm-001"
        finding.modifiers = ["nonReentrant"]

        # Record 2 FPs
        recorder.record(finding, "Has reentrancy guard")
        recorder.record(finding, "Has reentrancy guard")

        # Should get warning
        warnings = recorder.get_warnings(finding)
        assert len(warnings) > 0
        assert "similar" in warnings[0].lower()

    def test_no_warning_single_occurrence(self, tmp_path):
        recorder = FalsePositiveRecorder(tmp_path)

        finding = Mock()
        finding.pattern_id = "vm-001"
        finding.modifiers = []

        recorder.record(finding, "Some reason")

        warnings = recorder.get_warnings(finding)
        assert len(warnings) == 0  # Not enough occurrences

    def test_strong_warning_many_occurrences(self, tmp_path):
        recorder = FalsePositiveRecorder(tmp_path)

        finding = Mock()
        finding.pattern_id = "vm-001"
        finding.modifiers = ["nonReentrant"]

        # Record many FPs
        for _ in range(6):
            recorder.record(finding, "Has guard")

        warnings = recorder.get_warnings(finding)
        assert any("LIKELY" in w for w in warnings)
```
