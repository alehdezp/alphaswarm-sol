## Objective

Enable manual and automatic rollback of learning changes when degradation is detected. This is the SAFETY NET for the learning system.

---

## Before You Start

### Why Rollback is Critical

Without rollback:
- Bad learning data permanently degrades VKG
- No recovery path when things go wrong
- Users lose trust

With rollback:
- Can always return to known-good state
- Automatic protection against degradation
- Safe experimentation

```python
"""Rollback capability for learning changes."""
from dataclasses import dataclass
from datetime import datetime
from typing import Dict, List, Optional
import json
import shutil
from pathlib import Path


@dataclass
class LearningSnapshot:
    """A snapshot of learning state."""
    snapshot_id: str
    timestamp: datetime
    description: str
    confidence_values: Dict[str, float]  # pattern_id -> confidence
    is_baseline: bool = False


class VersionManager:
    """Manage learning state versions for rollback."""

    MAX_SNAPSHOTS = 20  # Keep last 20 snapshots

    def __init__(self, storage_path: Path):
        self.storage_path = storage_path
        self.snapshots_dir = storage_path / "snapshots"
        self.snapshots_dir.mkdir(parents=True, exist_ok=True)
        self._snapshots: List[LearningSnapshot] = []
        self._load_manifest()

    def create_snapshot(
        self,
        description: str,
        confidence_values: Dict[str, float],
        is_baseline: bool = False,
    ) -> str:
        """Create a new snapshot of learning state."""
        snapshot_id = f"snap_{datetime.now().strftime('%Y%m%d_%H%M%S')}"

        snapshot = LearningSnapshot(
            snapshot_id=snapshot_id,
            timestamp=datetime.now(),
            description=description,
            confidence_values=confidence_values,
            is_baseline=is_baseline,
        )

        # Save snapshot data
        snapshot_file = self.snapshots_dir / f"{snapshot_id}.json"
        with open(snapshot_file, "w") as f:
            json.dump({
                "snapshot_id": snapshot.snapshot_id,
                "timestamp": snapshot.timestamp.isoformat(),
                "description": snapshot.description,
                "confidence_values": snapshot.confidence_values,
                "is_baseline": snapshot.is_baseline,
            }, f, indent=2)

        self._snapshots.append(snapshot)
        self._save_manifest()
        self._cleanup_old_snapshots()

        return snapshot_id

    def get_snapshot(self, snapshot_id: str) -> Optional[LearningSnapshot]:
        """Get a specific snapshot."""
        for snap in self._snapshots:
            if snap.snapshot_id == snapshot_id:
                return snap
        return None

    def get_baseline(self) -> Optional[LearningSnapshot]:
        """Get the baseline snapshot."""
        for snap in self._snapshots:
            if snap.is_baseline:
                return snap
        return None

    def list_snapshots(self) -> List[LearningSnapshot]:
        """List all snapshots."""
        return list(self._snapshots)

    def rollback_to(self, snapshot_id: str) -> Dict[str, float]:
        """Rollback to a specific snapshot.

        Returns the confidence values from that snapshot.
        """
        snapshot = self.get_snapshot(snapshot_id)
        if snapshot is None:
            raise ValueError(f"Snapshot not found: {snapshot_id}")

        return snapshot.confidence_values

    def _load_manifest(self) -> None:
        """Load snapshot manifest."""
        manifest_file = self.storage_path / "snapshot_manifest.json"
        if not manifest_file.exists():
            return

        with open(manifest_file, "r") as f:
            data = json.load(f)

        self._snapshots = []
        for item in data["snapshots"]:
            self._snapshots.append(LearningSnapshot(
                snapshot_id=item["snapshot_id"],
                timestamp=datetime.fromisoformat(item["timestamp"]),
                description=item["description"],
                confidence_values=item.get("confidence_values", {}),
                is_baseline=item.get("is_baseline", False),
            ))

    def _save_manifest(self) -> None:
        """Save snapshot manifest."""
        manifest_file = self.storage_path / "snapshot_manifest.json"
        data = {
            "snapshots": [
                {
                    "snapshot_id": s.snapshot_id,
                    "timestamp": s.timestamp.isoformat(),
                    "description": s.description,
                    "is_baseline": s.is_baseline,
                }
                for s in self._snapshots
            ]
        }
        with open(manifest_file, "w") as f:
            json.dump(data, f, indent=2)

    def _cleanup_old_snapshots(self) -> None:
        """Remove old snapshots beyond limit."""
        # Keep baseline + most recent
        non_baseline = [s for s in self._snapshots if not s.is_baseline]
        if len(non_baseline) > self.MAX_SNAPSHOTS:
            to_remove = non_baseline[:-self.MAX_SNAPSHOTS]
            for snap in to_remove:
                snapshot_file = self.snapshots_dir / f"{snap.snapshot_id}.json"
                if snapshot_file.exists():
                    snapshot_file.unlink()
                self._snapshots.remove(snap)
            self._save_manifest()


class AutoRollback:
    """Automatic rollback on degradation detection."""

    def __init__(
        self,
        version_manager: VersionManager,
        degradation_threshold: float = 0.10,  # 10% precision drop
    ):
        self.version_manager = version_manager
        self.threshold = degradation_threshold

    def check_degradation(
        self,
        pattern_id: str,
        current_precision: float,
    ) -> Optional[str]:
        """Check if pattern has degraded and rollback if needed.

        Returns snapshot_id if rollback occurred, None otherwise.
        """
        baseline = self.version_manager.get_baseline()
        if baseline is None:
            return None

        baseline_precision = baseline.confidence_values.get(pattern_id)
        if baseline_precision is None:
            return None

        # Check for significant degradation
        if baseline_precision - current_precision > self.threshold:
            # Rollback
            values = self.version_manager.rollback_to(baseline.snapshot_id)
            return baseline.snapshot_id

        return None

    def create_alert(
        self,
        pattern_id: str,
        current: float,
        baseline: float,
    ) -> str:
        """Create degradation alert message."""
        return (
            f"DEGRADATION DETECTED: Pattern {pattern_id}\n"
            f"  Current precision: {current:.0%}\n"
            f"  Baseline precision: {baseline:.0%}\n"
            f"  Drop: {baseline - current:.0%}\n"
            f"  Threshold: {self.threshold:.0%}\n"
            f"  Action: Automatic rollback triggered"
        )


def rollback(pattern_id: str, snapshot_id: str, storage_path: Path) -> float:
    """Convenience function to rollback a pattern."""
    manager = VersionManager(storage_path)
    values = manager.rollback_to(snapshot_id)
    return values.get(pattern_id, 0.7)  # Default if not found
```

## CLI Commands

Add to `src/true_vkg/cli/learn.py`:

```python
@learn.command("rollback")
@click.argument("pattern_id")
@click.option("--snapshot", help="Specific snapshot ID to rollback to")
@click.option("--baseline", is_flag=True, help="Rollback to baseline")
def rollback_pattern(pattern_id: str, snapshot: str, baseline: bool):
    """Rollback learning for a pattern."""
    manager = VersionManager(Path(".vrs/learning"))

    if baseline:
        snap = manager.get_baseline()
        if snap is None:
            click.echo("Error: No baseline snapshot found", err=True)
            return
        snapshot = snap.snapshot_id

    if not snapshot:
        click.echo("Error: Specify --snapshot ID or --baseline", err=True)
        return

    values = manager.rollback_to(snapshot)
    if pattern_id in values:
        click.echo(f"Rolled back {pattern_id} to {values[pattern_id]:.2f}")
    else:
        click.echo(f"Warning: Pattern {pattern_id} not in snapshot")
```

"""Tests for rollback capability."""
import pytest
from true_vkg.learning.rollback import VersionManager, AutoRollback

class TestVersionManager:
    def test_create_and_retrieve_snapshot(self, tmp_path):
        manager = VersionManager(tmp_path)

        values = {"vm-001": 0.85, "auth-003": 0.75}
        snapshot_id = manager.create_snapshot("Test", values)

        snapshot = manager.get_snapshot(snapshot_id)
        assert snapshot is not None
        assert snapshot.confidence_values == values

    def test_rollback_restores_values(self, tmp_path):
        manager = VersionManager(tmp_path)

        original = {"vm-001": 0.90}
        snapshot_id = manager.create_snapshot("Original", original, is_baseline=True)

        # Rollback
        restored = manager.rollback_to(snapshot_id)
        assert restored["vm-001"] == 0.90

    def test_baseline_retrieval(self, tmp_path):
        manager = VersionManager(tmp_path)

        manager.create_snapshot("Not baseline", {"vm-001": 0.80})
        manager.create_snapshot("Baseline", {"vm-001": 0.90}, is_baseline=True)

        baseline = manager.get_baseline()
        assert baseline is not None
        assert baseline.confidence_values["vm-001"] == 0.90

class TestAutoRollback:
    def test_triggers_on_degradation(self, tmp_path):
        manager = VersionManager(tmp_path)
        manager.create_snapshot("Baseline", {"vm-001": 0.90}, is_baseline=True)

        auto = AutoRollback(manager, degradation_threshold=0.10)

        # 15% drop - should trigger
        result = auto.check_degradation("vm-001", 0.75)
        assert result is not None

    def test_no_trigger_within_threshold(self, tmp_path):
        manager = VersionManager(tmp_path)
        manager.create_snapshot("Baseline", {"vm-001": 0.90}, is_baseline=True)

        auto = AutoRollback(manager, degradation_threshold=0.10)

        # 5% drop - should not trigger
        result = auto.check_degradation("vm-001", 0.85)
        assert result is None
