## Why Empirical Bounds Matter

### The Problem with Arbitrary Bounds

Bad:
```python
# Arbitrary - no data backing
lower_bound = 0.2
upper_bound = 0.98
```

Good:
```python
# Data-driven - based on observed performance
observed_precision = 0.83
lower_bound = max(0.2, observed_precision - 0.15)  # 0.68
upper_bound = min(0.98, observed_precision + 0.05)  # 0.88
```

### Bound Properties

1. **Lower Bound**: Prevents confidence death spiral
   - Even terrible pattern shouldn't go below this
   - Should be based on worst-case acceptable precision

2. **Upper Bound**: Prevents overconfidence
   - Even great pattern shouldn't exceed this
   - Accounts for unseen edge cases

3. **Initial Value**: Starting confidence
   - Should match observed precision
   - Baseline for improvement measurement

```python
"""Empirical confidence bound derivation and management."""
from dataclasses import dataclass
from pathlib import Path
from typing import Dict, Optional, Tuple
import json
import statistics


@dataclass(frozen=True)
class PatternBounds:
    """Immutable confidence bounds for a pattern."""
    pattern_id: str
    lower: float
    upper: float
    initial: float
    observed_precision: float
    sample_size: int

    def clamp(self, value: float) -> float:
        """Clamp a confidence value to these bounds."""
        return max(self.lower, min(self.upper, value))

    def to_dict(self) -> Dict:
        return {
            "pattern_id": self.pattern_id,
            "lower": self.lower,
            "upper": self.upper,
            "initial": self.initial,
            "observed_precision": self.observed_precision,
            "sample_size": self.sample_size,
        }


class BoundsCalculator:
    """Calculate confidence bounds from benchmark data."""

    # Minimum samples for tight bounds
    MIN_SAMPLES_TIGHT = 20

    # Minimum samples for any bounds
    MIN_SAMPLES_ANY = 5

    # Absolute limits
    ABSOLUTE_MIN = 0.15
    ABSOLUTE_MAX = 0.98

    def __init__(self, baseline_data: Dict[str, Dict]):
        """
        Args:
            baseline_data: Dict from pattern_baseline.json
        """
        self.baselines = baseline_data

    def calculate_bounds(self, pattern_id: str) -> PatternBounds:
        """Calculate bounds for a single pattern."""
        if pattern_id not in self.baselines:
            return self._default_bounds(pattern_id)

        data = self.baselines[pattern_id]
        precision = data["precision"]
        sample_size = data["sample_size"]

        if sample_size < self.MIN_SAMPLES_ANY:
            return self._default_bounds(pattern_id)

        # Calculate bounds based on precision and sample size
        lower, upper = self._calculate_range(precision, sample_size)

        return PatternBounds(
            pattern_id=pattern_id,
            lower=round(lower, 3),
            upper=round(upper, 3),
            initial=round(precision, 3),
            observed_precision=round(precision, 3),
            sample_size=sample_size,
        )

    def _calculate_range(
        self,
        precision: float,
        sample_size: int
    ) -> Tuple[float, float]:
        """Calculate lower/upper bounds.

        Uses sample size to determine how tight bounds should be.
        More samples = tighter bounds.
        """
        # Base margin decreases with sample size
        if sample_size >= self.MIN_SAMPLES_TIGHT:
            # Tight bounds for well-tested patterns
            lower_margin = 0.10
            upper_margin = 0.05
        else:
            # Wider bounds for less tested patterns
            lower_margin = 0.20
            upper_margin = 0.10

        # Calculate bounds
        lower = max(self.ABSOLUTE_MIN, precision - lower_margin)
        upper = min(self.ABSOLUTE_MAX, precision + upper_margin)

        # Ensure valid range
        if lower >= upper:
            lower = precision * 0.7
            upper = min(self.ABSOLUTE_MAX, precision * 1.1)

        return lower, upper

    def _default_bounds(self, pattern_id: str) -> PatternBounds:
        """Default bounds for patterns without data."""
        return PatternBounds(
            pattern_id=pattern_id,
            lower=0.30,
            upper=0.90,
            initial=0.70,
            observed_precision=0.0,  # Unknown
            sample_size=0,
        )

    def calculate_all(self) -> Dict[str, PatternBounds]:
        """Calculate bounds for all patterns in baseline."""
        return {
            pattern_id: self.calculate_bounds(pattern_id)
            for pattern_id in self.baselines.keys()
        }


class BoundsManager:
    """Manage and persist confidence bounds."""

    def __init__(self, bounds_path: Path, baseline_path: Path):
        self.bounds_path = bounds_path
        self.baseline_path = baseline_path
        self._bounds: Dict[str, PatternBounds] = {}
        self._load()

    def _load(self) -> None:
        """Load bounds from file."""
        if self.bounds_path.exists():
            with open(self.bounds_path, "r") as f:
                data = json.load(f)
            self._bounds = {
                k: PatternBounds(**v) for k, v in data.items()
            }

    def get(self, pattern_id: str) -> PatternBounds:
        """Get bounds for a pattern."""
        if pattern_id not in self._bounds:
            # Generate on demand
            self._bounds[pattern_id] = self._generate_bounds(pattern_id)
        return self._bounds[pattern_id]

    def _generate_bounds(self, pattern_id: str) -> PatternBounds:
        """Generate bounds from baseline data."""
        if not self.baseline_path.exists():
            return BoundsCalculator({})._default_bounds(pattern_id)

        with open(self.baseline_path, "r") as f:
            baselines = json.load(f)

        calculator = BoundsCalculator(baselines)
        return calculator.calculate_bounds(pattern_id)

    def clamp(self, pattern_id: str, value: float) -> float:
        """Clamp a confidence value for a pattern."""
        bounds = self.get(pattern_id)
        return bounds.clamp(value)

    def regenerate(self) -> int:
        """Regenerate all bounds from baseline.

        Returns count of bounds generated.
        """
        if not self.baseline_path.exists():
            return 0

        with open(self.baseline_path, "r") as f:
            baselines = json.load(f)

        calculator = BoundsCalculator(baselines)
        self._bounds = calculator.calculate_all()
        self._save()
        return len(self._bounds)

    def _save(self) -> None:
        """Save bounds to file."""
        data = {k: v.to_dict() for k, v in self._bounds.items()}
        with open(self.bounds_path, "w") as f:
            json.dump(data, f, indent=2)

    def summary(self) -> str:
        """Generate summary of all bounds."""
        lines = ["# Confidence Bounds Summary", ""]
        lines.append(f"Total patterns: {len(self._bounds)}")
        lines.append("")
        lines.append("| Pattern | Lower | Initial | Upper | Samples |")
        lines.append("|---------|-------|---------|-------|---------|")

        for bounds in sorted(self._bounds.values(), key=lambda b: b.pattern_id):
            lines.append(
                f"| {bounds.pattern_id} | {bounds.lower:.2f} | "
                f"{bounds.initial:.2f} | {bounds.upper:.2f} | "
                f"{bounds.sample_size} |"
            )

        return "\n".join(lines)


# Convenience functions
def load_bounds(
    bounds_path: Optional[Path] = None,
    baseline_path: Optional[Path] = None
) -> BoundsManager:
    """Load bounds manager with default paths."""
    bounds_path = bounds_path or Path("benchmarks/confidence_bounds.json")
    baseline_path = baseline_path or Path("benchmarks/pattern_baseline.json")
    return BoundsManager(bounds_path, baseline_path)


def regenerate_bounds(
    baseline_path: Optional[Path] = None,
    output_path: Optional[Path] = None
) -> int:
    """Regenerate bounds from baseline data."""
    manager = load_bounds(output_path, baseline_path)
    return manager.regenerate()
```

## Integration with Learning

The bounds are used by the learning system to constrain confidence updates:

```python
# In learning pipeline
from .bounds import load_bounds

bounds_manager = load_bounds()

def update_confidence(pattern_id: str, current: float, adjustment: float) -> float:
    """Update confidence with bound clamping."""
    new_value = current + adjustment
    return bounds_manager.clamp(pattern_id, new_value)
```

```python
"""Tests for confidence bound derivation."""
import pytest
from pathlib import Path
import json
import tempfile

class TestBoundsCalculator:
    def test_high_precision_tight_bounds(self):
        """High precision patterns should have tight upper bounds."""
        calculator = BoundsCalculator({
            "good-pattern": {
                "precision": 0.95,
                "sample_size": 30,
            }
        })
        bounds = calculator.calculate_bounds("good-pattern")

        assert bounds.upper <= 0.98  # Can't exceed absolute max
        assert bounds.upper - bounds.initial <= 0.05  # Tight upper

    def test_low_precision_wider_bounds(self):
        """Low precision patterns should have wider bounds."""
        calculator = BoundsCalculator({
            "poor-pattern": {
                "precision": 0.50,
                "sample_size": 30,
            }
        })
        bounds = calculator.calculate_bounds("poor-pattern")

        assert bounds.lower >= 0.15  # Above absolute min
        assert bounds.initial - bounds.lower >= 0.10  # Room to drop

    def test_low_samples_default_bounds(self):
        """Patterns with few samples should get default bounds."""
        calculator = BoundsCalculator({
            "new-pattern": {
                "precision": 0.90,
                "sample_size": 3,  # Below minimum
            }

        bounds = calculator.calculate_bounds("new-pattern")

        assert bounds.sample_size == 0  # Indicates default
        assert bounds.lower == 0.30
```
