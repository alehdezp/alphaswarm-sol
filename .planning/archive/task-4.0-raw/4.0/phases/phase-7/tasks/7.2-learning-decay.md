## Why Decay Matters

### The Problem Without Decay

```
Day 0: Pattern gets 10 FPs in one buggy project
Day 365: Confidence still depressed from year-old FPs
```

### The Solution: Exponential Decay

```python
# Half-life of 30 days
decay_factor = exp(-0.693 * days_since / 30)

# Day 0:  adjustment * 1.0    = full weight
# Day 30: adjustment * 0.5    = half weight
# Day 60: adjustment * 0.25   = quarter weight
# Day 90: adjustment * 0.125  = negligible
```

```python
"""Time-based decay for learning adjustments."""
import math
from datetime import datetime, timedelta
from dataclasses import dataclass
from typing import Optional


@dataclass
class DecayConfig:
    """Configuration for decay calculations."""
    half_life_days: float = 30.0  # Days until adjustment halved
    min_factor: float = 0.01     # Minimum decay factor (below = zero)
    max_age_days: float = 180.0  # Events older than this are ignored


class DecayCalculator:
    """Calculate time-based decay factors."""

    # Decay constant for exponential decay
    # ln(2) / half_life gives correct half-life behavior
    LN2 = 0.693147

    def __init__(self, config: Optional[DecayConfig] = None):
        self.config = config or DecayConfig()
        self._decay_constant = self.LN2 / self.config.half_life_days

    def calculate_factor(self, event_time: datetime, now: Optional[datetime] = None) -> float:
        """Calculate decay factor for an event.

        Args:
            event_time: When the event occurred
            now: Current time (defaults to now)

        Returns:
            Decay factor between 0 and 1
        """
        now = now or datetime.now()
        age_days = (now - event_time).total_seconds() / 86400

        # Events too old are ignored
        if age_days > self.config.max_age_days:
            return 0.0

        # Calculate exponential decay
        factor = math.exp(-self._decay_constant * age_days)

        # Apply minimum threshold
        if factor < self.config.min_factor:
            return 0.0

        return factor

    def apply_decay(
        self,
        base_adjustment: float,
        event_time: datetime,
        now: Optional[datetime] = None
    ) -> float:
        """Apply decay to an adjustment value.

        Args:
            base_adjustment: Original adjustment (e.g., -0.05 for FP)
            event_time: When the event occurred

        Returns:
            Decayed adjustment value
        """
        factor = self.calculate_factor(event_time, now)
        return base_adjustment * factor

    def is_relevant(self, event_time: datetime, now: Optional[datetime] = None) -> bool:
        """Check if an event is still relevant (not fully decayed)."""
        return self.calculate_factor(event_time, now) > 0.0

    def days_until_negligible(self, threshold: float = 0.01) -> float:
        """Calculate days until factor falls below threshold.

        Returns:
            Number of days until decay factor < threshold
        """
        # factor = exp(-k * t)
        # threshold = exp(-k * t)
        # ln(threshold) = -k * t
        # t = -ln(threshold) / k
        return -math.log(threshold) / self._decay_constant


# Default instance
_default_calculator: Optional[DecayCalculator] = None


def get_decay_calculator() -> DecayCalculator:
    """Get default decay calculator singleton."""
    global _default_calculator
    if _default_calculator is None:
        _default_calculator = DecayCalculator()
    return _default_calculator


def apply_decay(
    adjustment: float,
    event_time: datetime
) -> float:
    """Convenience function to apply decay."""
    return get_decay_calculator().apply_decay(adjustment, event_time)
```

## Test Requirements

```python
"""Tests for learning decay."""
import pytest
from datetime import datetime, timedelta
from true_vkg.learning.decay import DecayCalculator, DecayConfig

class TestDecayCalculator:
    def test_no_decay_at_zero_days(self):
        calc = DecayCalculator()
        now = datetime.now()
        factor = calc.calculate_factor(now, now)
        assert factor == pytest.approx(1.0, abs=0.01)

    def test_half_decay_at_half_life(self):
        config = DecayConfig(half_life_days=30)
        calc = DecayCalculator(config)
        now = datetime.now()
        past = now - timedelta(days=30)
        factor = calc.calculate_factor(past, now)
        assert factor == pytest.approx(0.5, abs=0.01)

    def test_quarter_decay_at_two_half_lives(self):
        config = DecayConfig(half_life_days=30)
        calc = DecayCalculator(config)
        now = datetime.now()
        past = now - timedelta(days=60)
        factor = calc.calculate_factor(past, now)
        assert factor == pytest.approx(0.25, abs=0.01)

    def test_zero_at_max_age(self):
        config = DecayConfig(max_age_days=180)
        calc = DecayCalculator(config)
        now = datetime.now()
        past = now - timedelta(days=200)
        factor = calc.calculate_factor(past, now)
        assert factor == 0.0

    def test_apply_decay(self):
        calc = DecayCalculator(DecayConfig(half_life_days=30))
        now = datetime.now()
        past = now - timedelta(days=30)

        decayed = calc.apply_decay(-0.05, past, now)
        assert decayed == pytest.approx(-0.025, abs=0.001)
```
