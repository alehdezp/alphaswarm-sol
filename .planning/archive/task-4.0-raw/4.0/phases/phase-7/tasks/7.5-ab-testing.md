"""A/B testing infrastructure for pattern configurations."""
from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict, List, Optional
from enum import Enum
import hashlib
import json
from pathlib import Path
import statistics


class Variant(Enum):
    """A/B test variants."""
    CONTROL = "A"
    TREATMENT = "B"


@dataclass
class ABTestConfig:
    """Configuration for an A/B test."""
    test_id: str
    pattern_id: str
    treatment_config: Dict           # The variant B configuration
    traffic_fraction: float = 0.10   # 10% to treatment
    min_samples: int = 20            # Min samples for significance
    start_time: datetime = field(default_factory=datetime.now)
    end_time: Optional[datetime] = None


@dataclass
class ABTestResult:
    """Results of an A/B test."""
    test_id: str
    control_samples: int
    treatment_samples: int
    control_precision: float
    treatment_precision: float
    control_verdicts: Dict[str, int]   # {confirmed: N, rejected: N}
    treatment_verdicts: Dict[str, int]
    is_significant: bool
    p_value: Optional[float]
    winner: Optional[Variant]


class PatternABTest:
    """Manage A/B test for a pattern."""

    def __init__(self, config: ABTestConfig, storage_path: Path):
        self.config = config
        self.storage_path = storage_path
        self._results: Dict[str, Dict] = {"A": {}, "B": {}}
        self._load()

    def get_variant(self, finding_id: str) -> Variant:
        """Determine variant for a finding.

        Uses consistent hashing so same finding always gets same variant.
        """
        # Consistent hash assignment
        hash_input = f"{self.config.test_id}:{finding_id}"
        hash_value = int(hashlib.md5(hash_input.encode()).hexdigest(), 16)
        fraction = (hash_value % 1000) / 1000.0

        if fraction < self.config.traffic_fraction:
            return Variant.TREATMENT
        return Variant.CONTROL

    def record_verdict(
        self,
        finding_id: str,
        variant: Variant,
        verdict: str,  # "confirmed" or "rejected"
    ) -> None:
        """Record a verdict for a finding."""
        var_key = variant.value
        self._results[var_key][finding_id] = {
            "verdict": verdict,
            "timestamp": datetime.now().isoformat(),
        }
        self._save()

    def get_results(self) -> ABTestResult:
        """Calculate test results."""
        control = self._results["A"]
        treatment = self._results["B"]

        control_verdicts = self._count_verdicts(control)
        treatment_verdicts = self._count_verdicts(treatment)

        control_precision = self._calculate_precision(control_verdicts)
        treatment_precision = self._calculate_precision(treatment_verdicts)

        # Check statistical significance
        is_significant, p_value = self._check_significance(
            control_verdicts, treatment_verdicts
        )

        # Determine winner
        winner = None
        if is_significant:
            if treatment_precision > control_precision:
                winner = Variant.TREATMENT
            elif control_precision > treatment_precision:
                winner = Variant.CONTROL

        return ABTestResult(
            test_id=self.config.test_id,
            control_samples=len(control),
            treatment_samples=len(treatment),
            control_precision=control_precision,
            treatment_precision=treatment_precision,
            control_verdicts=control_verdicts,
            treatment_verdicts=treatment_verdicts,
            is_significant=is_significant,
            p_value=p_value,
            winner=winner,
        )

    def is_complete(self) -> bool:
        """Check if test has enough samples."""
        return (
            len(self._results["A"]) >= self.config.min_samples and
            len(self._results["B"]) >= max(5, self.config.min_samples * self.config.traffic_fraction)
        )

    def _count_verdicts(self, results: Dict) -> Dict[str, int]:
        counts = {"confirmed": 0, "rejected": 0}
        for r in results.values():
            verdict = r["verdict"]
            if verdict in counts:
                counts[verdict] += 1
        return counts

    def _calculate_precision(self, verdicts: Dict[str, int]) -> float:
        tp = verdicts.get("confirmed", 0)
        fp = verdicts.get("rejected", 0)
        if tp + fp == 0:
            return 0.5  # No data
        return tp / (tp + fp)

    def _check_significance(
        self,
        control: Dict[str, int],
        treatment: Dict[str, int]
    ) -> tuple[bool, Optional[float]]:
        """Check if difference is statistically significant.

        Uses simple proportion comparison - could use chi-square or
        Fisher's exact test for more rigorous analysis.
        """
        # Simple approach: require min samples and 10% difference
        control_total = sum(control.values())
        treatment_total = sum(treatment.values())

        if control_total < 10 or treatment_total < 5:
            return False, None

        control_precision = self._calculate_precision(control)
        treatment_precision = self._calculate_precision(treatment)

        diff = abs(control_precision - treatment_precision)

        # Very simple significance check
        # In production, use proper statistical test
        if diff > 0.15 and min(control_total, treatment_total) >= 10:
            return True, 0.05  # Placeholder p-value
        return False, None

    def _load(self) -> None:
        """Load results from storage."""
        results_file = self.storage_path / f"ab_test_{self.config.test_id}.json"
        if results_file.exists():
            with open(results_file, "r") as f:
                self._results = json.load(f)

    def _save(self) -> None:
        """Save results to storage."""
        self.storage_path.mkdir(parents=True, exist_ok=True)
        results_file = self.storage_path / f"ab_test_{self.config.test_id}.json"
        with open(results_file, "w") as f:
            json.dump(self._results, f, indent=2)


class ABTestManager:
    """Manage multiple A/B tests."""

    def __init__(self, storage_path: Path):
        self.storage_path = storage_path
        self._tests: Dict[str, PatternABTest] = {}

    def create_test(
        self,
        pattern_id: str,
        treatment_config: Dict,
        traffic_fraction: float = 0.10,
    ) -> str:
        """Create a new A/B test."""
        test_id = f"test_{pattern_id}_{datetime.now().strftime('%Y%m%d')}"
        config = ABTestConfig(
            test_id=test_id,
            pattern_id=pattern_id,
            treatment_config=treatment_config,
            traffic_fraction=traffic_fraction,
        )
        self._tests[test_id] = PatternABTest(config, self.storage_path)
        return test_id

    def get_test(self, test_id: str) -> Optional[PatternABTest]:
        return self._tests.get(test_id)

    def get_active_test(self, pattern_id: str) -> Optional[PatternABTest]:
        """Get active test for a pattern."""
        for test in self._tests.values():
            if test.config.pattern_id == pattern_id and not test.config.end_time:
                return test
        return None

    def end_test(self, test_id: str) -> Optional[ABTestResult]:
        """End a test and return results."""
        test = self._tests.get(test_id)
        if test:
            test.config.end_time = datetime.now()
            return test.get_results()
        return None

"""Tests for A/B testing."""
import pytest
from true_vkg.learning.ab_testing import (
    PatternABTest, ABTestConfig, ABTestManager, Variant
)

class TestPatternABTest:
    def test_consistent_assignment(self, tmp_path):
        config = ABTestConfig(
            test_id="test1",
            pattern_id="vm-001",
            treatment_config={},
            traffic_fraction=0.10,
        )
        test = PatternABTest(config, tmp_path)

        # Same finding should always get same variant
        finding_id = "VKG-001"
        v1 = test.get_variant(finding_id)
        v2 = test.get_variant(finding_id)
        assert v1 == v2

    def test_traffic_split(self, tmp_path):
        config = ABTestConfig(
            test_id="test1",
            pattern_id="vm-001",
            treatment_config={},
            traffic_fraction=0.10,
        )
        test = PatternABTest(config, tmp_path)

        # Check rough traffic split
        treatment_count = 0
        for i in range(1000):
            if test.get_variant(f"finding-{i}") == Variant.TREATMENT:
                treatment_count += 1

        # Should be roughly 10% +/- 5%
        assert 50 < treatment_count < 150

    def test_record_and_results(self, tmp_path):
        config = ABTestConfig(
            test_id="test1",
            pattern_id="vm-001",
            treatment_config={},
        )
        test = PatternABTest(config, tmp_path)

        # Record some verdicts
        test.record_verdict("f1", Variant.CONTROL, "confirmed")
        test.record_verdict("f2", Variant.CONTROL, "rejected")
        test.record_verdict("f3", Variant.TREATMENT, "confirmed")

        results = test.get_results()
        assert results.control_samples == 2
        assert results.treatment_samples == 1
