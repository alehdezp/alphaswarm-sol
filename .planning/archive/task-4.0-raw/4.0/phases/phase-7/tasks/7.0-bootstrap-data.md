- No way to measure improvement
- No bounds for confidence adjustments
- No fallback when learning fails

```python
"""Bootstrap confidence bounds from benchmark data."""
from dataclasses import dataclass
from pathlib import Path
from typing import Dict, List, Any
import json

from ..kg.builder import build_graph
from ..queries.patterns import PatternMatcher


@dataclass
class PatternBaseline:
    """Baseline metrics for a single pattern."""
    pattern_id: str
    true_positives: int
    false_positives: int
    false_negatives: int
    precision: float
    recall: float
    f1_score: float
    sample_size: int


@dataclass
class ConfidenceBounds:
    """Confidence bounds for a pattern."""
    pattern_id: str
    lower_bound: float      # Never go below this
    upper_bound: float      # Never exceed this
    initial: float          # Starting confidence
    observed_precision: float


class BootstrapGenerator:
    """Generate bootstrap data from test contracts."""

    # Minimum sample size for reliable bounds
    MIN_SAMPLES = 5

    # Default bounds when insufficient data
    DEFAULT_LOWER = 0.3
    DEFAULT_UPPER = 0.95
    DEFAULT_INITIAL = 0.7

    def __init__(
        self,
        test_projects_path: Path,
        patterns_path: Path,
    ):
        self.test_projects = test_projects_path
        self.patterns_path = patterns_path

    def generate_baselines(self) -> Dict[str, PatternBaseline]:
        """Generate baseline metrics for all patterns.

        Returns:
            Dict mapping pattern_id to PatternBaseline
        """
        baselines = {}

        # Collect results from all test projects
        results = self._run_on_test_projects()

        # Aggregate per pattern
        for pattern_id, outcomes in results.items():
            tp = sum(1 for o in outcomes if o["expected"] and o["detected"])
            fp = sum(1 for o in outcomes if not o["expected"] and o["detected"])
            fn = sum(1 for o in outcomes if o["expected"] and not o["detected"])

            precision = tp / (tp + fp) if (tp + fp) > 0 else 0.5
            recall = tp / (tp + fn) if (tp + fn) > 0 else 0.5
            f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

            baselines[pattern_id] = PatternBaseline(
                pattern_id=pattern_id,
                true_positives=tp,
                false_positives=fp,
                false_negatives=fn,
                precision=precision,
                recall=recall,
                f1_score=f1,
                sample_size=len(outcomes),
            )

        return baselines

    def generate_confidence_bounds(
        self,
        baselines: Dict[str, PatternBaseline]
    ) -> Dict[str, ConfidenceBounds]:
        """Generate confidence bounds from baselines.

        Strategy:
        - Use observed precision as initial confidence
        - Set bounds based on sample size
        - More samples = tighter bounds
        """
        bounds = {}

        for pattern_id, baseline in baselines.items():
            if baseline.sample_size < self.MIN_SAMPLES:
                # Insufficient data - use defaults
                bounds[pattern_id] = ConfidenceBounds(
                    pattern_id=pattern_id,
                    lower_bound=self.DEFAULT_LOWER,
                    upper_bound=self.DEFAULT_UPPER,
                    initial=self.DEFAULT_INITIAL,
                    observed_precision=baseline.precision,
                )
            else:
                # Calculate bounds from observed data
                # Wider bounds for less precise patterns
                precision = baseline.precision

                # Lower bound: can't drop below half of precision
                lower = max(0.2, precision * 0.5)

                # Upper bound: can't exceed precision + margin
                # Smaller margin for higher precision
                margin = 0.1 if precision > 0.8 else 0.15
                upper = min(0.98, precision + margin)

                bounds[pattern_id] = ConfidenceBounds(
                    pattern_id=pattern_id,
                    lower_bound=round(lower, 2),
                    upper_bound=round(upper, 2),
                    initial=round(precision, 2),
                    observed_precision=round(precision, 2),
                )

        return bounds

    def _run_on_test_projects(self) -> Dict[str, List[Dict]]:
        """Run patterns on test projects and collect outcomes.

        Returns:
            Dict mapping pattern_id to list of outcome dicts
        """
        results = {}

        # Find all test projects with MANIFEST.yaml
        for manifest_path in self.test_projects.glob("**/MANIFEST.yaml"):
            project_path = manifest_path.parent
            outcomes = self._run_on_project(project_path)
            for pattern_id, project_outcomes in outcomes.items():
                if pattern_id not in results:
                    results[pattern_id] = []
                results[pattern_id].extend(project_outcomes)

        return results

    def _run_on_project(self, project_path: Path) -> Dict[str, List[Dict]]:
        """Run patterns on a single project."""
        # Load manifest for ground truth
        manifest = self._load_manifest(project_path / "MANIFEST.yaml")

        # Build graph
        try:
            graph = build_graph(str(project_path))
        except Exception as e:
            print(f"Warning: Failed to build graph for {project_path}: {e}")
            return {}

        # Run pattern matching
        matcher = PatternMatcher(graph)
        matches = matcher.match_all()

        # Compare to ground truth
        outcomes = {}
        for result in matches:
            pattern_id = result.pattern_id
            if pattern_id not in outcomes:
                outcomes[pattern_id] = []

            for match in result.matches:
                expected = self._is_expected(manifest, pattern_id, match)
                outcomes[pattern_id].append({
                    "expected": expected,
                    "detected": True,
                    "function": match.function_id,
                })

        return outcomes

    def _load_manifest(self, path: Path) -> Dict:
        """Load project manifest."""
        import yaml
        with open(path, "r") as f:
            return yaml.safe_load(f)

    def _is_expected(self, manifest: Dict, pattern_id: str, match) -> bool:
        """Check if match is expected based on manifest."""
        # TODO: Implement based on actual manifest structure
        # This needs to map pattern findings to manifest ground truth
        return True  # Default - treat as expected

    def save_baselines(
        self,
        baselines: Dict[str, PatternBaseline],
        output_path: Path
    ) -> None:
        """Save baselines to JSON."""
        data = {
            pattern_id: {
                "pattern_id": b.pattern_id,
                "true_positives": b.true_positives,
                "false_positives": b.false_positives,
                "false_negatives": b.false_negatives,
                "precision": b.precision,
                "recall": b.recall,
                "f1_score": b.f1_score,
                "sample_size": b.sample_size,
            }
            for pattern_id, b in baselines.items()
        }
        with open(output_path, "w") as f:
            json.dump(data, f, indent=2)

    def save_confidence_bounds(
        self,
        bounds: Dict[str, ConfidenceBounds],
        output_path: Path
    ) -> None:
        """Save confidence bounds to JSON."""
        data = {
            pattern_id: {
                "pattern_id": b.pattern_id,
                "lower_bound": b.lower_bound,
                "upper_bound": b.upper_bound,
                "initial": b.initial,
                "observed_precision": b.observed_precision,
            }
            for pattern_id, b in bounds.items()
        }
        with open(output_path, "w") as f:
            json.dump(data, f, indent=2)


def generate_bootstrap_data(
    test_projects: Path,
    patterns: Path,
    output_dir: Path
) -> None:
    """Generate all bootstrap data.

    Usage:
        generate_bootstrap_data(
            Path("tests/projects"),
            Path("patterns/core"),
            Path("benchmarks")
        )
    """
    output_dir.mkdir(parents=True, exist_ok=True)

    generator = BootstrapGenerator(test_projects, patterns)

    print("Generating baselines...")
    baselines = generator.generate_baselines()
    generator.save_baselines(baselines, output_dir / "pattern_baseline.json")
    print(f"  Saved {len(baselines)} pattern baselines")

    print("Generating confidence bounds...")
    bounds = generator.generate_confidence_bounds(baselines)
    generator.save_confidence_bounds(bounds, output_dir / "confidence_bounds.json")
    print(f"  Saved {len(bounds)} confidence bounds")

    print("Bootstrap complete!")


# CLI entry point
if __name__ == "__main__":
    from pathlib import Path
    generate_bootstrap_data(
        Path("tests/projects"),
        Path("patterns/core"),
        Path("benchmarks")
    )
```

```python
"""Tests for bootstrap data generation."""
import pytest
from pathlib import Path
import json
import tempfile

from true_vkg.learning.bootstrap import (
    BootstrapGenerator, generate_bootstrap_data,
    PatternBaseline, ConfidenceBounds
)

class TestBootstrapGenerator:
    def test_bounds_within_limits(self):
        """All bounds should be reasonable."""
        generator = BootstrapGenerator(
            Path("tests/projects"),
            Path("patterns/core")
        )
        baselines = generator.generate_baselines()
        bounds = generator.generate_confidence_bounds(baselines)

        for b in bounds.values():
            assert 0.0 <= b.lower_bound <= 1.0
            assert 0.0 <= b.upper_bound <= 1.0
            assert b.lower_bound < b.upper_bound
            assert b.lower_bound <= b.initial <= b.upper_bound

    def test_low_sample_uses_defaults(self):
        """Patterns with few samples should use defaults."""
        baseline = PatternBaseline(
            pattern_id="test",
            true_positives=2,
            false_positives=1,
            false_negatives=0,
            precision=0.67,
            recall=1.0,
            f1_score=0.8,
            sample_size=3  # Below MIN_SAMPLES
        )
        generator = BootstrapGenerator(Path("."), Path("."))
        bounds = generator.generate_confidence_bounds({"test": baseline})

        assert bounds["test"].lower_bound == generator.DEFAULT_LOWER
        assert bounds["test"].upper_bound == generator.DEFAULT_UPPER

    def test_json_output_valid(self, tmp_path):
        """Output files should be valid JSON."""
        generate_bootstrap_data(
            Path("tests/projects"),
            Path("patterns/core"),
            tmp_path
        )

        # Verify files exist and are valid JSON
        baseline_path = tmp_path / "pattern_baseline.json"
        bounds_path = tmp_path / "confidence_bounds.json"

        assert baseline_path.exists()
        assert bounds_path.exists()

        # Should not raise
        with open(baseline_path) as f:
            json.load(f)
        with open(bounds_path) as f:
            json.load(f)
```
