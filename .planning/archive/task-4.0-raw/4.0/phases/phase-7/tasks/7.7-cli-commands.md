"""CLI commands for learning control."""
import json
import sys
from pathlib import Path
from typing import Optional

import click


# Default learning data location
DEFAULT_LEARNING_DIR = Path(".vrs/learning")


@click.group()
def learn():
    """Manage conservative learning system."""
    pass


@learn.command("enable")
def enable_learning():
    """Enable learning system."""
    config_path = DEFAULT_LEARNING_DIR / "config.json"
    config = _load_config(config_path)
    config["enabled"] = True
    _save_config(config_path, config)
    click.echo("Learning ENABLED")
    click.echo("  Verdicts will now update pattern confidence")


@learn.command("disable")
def disable_learning():
    """Disable learning system."""
    config_path = DEFAULT_LEARNING_DIR / "config.json"
    config = _load_config(config_path)
    config["enabled"] = False
    _save_config(config_path, config)
    click.echo("Learning DISABLED")
    click.echo("  Verdicts will not affect pattern confidence")


@learn.command("status")
def learning_status():
    """Show learning system status."""
    config_path = DEFAULT_LEARNING_DIR / "config.json"
    config = _load_config(config_path)

    from ..learning.events import EventStore
    store = EventStore(DEFAULT_LEARNING_DIR)
    recent = store.get_recent_events(days=30)

    click.echo(f"\n## Learning Status")
    click.echo(f"  Enabled: {config.get('enabled', False)}")
    click.echo(f"  Decay half-life: {config.get('decay_half_life_days', 30)} days")
    click.echo(f"  Auto-rollback threshold: {config.get('auto_rollback_threshold', 0.10):.0%}")
    click.echo(f"\n## Recent Activity (30 days)")
    click.echo(f"  Total events: {len(recent)}")

    # Count by type
    by_type = {}
    for e in recent:
        t = e.event_type.value
        by_type[t] = by_type.get(t, 0) + 1

    for t, count in by_type.items():
        click.echo(f"  {t}: {count}")


@learn.command("stats")
@click.argument("pattern_id", required=False)
def learning_stats(pattern_id: Optional[str]):
    """Show learning stats, optionally for a specific pattern."""
    from ..learning.events import EventStore
    from ..learning.bounds import load_bounds

    store = EventStore(DEFAULT_LEARNING_DIR)
    bounds = load_bounds()

    if pattern_id:
        _show_pattern_stats(pattern_id, store, bounds)
    else:
        _show_all_stats(store, bounds)


def _show_pattern_stats(pattern_id: str, store, bounds):
    """Show stats for one pattern."""
    events = store.get_events_for_pattern(pattern_id)
    counts = store.count_by_type(pattern_id)
    pattern_bounds = bounds.get(pattern_id)

    click.echo(f"\n## Pattern: {pattern_id}")
    click.echo(f"\n### Confidence Bounds")
    click.echo(f"  Lower: {pattern_bounds.lower:.2f}")
    click.echo(f"  Current: {pattern_bounds.initial:.2f}")
    click.echo(f"  Upper: {pattern_bounds.upper:.2f}")

    click.echo(f"\n### Learning Events")
    click.echo(f"  Confirmed: {counts.get('confirmed', 0)}")
    click.echo(f"  Rejected: {counts.get('rejected', 0)}")
    click.echo(f"  Total: {len(events)}")

    # Recent events
    click.echo(f"\n### Recent Events")
    for event in events[-5:]:
        click.echo(f"  {event.timestamp.strftime('%Y-%m-%d')}: {event.event_type.value} - {event.reason[:50]}")


def _show_all_stats(store, bounds):
    """Show summary stats for all patterns."""
    recent = store.get_recent_events(days=30)

    # Group by pattern
    by_pattern = {}
    for e in recent:
        if e.pattern_id not in by_pattern:
            by_pattern[e.pattern_id] = []
        by_pattern[e.pattern_id].append(e)

    click.echo(f"\n## Learning Stats (30 days)")
    click.echo(f"\n| Pattern | Confirmed | Rejected | Current |")
    click.echo(f"|---------|-----------|----------|---------|")

    for pattern_id, events in sorted(by_pattern.items()):
        confirmed = sum(1 for e in events if e.event_type.value == "confirmed")
        rejected = sum(1 for e in events if e.event_type.value == "rejected")
        current = bounds.get(pattern_id).initial
        click.echo(f"| {pattern_id[:20]} | {confirmed} | {rejected} | {current:.2f} |")


@learn.command("history")
@click.option("--limit", default=20, help="Max events to show")
@click.option("--pattern", help="Filter by pattern ID")
def learning_history(limit: int, pattern: Optional[str]):
    """Show learning history."""
    from ..learning.events import EventStore

    store = EventStore(DEFAULT_LEARNING_DIR)

    if pattern:
        events = store.get_events_for_pattern(pattern)
    else:
        events = store.get_recent_events(days=90)

    events = events[-limit:]

    click.echo(f"\n## Learning History ({len(events)} events)")
    for event in events:
        click.echo(
            f"  {event.timestamp.strftime('%Y-%m-%d %H:%M')} | "
            f"{event.pattern_id} | {event.event_type.value} | "
            f"{event.reason[:40]}"
        )


@learn.command("reset")
@click.argument("pattern_id")
@click.option("--confirm", is_flag=True, help="Confirm reset")
def reset_pattern(pattern_id: str, confirm: bool):
    """Reset learning for a pattern to baseline."""
    if not confirm:
        click.echo(f"This will reset {pattern_id} to baseline confidence.")
        click.echo("Run with --confirm to proceed.")
        return

    from ..learning.rollback import VersionManager

    manager = VersionManager(DEFAULT_LEARNING_DIR)
    baseline = manager.get_baseline()

    if baseline and pattern_id in baseline.confidence_values:
        value = baseline.confidence_values[pattern_id]
        click.echo(f"Reset {pattern_id} to baseline: {value:.2f}")
    else:
        click.echo(f"No baseline found for {pattern_id}")


@learn.command("export")
@click.argument("output_file", type=click.Path())

    }

    click.echo(f"Exported to {output_file}")

def _load_config(path: Path) -> dict:
    """Load learning config."""
    path.parent.mkdir(parents=True, exist_ok=True)
    if path.exists():
        with open(path, "r") as f:
            return json.load(f)
    return {
        "enabled": False,
        "decay_half_life_days": 30,
        "auto_rollback_threshold": 0.10,
    }


def _save_config(path: Path, config: dict) -> None:
    """Save learning config."""
    path.parent.mkdir(parents=True, exist_ok=True)
    with open(path, "w") as f:
        json.dump(config, f, indent=2)
