## 2. CONTEXT

### 2.1 What is a Grimoire?

A Grimoire is a **per-vulnerability testing playbook** that encodes expert knowledge for verification:

```
Finding: "Potential reentrancy in withdraw()"
                │
                ▼
┌─────────────────────────────────────┐
│     Reentrancy Grimoire             │
│                                     │
│  Step 1: Analyze call pattern       │
│  Step 2: Generate Foundry test      │
│  Step 3: Run forge test             │
│  Step 4: Interpret results          │
│                                     │
│  Output: CONFIRMED/REJECTED         │
└─────────────────────────────────────┘
                │
                ▼
        Structured Verdict
```

### 3.2 Base Classes

```python
# src/true_vkg/grimoires/base.py

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from enum import Enum
from typing import Dict, Any, List, Optional
import yaml

class StepType(Enum):
    ANALYSIS = "analysis"
    CODE_GENERATION = "code_generation"
    BASH = "bash"
    DECISION = "decision"

@dataclass
class GrimoireStep:
    """A single step in a grimoire."""
    name: str
    type: StepType
    description: Optional[str] = None
    prompt: Optional[str] = None  # For analysis/code_generation
    command: Optional[str] = None  # For bash
    output_file: Optional[str] = None  # For code_generation
    condition: Optional[str] = None  # For decision

@dataclass
class GrimoireContext:
    """Context passed to a grimoire execution."""
    finding: Any  # Finding being verified
    code: str  # Source code
    graph_context: Dict[str, Any]  # From knowledge graph
    working_dir: str  # Directory for generated files

@dataclass
class StepResult:
    """Result of a single step execution."""
    step_name: str
    success: bool
    output: str
    tokens_used: int = 0
    cost_usd: float = 0.0
    error: Optional[str] = None

@dataclass
class GrimoireResult:
    """Final result of grimoire execution."""
    grimoire_id: str
    verdict: str  # CONFIRMED, REJECTED, UNCERTAIN
    confidence: float  # 0-100
    evidence: List[str]
    reasoning: str

    # Generated artifacts
    test_file: Optional[str] = None
    test_result: Optional[str] = None

    # Execution info
    steps_executed: List[StepResult] = field(default_factory=list)
    total_tokens: int = 0
    total_cost_usd: float = 0.0
    iterations: int = 1

    # Error info
    error: Optional[str] = None

class Grimoire(ABC):
    """Base class for all grimoires."""

    def __init__(self, definition: Dict[str, Any], llm_provider=None):
        """
        Initialize grimoire from YAML definition.

        Args:
            definition: Parsed YAML grimoire definition
            llm_provider: LLM provider for analysis steps
        """
        self.id = definition["id"]
        self.name = definition["name"]
        self.description = definition.get("description", "")
        self.vulnerability_classes = definition.get("vulnerability_classes", [])
        self.required_context = definition.get("required_context", [])
        self.steps = self._parse_steps(definition.get("steps", []))
        self.output_schema = definition.get("output_schema", {})
        self.tools_needed = definition.get("tools_needed", [])
        self.max_budget_usd = definition.get("max_budget_usd", 1.50)
        self.max_iterations = definition.get("max_iterations", 3)

        self.llm_provider = llm_provider
        self.total_cost = 0.0

    def _parse_steps(self, steps: List[Dict]) -> List[GrimoireStep]:
        """Parse step definitions from YAML."""
        return [
            GrimoireStep(
                name=s["name"],
                type=StepType(s["type"]),
                description=s.get("description"),
                prompt=s.get("prompt"),
                command=s.get("command"),
                output_file=s.get("output_file"),
                condition=s.get("condition"),
            )
            for s in steps
        ]

    async def execute(self, context: GrimoireContext) -> GrimoireResult:
        """
        Execute the grimoire.

        Args:
            context: Execution context with finding and code

        Returns:
            GrimoireResult with verdict and evidence
        """
        step_results = []

        for step in self.steps:
            # Check budget
            if self.total_cost >= self.max_budget_usd:
                return self._budget_exceeded_result(step_results)

            # Execute step
            result = await self._execute_step(step, context, step_results)
            step_results.append(result)

            # Check for early termination
            if not result.success and step.type != StepType.DECISION:
                return self._step_failed_result(step_results, result)

        # Parse final result from steps
        return self._finalize_result(step_results, context)

    async def _execute_step(
        self,
        step: GrimoireStep,
        context: GrimoireContext,
        previous_results: List[StepResult],
    ) -> StepResult:
        """Execute a single step."""
        try:
            if step.type == StepType.ANALYSIS:
                return await self._execute_analysis(step, context, previous_results)
            elif step.type == StepType.CODE_GENERATION:
                return await self._execute_code_gen(step, context, previous_results)
            elif step.type == StepType.BASH:
                return await self._execute_bash(step, context)
            elif step.type == StepType.DECISION:
                return self._execute_decision(step, previous_results)
            else:
                raise ValueError(f"Unknown step type: {step.type}")
        except Exception as e:
            return StepResult(
                step_name=step.name,
                success=False,
                output="",
                error=str(e),
            )

    async def _execute_analysis(
        self,
        step: GrimoireStep,
        context: GrimoireContext,
        previous_results: List[StepResult],
    ) -> StepResult:
        """Execute an analysis step using LLM."""
        # Format prompt with context
        prompt = self._format_prompt(step.prompt, context, previous_results)

        # Call LLM
        response = await self.llm_provider.analyze(
            context=context.code,
            prompt=prompt,
        )

        cost = response.get("cost_usd", 0.0)
        self.total_cost += cost

        return StepResult(
            step_name=step.name,
            success=True,
            output=response.get("content", ""),
            tokens_used=response.get("tokens", 0),
            cost_usd=cost,
        )

    async def _execute_code_gen(
        self,
        step: GrimoireStep,
        context: GrimoireContext,
        previous_results: List[StepResult],
    ) -> StepResult:
        """Execute a code generation step."""
        # Similar to analysis but writes to file
        prompt = self._format_prompt(step.prompt, context, previous_results)

        response = await self.llm_provider.analyze(
            context=context.code,
            prompt=prompt,
        )

        code = response.get("content", "")

        # Extract code block if present
        if "```" in code:
            code = self._extract_code_block(code)

        # Write to file
        if step.output_file:
            output_path = f"{context.working_dir}/{step.output_file}"
            with open(output_path, "w") as f:
                f.write(code)

        cost = response.get("cost_usd", 0.0)
        self.total_cost += cost

        return StepResult(
            step_name=step.name,
            success=True,
            output=code,
            tokens_used=response.get("tokens", 0),
            cost_usd=cost,
        )

    async def _execute_bash(
        self,
        step: GrimoireStep,
        context: GrimoireContext,
    ) -> StepResult:
        """Execute a bash command."""
        import subprocess

        command = step.command.format(
            working_dir=context.working_dir,
            **context.graph_context,
        )

        try:
            result = subprocess.run(
                command,
                shell=True,
                capture_output=True,
                text=True,
                timeout=120,
                cwd=context.working_dir,
            )

            success = result.returncode == 0
            output = result.stdout if success else f"{result.stdout}\n{result.stderr}"

            return StepResult(
                step_name=step.name,
                success=success,
                output=output,
            )
        except subprocess.TimeoutExpired:
            return StepResult(
                step_name=step.name,
                success=False,
                output="",
                error="Command timed out",
            )

    def _execute_decision(
        self,
        step: GrimoireStep,
        previous_results: List[StepResult],
    ) -> StepResult:
        """Execute a decision step (conditional branching)."""
        # Evaluate condition based on previous results
        # This is simple for now - just checks if previous succeeded
        condition_met = all(r.success for r in previous_results)

        return StepResult(
            step_name=step.name,
            success=True,
            output=f"Condition: {condition_met}",
        )

    def _format_prompt(
        self,
        template: str,
        context: GrimoireContext,
        previous_results: List[StepResult],
    ) -> str:
        """Format a prompt template with context."""
        # Build replacement dict
        replacements = {
            "code": context.code,
            "function_code": context.code,
            **context.graph_context,
        }

        # Add previous step outputs
        for result in previous_results:
            replacements[f"{result.step_name}_output"] = result.output

        # Replace placeholders
        formatted = template
        for key, value in replacements.items():
            formatted = formatted.replace(f"{{{key}}}", str(value))

        return formatted

    def _extract_code_block(self, text: str) -> str:
        """Extract code from markdown code block."""
        import re
        match = re.search(r"```(?:\w+)?\n(.*?)```", text, re.DOTALL)
        if match:
            return match.group(1).strip()
        return text

    def _finalize_result(
        self,
        step_results: List[StepResult],
        context: GrimoireContext,
    ) -> GrimoireResult:
        """Parse final result from step outputs."""
        # Default implementation - subclasses can override
        last_output = step_results[-1].output if step_results else ""

        # Try to parse as JSON
        try:
            import json
            parsed = json.loads(last_output)
            return GrimoireResult(
                grimoire_id=self.id,
                verdict=parsed.get("verdict", "UNCERTAIN"),
                confidence=parsed.get("confidence", 50),
                evidence=parsed.get("evidence", []),
                reasoning=parsed.get("reasoning", ""),
                steps_executed=step_results,
                total_tokens=sum(r.tokens_used for r in step_results),
                total_cost_usd=self.total_cost,
            )
        except json.JSONDecodeError:
            # Fallback
            return GrimoireResult(
                grimoire_id=self.id,
                verdict="UNCERTAIN",
                confidence=50,
                evidence=[],
                reasoning=last_output[:500],
                steps_executed=step_results,
                total_tokens=sum(r.tokens_used for r in step_results),
                total_cost_usd=self.total_cost,
            )

    def _budget_exceeded_result(
        self,
        step_results: List[StepResult],
    ) -> GrimoireResult:
        """Return result when budget exceeded."""
        return GrimoireResult(
            grimoire_id=self.id,
            verdict="UNCERTAIN",
            confidence=0,
            evidence=[],
            reasoning="Budget exceeded before completion",
            steps_executed=step_results,
            total_tokens=sum(r.tokens_used for r in step_results),
            total_cost_usd=self.total_cost,
            error="Budget exceeded",
        )

    def _step_failed_result(
        self,
        step_results: List[StepResult],
        failed_step: StepResult,
    ) -> GrimoireResult:
        """Return result when step fails."""
        return GrimoireResult(
            grimoire_id=self.id,
            verdict="UNCERTAIN",
            confidence=0,
            evidence=[],
            reasoning=f"Step '{failed_step.step_name}' failed: {failed_step.error}",
            steps_executed=step_results,
            total_tokens=sum(r.tokens_used for r in step_results),
            total_cost_usd=self.total_cost,
            error=failed_step.error,
        )
```

### 3.3 Grimoire Loader

```python
# src/true_vkg/grimoires/loader.py

import os
from pathlib import Path
from typing import Dict, Optional
import yaml

from true_vkg.grimoires.base import Grimoire

class GrimoireLoader:
    """Load grimoires from YAML files."""

    def __init__(self, grimoire_dir: str = "grimoires"):
        """
        Initialize loader with grimoire directory.

        Args:
            grimoire_dir: Directory containing grimoire YAML files
        """
        self.grimoire_dir = Path(grimoire_dir)
        self._cache: Dict[str, Grimoire] = {}

    def load(self, grimoire_id: str, llm_provider=None) -> Optional[Grimoire]:
        """
        Load a grimoire by ID.

        Args:
            grimoire_id: ID of grimoire (e.g., "grimoire-reentrancy")
            llm_provider: LLM provider for analysis steps

        Returns:
            Loaded Grimoire or None if not found
        """
        # Check cache
        cache_key = f"{grimoire_id}:{id(llm_provider)}"
        if cache_key in self._cache:
            return self._cache[cache_key]

        # Find YAML file
        yaml_path = self._find_grimoire_file(grimoire_id)
        if not yaml_path:
            return None

        # Load and parse
        with open(yaml_path) as f:
            definition = yaml.safe_load(f)

        # Create grimoire instance
        grimoire = Grimoire(definition, llm_provider)
        self._cache[cache_key] = grimoire

        return grimoire

    def _find_grimoire_file(self, grimoire_id: str) -> Optional[Path]:
        """Find YAML file for grimoire ID."""
        # Try exact ID match
        for suffix in [".yaml", ".yml"]:
            path = self.grimoire_dir / f"{grimoire_id}{suffix}"
            if path.exists():
                return path

        # Try without "grimoire-" prefix
        short_id = grimoire_id.replace("grimoire-", "")
        for suffix in [".yaml", ".yml"]:
            path = self.grimoire_dir / f"{short_id}{suffix}"
            if path.exists():
                return path

        return None

    def list_available(self) -> List[str]:
        """List all available grimoire IDs."""
        grimoires = []
        for path in self.grimoire_dir.glob("*.yaml"):
            with open(path) as f:
                definition = yaml.safe_load(f)
                grimoires.append(definition.get("id", path.stem))
        return grimoires

    def get_for_vulnerability(self, vuln_class: str) -> Optional[str]:
        """
        Get grimoire ID that handles a vulnerability class.

        Args:
            vuln_class: Vulnerability class (e.g., "reentrancy-classic")

        Returns:
            Grimoire ID or None
        """
        for path in self.grimoire_dir.glob("*.yaml"):
            with open(path) as f:
                definition = yaml.safe_load(f)
                classes = definition.get("vulnerability_classes", [])
                if vuln_class in classes:
                    return definition.get("id")
        return None
```
