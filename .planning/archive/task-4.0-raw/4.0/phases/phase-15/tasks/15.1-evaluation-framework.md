## Objective

Define a rigorous, repeatable framework for evaluating novel solutions and making INTEGRATE/DEFER/CUT decisions.

---

## Why This Matters

Without a framework:
- Decisions are subjective and inconsistent
- We might integrate broken solutions
- We might cut valuable solutions
- Future evaluations won't be comparable

## Implementation Steps

### Step 1: Define Evaluation Criteria (1h)

The framework uses 5 criteria, each scored 0-5:

| Criterion | Description | Scoring |
|-----------|-------------|---------|
| **Works** | Does it pass tests and work on real code? | 0=broken, 5=flawless |
| **Value** | Does it improve detection or workflow? | 0=no value, 5=major improvement |
| **Complexity** | How much code/deps/maintenance? | 0=nightmare, 5=trivial |
| **Cost** | Ongoing operational cost? | 0=expensive, 5=free |
| **Demand** | Do users want this? | 0=nobody, 5=everyone |

```python
from dataclasses import dataclass, field
from typing import List, Optional
from enum import Enum
import subprocess
import json
from pathlib import Path

class Decision(Enum):
    INTEGRATE = "INTEGRATE"
    DEFER = "DEFER"
    CUT = "CUT"

@dataclass
class EvaluationCriteria:
    """Weights for evaluation criteria."""
    works_weight: float = 0.25
    value_weight: float = 0.30
    complexity_weight: float = 0.15
    cost_weight: float = 0.15
    demand_weight: float = 0.15

    def validate(self) -> bool:
        """Weights should sum to 1.0."""
        total = (self.works_weight + self.value_weight +
                 self.complexity_weight + self.cost_weight + self.demand_weight)
        return abs(total - 1.0) < 0.01

@dataclass
class TestResults:
    """Results from running solution tests."""
    passed: int
    failed: int
    skipped: int
    errors: int
    total: int

    @property
    def pass_rate(self) -> float:
        if self.total == 0:
            return 0.0
        return self.passed / self.total

@dataclass
class SolutionEvaluation:
    """Complete evaluation of a novel solution."""
    solution_name: str
    solution_path: str

    # Test results
    test_results: TestResults

    # Scores (0-5)
    works_score: int
    value_score: int
    complexity_score: int  # Inverted: 5 = low complexity (good)
    cost_score: int  # Inverted: 5 = no cost (good)
    demand_score: int

    # Computed
    weighted_score: float = 0.0
    decision: Decision = Decision.DEFER
    rationale: str = ""

    # Evidence
    evidence: List[str] = field(default_factory=list)

    def compute_weighted_score(self, criteria: EvaluationCriteria) -> float:
        """Compute weighted total score."""
        self.weighted_score = (
            self.works_score * criteria.works_weight +
            self.value_score * criteria.value_weight +
            self.complexity_score * criteria.complexity_weight +
            self.cost_score * criteria.cost_weight +
            self.demand_score * criteria.demand_weight
        )
        return self.weighted_score

    def make_decision(self) -> Decision:
        """Make integration decision based on score."""
        if self.works_score == 0:
            self.decision = Decision.CUT
            self.rationale = "Solution does not work (tests failing)"
        elif self.weighted_score >= 3.5:
            self.decision = Decision.INTEGRATE
            self.rationale = f"High value solution (score: {self.weighted_score:.2f})"
        elif self.weighted_score >= 2.0:
            self.decision = Decision.DEFER
            self.rationale = f"Promising but needs work (score: {self.weighted_score:.2f})"
        else:
            self.decision = Decision.CUT
            self.rationale = f"Low value or high complexity (score: {self.weighted_score:.2f})"
        return self.decision

class SolutionEvaluator:
    """Evaluates novel solutions against the framework."""

    def __init__(self, criteria: Optional[EvaluationCriteria] = None):
        self.criteria = criteria or EvaluationCriteria()
        assert self.criteria.validate(), "Criteria weights must sum to 1.0"

    def run_tests(self, solution_name: str) -> TestResults:
        """Run pytest for a solution and capture results."""
        test_file = f"tests/test_{solution_name}.py"

        try:
            result = subprocess.run(
                ["uv", "run", "pytest", test_file, "-v", "--tb=no", "-q"],
                capture_output=True,
                text=True,
                timeout=300
            )

            # Parse pytest output
            output = result.stdout + result.stderr
            # Example: "47 passed, 2 failed, 1 skipped"
            passed = failed = skipped = errors = 0

            for line in output.split('\n'):
                if 'passed' in line:
                    import re
                    matches = re.findall(r'(\d+) passed', line)
                    if matches:
                        passed = int(matches[0])
                    matches = re.findall(r'(\d+) failed', line)
                    if matches:
                        failed = int(matches[0])
                    matches = re.findall(r'(\d+) skipped', line)
                    if matches:
                        skipped = int(matches[0])
                    matches = re.findall(r'(\d+) error', line)
                    if matches:
                        errors = int(matches[0])

            return TestResults(
                passed=passed,
                failed=failed,
                skipped=skipped,
                errors=errors,
                total=passed + failed + skipped + errors
            )

        except subprocess.TimeoutExpired:
            return TestResults(0, 0, 0, 1, 1)
        except Exception as e:
            return TestResults(0, 0, 0, 1, 1)

    def evaluate(
        self,
        solution_name: str,
        value_score: int,
        complexity_score: int,
        cost_score: int,
        demand_score: int,
        evidence: Optional[List[str]] = None
    ) -> SolutionEvaluation:
        """
        Evaluate a solution.

        Args:
            solution_name: Name of the solution (e.g., "evolution")
            value_score: 0-5 score for real-world value
            complexity_score: 0-5 score (5 = low complexity, good)
            cost_score: 0-5 score (5 = no cost, good)
            demand_score: 0-5 score for user demand
            evidence: List of evidence supporting scores

        Returns:
            Complete SolutionEvaluation
        """
        # Run tests
        test_results = self.run_tests(solution_name)

        # Calculate works score from test pass rate
        works_score = int(test_results.pass_rate * 5)

        evaluation = SolutionEvaluation(
            solution_name=solution_name,
            solution_path=f"src/true_vkg/{solution_name}/",
            test_results=test_results,
            works_score=works_score,
            value_score=value_score,
            complexity_score=complexity_score,
            cost_score=cost_score,
            demand_score=demand_score,
            evidence=evidence or []
        )

        evaluation.compute_weighted_score(self.criteria)
        evaluation.make_decision()

        return evaluation

def generate_evaluation_report(evaluations: List[SolutionEvaluation]) -> str:
    """Generate markdown report from evaluations."""
    lines = [
        "# Novel Solutions Evaluation Report",
        "",
        "## Summary",
        "",
        "| Solution | Works | Value | Complexity | Cost | Demand | **Score** | **Decision** |",
        "|----------|-------|-------|------------|------|--------|-----------|--------------|",
    ]

    # Sort by score
    sorted_evals = sorted(evaluations, key=lambda e: e.weighted_score, reverse=True)

    for e in sorted_evals:
        decision_emoji = {
            Decision.INTEGRATE: "INTEGRATE",
            Decision.DEFER: "DEFER",
            Decision.CUT: "CUT"
        }[e.decision]
        lines.append(
            f"| {e.solution_name} | {e.works_score} | {e.value_score} | "
            f"{e.complexity_score} | {e.cost_score} | {e.demand_score} | "
            f"**{e.weighted_score:.2f}** | {decision_emoji} |"
        )

    lines.extend([
        "",
        "## Decision Summary",
        "",
        f"- **INTEGRATE:** {sum(1 for e in evaluations if e.decision == Decision.INTEGRATE)}",
        f"- **DEFER:** {sum(1 for e in evaluations if e.decision == Decision.DEFER)}",
        f"- **CUT:** {sum(1 for e in evaluations if e.decision == Decision.CUT)}",
        "",
        "## Detailed Evaluations",
        ""
    ])

    for e in sorted_evals:
        lines.extend([
            f"### {e.solution_name}",
            "",
            f"**Decision:** {e.decision.value}",
            f"**Rationale:** {e.rationale}",
            "",
            f"- Tests: {e.test_results.passed}/{e.test_results.total} passing ({e.test_results.pass_rate:.0%})",
            f"- Location: `{e.solution_path}`",
            "",
            "Evidence:",
        ])
        for evidence in e.evidence:
            lines.append(f"- {evidence}")
        lines.append("")

    return "\n".join(lines)
```

### Step 3: Create Evaluation Script (30 min)

```python
#!/usr/bin/env python
"""Evaluate novel solutions using the framework."""

import argparse
from pathlib import Path
from true_vkg.evaluation.framework import (
    SolutionEvaluator,
    generate_evaluation_report
)

# Default scores based on research phase findings
DEFAULT_SCORES = {
    "evolution": {"value": 4, "complexity": 3, "cost": 5, "demand": 3},
    "crosschain": {"value": 2, "complexity": 2, "cost": 5, "demand": 2},
    "adversarial": {"value": 4, "complexity": 3, "cost": 2, "demand": 4},
    "streaming": {"value": 2, "complexity": 2, "cost": 5, "demand": 2},
    "collab": {"value": 2, "complexity": 1, "cost": 4, "demand": 2},
    "predictive": {"value": 3, "complexity": 3, "cost": 4, "demand": 3},
    "swarm": {"value": 2, "complexity": 1, "cost": 3, "demand": 2},
    "invariants": {"value": 4, "complexity": 2, "cost": 5, "demand": 4},
    "similarity": {"value": 4, "complexity": 4, "cost": 5, "demand": 4},
}

def main():
    parser = argparse.ArgumentParser(description="Evaluate novel solutions")
    parser.add_argument("--output", type=Path, default=Path("docs/evaluation/novel-solutions.md"))
    parser.add_argument("--solution", type=str, help="Evaluate single solution")
    args = parser.parse_args()

    evaluator = SolutionEvaluator()
    evaluations = []

    solutions = [args.solution] if args.solution else list(DEFAULT_SCORES.keys())

    for solution in solutions:
        scores = DEFAULT_SCORES.get(solution, {"value": 2, "complexity": 2, "cost": 3, "demand": 2})
        print(f"Evaluating {solution}...")

        eval_result = evaluator.evaluate(
            solution_name=solution,
            value_score=scores["value"],
            complexity_score=scores["complexity"],
            cost_score=scores["cost"],
            demand_score=scores["demand"],
            evidence=[f"Scores from research phase findings"]
        )
        evaluations.append(eval_result)
        print(f"  Score: {eval_result.weighted_score:.2f} -> {eval_result.decision.value}")

    # Generate report
    report = generate_evaluation_report(evaluations)
    args.output.parent.mkdir(parents=True, exist_ok=True)
    args.output.write_text(report)
    print(f"\nReport saved to {args.output}")

if __name__ == "__main__":
    main()
```

import pytest
from true_vkg.evaluation.framework import (
    SolutionEvaluator,
    SolutionEvaluation,
    EvaluationCriteria,
    TestResults,
    Decision,
    generate_evaluation_report
)

class TestEvaluationFramework:
    """Tests for evaluation framework."""

    def test_criteria_validation(self):
        """Criteria weights should sum to 1.0."""
        criteria = EvaluationCriteria()
        assert criteria.validate()

    def test_invalid_criteria_rejected(self):
        """Invalid criteria should fail validation."""
        criteria = EvaluationCriteria(works_weight=0.5, value_weight=0.5)
        # Other weights are defaults, so won't sum to 1.0 correctly
        # Actually this might pass - need to check
        criteria = EvaluationCriteria(works_weight=0.1, value_weight=0.1,
                                       complexity_weight=0.1, cost_weight=0.1,
                                       demand_weight=0.1)
        assert not criteria.validate()  # Sums to 0.5

    def test_decision_integrate(self):
        """High score should result in INTEGRATE."""
        eval = SolutionEvaluation(
            solution_name="test",
            solution_path="src/",
            test_results=TestResults(10, 0, 0, 0, 10),
            works_score=5,
            value_score=5,
            complexity_score=4,
            cost_score=5,
            demand_score=4
        )
        eval.compute_weighted_score(EvaluationCriteria())
        eval.make_decision()

        assert eval.decision == Decision.INTEGRATE
        assert eval.weighted_score >= 3.5

    def test_decision_cut_broken(self):
        """Broken solution should be CUT."""
        eval = SolutionEvaluation(
            solution_name="broken",
            solution_path="src/",
            test_results=TestResults(0, 10, 0, 0, 10),
            works_score=0,  # Broken
            value_score=5,
            complexity_score=5,
            cost_score=5,
            demand_score=5
        )
        eval.compute_weighted_score(EvaluationCriteria())
        eval.make_decision()

        assert eval.decision == Decision.CUT
        assert "does not work" in eval.rationale.lower()

    def test_report_generation(self):
        """Report should be valid markdown."""
        evals = [
            SolutionEvaluation(
                solution_name="good",
                solution_path="src/",
                test_results=TestResults(10, 0, 0, 0, 10),
                works_score=5, value_score=4, complexity_score=4,
                cost_score=5, demand_score=4,
                weighted_score=4.0,
                decision=Decision.INTEGRATE,
                rationale="Good solution",
                evidence=["Evidence 1"]
            ),
            SolutionEvaluation(
                solution_name="bad",
                solution_path="src/",
                test_results=TestResults(2, 8, 0, 0, 10),
                works_score=1, value_score=2, complexity_score=2,
                cost_score=2, demand_score=1,
                weighted_score=1.5,
                decision=Decision.CUT,
                rationale="Poor solution",
                evidence=["Evidence 2"]
            )
        ]

        report = generate_evaluation_report(evals)

        assert "# Novel Solutions Evaluation Report" in report
        assert "good" in report
        assert "bad" in report
        assert "INTEGRATE" in report
        assert "CUT" in report
