```python
"""Metric event types for recording."""
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional
from enum import Enum


class EventType(Enum):
    """Types of metric events."""
    DETECTION = "detection"
    TIMING = "timing"
    SCAFFOLD = "scaffold"
    VERDICT = "verdict"


@dataclass
class DetectionEvent:
    """Records whether a vulnerability was detected."""
    event_id: str
    event_type: EventType = field(default=EventType.DETECTION, init=False)
    timestamp: datetime = field(default_factory=datetime.now)

    # Detection context
    contract_id: str
    pattern_id: str
    function_name: str
    line_number: int

    # Outcome
    expected: bool  # Was this in MANIFEST.yaml?
    detected: bool  # Did BSKG find it?

    def to_dict(self) -> dict:
        return {
            "event_id": self.event_id,
            "event_type": self.event_type.value,
            "timestamp": self.timestamp.isoformat(),
            "contract_id": self.contract_id,
            "pattern_id": self.pattern_id,
            "function_name": self.function_name,
            "line_number": self.line_number,
            "expected": self.expected,
            "detected": self.detected,
        }


@dataclass
class TimingEvent:
    """Records timing for operations."""
    event_id: str
    event_type: EventType = field(default=EventType.TIMING, init=False)
    timestamp: datetime = field(default_factory=datetime.now)

    # Timing context
    operation: str  # "scan", "build_graph", "query"
    contract_id: str
    duration_seconds: float

    def to_dict(self) -> dict:
        return {
            "event_id": self.event_id,
            "event_type": self.event_type.value,
            "timestamp": self.timestamp.isoformat(),
            "operation": self.operation,
            "contract_id": self.contract_id,
            "duration_seconds": self.duration_seconds,
        }


@dataclass
class ScaffoldEvent:
    """Records scaffold compilation result."""
    event_id: str
    event_type: EventType = field(default=EventType.SCAFFOLD, init=False)
    timestamp: datetime = field(default_factory=datetime.now)

    # Scaffold context
    finding_id: str
    pattern_id: str

    # Outcome
    compiled: bool
    error_message: Optional[str] = None

    def to_dict(self) -> dict:
        return {
            "event_id": self.event_id,
            "event_type": self.event_type.value,
            "timestamp": self.timestamp.isoformat(),
            "finding_id": self.finding_id,
            "pattern_id": self.pattern_id,
            "compiled": self.compiled,
            "error_message": self.error_message,
        }


@dataclass
class VerdictEvent:
    """Records LLM verdict on finding."""
    event_id: str
    event_type: EventType = field(default=EventType.VERDICT, init=False)
    timestamp: datetime = field(default_factory=datetime.now)

    # Verdict context
    finding_id: str
    pattern_id: str

    # Outcome
    verdict: str  # "confirmed", "rejected", "uncertain"
    auto_resolved: bool  # Was human escalation needed?
    tokens_used: int

    def to_dict(self) -> dict:
        return {
            "event_id": self.event_id,
            "event_type": self.event_type.value,
            "timestamp": self.timestamp.isoformat(),
            "finding_id": self.finding_id,
            "pattern_id": self.pattern_id,
            "verdict": self.verdict,
            "auto_resolved": self.auto_resolved,
            "tokens_used": self.tokens_used,
        }
```

```python
"""Event storage for metrics."""
import json
import uuid
from datetime import datetime
from pathlib import Path
from typing import List, Optional
from .events import DetectionEvent, TimingEvent, ScaffoldEvent, VerdictEvent, EventType


class EventStore:
    """Stores metric events to disk."""

    def __init__(self, storage_path: Path):
        self.storage_path = Path(storage_path)
        self.storage_path.mkdir(parents=True, exist_ok=True)

    def _get_today_file(self) -> Path:
        """Get today's event file."""
        today = datetime.now().strftime("%Y-%m-%d")
        return self.storage_path / f"events-{today}.json"

    def _load_events(self, filepath: Path) -> List[dict]:
        """Load events from file."""
        if not filepath.exists():
            return []
        with open(filepath, "r") as f:
            return json.load(f)

    def _save_events(self, filepath: Path, events: List[dict]):
        """Save events to file."""
        with open(filepath, "w") as f:
            json.dump(events, f, indent=2)

    def _generate_id(self) -> str:
        """Generate unique event ID."""
        return f"evt-{uuid.uuid4().hex[:8]}"

    def record_detection(
        self,
        contract_id: str,
        pattern_id: str,
        function_name: str,
        line_number: int,
        expected: bool,
        detected: bool
    ) -> str:
        """Record a detection event."""
        event = DetectionEvent(
            event_id=self._generate_id(),
            contract_id=contract_id,
            pattern_id=pattern_id,
            function_name=function_name,
            line_number=line_number,
            expected=expected,
            detected=detected
        )

        events = self._load_events(self._get_today_file())
        events.append(event.to_dict())
        self._save_events(self._get_today_file(), events)

        return event.event_id

    def record_timing(
        self,
        operation: str,
        contract_id: str,
        duration_seconds: float
    ) -> str:
        """Record a timing event."""
        event = TimingEvent(
            event_id=self._generate_id(),
            operation=operation,
            contract_id=contract_id,
            duration_seconds=duration_seconds
        )

        events = self._load_events(self._get_today_file())
        events.append(event.to_dict())
        self._save_events(self._get_today_file(), events)

        return event.event_id

    def record_scaffold(
        self,
        finding_id: str,
        pattern_id: str,
        compiled: bool,
        error_message: Optional[str] = None
    ) -> str:
        """Record a scaffold compilation event."""
        event = ScaffoldEvent(
            event_id=self._generate_id(),
            finding_id=finding_id,
            pattern_id=pattern_id,
            compiled=compiled,
            error_message=error_message
        )

        events = self._load_events(self._get_today_file())
        events.append(event.to_dict())
        self._save_events(self._get_today_file(), events)

        return event.event_id

    def record_verdict(
        self,
        finding_id: str,
        pattern_id: str,
        verdict: str,
        auto_resolved: bool,
        tokens_used: int
    ) -> str:
        """Record an LLM verdict event."""
        event = VerdictEvent(
            event_id=self._generate_id(),
            finding_id=finding_id,
            pattern_id=pattern_id,
            verdict=verdict,
            auto_resolved=auto_resolved,
            tokens_used=tokens_used
        )

        events = self._load_events(self._get_today_file())
        events.append(event.to_dict())
        self._save_events(self._get_today_file(), events)

        return event.event_id

    def get_events(
        self,
        event_type: Optional[EventType] = None,
        days: int = 1
    ) -> List[dict]:
        """Get events from the last N days."""
        all_events = []

        for i in range(days):
            date = datetime.now() - timedelta(days=i)
            filepath = self.storage_path / f"events-{date.strftime('%Y-%m-%d')}.json"
            all_events.extend(self._load_events(filepath))

        if event_type:
            all_events = [e for e in all_events if e["event_type"] == event_type.value]

        return all_events
```

### Step 3: Create Recorder Facade

**File:** `src/true_vkg/metrics/recorder.py`

```python
"""High-level recorder for metric events."""
from pathlib import Path
from typing import Optional
from .event_store import EventStore
from ..config import METRICS_CONFIG


class MetricsRecorder:
    """Singleton recorder for metric events.

    Usage:
        from true_vkg.metrics.recorder import recorder

        recorder.detection(
            contract_id="test.sol",
            pattern_id="vm-001",
            function_name="withdraw",
            line_number=42,
            expected=True,
            detected=True
        )
    """

    _instance: Optional["MetricsRecorder"] = None

    def __init__(self):
        storage_path = Path(METRICS_CONFIG.get("storage_path", ".vrs/metrics"))
        self.store = EventStore(storage_path / "events")

    @classmethod
    def get_instance(cls) -> "MetricsRecorder":
        """Get or create singleton instance."""
        if cls._instance is None:
            cls._instance = cls()
        return cls._instance

    def detection(
        self,
        contract_id: str,
        pattern_id: str,
        function_name: str,
        line_number: int,
        expected: bool,
        detected: bool
    ):
        """Record detection event."""
        self.store.record_detection(
            contract_id=contract_id,
            pattern_id=pattern_id,
            function_name=function_name,
            line_number=line_number,
            expected=expected,
            detected=detected
        )

    def timing(
        self,
        operation: str,
        contract_id: str,
        duration_seconds: float
    ):
        """Record timing event."""
        self.store.record_timing(
            operation=operation,
            contract_id=contract_id,
            duration_seconds=duration_seconds
        )

    def scaffold(
        self,
        finding_id: str,
        pattern_id: str,
        compiled: bool,
        error_message: Optional[str] = None
    ):
        """Record scaffold event."""
        self.store.record_scaffold(
            finding_id=finding_id,
            pattern_id=pattern_id,
            compiled=compiled,
            error_message=error_message
        )

    def verdict(
        self,
        finding_id: str,
        pattern_id: str,
        verdict: str,
        auto_resolved: bool,
        tokens_used: int
    ):
        """Record verdict event."""
        self.store.record_verdict(
            finding_id=finding_id,
            pattern_id=pattern_id,
            verdict=verdict,
            auto_resolved=auto_resolved,
            tokens_used=tokens_used
        )


# Global recorder instance
recorder = MetricsRecorder.get_instance()
```

```python
import tempfile
from pathlib import Path
from true_vkg.metrics.event_store import EventStore
from true_vkg.metrics.events import EventType


class TestMetricRecording:
    """Test metric recording infrastructure."""

    def test_record_detection_event(self, tmp_path):
        """Can record detection events."""
        store = EventStore(tmp_path)

        event_id = store.record_detection(
            contract_id="test.sol",
            pattern_id="vm-001",
            function_name="withdraw",
            line_number=42,
            expected=True,
            detected=True
        )

        assert event_id.startswith("evt-")

        events = store.get_events(EventType.DETECTION)
        assert len(events) == 1
        assert events[0]["contract_id"] == "test.sol"
        assert events[0]["detected"] is True

    def test_record_timing_event(self, tmp_path):
        """Can record timing events."""
        store = EventStore(tmp_path)

        store.record_timing(
            operation="scan",
            contract_id="test.sol",
            duration_seconds=2.5
        )

        events = store.get_events(EventType.TIMING)
        assert len(events) == 1
        assert events[0]["duration_seconds"] == 2.5

    def test_events_persisted_to_file(self, tmp_path):
        """Events written to daily JSON file."""
        store = EventStore(tmp_path)
        store.record_detection(
            contract_id="test.sol",
            pattern_id="vm-001",
            function_name="withdraw",
            line_number=42,
            expected=True,
            detected=True
        )

        # Check file exists
        files = list(tmp_path.glob("events-*.json"))
        assert len(files) == 1
```
