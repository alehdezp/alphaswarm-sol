### Step 1: Create Metric Calculator

**File:** `src/true_vkg/metrics/calculator.py`

```python
"""Metric calculation engine."""
from collections import defaultdict
from datetime import datetime, timedelta
from pathlib import Path
from typing import Dict, List, Optional
from .types import MetricName, MetricValue, MetricSnapshot, MetricStatus
from .definitions import METRIC_DEFINITIONS, get_available_metrics
from .event_store import EventStore
from .events import EventType


class MetricCalculator:
    """Calculates BSKG metrics from recorded events."""

    def __init__(self, event_store: EventStore):
        self.store = event_store

    def calculate_all(self, days: int = 7) -> MetricSnapshot:
        """Calculate all available metrics from recent events.

        Args:
            days: Number of days of events to include

        Returns:
            MetricSnapshot with all calculated metrics
        """
        available = get_available_metrics()
        metrics: Dict[MetricName, MetricValue] = {}

        for name in available:
            try:
                value = self._calculate_metric(name, days)
                if value is not None:
                    metrics[name] = value
            except Exception as e:
                # Log but don't fail on individual metric calculation
                print(f"Warning: Could not calculate {name.value}: {e}")

        return MetricSnapshot(
            timestamp=datetime.now(),
            version=self._get_version(),
            metrics=metrics
        )

    def _calculate_metric(
        self,
        name: MetricName,
        days: int
    ) -> Optional[MetricValue]:
        """Calculate a single metric."""
        defn = METRIC_DEFINITIONS[name]

        if name == MetricName.DETECTION_RATE:
            return self._calc_detection_rate(defn, days)
        elif name == MetricName.FALSE_POSITIVE_RATE:
            return self._calc_fp_rate(defn, days)
        elif name == MetricName.PATTERN_PRECISION:
            return self._calc_pattern_precision(defn, days)
        elif name == MetricName.TIME_TO_DETECTION:
            return self._calc_time_to_detection(defn, days)
        elif name == MetricName.SCAFFOLD_COMPILE_RATE:
            return self._calc_scaffold_compile_rate(defn, days)
        elif name == MetricName.LLM_AUTONOMY:
            return self._calc_llm_autonomy(defn, days)
        elif name == MetricName.TOKEN_EFFICIENCY:
            return self._calc_token_efficiency(defn, days)
        elif name == MetricName.ESCALATION_RATE:
            return self._calc_escalation_rate(defn, days)
        else:
            return None

    def _calc_detection_rate(self, defn, days: int) -> MetricValue:
        """Calculate detection rate: detected / expected."""
        events = self.store.get_events(EventType.DETECTION, days=days)

        if not events:
            return self._make_unknown(defn)

        # Count expected vulnerabilities that were detected
        expected_events = [e for e in events if e["expected"]]
        if not expected_events:
            return self._make_unknown(defn)

        detected_count = sum(1 for e in expected_events if e["detected"])
        rate = detected_count / len(expected_events)

        return self._make_value(defn, rate)

    def _calc_fp_rate(self, defn, days: int) -> MetricValue:
        """Calculate false positive rate: FP / (FP + TP)."""
        events = self.store.get_events(EventType.DETECTION, days=days)

        if not events:
            return self._make_unknown(defn)

        # TP: expected AND detected
        tp = sum(1 for e in events if e["expected"] and e["detected"])
        # FP: NOT expected AND detected
        fp = sum(1 for e in events if not e["expected"] and e["detected"])

        if tp + fp == 0:
            return self._make_unknown(defn)

        rate = fp / (fp + tp)
        return self._make_value(defn, rate)

    def _calc_pattern_precision(self, defn, days: int) -> MetricValue:
        """Calculate average precision per pattern."""
        events = self.store.get_events(EventType.DETECTION, days=days)

        if not events:
            return self._make_unknown(defn)

        # Group by pattern
        by_pattern: Dict[str, List[dict]] = defaultdict(list)
        for e in events:
            by_pattern[e["pattern_id"]].append(e)

        # Calculate precision per pattern
        precisions = []
        for pattern_id, pattern_events in by_pattern.items():
            tp = sum(1 for e in pattern_events if e["expected"] and e["detected"])
            fp = sum(1 for e in pattern_events if not e["expected"] and e["detected"])

            if tp + fp > 0:
                precisions.append(tp / (tp + fp))

        if not precisions:
            return self._make_unknown(defn)

        avg_precision = sum(precisions) / len(precisions)
        return self._make_value(defn, avg_precision)

    def _calc_time_to_detection(self, defn, days: int) -> MetricValue:
        """Calculate average scan time."""
        events = self.store.get_events(EventType.TIMING, days=days)

        # Filter to scan operations
        scan_events = [e for e in events if e["operation"] == "scan"]

        if not scan_events:
            return self._make_unknown(defn)

        avg_time = sum(e["duration_seconds"] for e in scan_events) / len(scan_events)
        return self._make_value(defn, avg_time)

    def _calc_scaffold_compile_rate(self, defn, days: int) -> MetricValue:
        """Calculate scaffold compilation rate."""
        events = self.store.get_events(EventType.SCAFFOLD, days=days)

        if not events:
            return self._make_unknown(defn)

        compiled = sum(1 for e in events if e["compiled"])
        rate = compiled / len(events)
        return self._make_value(defn, rate)

    def _calc_llm_autonomy(self, defn, days: int) -> MetricValue:
        """Calculate LLM autonomy rate."""
        events = self.store.get_events(EventType.VERDICT, days=days)

        if not events:
            return self._make_unknown(defn)

        auto_resolved = sum(1 for e in events if e["auto_resolved"])
        rate = auto_resolved / len(events)
        return self._make_value(defn, rate)

    def _calc_token_efficiency(self, defn, days: int) -> MetricValue:
        """Calculate average tokens per resolution."""
        events = self.store.get_events(EventType.VERDICT, days=days)

        if not events:
            return self._make_unknown(defn)

        avg_tokens = sum(e["tokens_used"] for e in events) / len(events)
        return self._make_value(defn, avg_tokens)

    def _calc_escalation_rate(self, defn, days: int) -> MetricValue:
        """Calculate human escalation rate."""
        events = self.store.get_events(EventType.VERDICT, days=days)

        if not events:
            return self._make_unknown(defn)

        escalated = sum(1 for e in events if not e["auto_resolved"])
        rate = escalated / len(events)
        return self._make_value(defn, rate)

    def _make_value(self, defn, value: float) -> MetricValue:
        """Create MetricValue from definition and calculated value."""
        mv = MetricValue(
            name=defn.name,
            value=value,
            target=defn.target,
            threshold_warning=defn.threshold_warning,
            threshold_critical=defn.threshold_critical,
            unit=defn.unit,
            timestamp=datetime.now()
        )
        mv.status = mv.evaluate_status()
        return mv

    def _make_unknown(self, defn) -> MetricValue:
        """Create unknown MetricValue when no data available."""
        return MetricValue(
            name=defn.name,
            value=0.0,
            target=defn.target,
            threshold_warning=defn.threshold_warning,
            threshold_critical=defn.threshold_critical,
            unit=defn.unit,
            status=MetricStatus.UNKNOWN,
            timestamp=datetime.now()
        )

    def _get_version(self) -> str:
        """Get BSKG version."""
        try:
            from true_vkg import __version__
            return __version__
        except ImportError:
            return "unknown"


def create_calculator(storage_path: Optional[Path] = None) -> MetricCalculator:
    """Factory function to create calculator with default storage."""
    from ..config import METRICS_CONFIG

    if storage_path is None:
        storage_path = Path(METRICS_CONFIG.get("storage_path", ".vrs/metrics"))

    store = EventStore(storage_path / "events")
    return MetricCalculator(store)
```

### Step 2: Create High-Level Tracker

**File:** `src/true_vkg/metrics/tracker.py`

```python
"""High-level metrics tracker combining recording and calculation."""
from pathlib import Path
from typing import Optional
from .recorder import MetricsRecorder
from .calculator import MetricCalculator, create_calculator
from .types import MetricSnapshot


class MetricsTracker:
    """Unified interface for recording and calculating metrics.

    Usage:
        tracker = MetricsTracker()

        # Record events
        tracker.record_detection(...)

        # Calculate metrics
        snapshot = tracker.calculate_metrics()
        print(snapshot.metrics[MetricName.DETECTION_RATE].value)
    """

    def __init__(self, storage_path: Optional[Path] = None):
        self.recorder = MetricsRecorder()
        self.calculator = create_calculator(storage_path)

    def record_detection(
        self,
        contract_id: str,
        pattern_id: str,
        function_name: str,
        line_number: int,
        expected: bool,
        detected: bool
    ):
        """Record a detection event."""
        self.recorder.detection(
            contract_id=contract_id,
            pattern_id=pattern_id,
            function_name=function_name,
            line_number=line_number,
            expected=expected,
            detected=detected
        )

    def record_timing(
        self,
        operation: str,
        contract_id: str,
        duration_seconds: float
    ):
        """Record a timing event."""
        self.recorder.timing(
            operation=operation,
            contract_id=contract_id,
            duration_seconds=duration_seconds
        )

    def record_scaffold(
        self,
        finding_id: str,
        pattern_id: str,
        compiled: bool,
        error_message: Optional[str] = None
    ):
        """Record a scaffold compilation event."""
        self.recorder.scaffold(
            finding_id=finding_id,
            pattern_id=pattern_id,
            compiled=compiled,
            error_message=error_message
        )

    def record_verdict(
        self,
        finding_id: str,
        pattern_id: str,
        verdict: str,
        auto_resolved: bool,
        tokens_used: int
    ):
        """Record an LLM verdict event."""
        self.recorder.verdict(
            finding_id=finding_id,
            pattern_id=pattern_id,
            verdict=verdict,
            auto_resolved=auto_resolved,
            tokens_used=tokens_used
        )

    def calculate_metrics(self, days: int = 7) -> MetricSnapshot:
        """Calculate all metrics from recent events."""
        return self.calculator.calculate_all(days=days)
```

```python
from datetime import datetime
from true_vkg.metrics.calculator import MetricCalculator
from true_vkg.metrics.event_store import EventStore
from true_vkg.metrics.types import MetricName, MetricStatus


class TestMetricCalculation:
    """Test metric calculation."""

    def test_detection_rate_calculation(self, tmp_path):
        """Detection rate calculated correctly."""
        store = EventStore(tmp_path)

        # Record 10 expected vulnerabilities, 8 detected
        for i in range(10):
            store.record_detection(
                contract_id="test.sol",
                pattern_id="vm-001",
                function_name=f"func{i}",
                line_number=i,
                expected=True,
                detected=i < 8  # First 8 detected
            )

        calc = MetricCalculator(store)
        snapshot = calc.calculate_all()

        assert MetricName.DETECTION_RATE in snapshot.metrics
        rate = snapshot.metrics[MetricName.DETECTION_RATE]
        assert rate.value == 0.8  # 8/10
        assert rate.status == MetricStatus.OK  # >= 80% target

    def test_fp_rate_calculation(self, tmp_path):
        """False positive rate calculated correctly."""
        store = EventStore(tmp_path)

        # 8 TP (expected and detected)
        for i in range(8):
            store.record_detection(
                contract_id="test.sol",
                pattern_id="vm-001",
                function_name=f"tp{i}",
                line_number=i,
                expected=True,
                detected=True
            )

        # 2 FP (not expected but detected)
        for i in range(2):
            store.record_detection(
                contract_id="test.sol",
                pattern_id="vm-001",
                function_name=f"fp{i}",
                line_number=100 + i,
                expected=False,
                detected=True
            )

        calc = MetricCalculator(store)
        snapshot = calc.calculate_all()

        fp_rate = snapshot.metrics[MetricName.FALSE_POSITIVE_RATE]
        assert abs(fp_rate.value - 0.2) < 0.001  # 2/(2+8) = 0.2

    def test_pattern_precision_per_pattern(self, tmp_path):
        """Pattern precision calculated per pattern then averaged."""
        store = EventStore(tmp_path)

        # Pattern A: 4 TP, 0 FP = 100% precision
        for i in range(4):
            store.record_detection(
                contract_id="test.sol",
                pattern_id="pattern-a",
                function_name=f"a{i}",
                line_number=i,
                expected=True,
                detected=True
            )

        # Pattern B: 3 TP, 3 FP = 50% precision
        for i in range(3):
            store.record_detection(
                contract_id="test.sol",
                pattern_id="pattern-b",
                function_name=f"b-tp{i}",
                line_number=100 + i,
                expected=True,
                detected=True
            )
        for i in range(3):
            store.record_detection(
                contract_id="test.sol",
                pattern_id="pattern-b",
                function_name=f"b-fp{i}",
                line_number=200 + i,
                expected=False,
                detected=True
            )

        calc = MetricCalculator(store)
        snapshot = calc.calculate_all()

        precision = snapshot.metrics[MetricName.PATTERN_PRECISION]
        # Average of 100% and 50% = 75%
        assert abs(precision.value - 0.75) < 0.001

    def test_unknown_when_no_data(self, tmp_path):
        """Returns UNKNOWN status when no events recorded."""
        store = EventStore(tmp_path)
        calc = MetricCalculator(store)
        snapshot = calc.calculate_all()

        for metric in snapshot.metrics.values():
            assert metric.status == MetricStatus.UNKNOWN
```
