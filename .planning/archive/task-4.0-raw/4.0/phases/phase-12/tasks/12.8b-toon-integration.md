## 2. CONTEXT

### 2.1 What is TOON?

TOON is a token-efficient serialization format defined in Phase 9.8. Instead of:

```json
{
  "function": {
    "name": "withdraw",
    "visibility": "public",
    "modifiers": ["nonReentrant"]
  }
}
```

TOON produces:

```
F:withdraw|V:public|M:nonReentrant
```

This reduces tokens significantly.

### Step 2: Implement Minimal TOON Encoder (1.5h)

**File:** `src/true_vkg/llm/toon.py`

```python
"""Token-Optimized Object Notation (TOON) encoding."""

from typing import Dict, Any, List
import re

# Abbreviations for common keys
ABBREVIATIONS = {
    "function_name": "F",
    "visibility": "V",
    # ... (see above)
}

REVERSE_ABBREVIATIONS = {v: k for k, v in ABBREVIATIONS.items()}

class TOONEncoder:
    """Encode/decode TOON format."""

    def __init__(self, abbreviations: Dict[str, str] = None):
        self.abbrevs = abbreviations or ABBREVIATIONS
        self.reverse = {v: k for k, v in self.abbrevs.items()}

    def encode(self, data: Dict[str, Any]) -> str:
        """Encode dict to TOON."""
        parts = []
        for key, value in data.items():
            abbrev = self.abbrevs.get(key, key)
            encoded_value = self._encode_value(value)
            parts.append(f"{abbrev}:{encoded_value}")
        return "|".join(parts)

    def _encode_value(self, value: Any) -> str:
        """Encode a single value."""
        if isinstance(value, bool):
            return "1" if value else "0"
        if isinstance(value, (int, float)):
            return str(value)
        if isinstance(value, str):
            # Escape pipes and colons
            return value.replace("|", "\\|").replace(":", "\\:")
        if isinstance(value, list):
            return ",".join(self._encode_value(v) for v in value)
        if isinstance(value, dict):
            # Nested dict: use recursive encoding
            return "{" + self.encode(value) + "}"
        return str(value)

    def decode(self, toon_str: str) -> Dict[str, Any]:
        """Decode TOON string to dict."""
        result = {}
        # Split on unescaped pipes
        parts = re.split(r'(?<!\\)\|', toon_str)
        for part in parts:
            if ':' in part:
                key, value = part.split(':', 1)
                full_key = self.reverse.get(key, key)
                result[full_key] = self._decode_value(value)
        return result

    def _decode_value(self, value: str) -> Any:
        """Decode a single value."""
        # Unescape
        value = value.replace("\\|", "|").replace("\\:", ":")

        # Try int
        if value.isdigit():
            return int(value)

        # Try float
        try:
            return float(value)
        except ValueError:
            pass

        # Check for list
        if ',' in value and not value.startswith('{'):
            return value.split(',')

        # Check for nested dict
        if value.startswith('{') and value.endswith('}'):
            return self.decode(value[1:-1])

        return value
```

### Step 3: Integrate with SubagentManager (1h)

**File:** `src/true_vkg/llm/routing.py` (add to existing)

```python
from true_vkg.llm.toon import TOONEncoder

@dataclass
class PreparedTask:
    """Task prepared for execution."""
    original: SubagentTask
    routing: RoutingDecision
    encoded_context: str  # TOON-encoded context
    token_estimate: int

class SubagentManager:
    """Manages subagent task routing and context encoding."""

    def __init__(self, config: LLMConfig, use_toon: bool = True):
        self.router = TaskRouter(config)
        self.use_toon = use_toon
        self.toon_encoder = TOONEncoder() if use_toon else None

    def prepare_task(self, task: SubagentTask) -> PreparedTask:
        """Prepare a task for execution."""
        decision = self.router.route(task)

        if self.use_toon and self.toon_encoder:
            encoded_context = self.toon_encoder.encode(task.context)
        else:
            # Fallback to JSON
            import json
            encoded_context = json.dumps(task.context)

        return PreparedTask(
            original=task,
            routing=decision,
            encoded_context=encoded_context,
            token_estimate=self._estimate_tokens(encoded_context),
        )

    def _estimate_tokens(self, text: str) -> int:
        """Rough token estimate (4 chars per token)."""
        return len(text) // 4
```

```python
# tests/test_toon.py

def test_encode_simple():
    encoder = TOONEncoder()
    data = {"function_name": "withdraw", "visibility": "public"}
    encoded = encoder.encode(data)
    assert "F:withdraw" in encoded
    assert "V:public" in encoded

def test_roundtrip():
    encoder = TOONEncoder()
    original = {
        "function_name": "withdraw",
        "visibility": "public",
        "modifiers": ["nonReentrant", "onlyOwner"],
    }
    encoded = encoder.encode(original)
    decoded = encoder.decode(encoded)
    assert decoded == original

def test_token_savings():
    encoder = TOONEncoder()
    import json

    data = {
        "function_name": "withdraw",
        "visibility": "public",
        "has_reentrancy_guard": False,
        "external_calls": ["transfer", "call"],
    }

    json_encoded = json.dumps(data)
    toon_encoded = encoder.encode(data)

    # TOON should be smaller
    assert len(toon_encoded) < len(json_encoded)

    # Should save at least 20%
    savings = (len(json_encoded) - len(toon_encoded)) / len(json_encoded)
    assert savings >= 0.20
```
