### 3.3 Cost Tracking

```python
@dataclass
class CostTracker:
    """Track costs during batch execution."""
    max_cost_usd: float
    current_cost_usd: float = 0.0
    tasks_completed: int = 0
    tasks_skipped: int = 0  # Due to cost limit

    def can_proceed(self, estimated_cost: float) -> bool:
        """Check if we can proceed with next task."""
        return (self.current_cost_usd + estimated_cost) <= self.max_cost_usd

    def record(self, actual_cost: float):
        """Record cost of completed task."""
        self.current_cost_usd += actual_cost
        self.tasks_completed += 1

@dataclass
class BatchSummary:
    """Summary of batch execution."""
    total_tasks: int
    successful: int
    failed: int
    skipped_cost: int  # Skipped due to cost limit

    total_tokens: int
    total_cost_usd: float
    total_latency_ms: int

    providers_used: Dict[str, int]  # Provider -> count
    tiers_used: Dict[str, int]  # Tier -> count
```

```python
async def dispatch(self, task: SubagentTask) -> SubagentResult:
    """Execute a single subagent task."""
    import time
    import uuid

    task_id = str(uuid.uuid4())[:8]
    start_time = time.time()

    try:
        # Prepare task (routing + encoding)
        prepared = self.prepare_task(task)

        # Get provider
        provider = self._get_provider(prepared.routing.provider)

        # Execute
        response = await provider.analyze(
            context=prepared.encoded_context,
            prompt=task.prompt,
            model=prepared.routing.model,
        )

        latency_ms = int((time.time() - start_time) * 1000)

        return SubagentResult(
            task_type=task.task_type,
            task_id=task_id,
            provider=prepared.routing.provider,
            model=prepared.routing.model,
            tier=prepared.routing.tier,
            success=True,
            verdict=response.get("verdict"),
            confidence=response.get("confidence"),
            reasoning=response.get("reasoning"),
            evidence=response.get("evidence", []),
            raw_response=str(response),
            tokens_used=response.get("tokens", 0),
            cost_usd=self._calculate_cost(response.get("tokens", 0), prepared.routing),
            latency_ms=latency_ms,
        )

    except Exception as e:
        latency_ms = int((time.time() - start_time) * 1000)
        return SubagentResult(
            task_type=task.task_type,
            task_id=task_id,
            provider=prepared.routing.provider if prepared else "unknown",
            model=prepared.routing.model if prepared else "unknown",
            tier=prepared.routing.tier if prepared else ModelTier.MEDIUM,
            success=False,
            error=str(e),
            latency_ms=latency_ms,
        )
```

### Step 3: Implement Batch Dispatch (1h)

```python
async def dispatch_batch(
    self,
    tasks: List[SubagentTask],
    parallel: int = 3,
    max_cost_usd: float = 10.0,
    fail_fast: bool = False,
) -> List[SubagentResult]:
    """Execute multiple tasks with concurrency control."""
    import asyncio

    semaphore = asyncio.Semaphore(parallel)
    cost_tracker = CostTracker(max_cost_usd=max_cost_usd)
    results: List[SubagentResult] = [None] * len(tasks)
    failed = False

    async def dispatch_with_control(index: int, task: SubagentTask):
        nonlocal failed

        # Check fail fast
        if fail_fast and failed:
            return

        # Check cost limit
        estimated_cost = self._estimate_cost(task)
        if not cost_tracker.can_proceed(estimated_cost):
            results[index] = SubagentResult(
                task_type=task.task_type,
                task_id=f"skipped-{index}",
                provider="none",
                model="none",
                tier=ModelTier.MEDIUM,
                success=False,
                error="Cost limit exceeded",
            )
            cost_tracker.tasks_skipped += 1
            return

        async with semaphore:
            result = await self.dispatch(task)
            results[index] = result

            if result.success:
                cost_tracker.record(result.cost_usd)
            elif fail_fast:
                failed = True

    # Execute all tasks
    await asyncio.gather(
        *[dispatch_with_control(i, task) for i, task in enumerate(tasks)],
        return_exceptions=True,
    )

    return results
```

```python
def get_batch_summary(self, results: List[SubagentResult]) -> BatchSummary:
    """Get summary statistics for a batch run."""
    providers_used: Dict[str, int] = {}
    tiers_used: Dict[str, int] = {}

    successful = 0
    failed = 0
    skipped = 0
    total_tokens = 0
    total_cost = 0.0
    total_latency = 0

    for result in results:
        if result.success:
            successful += 1
            total_tokens += result.tokens_used
            total_cost += result.cost_usd
            total_latency += result.latency_ms

            providers_used[result.provider] = providers_used.get(result.provider, 0) + 1
            tiers_used[result.tier.value] = tiers_used.get(result.tier.value, 0) + 1
        elif "Cost limit" in (result.error or ""):
            skipped += 1
        else:
            failed += 1

    return BatchSummary(
        total_tasks=len(results),
        successful=successful,
        failed=failed,
        skipped_cost=skipped,
        total_tokens=total_tokens,
        total_cost_usd=total_cost,
        total_latency_ms=total_latency,
        providers_used=providers_used,
        tiers_used=tiers_used,
    )
```

async def test_batch_respects_parallel_limit():
    """Batch should respect parallel limit."""

    tasks = [

        await manager.dispatch_batch(tasks, parallel=3)
