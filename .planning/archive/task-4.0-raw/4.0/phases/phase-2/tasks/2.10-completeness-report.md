## CRITICAL CONTEXT

**Why this task matters:**
Users MUST know when BSKG analysis is incomplete. Silent failures hide vulnerabilities. If BSKG skips a contract due to inline assembly but doesn't tell the user, that contract could contain critical vulnerabilities that are never reviewed.

**What failure looks like:**
- BSKG analyzes 10/15 contracts silently
- User thinks 15 contracts analyzed, misses 5
- Critical vulnerability in skipped contract goes undetected
- Protocol gets exploited

**Real-world scenario:**
```
Project: 20 contracts
VKG output: "Found 3 vulnerabilities"
Reality: 5 contracts failed to parse, 3 of those have vulnerabilities
User: "VKG said only 3 issues, we're safe!"
```

---

## 1. OBJECTIVE

Generate a completeness report on EVERY analysis run that tells users exactly:
1. What was analyzed
2. What was skipped
3. Why it was skipped
4. What they should do about it

### Subtask 2.10b: Implement CompletenessReport Class (2h)

**Deliverable:** `src/true_vkg/report/completeness.py`

```python
"""
Analysis Completeness Report

Tracks what was analyzed, skipped, and why.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from enum import Enum
from pathlib import Path
from typing import Any
import json


class CompletionStatus(Enum):
    COMPLETE = "complete"
    PARTIAL = "partial"
    FAILED = "failed"


class SkipReason(Enum):
    INLINE_ASSEMBLY = "inline_assembly"
    PROXY_UNRESOLVED = "proxy_unresolved"
    PARSE_ERROR = "parse_error"
    UNSUPPORTED_SOLC = "unsupported_solc"
    COMPILATION_ERROR = "compilation_error"
    TIMEOUT = "timeout"


@dataclass
class SkippedContract:
    contract: str
    reason: SkipReason
    details: str = ""


@dataclass
class UnsupportedFeature:
    feature: str
    occurrences: int
    locations: list[str] = field(default_factory=list)


@dataclass
class BuildInfo:
    solc_version: str = ""
    framework: str = ""
    optimizer_enabled: bool = False
    slither_version: str = ""


@dataclass
class CompletenessReport:
    """Tracks analysis completeness."""

    contracts_analyzed: list[str] = field(default_factory=list)
    skipped: list[SkippedContract] = field(default_factory=list)
    unsupported_features: list[UnsupportedFeature] = field(default_factory=list)
    warnings: list[str] = field(default_factory=list)
    build_info: BuildInfo = field(default_factory=BuildInfo)

    @property
    def status(self) -> CompletionStatus:
        if not self.contracts_analyzed and self.skipped:
            return CompletionStatus.FAILED
        if self.skipped:
            return CompletionStatus.PARTIAL
        return CompletionStatus.COMPLETE

    @property
    def coverage_pct(self) -> float:
        total = len(self.contracts_analyzed) + len(self.skipped)
        if total == 0:
            return 0.0
        return (len(self.contracts_analyzed) / total) * 100

    @property
    def contracts_total(self) -> int:
        return len(self.contracts_analyzed) + len(self.skipped)

    def add_analyzed(self, contract: str) -> None:
        self.contracts_analyzed.append(contract)

    def add_skipped(
        self,
        contract: str,
        reason: SkipReason,
        details: str = ""
    ) -> None:
        self.skipped.append(SkippedContract(contract, reason, details))

    def add_unsupported_feature(
        self,
        feature: str,
        location: str
    ) -> None:
        # Check if feature already tracked
        for uf in self.unsupported_features:
            if uf.feature == feature:
                uf.occurrences += 1
                uf.locations.append(location)
                return
        self.unsupported_features.append(
            UnsupportedFeature(feature, 1, [location])
        )

    def add_warning(self, warning: str) -> None:
        self.warnings.append(warning)

    def to_dict(self) -> dict[str, Any]:
        return {
            "status": self.status.value,
            "coverage_pct": round(self.coverage_pct, 1),
            "contracts_analyzed": len(self.contracts_analyzed),
            "contracts_skipped": len(self.skipped),
            "contracts_total": self.contracts_total,
            "skipped_details": [
                {
                    "contract": s.contract,
                    "reason": s.reason.value,
                    "details": s.details,
                }
                for s in self.skipped
            ],
            "unsupported_features": [
                {
                    "feature": uf.feature,
                    "occurrences": uf.occurrences,
                    "locations": uf.locations[:5],  # Limit for brevity
                }
                for uf in self.unsupported_features
            ],
            "warnings": self.warnings,
            "build_info": {
                "solc_version": self.build_info.solc_version,
                "framework": self.build_info.framework,
                "optimizer_enabled": self.build_info.optimizer_enabled,
                "slither_version": self.build_info.slither_version,
            },
        }

    def to_json(self) -> str:
        return json.dumps(self.to_dict(), indent=2)


def generate_completeness_report(
    analyzed: list[str],
    skipped: list[tuple[str, SkipReason, str]],
    build_info: dict[str, Any] | None = None,
) -> CompletenessReport:
    """Factory function for completeness report."""
    report = CompletenessReport()

    for contract in analyzed:
        report.add_analyzed(contract)

    for contract, reason, details in skipped:
        report.add_skipped(contract, reason, details)

    if build_info:
        report.build_info = BuildInfo(**build_info)

    return report
```

### Subtask 2.10c: Integrate with Builder (2h)

**Modify:** `src/true_vkg/kg/builder.py`

```python
# Add to VKGBuilder class

def build(self, path: Path) -> tuple[KnowledgeGraph, CompletenessReport]:
    """Build graph and return completeness report."""
    report = CompletenessReport()

    # Track build info
    report.build_info.framework = self._detect_framework(path)
    report.build_info.solc_version = self._get_solc_version()

    for contract_path in self._discover_contracts(path):
        try:
            contract_graph = self._analyze_contract(contract_path)
            graph.merge(contract_graph)
            report.add_analyzed(contract_path.name)

        except InlineAssemblyError as e:
            report.add_skipped(
                contract_path.name,
                SkipReason.INLINE_ASSEMBLY,
                str(e)
            )
            report.add_unsupported_feature(
                "inline_assembly",
                str(contract_path)
            )

        except ProxyResolutionError as e:
            report.add_skipped(
                contract_path.name,
                SkipReason.PROXY_UNRESOLVED,
                str(e)
            )
            report.add_warning(f"Proxy at {e.address} not resolved")

        except SlitherParseError as e:
            report.add_skipped(
                contract_path.name,
                SkipReason.PARSE_ERROR,
                str(e)
            )

        except UnsupportedSolcVersion as e:
            report.add_skipped(
                contract_path.name,
                SkipReason.UNSUPPORTED_SOLC,
                f"Requires {e.required}, have {e.available}"
            )

        except TimeoutError:
            report.add_skipped(
                contract_path.name,
                SkipReason.TIMEOUT,
                f"Analysis exceeded {self.timeout}s"
            )

    return graph, report
```

```python
@click.command()
def analyze(path: str, verbose: bool = False):
    """Analyze contracts and show completeness."""
    graph, report = builder.build(Path(path))

    # Always show completeness summary
    click.echo(_format_completeness_summary(report))

    if report.status == CompletionStatus.PARTIAL:
        click.echo()
        click.echo(click.style("WARNINGS:", fg="yellow", bold=True))
        for s in report.skipped:
            click.echo(f"  [!] {s.contract}: {s.reason.value}")
            if verbose and s.details:
                click.echo(f"      {s.details}")

    if report.unsupported_features:
        click.echo()
        click.echo("Unsupported features detected:")
        for uf in report.unsupported_features:
            click.echo(f"  - {uf.feature}: {uf.occurrences} occurrence(s)")

    # Continue with findings...

def _format_completeness_summary(report: CompletenessReport) -> str:
    """Format completeness for CLI output."""
    status_colors = {
        CompletionStatus.COMPLETE: "green",
        CompletionStatus.PARTIAL: "yellow",
        CompletionStatus.FAILED: "red",
    }

    status_text = {
        CompletionStatus.COMPLETE: "COMPLETE",
        CompletionStatus.PARTIAL: "PARTIAL COVERAGE",
        CompletionStatus.FAILED: "FAILED",
    }

    color = status_colors[report.status]
    text = status_text[report.status]

    return (
        f"\nAnalysis {click.style(text, fg=color, bold=True)}\n"
        f"Coverage: {report.coverage_pct:.1f}% "
        f"({len(report.contracts_analyzed)}/{report.contracts_total} contracts)\n"
        f"Build: {report.build_info.framework} | "
        f"Solc {report.build_info.solc_version}"
    )
```

```python
import pytest
from true_vkg.report.completeness import (
    CompletenessReport,
    CompletionStatus,
    SkipReason,
)


class TestCompletenessReport:
    def test_complete_status(self):
        report = CompletenessReport()
        report.add_analyzed("A.sol")
        report.add_analyzed("B.sol")
        assert report.status == CompletionStatus.COMPLETE
        assert report.coverage_pct == 100.0

    def test_partial_status(self):
        report = CompletenessReport()
        report.add_analyzed("A.sol")
        report.add_skipped("B.sol", SkipReason.INLINE_ASSEMBLY)
        assert report.status == CompletionStatus.PARTIAL
        assert report.coverage_pct == 50.0

    def test_failed_status(self):
        report = CompletenessReport()
        report.add_skipped("A.sol", SkipReason.PARSE_ERROR)
        assert report.status == CompletionStatus.FAILED
        assert report.coverage_pct == 0.0

        report = CompletenessReport()
```
