## Design

### Context Layers (Priority Order)

```
Priority 1: Vulnerable Function (ALWAYS include)
           - Full function code
           - NatSpec comments
           - Access modifiers

Priority 2: Direct Context (include if budget allows)
           - Modifiers applied to function
           - Internal functions called
           - State variables accessed

Priority 3: Extended Context (include if budget remains)
           - Functions calling this function
           - Related state variables
           - Inherited contracts

Priority 4: Ecosystem Context (last priority)
           - External contracts called
           - Interface definitions
           - Related events
```

### Token Budget Management

```python
class TokenBudget:

            "ecosystem": 0.1,            # 10% for ecosystem
        }
```

```python
"""
Context Slicing for LLM Analysis

Extracts minimal, relevant context for LLM analysis while respecting token budgets.
Uses priority-based inclusion to ensure most important context is always present.
"""

import tiktoken
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Set
from enum import IntEnum
import logging

from true_vkg.kg.schema import KnowledgeGraph, FunctionNode, StateVariableNode

logger = logging.getLogger(__name__)


class ContextPriority(IntEnum):
    """Priority levels for context inclusion."""
    CRITICAL = 1    # Always include (vulnerable function)
    HIGH = 2        # Include if budget allows (modifiers, direct calls)
    MEDIUM = 3      # Include if budget remains (callers, state)
    LOW = 4         # Last priority (ecosystem)


@dataclass
class ContextItem:
    """Single item of context to include."""
    content: str
    priority: ContextPriority
    label: str
    tokens: int = 0
    included: bool = False


@dataclass
class SlicedContext:
    """Result of context slicing."""
    content: str
    tokens_used: int
    tokens_budget: int
    items_included: List[str]
    items_excluded: List[str]

    def to_dict(self) -> dict:
        return {
            "tokens_used": self.tokens_used,
            "tokens_budget": self.tokens_budget,
            "included": self.items_included,
            "excluded": self.items_excluded,
            "utilization": self.tokens_used / self.tokens_budget if self.tokens_budget > 0 else 0,
        }


class TokenEstimator:
    """Estimate token counts for text."""

    def __init__(self, model: str = "cl100k_base"):
        try:
            self.encoding = tiktoken.get_encoding(model)
        except Exception:
            self.encoding = None

    def count(self, text: str) -> int:
        """Count tokens in text."""
        if self.encoding:
            return len(self.encoding.encode(text))
        # Fallback: rough estimate (4 chars per token)
        return len(text) // 4


class ContextSlicer:
    """
    Extract relevant context for LLM analysis.

    Features:
    - Priority-based inclusion
    - Token budget management
    - Call graph traversal
    - State variable context
    """

    def __init__(self, token_budget: int = 4000):
        self.token_budget = token_budget
        self.estimator = TokenEstimator()

    def slice(
        self,
        function: FunctionNode,
        graph: KnowledgeGraph,
        finding_context: str = "",
    ) -> SlicedContext:
        """
        Extract context for analyzing a function.

        Args:
            function: The function to analyze
            graph: Knowledge graph with relationships
            finding_context: Additional context about the finding

        Returns:
            SlicedContext with prioritized content
        """
        items: List[ContextItem] = []

        # Priority 1: Vulnerable function (CRITICAL)
        items.append(ContextItem(
            content=self._format_function(function),
            priority=ContextPriority.CRITICAL,
            label=f"Function: {function.name}",
        ))

        # Priority 1: Finding context (CRITICAL)
        if finding_context:
            items.append(ContextItem(
                content=f"## Finding\n{finding_context}",
                priority=ContextPriority.CRITICAL,
                label="Finding context",
            ))

        # Priority 2: Modifiers (HIGH)
        for modifier in self._get_modifiers(function, graph):
            items.append(ContextItem(
                content=self._format_modifier(modifier),
                priority=ContextPriority.HIGH,
                label=f"Modifier: {modifier.name}",
            ))

        # Priority 2: Called functions (HIGH)
        for called in self._get_called_functions(function, graph):
            items.append(ContextItem(
                content=self._format_function(called),
                priority=ContextPriority.HIGH,
                label=f"Called: {called.name}",
            ))

        # Priority 3: State variables (MEDIUM)
        for var in self._get_state_variables(function, graph):
            items.append(ContextItem(
                content=self._format_state_variable(var),
                priority=ContextPriority.MEDIUM,
                label=f"State: {var.name}",
            ))

        # Priority 3: Callers (MEDIUM) - who calls this function
        for caller in self._get_callers(function, graph):
            items.append(ContextItem(
                content=self._format_function_summary(caller),
                priority=ContextPriority.MEDIUM,
                label=f"Caller: {caller.name}",
            ))

        # Priority 4: External contracts (LOW)
        for external in self._get_external_calls(function, graph):
            items.append(ContextItem(
                content=self._format_external_call(external),
                priority=ContextPriority.LOW,
                label=f"External: {external}",
            ))

        # Estimate tokens for all items
        for item in items:
            item.tokens = self.estimator.count(item.content)

        # Select items within budget
        selected = self._select_within_budget(items)

        # Build final context
        content_parts = []
        included = []
        excluded = []
        tokens_used = 0

        for item in items:
            if item.included:
                content_parts.append(item.content)
                included.append(item.label)
                tokens_used += item.tokens
            else:
                excluded.append(f"{item.label} ({item.tokens} tokens)")

        return SlicedContext(
            content="\n\n".join(content_parts),
            tokens_used=tokens_used,
            tokens_budget=self.token_budget,
            items_included=included,
            items_excluded=excluded,
        )

    def _select_within_budget(self, items: List[ContextItem]) -> List[ContextItem]:
        """Select items within token budget, prioritizing by priority."""
        # Sort by priority (lower = higher priority)
        sorted_items = sorted(items, key=lambda x: x.priority)

        remaining = self.token_budget

        for item in sorted_items:
            if item.tokens <= remaining:
                item.included = True
                remaining -= item.tokens
            elif item.priority == ContextPriority.CRITICAL:
                # Critical items must be included, even if over budget
                item.included = True
                remaining -= item.tokens
                logger.warning(f"Critical item exceeds budget: {item.label}")

        return sorted_items

    def _format_function(self, func: FunctionNode) -> str:
        """Format function for context."""
        parts = [f"## Function: {func.name}"]

        if hasattr(func, 'natspec') and func.natspec:
            parts.append(f"/**\n{func.natspec}\n*/")

        parts.append(f"Visibility: {func.visibility}")
        parts.append(f"Modifiers: {', '.join(func.modifiers) if func.modifiers else 'none'}")

        if hasattr(func, 'source_code') and func.source_code:
            parts.append(f"```solidity\n{func.source_code}\n```")

        return "\n".join(parts)

    def _format_function_summary(self, func: FunctionNode) -> str:
        """Format function summary (no code, just signature)."""
        return f"- {func.visibility} {func.name}({', '.join(func.parameters) if hasattr(func, 'parameters') else ''})"

    def _format_modifier(self, modifier) -> str:
        """Format modifier for context."""
        parts = [f"## Modifier: {modifier.name}"]
        if hasattr(modifier, 'source_code') and modifier.source_code:
            parts.append(f"```solidity\n{modifier.source_code}\n```")
        return "\n".join(parts)

    def _format_state_variable(self, var: StateVariableNode) -> str:
        """Format state variable for context."""
        return f"- {var.visibility} {var.type_name} {var.name}"

    def _format_external_call(self, call: str) -> str:
        """Format external call for context."""
        return f"- External call: {call}"

    def _get_modifiers(self, func: FunctionNode, graph: KnowledgeGraph) -> List:
        """Get modifiers applied to function."""
        modifiers = []
        if hasattr(func, 'modifiers') and func.modifiers:
            for mod_name in func.modifiers:
                # Look up modifier in graph
                mod_nodes = [n for n in graph.nodes if getattr(n, 'name', '') == mod_name and getattr(n, 'node_type', '') == 'Modifier']
                modifiers.extend(mod_nodes)
        return modifiers

    def _get_called_functions(self, func: FunctionNode, graph: KnowledgeGraph) -> List[FunctionNode]:
        """Get functions called by this function."""
        called = []
        for edge in graph.edges:
            if edge.source_id == func.id and edge.edge_type == 'CALLS':
                target = graph.get_node(edge.target_id)
                if target and hasattr(target, 'node_type') and target.node_type == 'Function':
                    called.append(target)
        return called[:5]  # Limit to 5 most relevant

    def _get_callers(self, func: FunctionNode, graph: KnowledgeGraph) -> List[FunctionNode]:
        """Get functions that call this function."""
        callers = []
        for edge in graph.edges:
            if edge.target_id == func.id and edge.edge_type == 'CALLS':
                source = graph.get_node(edge.source_id)
                if source and hasattr(source, 'node_type') and source.node_type == 'Function':
                    callers.append(source)
        return callers[:3]  # Limit to 3 most relevant

    def _get_state_variables(self, func: FunctionNode, graph: KnowledgeGraph) -> List[StateVariableNode]:
        """Get state variables accessed by function."""
        variables = []
        for edge in graph.edges:
            if edge.source_id == func.id and edge.edge_type in ['READS', 'WRITES']:
                target = graph.get_node(edge.target_id)
                if target and hasattr(target, 'node_type') and target.node_type == 'StateVariable':
                    variables.append(target)
        return variables[:10]  # Limit to 10

    def _get_external_calls(self, func: FunctionNode, graph: KnowledgeGraph) -> List[str]:
        """Get external calls made by function."""
        calls = []
        if hasattr(func, 'external_calls'):
            calls = func.external_calls[:5]
        return calls


def slice_context_for_finding(
    finding: dict,
    graph: KnowledgeGraph,
    budget: int = 4000,
) -> SlicedContext:
    """
    Convenience function to slice context for a finding.

    Args:
        finding: Finding dict with 'function_id' and 'description'
        graph: Knowledge graph
        budget: Token budget

    Returns:
        SlicedContext with relevant code
    """
    slicer = ContextSlicer(token_budget=budget)

    func = graph.get_node(finding.get('function_id'))
    if not func:
        return SlicedContext(
            content="Error: Function not found in graph",
            tokens_used=0,
            tokens_budget=budget,
            items_included=[],
            items_excluded=["Function not found"],
        )

    return slicer.slice(
        function=func,
        graph=graph,
        finding_context=finding.get('description', ''),
    )
```

```python
        critical_item = next(i for i in selected if i.label == "Critical")

            items_excluded=["caller"],

        d = context.to_dict()

        assert d["utilization"] == 0.1
```
