## Objective

## Context

### Tier System

```
Tier A: Deterministic (patterns, properties)
        Always runs first, no API costs
        |
        v
Tier B: LLM-Enhanced (context-aware verification)
        Only for uncertain findings (0.3-0.9 confidence)
        Adds cost, adds context-awareness
```

### Confidence Routing

| Confidence | Action | Rationale |
|------------|--------|-----------|
| >= 0.9 | Auto-confirm | High certainty, LLM adds noise |
| 0.3 - 0.9 | Send to Tier B | Uncertain, LLM can help |
| <= 0.3 | Auto-dismiss | Low signal, not worth LLM cost |

---

## Design

"""
Tier B Analysis Workflow

End-to-end pipeline for LLM-enhanced vulnerability verification.
Routes findings through confidence-based filtering and LLM analysis.
"""

import asyncio
from dataclasses import dataclass, field
from enum import Enum
from typing import List, Optional, Dict, Any
import logging
import json

from .client import UnifiedLLMClient, LLMResult, ModelTier
from .slicer import ContextSlicer, slice_context_for_finding
from .limits import RateLimiter, LLMLimits, LimitExceededError

logger = logging.getLogger(__name__)


class RoutingDecision(Enum):
    """Routing decision for a finding."""
    AUTO_CONFIRM = "auto_confirm"      # High confidence, skip LLM
    AUTO_DISMISS = "auto_dismiss"      # Low confidence, skip LLM
    NEEDS_VERIFICATION = "verify"      # Send to LLM


@dataclass
class TierAFinding:
    """Finding from Tier A (deterministic) analysis."""
    id: str
    pattern_id: str
    function_id: str
    description: str
    severity: str
    confidence: float  # 0.0 - 1.0
    evidence: List[Dict[str, Any]] = field(default_factory=list)

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "pattern_id": self.pattern_id,
            "function_id": self.function_id,
            "description": self.description,
            "severity": self.severity,
            "confidence": self.confidence,
            "evidence": self.evidence,
        }


@dataclass
class TierBResult:
    """Result from Tier B (LLM) verification."""
    finding_id: str
    routing: RoutingDecision
    verdict: Optional[str] = None         # VULNERABLE, SAFE, UNCERTAIN
    llm_confidence: Optional[float] = None # 0-100
    reasoning: Optional[str] = None
    evidence_refs: List[str] = field(default_factory=list)
    tokens_used: int = 0
    cost_usd: float = 0.0
    error: Optional[str] = None

    @property
    def final_verdict(self) -> str:
        """Get final verdict based on routing and LLM result."""
        if self.routing == RoutingDecision.AUTO_CONFIRM:
            return "VULNERABLE"
        elif self.routing == RoutingDecision.AUTO_DISMISS:
            return "DISMISSED"
        else:
            return self.verdict or "UNCERTAIN"

    def to_dict(self) -> dict:
        return {
            "finding_id": self.finding_id,
            "routing": self.routing.value,
            "verdict": self.final_verdict,
            "llm_confidence": self.llm_confidence,
            "reasoning": self.reasoning,
            "evidence_refs": self.evidence_refs,
            "tokens_used": self.tokens_used,
            "cost_usd": self.cost_usd,
            "error": self.error,
        }


@dataclass
class TierBConfig:
    """Configuration for Tier B workflow."""
    auto_confirm_threshold: float = 0.9
    auto_dismiss_threshold: float = 0.3
    token_budget: int = 4000
    max_retries: int = 3
    model_tier: ModelTier = ModelTier.DEFAULT


class ConfidenceRouter:
    """Routes findings based on confidence thresholds."""

    def __init__(self, config: TierBConfig):
        self.config = config

    def route(self, finding: TierAFinding) -> RoutingDecision:
        """Determine routing for a finding."""
        if finding.confidence >= self.config.auto_confirm_threshold:
            logger.debug(f"Auto-confirming {finding.id}: confidence {finding.confidence:.2f}")
            return RoutingDecision.AUTO_CONFIRM

        if finding.confidence <= self.config.auto_dismiss_threshold:
            logger.debug(f"Auto-dismissing {finding.id}: confidence {finding.confidence:.2f}")
            return RoutingDecision.AUTO_DISMISS

        logger.debug(f"Sending {finding.id} to Tier B: confidence {finding.confidence:.2f}")
        return RoutingDecision.NEEDS_VERIFICATION


class PromptGenerator:
    """Generates verification prompts for LLM."""

    VERIFICATION_TEMPLATE = """You are a security auditor verifying a potential vulnerability in Solidity code.

## Finding to Verify
Pattern: {pattern_id}
Severity: {severity}
Description: {description}

## Code Context
{context}

## Your Task
Analyze the code and determine if this is a real vulnerability or a false positive.

Consider:
1. Is the vulnerability condition actually reachable?
2. Are there mitigating factors (guards, checks, access control)?
3. What is the actual impact if exploited?

## Response Format (JSON only)
{{
    "verdict": "VULNERABLE" | "SAFE" | "UNCERTAIN",
    "confidence": 0-100,
    "reasoning": "Detailed explanation of your analysis",
    "evidence": ["Specific code references supporting your verdict"]
}}"""

    def generate(
        self,
        finding: TierAFinding,
        context: str,
    ) -> str:
        """Generate verification prompt."""
        return self.VERIFICATION_TEMPLATE.format(
            pattern_id=finding.pattern_id,
            severity=finding.severity,
            description=finding.description,
            context=context,
        )


class ResponseParser:
    """Parses LLM responses into structured results."""

    VALID_VERDICTS = {"VULNERABLE", "SAFE", "UNCERTAIN"}

    def parse(self, response: LLMResult, finding_id: str) -> TierBResult:
        """Parse LLM response into TierBResult."""
        result = TierBResult(
            finding_id=finding_id,
            routing=RoutingDecision.NEEDS_VERIFICATION,
            tokens_used=response.input_tokens + response.output_tokens,
            cost_usd=response.cost_usd,
        )

        try:
            # Try to parse JSON from response
            parsed = response.parsed
            if parsed is None:
                # Try manual extraction
                parsed = self._extract_json(response.content)

            if parsed:
                result.verdict = self._validate_verdict(parsed.get("verdict"))
                result.llm_confidence = self._validate_confidence(parsed.get("confidence"))
                result.reasoning = parsed.get("reasoning", "")
                result.evidence_refs = parsed.get("evidence", [])
            else:
                result.error = "Failed to parse JSON from response"
                result.verdict = "UNCERTAIN"

        except Exception as e:
            result.error = str(e)
            result.verdict = "UNCERTAIN"
            logger.warning(f"Failed to parse response for {finding_id}: {e}")

        return result

    def _extract_json(self, content: str) -> Optional[dict]:
        """Extract JSON from response content."""
        import re

        # Try to find JSON in response
        json_match = re.search(r'\{[^{}]*\}', content, re.DOTALL)
        if json_match:
            try:
                return json.loads(json_match.group())
            except json.JSONDecodeError:
                pass

        return None

    def _validate_verdict(self, verdict: Any) -> str:
        """Validate and normalize verdict."""
        if isinstance(verdict, str):
            verdict = verdict.upper()
            if verdict in self.VALID_VERDICTS:
                return verdict
        return "UNCERTAIN"

    def _validate_confidence(self, confidence: Any) -> Optional[float]:
        """Validate confidence value."""
        if confidence is None:
            return None
        try:
            conf = float(confidence)
            return max(0, min(100, conf))
        except (TypeError, ValueError):
            return None


class TierBWorkflow:
    """
    End-to-end Tier B analysis workflow.

    Orchestrates:
    - Confidence-based routing
    - Context slicing
    - LLM verification
    - Result parsing
    """

    def __init__(
        self,
        llm_client: UnifiedLLMClient = None,
        config: TierBConfig = None,
        limiter: RateLimiter = None,
    ):
        self.config = config or TierBConfig()
        self.llm_client = llm_client
        self.limiter = limiter or RateLimiter(LLMLimits())

        self.router = ConfidenceRouter(self.config)
        self.slicer = ContextSlicer(self.config.token_budget)
        self.prompt_generator = PromptGenerator()
        self.response_parser = ResponseParser()

        # Stats
        self.stats = {
            "auto_confirmed": 0,
            "auto_dismissed": 0,
            "llm_verified": 0,
            "llm_vulnerable": 0,
            "llm_safe": 0,
            "llm_uncertain": 0,
            "errors": 0,
        }

    async def analyze(
        self,
        finding: TierAFinding,
        graph: Any,  # KnowledgeGraph
    ) -> TierBResult:
        """
        Analyze a single finding with Tier B workflow.

        Args:
            finding: Tier A finding to verify
            graph: Knowledge graph for context extraction

        Returns:
            TierBResult with verdict and metadata
        """
        # Route based on confidence
        routing = self.router.route(finding)

        if routing == RoutingDecision.AUTO_CONFIRM:
            self.stats["auto_confirmed"] += 1
            return TierBResult(
                finding_id=finding.id,
                routing=routing,
                verdict="VULNERABLE",
                llm_confidence=None,
                reasoning="Auto-confirmed: High Tier A confidence",
            )

        if routing == RoutingDecision.AUTO_DISMISS:
            self.stats["auto_dismissed"] += 1
            return TierBResult(
                finding_id=finding.id,
                routing=routing,
                verdict="DISMISSED",
                llm_confidence=None,
                reasoning="Auto-dismissed: Low Tier A confidence",
            )

        # Need LLM verification
        return await self._llm_verify(finding, graph)

    async def _llm_verify(
        self,
        finding: TierAFinding,
        graph: Any,
    ) -> TierBResult:
        """Verify finding using LLM."""
        if self.llm_client is None:
            return TierBResult(
                finding_id=finding.id,
                routing=RoutingDecision.NEEDS_VERIFICATION,
                verdict="UNCERTAIN",
                error="LLM client not configured",
            )

        # Slice context
        context_result = slice_context_for_finding(
            finding.to_dict(),
            graph,
            budget=self.config.token_budget,
        )

        # Generate prompt
        prompt = self.prompt_generator.generate(finding, context_result.content)

        # Estimate tokens and check limits
        estimated_tokens = context_result.tokens_used + 500  # Prompt overhead
        try:
            self.limiter.require_capacity(estimated_tokens)
        except LimitExceededError as e:
            return TierBResult(
                finding_id=finding.id,
                routing=RoutingDecision.NEEDS_VERIFICATION,
                verdict="UNCERTAIN",
                error=str(e),
            )

        # Call LLM with retries
        for attempt in range(self.config.max_retries):
            try:
                response = await self.llm_client.analyze(
                    context=context_result.content,
                    prompt=prompt,
                    tier=self.config.model_tier,
                )

                # Record usage
                self.limiter.record_usage(
                    tokens=response.input_tokens + response.output_tokens,
                    cost_usd=response.cost_usd,
                )

                # Parse response
                result = self.response_parser.parse(response, finding.id)

                # Update stats
                self.stats["llm_verified"] += 1
                if result.verdict == "VULNERABLE":
                    self.stats["llm_vulnerable"] += 1
                elif result.verdict == "SAFE":
                    self.stats["llm_safe"] += 1
                else:
                    self.stats["llm_uncertain"] += 1

                return result

            except Exception as e:
                logger.warning(f"LLM attempt {attempt + 1} failed: {e}")
                if attempt == self.config.max_retries - 1:
                    self.stats["errors"] += 1
                    return TierBResult(
                        finding_id=finding.id,
                        routing=RoutingDecision.NEEDS_VERIFICATION,
                        verdict="UNCERTAIN",
                        error=f"Max retries exceeded: {e}",
                    )

        # Should not reach here
        return TierBResult(
            finding_id=finding.id,
            routing=RoutingDecision.NEEDS_VERIFICATION,
            verdict="UNCERTAIN",
            error="Unexpected error in LLM verification",
        )

    async def analyze_batch(
        self,
        findings: List[TierAFinding],
        graph: Any,
    ) -> List[TierBResult]:
        """
        Analyze multiple findings.

        Args:
            findings: List of Tier A findings
            graph: Knowledge graph

        Returns:
            List of TierBResults
        """
        results = []
        for finding in findings:
            result = await self.analyze(finding, graph)
            results.append(result)
        return results

    def get_stats(self) -> dict:
        """Get workflow statistics."""
        return {
            **self.stats,
            "total_processed": sum([
                self.stats["auto_confirmed"],
                self.stats["auto_dismissed"],
                self.stats["llm_verified"],
            ]),
            "llm_usage": self.limiter.get_usage_summary(),
        }

"""
Confidence Thresholds and Calculations

Manages confidence routing thresholds for Tier B workflow.
"""

from dataclasses import dataclass
from typing import Dict, Any


@dataclass
class ConfidenceThresholds:
    """Configurable confidence thresholds."""
    auto_confirm: float = 0.9    # >= this: auto-confirm
    auto_dismiss: float = 0.3    # <= this: auto-dismiss
    high_confidence: float = 0.8  # Classification boundary
    medium_confidence: float = 0.5

    def classify(self, confidence: float) -> str:
        """Classify confidence level."""
        if confidence >= self.auto_confirm:
            return "very_high"
        elif confidence >= self.high_confidence:
            return "high"
        elif confidence >= self.medium_confidence:
            return "medium"
        elif confidence > self.auto_dismiss:
            return "low"
        else:
            return "very_low"

    def should_verify(self, confidence: float) -> bool:
        """Check if finding should go to Tier B."""
        return self.auto_dismiss < confidence < self.auto_confirm

    @classmethod
    def from_config(cls, config: Dict[str, Any]) -> "ConfidenceThresholds":
        """Create from config dict."""
        return cls(
            auto_confirm=config.get("auto_confirm_threshold", 0.9),
            auto_dismiss=config.get("auto_dismiss_threshold", 0.3),
            high_confidence=config.get("high_confidence", 0.8),
            medium_confidence=config.get("medium_confidence", 0.5),
        )
