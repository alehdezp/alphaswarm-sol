## Objective

Establish formal input/output contract for LLM interactions with schema validation.

---

## Context

### Why Contracts Matter

Without contracts:
- LLM output varies unpredictably
- Parsing breaks on edge cases
- No way to detect malformed responses
- Hard to debug failures

### Reference

See existing spec: `./docs/contracts/llm-prompt-contract.md`

### 3. Create: `src/true_vkg/llm/contract.py`

```python
"""
LLM Prompt Contract

Enforces input/output schemas for LLM interactions.
"""

import json
import os
from typing import Dict, Any, Optional, Tuple
from dataclasses import dataclass
import logging

try:
    import jsonschema
    from jsonschema import validate, ValidationError
    HAS_JSONSCHEMA = True
except ImportError:
    HAS_JSONSCHEMA = False
    ValidationError = Exception

logger = logging.getLogger(__name__)


# Schema file paths
SCHEMA_DIR = os.path.join(os.path.dirname(__file__), "..", "..", "..", "schemas")
INPUT_SCHEMA_PATH = os.path.join(SCHEMA_DIR, "llm_input.json")
RESPONSE_SCHEMA_PATH = os.path.join(SCHEMA_DIR, "llm_response.json")


@dataclass
class ContractViolation:
    """Details of a contract violation."""
    schema_type: str  # "input" or "response"
    message: str
    path: str
    value: Any


class PromptContract:
    """
    Enforces LLM prompt contracts via JSON Schema validation.

    Features:
    - Input validation before sending to LLM
    - Output validation after receiving response
    - Detailed error messages for violations
    - Recovery suggestions for common issues
    """

    def __init__(self, schema_dir: str = None):
        self.schema_dir = schema_dir or SCHEMA_DIR
        self._input_schema: Optional[Dict] = None
        self._response_schema: Optional[Dict] = None
        self._load_schemas()

    def _load_schemas(self):
        """Load JSON schemas from files."""
        try:
            input_path = os.path.join(self.schema_dir, "llm_input.json")
            response_path = os.path.join(self.schema_dir, "llm_response.json")

            if os.path.exists(input_path):
                with open(input_path) as f:
                    self._input_schema = json.load(f)

            if os.path.exists(response_path):
                with open(response_path) as f:
                    self._response_schema = json.load(f)

        except Exception as e:
            logger.warning(f"Failed to load schemas: {e}")

    def validate_input(self, input_data: Dict[str, Any]) -> Tuple[bool, Optional[ContractViolation]]:
        """
        Validate input against schema.

        Args:
            input_data: Input data to validate

        Returns:
            Tuple of (is_valid, violation_or_none)
        """
        if not HAS_JSONSCHEMA:
            logger.warning("jsonschema not installed, skipping input validation")
            return True, None

        if self._input_schema is None:
            logger.warning("Input schema not loaded")
            return True, None

        try:
            validate(instance=input_data, schema=self._input_schema)
            return True, None
        except ValidationError as e:
            violation = ContractViolation(
                schema_type="input",
                message=e.message,
                path=".".join(str(p) for p in e.path),
                value=e.instance,
            )
            return False, violation

    def validate_response(self, response_data: Dict[str, Any]) -> Tuple[bool, Optional[ContractViolation]]:
        """
        Validate LLM response against schema.

        Args:
            response_data: Response data to validate

        Returns:
            Tuple of (is_valid, violation_or_none)
        """
        if not HAS_JSONSCHEMA:
            logger.warning("jsonschema not installed, skipping response validation")
            return True, None

        if self._response_schema is None:
            logger.warning("Response schema not loaded")
            return True, None

        try:
            validate(instance=response_data, schema=self._response_schema)
            return True, None
        except ValidationError as e:
            violation = ContractViolation(
                schema_type="response",
                message=e.message,
                path=".".join(str(p) for p in e.path),
                value=e.instance,
            )
            return False, violation

    def enforce_input(self, input_data: Dict[str, Any]):
        """
        Enforce input contract, raising on violation.

        Args:
            input_data: Input to enforce

        Raises:
            ContractViolationError: If input violates contract
        """
        is_valid, violation = self.validate_input(input_data)
        if not is_valid:
            raise ContractViolationError(violation)

    def enforce_response(self, response_data: Dict[str, Any]):
        """
        Enforce response contract, raising on violation.

        Args:
            response_data: Response to enforce

        Raises:
            ContractViolationError: If response violates contract
        """
        is_valid, violation = self.validate_response(response_data)
        if not is_valid:
            raise ContractViolationError(violation)

    def fix_response(self, response_data: Dict[str, Any]) -> Dict[str, Any]:
        """
        Attempt to fix common response issues.

        Args:
            response_data: Potentially invalid response

        Returns:
            Fixed response (best effort)
        """
        fixed = response_data.copy()

        # Normalize verdict
        if "verdict" in fixed:
            verdict = str(fixed["verdict"]).upper()
            if verdict not in ("VULNERABLE", "SAFE", "UNCERTAIN"):
                fixed["verdict"] = "UNCERTAIN"
            else:
                fixed["verdict"] = verdict

        # Ensure confidence in range
        if "confidence" in fixed:
            try:
                conf = float(fixed["confidence"])
                fixed["confidence"] = max(0, min(100, conf))
            except (TypeError, ValueError):
                fixed["confidence"] = 50

        # Ensure reasoning exists
        if not fixed.get("reasoning"):
            fixed["reasoning"] = "No reasoning provided"

        # Ensure evidence is array
        if "evidence" in fixed and not isinstance(fixed["evidence"], list):
            if isinstance(fixed["evidence"], str):
                fixed["evidence"] = [fixed["evidence"]]
            else:
                fixed["evidence"] = []

        return fixed


class ContractViolationError(Exception):
    """Raised when a contract is violated."""

    def __init__(self, violation: ContractViolation):
        self.violation = violation
        super().__init__(
            f"Contract violation ({violation.schema_type}): "
            f"{violation.message} at {violation.path}"
        )


def build_prompt_input(
    finding_id: str,
    pattern_id: str,
    description: str,
    code: str,
    prompt_type: str = "verify",
    severity: str = "medium",
    confidence: float = 0.5,
    modifiers: list = None,
    state_variables: list = None,
) -> Dict[str, Any]:
    """
    Build a contract-compliant prompt input.

    Args:
        finding_id: Finding identifier
        pattern_id: Pattern that triggered finding
        description: Finding description
        code: Code to analyze
        prompt_type: Type of analysis
        severity: Severity level
        confidence: Tier A confidence
        modifiers: Modifiers applied
        state_variables: State variables accessed

    Returns:
        Contract-compliant input dict
    """
    return {
        "prompt_type": prompt_type,
        "finding": {
            "id": finding_id,
            "pattern_id": pattern_id,
            "description": description,
            "severity": severity,
            "confidence": confidence,
        },
        "context": {
            "code": code,
            "modifiers": modifiers or [],
            "state_variables": state_variables or [],
        },
    }


# Global contract instance
_contract: Optional[PromptContract] = None


def get_contract() -> PromptContract:
    """Get global prompt contract."""
    global _contract
    if _contract is None:
        _contract = PromptContract()
    return _contract
```

    def test_validate_valid_input(self, contract):
        """Valid input passes validation."""
        input_data = {
            "prompt_type": "verify",
            "finding": {
                "id": "f1",
                "pattern_id": "reentrancy",
                "description": "test",
            },
            "context": {
                "code": "function test() {}",
            }
        }

        is_valid, violation = contract.validate_input(input_data)
        assert is_valid is True
        assert violation is None

    def test_validate_invalid_input(self, contract):
        """Invalid input fails validation."""
        input_data = {
            "prompt_type": "invalid_type",  # Invalid
            "finding": {},  # Missing required fields
            "context": {}
        }

        is_valid, violation = contract.validate_input(input_data)
        assert is_valid is False
        assert violation is not None

    def test_validate_valid_response(self, contract):
        """Valid response passes validation."""
        response = {
            "verdict": "VULNERABLE",
            "confidence": 85,
            "reasoning": "This is a detailed reasoning",
        }

        is_valid, violation = contract.validate_response(response)
        assert is_valid is True

    def test_validate_invalid_response(self, contract):
        """Invalid response fails validation."""
        response = {
            "verdict": "MAYBE",  # Invalid
            "confidence": 150,   # Out of range
            "reasoning": "short",  # Too short
        }

        is_valid, violation = contract.validate_response(response)
        assert is_valid is False

    def test_enforce_raises(self, contract):
        """Enforce raises on violation."""
        invalid_input = {"prompt_type": "bad"}

        with pytest.raises(ContractViolationError):
            contract.enforce_input(invalid_input)

    def test_fix_response(self, contract):
        """Fix response normalizes values."""
        bad_response = {
            "verdict": "vulnerable",  # Lowercase
            "confidence": 200,        # Out of range
            "reasoning": "",          # Empty
        }

        fixed = contract.fix_response(bad_response)

        assert fixed["verdict"] == "VULNERABLE"
        assert fixed["confidence"] == 100
        assert fixed["reasoning"] == "No reasoning provided"
