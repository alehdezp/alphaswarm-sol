## Design

### Unified Provider Interface

```python
class UnifiedLLMClient:
    """Unified client with failover support."""

    def __init__(self, config: LLMConfig):
        self.providers = self._init_providers(config)
        self.primary = config.default_provider
        self.fallback = config.fallback_provider

    async def analyze(
        self,
        context: str,
        prompt: str,
        tier: ModelTier = ModelTier.DEFAULT,
    ) -> LLMResult:
        """Analyze with automatic failover."""
        try:
            return await self._call_provider(self.primary, context, prompt, tier)
        except ProviderError:
            if self.fallback:
                return await self._call_provider(self.fallback, context, prompt, tier)
            raise
```

```python
"""
Unified LLM Client

Multi-provider abstraction with failover, tier selection, and consistent responses.
"""

import asyncio
from dataclasses import dataclass
from enum import Enum
from typing import Optional, Dict, Any, List
import logging

from .providers.base import LLMProvider, LLMResponse
from .config import LLMConfig, ProviderConfig

logger = logging.getLogger(__name__)


class ModelTier(Enum):
    """Model tier for task-appropriate selection."""
    CHEAP = "cheap"      # Fast, low-cost
    DEFAULT = "default"  # Balanced
    BEST = "best"        # Highest quality


class ProviderError(Exception):
    """Error from LLM provider."""
    def __init__(self, provider: str, error: str, retryable: bool = False):
        self.provider = provider
        self.error = error
        self.retryable = retryable
        super().__init__(f"{provider}: {error}")


@dataclass
class LLMResult:
    """Result from LLM analysis."""
    content: str
    provider: str
    model: str
    tier: ModelTier
    input_tokens: int
    output_tokens: int
    cost_usd: float
    latency_ms: int
    cached: bool = False

    # Parsed result (if structured output)
    parsed: Optional[Dict[str, Any]] = None
    parse_error: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "content": self.content,
            "provider": self.provider,
            "model": self.model,
            "tier": self.tier.value,
            "tokens": {
                "input": self.input_tokens,
                "output": self.output_tokens,
            },
            "cost_usd": self.cost_usd,
            "latency_ms": self.latency_ms,
            "cached": self.cached,
            "parsed": self.parsed,
        }


class UnifiedLLMClient:
    """
    Unified LLM client with multi-provider support.

    Features:
    - Multiple provider support
    - Automatic failover
    - Model tier selection
    - Response parsing
    - Usage tracking
    """

    def __init__(self, config: LLMConfig = None):
        self.config = config or LLMConfig.from_env()
        self.providers: Dict[str, LLMProvider] = {}
        self.primary_provider = self.config.default_provider
        self.fallback_provider = self.config.fallback_provider

        # Usage tracking
        self._total_cost = 0.0
        self._total_tokens = 0
        self._request_count = 0

        self._init_providers()

    def _init_providers(self):
        """Initialize configured providers."""
        for name, provider_config in self.config.providers.items():
            if not provider_config.enabled:
                continue

            try:
                provider = self._create_provider(name, provider_config)
                if provider:
                    self.providers[name] = provider
                    logger.info(f"Initialized provider: {name}")
            except Exception as e:
                logger.warning(f"Failed to init provider {name}: {e}")

    def _create_provider(
        self,
        name: str,
        config: ProviderConfig
    ) -> Optional[LLMProvider]:
        """Create provider instance."""
        from .providers import (
            AnthropicProvider,
            OpenAIProvider,
            GoogleProvider,
            OllamaProvider,
        )

        providers = {
            "anthropic": AnthropicProvider,
            "openai": OpenAIProvider,
            "google": GoogleProvider,
            "ollama": OllamaProvider,
        }

        provider_class = providers.get(name)
        if provider_class:
            return provider_class(config)
        return None

    async def analyze(
        self,
        context: str,
        prompt: str,
        tier: ModelTier = ModelTier.DEFAULT,
        provider: str = None,
        parse_json: bool = True,
    ) -> LLMResult:
        """
        Analyze with LLM.

        Args:
            context: Code/evidence context
            prompt: Analysis prompt
            tier: Model tier to use
            provider: Override provider selection
            parse_json: Attempt to parse JSON from response

        Returns:
            LLMResult with response and metadata
        """
        provider_name = provider or self.primary_provider

        if provider_name not in self.providers:
            raise ProviderError(
                provider_name,
                f"Provider not available. Available: {list(self.providers.keys())}"
            )

        try:
            result = await self._call_provider(
                provider_name, context, prompt, tier
            )

            if parse_json:
                result = self._try_parse_json(result)

            self._track_usage(result)
            return result

        except ProviderError as e:
            if e.retryable and self.fallback_provider:
                logger.warning(f"Primary failed, trying fallback: {e}")
                return await self._call_provider(
                    self.fallback_provider, context, prompt, tier
                )
            raise

    async def _call_provider(
        self,
        provider_name: str,
        context: str,
        prompt: str,
        tier: ModelTier,
    ) -> LLMResult:
        """Call a specific provider."""
        import time

        provider = self.providers[provider_name]
        model = self._get_model_for_tier(provider_name, tier)

        full_prompt = f"{context}\n\n{prompt}"
        start = time.monotonic()

        try:
            response = await provider.generate(
                prompt=full_prompt,
                max_tokens=4096,
                temperature=0.1,
                json_mode=True,
            )

            latency_ms = int((time.monotonic() - start) * 1000)

            return LLMResult(
                content=response.content,
                provider=provider_name,
                model=model,
                tier=tier,
                input_tokens=response.input_tokens,
                output_tokens=response.output_tokens,
                cost_usd=response.cost_usd,
                latency_ms=latency_ms,
                cached=response.cached,
            )

        except Exception as e:
            raise ProviderError(
                provider_name,
                str(e),
                retryable=self._is_retryable(e)
            )

    def _get_model_for_tier(self, provider: str, tier: ModelTier) -> str:
        """Get model name for provider and tier."""
        tier_models = {
            "anthropic": {
                ModelTier.CHEAP: "claude-3-haiku-20240307",
                ModelTier.DEFAULT: "claude-sonnet-4-20250514",
                ModelTier.BEST: "claude-opus-4-5-20251101",
            },
            "openai": {
                ModelTier.CHEAP: "gpt-4o-mini",
                ModelTier.DEFAULT: "gpt-4o",
                ModelTier.BEST: "o1",
            },
            "google": {
                ModelTier.CHEAP: "gemini-2.0-flash",
                ModelTier.DEFAULT: "gemini-pro",
                ModelTier.BEST: "gemini-pro",
            },
        }

        return tier_models.get(provider, {}).get(
            tier,
            self.config.providers[provider].default_model
        )

    def _try_parse_json(self, result: LLMResult) -> LLMResult:
        """Try to parse JSON from response."""
        import json

        try:
            # Try to extract JSON from response
            content = result.content.strip()

            # Handle markdown code blocks
            if content.startswith("```"):
                lines = content.split("\n")
                content = "\n".join(lines[1:-1])

            result.parsed = json.loads(content)
        except json.JSONDecodeError as e:
            result.parse_error = str(e)

        return result

    def _is_retryable(self, error: Exception) -> bool:
        """Check if error is retryable."""
        error_str = str(error).lower()
        retryable_patterns = [
            "rate limit",
            "timeout",
            "connection",
            "502",
            "503",
            "overloaded",
        ]
        return any(p in error_str for p in retryable_patterns)

    def _track_usage(self, result: LLMResult):
        """Track usage statistics."""
        self._total_cost += result.cost_usd
        self._total_tokens += result.input_tokens + result.output_tokens
        self._request_count += 1

    def get_usage(self) -> Dict[str, Any]:
        """Get usage statistics."""
        return {
            "total_cost_usd": self._total_cost,
            "total_tokens": self._total_tokens,
            "request_count": self._request_count,
            "available_providers": list(self.providers.keys()),
        }

    def get_available_providers(self) -> List[str]:
        """Get list of available providers."""
        return list(self.providers.keys())

    async def health_check(self) -> Dict[str, bool]:
        """Check health of all providers."""
        results = {}
        for name, provider in self.providers.items():
            try:
                healthy = await provider.health_check()
                results[name] = healthy
            except Exception:
                results[name] = False
        return results
```

"""LLM Configuration."""

import os
from dataclasses import dataclass, field
from typing import Dict, Optional
from enum import Enum


class ProviderType(Enum):
    ANTHROPIC = "anthropic"
    OPENAI = "openai"
    GOOGLE = "google"
    OLLAMA = "ollama"


@dataclass
class ProviderConfig:
    """Configuration for a single provider."""
    enabled: bool = False
    api_key: Optional[str] = None
    default_model: str = ""
    base_url: Optional[str] = None

    # Cost per 1M tokens
    cost_per_1m_input: float = 0.0
    cost_per_1m_output: float = 0.0


@dataclass
class LLMConfig:
    """LLM configuration."""
    default_provider: str = "anthropic"
    fallback_provider: Optional[str] = "ollama"
    providers: Dict[str, ProviderConfig] = field(default_factory=dict)

    # Limits
    max_tokens_per_request: int = 4096
    max_cost_per_run: float = 5.0

    @classmethod
    def from_env(cls) -> "LLMConfig":
        """Create config from environment variables."""
        config = cls()

        # Anthropic
        if os.environ.get("ANTHROPIC_API_KEY"):
            config.providers["anthropic"] = ProviderConfig(
                enabled=True,
                api_key=os.environ["ANTHROPIC_API_KEY"],
                default_model="claude-sonnet-4-20250514",
                cost_per_1m_input=3.0,
                cost_per_1m_output=15.0,
            )

        # OpenAI
        if os.environ.get("OPENAI_API_KEY"):
            config.providers["openai"] = ProviderConfig(
                enabled=True,
                api_key=os.environ["OPENAI_API_KEY"],
                default_model="gpt-4o",
                cost_per_1m_input=5.0,
                cost_per_1m_output=15.0,
            )

        # Google
        if os.environ.get("GOOGLE_API_KEY"):
            config.providers["google"] = ProviderConfig(
                enabled=True,
                api_key=os.environ["GOOGLE_API_KEY"],
                default_model="gemini-pro",
                cost_per_1m_input=1.25,
                cost_per_1m_output=5.0,
            )

        # Ollama (local, always available if installed)
        config.providers["ollama"] = ProviderConfig(
            enabled=True,
            default_model="qwen2.5-coder:7b",
            cost_per_1m_input=0.0,
            cost_per_1m_output=0.0,
        )

        return config
