## Objective

## Context

### The FP Problem

Pattern-based detection often flags code as vulnerable when mitigating factors exist:
- Access control is present but not detected
- The vulnerability condition is unreachable
- Business logic makes the pattern safe
- Guards exist in modifiers or called functions

### Goal

LLM can understand context that patterns cannot:
- Read modifiers and understand their effect
- Trace access control through multiple layers
- Understand business logic constraints

3. Unchecked user input affecting critical state
4. Missing validation on value transfers

    "is_false_positive": true | false,
    "confidence": 0-100,
    "reasoning": "Detailed explanation",
    "mitigation_found": "What protection exists" | null,

}}"""

"""
False Positive Filtering

Uses LLM to identify and filter false positives from Tier A findings.
Target: >= 20% FP reduction while preserving recall.
"""

from dataclasses import dataclass, field
from typing import List, Optional, Dict, Any
import logging
import json

from .client import UnifiedLLMClient, LLMResult, ModelTier
from .slicer import slice_context_for_finding
from .workflow import TierAFinding

logger = logging.getLogger(__name__)


FP_FILTER_PROMPT = """You are filtering false positives from vulnerability findings.

## Finding Details
- Pattern: {pattern_id}
- Severity: {severity}
- Description: {description}

## Evidence for Vulnerability
{tier_a_evidence}

## Code Context
{code_context}

## Your Task
Determine if this is a TRUE vulnerability or a FALSE POSITIVE.

FALSE POSITIVE indicators:
1. Access control is present (onlyOwner, require(msg.sender == ...), etc.)
2. Reentrancy guard is applied (nonReentrant modifier)
3. The vulnerability path is unreachable
4. Business logic prevents exploitation
5. Value bounds prevent meaningful exploitation

TRUE VULNERABILITY indicators:
1. No access control on state-changing function
2. External call before state update
3. Unchecked user input affecting critical state
4. Missing validation on value transfers

## Response (JSON only)
{{
    "is_false_positive": true | false,
    "confidence": 0-100,
    "reasoning": "Detailed explanation",
    "mitigation_found": "What protection exists, if any",
    "attack_path": "How to exploit, if vulnerable"
}}"""


@dataclass
class FPFilterResult:
    """Result of false positive filtering."""
    finding_id: str
    is_false_positive: bool
    confidence: float  # 0-100
    reasoning: str
    mitigation_found: Optional[str] = None
    attack_path: Optional[str] = None
    tokens_used: int = 0
    cost_usd: float = 0.0
    error: Optional[str] = None

    def to_dict(self) -> dict:
        return {
            "finding_id": self.finding_id,
            "is_false_positive": self.is_false_positive,
            "confidence": self.confidence,
            "reasoning": self.reasoning,
            "mitigation_found": self.mitigation_found,
            "attack_path": self.attack_path,
            "tokens_used": self.tokens_used,
            "cost_usd": self.cost_usd,
        }


@dataclass
class FPFilterStats:
    """Statistics for FP filtering."""
    total_findings: int = 0
    true_positives: int = 0
    false_positives: int = 0
    uncertain: int = 0
    errors: int = 0

    @property
    def fp_rate(self) -> float:
        """False positive rate."""
        if self.total_findings == 0:
            return 0.0
        return self.false_positives / self.total_findings

    @property
    def fp_reduction_estimate(self) -> float:
        """Estimated FP reduction (vs no filtering)."""
        if self.total_findings == 0:
            return 0.0
        # Assuming all findings without filter would be reported
        return self.false_positives / self.total_findings * 100


class FalsePositiveFilter:
    """
    Filters false positives using LLM analysis.

    Target: >= 20% FP reduction while preserving recall.
    """

    def __init__(
        self,
        llm_client: UnifiedLLMClient,
        model_tier: ModelTier = ModelTier.DEFAULT,
        token_budget: int = 4000,
    ):
        self.llm_client = llm_client
        self.model_tier = model_tier
        self.token_budget = token_budget
        self.stats = FPFilterStats()

    async def filter(
        self,
        finding: TierAFinding,
        graph: Any,
    ) -> FPFilterResult:
        """
        Filter a single finding for false positives.

        Args:
            finding: The finding to analyze
            graph: Knowledge graph for context

        Returns:
            FPFilterResult with determination
        """
        self.stats.total_findings += 1

        # Get context
        context_result = slice_context_for_finding(
            finding.to_dict(),
            graph,
            budget=self.token_budget,
        )

        # Format evidence
        evidence_text = self._format_evidence(finding.evidence)

        # Build prompt
        prompt = FP_FILTER_PROMPT.format(
            pattern_id=finding.pattern_id,
            severity=finding.severity,
            description=finding.description,
            tier_a_evidence=evidence_text,
            code_context=context_result.content,
        )

        try:
            response = await self.llm_client.analyze(
                context=context_result.content,
                prompt=prompt,
                tier=self.model_tier,
            )

            result = self._parse_response(response, finding.id)

            # Update stats
            if result.error:
                self.stats.errors += 1
            elif result.is_false_positive:
                self.stats.false_positives += 1
            else:
                self.stats.true_positives += 1

            return result

        except Exception as e:
            self.stats.errors += 1
            logger.error(f"FP filter failed for {finding.id}: {e}")
            return FPFilterResult(
                finding_id=finding.id,
                is_false_positive=False,  # Conservative: assume TP on error
                confidence=0,
                reasoning=f"Error during filtering: {e}",
                error=str(e),
            )

    async def filter_batch(
        self,
        findings: List[TierAFinding],
        graph: Any,
    ) -> List[FPFilterResult]:
        """Filter multiple findings."""
        results = []
        for finding in findings:
            result = await self.filter(finding, graph)
            results.append(result)
        return results

    def _format_evidence(self, evidence: List[Dict[str, Any]]) -> str:
        """Format evidence for prompt."""
        if not evidence:
            return "No specific evidence provided."

        lines = []
        for e in evidence[:5]:  # Limit to 5 pieces
            lines.append(f"- {e.get('description', str(e))}")
        return "\n".join(lines)

    def _parse_response(self, response: LLMResult, finding_id: str) -> FPFilterResult:
        """Parse LLM response into FPFilterResult."""
        try:
            parsed = response.parsed
            if parsed is None:
                # Try to extract JSON
                parsed = self._extract_json(response.content)

            if parsed:
                return FPFilterResult(
                    finding_id=finding_id,
                    is_false_positive=parsed.get("is_false_positive", False),
                    confidence=parsed.get("confidence", 50),
                    reasoning=parsed.get("reasoning", ""),
                    mitigation_found=parsed.get("mitigation_found"),
                    attack_path=parsed.get("attack_path"),
                    tokens_used=response.input_tokens + response.output_tokens,
                    cost_usd=response.cost_usd,
                )
            else:
                return FPFilterResult(
                    finding_id=finding_id,
                    is_false_positive=False,  # Conservative
                    confidence=0,
                    reasoning="Failed to parse response",
                    tokens_used=response.input_tokens + response.output_tokens,
                    cost_usd=response.cost_usd,
                    error="JSON parse failed",
                )

        except Exception as e:
            return FPFilterResult(
                finding_id=finding_id,
                is_false_positive=False,
                confidence=0,
                reasoning=str(e),
                error=str(e),
            )

    def _extract_json(self, content: str) -> Optional[dict]:
        """Extract JSON from response."""
        import re

        # Try to find JSON block
        json_match = re.search(r'\{[^{}]*\}', content, re.DOTALL)
        if json_match:
            try:
                return json.loads(json_match.group())
            except json.JSONDecodeError:
                pass

        return None

    def get_stats(self) -> dict:
        """Get filtering statistics."""
        return {
            "total_findings": self.stats.total_findings,
            "true_positives": self.stats.true_positives,
            "false_positives": self.stats.false_positives,
            "uncertain": self.stats.uncertain,
            "errors": self.stats.errors,
            "fp_rate": self.stats.fp_rate,
            "fp_reduction_estimate": f"{self.stats.fp_reduction_estimate:.1f}%",
        }


def apply_fp_filter(
    findings: List[TierAFinding],
    results: List[FPFilterResult],
    confidence_threshold: float = 70.0,
) -> List[TierAFinding]:
    """
    Apply FP filter results to findings.

    Only removes findings marked as FP with high confidence.

    Args:
        findings: Original findings
        results: FP filter results
        confidence_threshold: Minimum confidence to filter

    Returns:
        Filtered list of findings
    """
    result_map = {r.finding_id: r for r in results}

    filtered = []
    for finding in findings:
        result = result_map.get(finding.id)

        if result is None:
            # No result, keep finding
            filtered.append(finding)
        elif result.is_false_positive and result.confidence >= confidence_threshold:
            # High-confidence FP, remove
            logger.info(f"Filtered FP: {finding.id} (confidence: {result.confidence})")
        else:
            # Keep finding
            filtered.append(finding)

    return filtered
