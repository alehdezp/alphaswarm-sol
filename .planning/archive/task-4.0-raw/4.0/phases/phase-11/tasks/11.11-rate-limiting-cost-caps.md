## Objective

Prevent runaway costs and API abuse with rate limiting, cost caps, and usage monitoring.

---

## Context

### Why This Matters

Without limits:
1. User accidentally runs on 1000 files = $500+ bill
2. Infinite loop bug = API quota exhausted
3. Malicious input = denial of service via cost

### Meta-Critique Reference

Lines 477-480: "No discussion of rate limiting, cost caps, or abuse prevention."

## Design

### Limit Types

| Limit Type | Purpose | Default |
|------------|---------|---------|
| Max tokens per run | Prevent runaway | 100,000 |
| Max cost per run (USD) | Budget control | $5.00 |
| Max requests per minute | API protection | 60 |
| Max findings per run | Scope control | 200 |
| Max cost per day (USD) | Daily budget | $50.00 |

### Enforcement Points

```
User Request
    |
    +-- Check limits BEFORE sending to LLM
    |       |
    |       +-- Estimate tokens
    |       +-- Check budget remaining
    |       +-- Check rate limit
    |       +-- REJECT if any exceeded
    |
    +-- Send to LLM
    |
    +-- Record usage AFTER response
            |
            +-- Update token count
            +-- Update cost total
            +-- Update rate limit window
```

```python
"""
Rate Limiting and Cost Caps

Prevents runaway costs and API abuse.
"""

import time
from dataclasses import dataclass, field
from typing import Optional, Tuple, List
from datetime import datetime, timedelta
import logging

logger = logging.getLogger(__name__)


@dataclass
class LLMLimits:
    """LLM usage limits."""
    max_tokens_per_run: int = 100_000
    max_cost_per_run_usd: float = 5.00
    max_requests_per_minute: int = 60
    max_findings_per_run: int = 200
    max_cost_per_day_usd: float = 50.00

    @classmethod
    def from_config(cls, config: dict) -> "LLMLimits":
        """Create from config dict."""
        return cls(
            max_tokens_per_run=config.get("max_tokens_per_run", 100_000),
            max_cost_per_run_usd=config.get("max_cost_per_run_usd", 5.00),
            max_requests_per_minute=config.get("max_requests_per_minute", 60),
            max_findings_per_run=config.get("max_findings_per_run", 200),
            max_cost_per_day_usd=config.get("max_cost_per_day_usd", 50.00),
        )


@dataclass
class UsageStats:
    """Current usage statistics."""
    tokens_used: int = 0
    cost_usd: float = 0.0
    requests_made: int = 0
    findings_processed: int = 0
    request_times: List[float] = field(default_factory=list)

    def to_dict(self) -> dict:
        return {
            "tokens_used": self.tokens_used,
            "cost_usd": self.cost_usd,
            "requests_made": self.requests_made,
            "findings_processed": self.findings_processed,
        }


class LimitExceededError(Exception):
    """Raised when a limit is exceeded."""
    def __init__(self, limit_type: str, current: float, max_allowed: float, suggestion: str):
        self.limit_type = limit_type
        self.current = current
        self.max_allowed = max_allowed
        self.suggestion = suggestion
        super().__init__(
            f"{limit_type} limit exceeded: {current} >= {max_allowed}. {suggestion}"
        )


class RateLimiter:
    """Rate limiting and cost control for LLM API calls."""

    def __init__(self, limits: LLMLimits = None):
        self.limits = limits or LLMLimits()
        self.usage = UsageStats()
        self._daily_reset = datetime.now()

    def check_limits(self, estimated_tokens: int, estimated_cost: float = None) -> Tuple[bool, Optional[str]]:
        """
        Check if request is within limits.

        Args:
            estimated_tokens: Estimated token count for request
            estimated_cost: Estimated cost (calculated if not provided)

        Returns:
            Tuple of (allowed, error_message)
        """
        # Check token limit
        if self.usage.tokens_used + estimated_tokens > self.limits.max_tokens_per_run:
            return False, (
                f"Token limit exceeded: {self.usage.tokens_used}/{self.limits.max_tokens_per_run}. "
                f"Use --max-tokens to increase limit."
            )

        # Check cost limit
        if estimated_cost is None:
            # Estimate cost at $0.01 per 1K tokens (conservative)
            estimated_cost = estimated_tokens / 1000 * 0.01

        if self.usage.cost_usd + estimated_cost > self.limits.max_cost_per_run_usd:
            return False, (
                f"Cost limit exceeded: ${self.usage.cost_usd:.2f}/${self.limits.max_cost_per_run_usd:.2f}. "
                f"Use --max-cost to increase limit."
            )

        # Check rate limit
        now = time.time()
        self.usage.request_times = [t for t in self.usage.request_times if now - t < 60]
        if len(self.usage.request_times) >= self.limits.max_requests_per_minute:
            wait_time = 60 - (now - min(self.usage.request_times))
            return False, (
                f"Rate limit: {len(self.usage.request_times)} requests in last minute. "
                f"Wait {wait_time:.0f}s or use --no-rate-limit."
            )

        # Check findings limit
        if self.usage.findings_processed >= self.limits.max_findings_per_run:
            return False, (
                f"Findings limit exceeded: {self.usage.findings_processed}/{self.limits.max_findings_per_run}. "
                f"Use --max-findings to increase limit."
            )

        # Check daily limit
        if self._should_reset_daily():
            self._reset_daily()
        # TODO: Load daily usage from persistent storage

        return True, None

    def record_usage(self, tokens: int, cost_usd: float, finding_count: int = 1):
        """
        Record usage after successful request.

        Args:
            tokens: Tokens used
            cost_usd: Cost in USD
            finding_count: Number of findings processed
        """
        self.usage.tokens_used += tokens
        self.usage.cost_usd += cost_usd
        self.usage.requests_made += 1
        self.usage.findings_processed += finding_count
        self.usage.request_times.append(time.time())

        logger.debug(
            f"LLM usage: {self.usage.tokens_used} tokens, ${self.usage.cost_usd:.4f}, "
            f"{self.usage.requests_made} requests"
        )

    def require_capacity(self, estimated_tokens: int, estimated_cost: float = None):
        """
        Require capacity for request, raising if exceeded.

        Args:
            estimated_tokens: Estimated tokens
            estimated_cost: Estimated cost

        Raises:
            LimitExceededError: If any limit exceeded
        """
        allowed, error = self.check_limits(estimated_tokens, estimated_cost)
        if not allowed:
            # Parse error to determine limit type
            if "Token" in error:
                raise LimitExceededError(
                    "token", self.usage.tokens_used, self.limits.max_tokens_per_run,
                    "Use --max-tokens to increase or --skip-low-confidence"
                )
            elif "Cost" in error:
                raise LimitExceededError(
                    "cost", self.usage.cost_usd, self.limits.max_cost_per_run_usd,
                    "Use --max-cost to increase or reduce scope"
                )
            elif "Rate" in error:
                raise LimitExceededError(
                    "rate", len(self.usage.request_times), self.limits.max_requests_per_minute,
                    "Wait and retry, or use --no-rate-limit (careful!)"
                )
            elif "Findings" in error:
                raise LimitExceededError(
                    "findings", self.usage.findings_processed, self.limits.max_findings_per_run,
                    "Use --max-findings to increase or filter high-priority only"
                )
            else:
                raise LimitExceededError("unknown", 0, 0, error)

    def get_remaining(self) -> dict:
        """Get remaining capacity."""
        return {
            "tokens_remaining": self.limits.max_tokens_per_run - self.usage.tokens_used,
            "cost_remaining_usd": self.limits.max_cost_per_run_usd - self.usage.cost_usd,
            "requests_remaining": max(0, self.limits.max_requests_per_minute - len([
                t for t in self.usage.request_times if time.time() - t < 60
            ])),
            "findings_remaining": self.limits.max_findings_per_run - self.usage.findings_processed,
        }

    def get_usage_summary(self) -> dict:
        """Get usage summary."""
        return {
            "usage": self.usage.to_dict(),
            "remaining": self.get_remaining(),
            "limits": {
                "max_tokens": self.limits.max_tokens_per_run,
                "max_cost": self.limits.max_cost_per_run_usd,
                "max_requests_per_minute": self.limits.max_requests_per_minute,
                "max_findings": self.limits.max_findings_per_run,
            }
        }

    def _should_reset_daily(self) -> bool:
        """Check if daily counters should reset."""
        return datetime.now() - self._daily_reset > timedelta(days=1)

    def _reset_daily(self):
        """Reset daily counters."""
        self._daily_reset = datetime.now()
        # Note: Would persist to file for true daily limits


class UsageMonitor:
    """Monitors and reports LLM usage."""

    def __init__(self, limiter: RateLimiter):
        self.limiter = limiter
        self._start_time = time.time()

    def format_progress(self, current_finding: int, total_findings: int) -> str:
        """
        Format progress string for display.

        Example: "[12/100] Tokens: 23,456 | Cost: $0.12 | Rate: 8 req/min"
        """
        usage = self.limiter.usage

        elapsed = time.time() - self._start_time
        rate = usage.requests_made / elapsed * 60 if elapsed > 0 else 0

        return (
            f"[{current_finding}/{total_findings}] "
            f"Tokens: {usage.tokens_used:,} | "
            f"Cost: ${usage.cost_usd:.2f} | "
            f"Rate: {rate:.1f} req/min"
        )

    def format_summary(self) -> str:
        """Format final usage summary."""
        usage = self.limiter.usage
        elapsed = time.time() - self._start_time

        return f"""
LLM Usage Summary
================
Findings processed: {usage.findings_processed}
Requests made:      {usage.requests_made}
Tokens used:        {usage.tokens_used:,}
Total cost:         ${usage.cost_usd:.2f}
Time elapsed:       {elapsed:.1f}s
Avg per finding:    ${usage.cost_usd / max(1, usage.findings_processed):.4f}
"""


# Global rate limiter
_rate_limiter: Optional[RateLimiter] = None


def get_rate_limiter(limits: LLMLimits = None) -> RateLimiter:
    """Get global rate limiter."""
    global _rate_limiter
    if _rate_limiter is None:
        _rate_limiter = RateLimiter(limits)
    return _rate_limiter


def reset_rate_limiter():
    """Reset global rate limiter (for testing)."""
    global _rate_limiter
    _rate_limiter = None
```

```python
"""Tests for rate limiting and cost caps."""

import pytest
import time
from unittest.mock import patch

from true_vkg.llm.limits import (
    LLMLimits, RateLimiter, LimitExceededError,
    UsageMonitor, UsageStats
)


class TestLLMLimits:
    """Test limit configuration."""

    def test_default_limits(self):
        """Default limits are reasonable."""
        limits = LLMLimits()
        assert limits.max_tokens_per_run == 100_000
        assert limits.max_cost_per_run_usd == 5.0
        assert limits.max_requests_per_minute == 60

    def test_from_config(self):
        """Limits can be loaded from config."""
        config = {"max_tokens_per_run": 50_000, "max_cost_per_run_usd": 10.0}
        limits = LLMLimits.from_config(config)
        assert limits.max_tokens_per_run == 50_000
        assert limits.max_cost_per_run_usd == 10.0


class TestRateLimiter:
    """Test rate limiter."""

    def test_allows_under_limit(self):
        """Allows requests under limit."""
        limiter = RateLimiter()
        allowed, error = limiter.check_limits(1000)
        assert allowed is True
        assert error is None

    def test_blocks_over_token_limit(self):
        """Blocks requests over token limit."""
        limits = LLMLimits(max_tokens_per_run=1000)
        limiter = RateLimiter(limits)

        # Use up tokens
        limiter.record_usage(900, 0.01)

        # Try to use more
        allowed, error = limiter.check_limits(200)
        assert allowed is False
        assert "Token" in error

    def test_blocks_over_cost_limit(self):
        """Blocks requests over cost limit."""
        limits = LLMLimits(max_cost_per_run_usd=0.10)
        limiter = RateLimiter(limits)

        # Use up budget
        limiter.record_usage(10000, 0.09)

        # Try to use more
        allowed, error = limiter.check_limits(5000, 0.02)
        assert allowed is False
        assert "Cost" in error

    def test_rate_limiting(self):
        """Rate limiting enforced."""
        limits = LLMLimits(max_requests_per_minute=3)
        limiter = RateLimiter(limits)

        # Make 3 requests
        for _ in range(3):
            limiter.record_usage(100, 0.001)

        # 4th should be blocked
        allowed, error = limiter.check_limits(100)
        assert allowed is False
        assert "Rate" in error

    def test_require_capacity_raises(self):
        """require_capacity raises on limit."""
        limits = LLMLimits(max_tokens_per_run=100)
        limiter = RateLimiter(limits)

        with pytest.raises(LimitExceededError) as exc_info:
            limiter.require_capacity(200)

        assert exc_info.value.limit_type == "token"

    def test_get_remaining(self):
        """get_remaining shows correct values."""
        limits = LLMLimits(max_tokens_per_run=10000)
        limiter = RateLimiter(limits)

        limiter.record_usage(3000, 0.03)

        remaining = limiter.get_remaining()
        assert remaining["tokens_remaining"] == 7000


class TestUsageMonitor:
    """Test usage monitoring."""

    def test_format_progress(self):
        """Progress format is readable."""
        limiter = RateLimiter()
        limiter.record_usage(5000, 0.05)

        monitor = UsageMonitor(limiter)
        progress = monitor.format_progress(5, 20)

        assert "[5/20]" in progress
        assert "5,000" in progress or "5000" in progress
        assert "$0.05" in progress

    def test_format_summary(self):
        """Summary format is readable."""
        limiter = RateLimiter()
        limiter.record_usage(10000, 0.10, finding_count=5)

        monitor = UsageMonitor(limiter)
        summary = monitor.format_summary()

        assert "10,000" in summary or "10000" in summary
        assert "$0.10" in summary
        assert "5" in summary


class TestLimitExceededError:
    """Test limit exceeded error."""

    def test_error_message(self):
        """Error has helpful message."""
        error = LimitExceededError(
            "token", 10000, 5000,
            "Use --max-tokens to increase"
        )

        msg = str(error)
        assert "token" in msg.lower()
        assert "10000" in msg
        assert "5000" in msg
        assert "--max-tokens" in msg
```
