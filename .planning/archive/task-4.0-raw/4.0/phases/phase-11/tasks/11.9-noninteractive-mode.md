```python
"""
Noninteractive Batch Processing

Enables LLM analysis in CI/batch environments.
"""

import asyncio
import json
import sys
import time
from dataclasses import dataclass, field
from typing import List, Optional, Dict, Any, Callable
from pathlib import Path
import logging

from .workflow import TierBWorkflow, TierAFinding, TierBResult, TierBConfig
from .client import UnifiedLLMClient, ModelTier
from .limits import RateLimiter, LLMLimits

logger = logging.getLogger(__name__)


@dataclass
class BatchConfig:
    """Configuration for batch processing."""
    max_concurrent: int = 5
    max_retries: int = 3
    retry_delay_seconds: float = 1.0
    timeout_seconds: float = 120.0
    progress_interval: int = 10  # Log progress every N findings
    fail_fast: bool = False  # Stop on first error
    output_file: Optional[str] = None


@dataclass
class BatchResult:
    """Result of batch processing."""
    total_findings: int
    processed: int
    succeeded: int
    failed: int
    skipped: int
    results: List[TierBResult]
    errors: List[Dict[str, Any]]
    duration_seconds: float
    exit_code: int

    def to_dict(self) -> dict:
        return {
            "summary": {
                "total": self.total_findings,
                "processed": self.processed,
                "succeeded": self.succeeded,
                "failed": self.failed,
                "skipped": self.skipped,
                "duration_seconds": round(self.duration_seconds, 2),
                "exit_code": self.exit_code,
            },
            "results": [r.to_dict() for r in self.results],
            "errors": self.errors,
        }


class BatchProgress:
    """Progress tracking for batch jobs."""

    def __init__(self, total: int, callback: Callable[[str], None] = None):
        self.total = total
        self.processed = 0
        self.succeeded = 0
        self.failed = 0
        self.start_time = time.time()
        self.callback = callback or (lambda msg: logger.info(msg))

    def update(self, success: bool):
        """Update progress."""
        self.processed += 1
        if success:
            self.succeeded += 1
        else:
            self.failed += 1

    def log(self, interval: int = 10):
        """Log progress if at interval."""
        if self.processed % interval == 0 or self.processed == self.total:
            elapsed = time.time() - self.start_time
            rate = self.processed / elapsed if elapsed > 0 else 0
            remaining = (self.total - self.processed) / rate if rate > 0 else 0

            self.callback(
                f"Progress: {self.processed}/{self.total} "
                f"({self.succeeded} ok, {self.failed} fail) "
                f"| {rate:.1f}/s | ETA: {remaining:.0f}s"
            )


class NoninteractiveLLMRunner:
    """
    Runs LLM analysis in noninteractive mode.

    Features:
    - Batch processing with concurrency control
    - Automatic retries with backoff
    - Timeout handling
    - Progress logging
    - CI-compatible exit codes
    """

    def __init__(
        self,
        workflow: TierBWorkflow = None,
        config: BatchConfig = None,
    ):
        self.config = config or BatchConfig()
        self.workflow = workflow

    async def analyze_batch(
        self,
        findings: List[TierAFinding],
        graph: Any,
    ) -> BatchResult:
        """
        Analyze findings in batch mode.

        Args:
            findings: Findings to analyze
            graph: Knowledge graph

        Returns:
            BatchResult with all results and summary
        """
        start_time = time.time()
        results: List[TierBResult] = []
        errors: List[Dict[str, Any]] = []

        progress = BatchProgress(len(findings))

        for finding in findings:
            try:
                result = await self._analyze_with_retry(finding, graph)
                results.append(result)
                progress.update(success=result.error is None)

            except Exception as e:
                logger.error(f"Failed to analyze {finding.id}: {e}")
                errors.append({
                    "finding_id": finding.id,
                    "error": str(e),
                })
                progress.update(success=False)

                if self.config.fail_fast:
                    break

            progress.log(self.config.progress_interval)

        # Calculate summary
        duration = time.time() - start_time
        succeeded = len([r for r in results if r.error is None])
        failed = len(errors) + len([r for r in results if r.error is not None])

        # Determine exit code
        if failed > 0:
            exit_code = 1  # Some failures
        elif succeeded == 0 and len(findings) > 0:
            exit_code = 2  # No successes
        else:
            exit_code = 0  # All good

        batch_result = BatchResult(
            total_findings=len(findings),
            processed=len(results),
            succeeded=succeeded,
            failed=failed,
            skipped=len(findings) - len(results),
            results=results,
            errors=errors,
            duration_seconds=duration,
            exit_code=exit_code,
        )

        # Write output if configured
        if self.config.output_file:
            self._write_output(batch_result)

        return batch_result

    async def _analyze_with_retry(
        self,
        finding: TierAFinding,
        graph: Any,
    ) -> TierBResult:
        """Analyze with automatic retry."""
        last_error = None

        for attempt in range(self.config.max_retries):
            try:
                result = await asyncio.wait_for(
                    self.workflow.analyze(finding, graph),
                    timeout=self.config.timeout_seconds,
                )

                if result.error is None:
                    return result

                # Got a result but with error, may retry
                last_error = result.error

            except asyncio.TimeoutError:
                last_error = "Timeout"
                logger.warning(f"Attempt {attempt + 1} timed out for {finding.id}")

            except Exception as e:
                last_error = str(e)
                logger.warning(f"Attempt {attempt + 1} failed for {finding.id}: {e}")

            # Wait before retry (with backoff)
            if attempt < self.config.max_retries - 1:
                delay = self.config.retry_delay_seconds * (2 ** attempt)
                await asyncio.sleep(delay)

        # All retries exhausted
        return TierBResult(
            finding_id=finding.id,
            routing=None,  # type: ignore
            verdict="UNCERTAIN",
            error=f"Max retries exceeded: {last_error}",
        )

    def _write_output(self, result: BatchResult):
        """Write results to output file."""
        output_path = Path(self.config.output_file)
        output_path.parent.mkdir(parents=True, exist_ok=True)

        with open(output_path, 'w') as f:
            json.dump(result.to_dict(), f, indent=2)

        logger.info(f"Results written to {output_path}")


def run_noninteractive(
    findings_file: str,
    output_file: str,
    graph_path: str = None,
    max_concurrent: int = 5,
    timeout: float = 120.0,
) -> int:
    """
    Run BSKG analysis in noninteractive mode.

    Args:
        findings_file: Path to JSON file with findings
        output_file: Path to write results
        graph_path: Path to knowledge graph (optional)
        max_concurrent: Max concurrent analyses
        timeout: Timeout per finding

    Returns:
        Exit code (0=success, 1=failures, 2=no successes)
    """
    import asyncio

    # Load findings
    with open(findings_file) as f:
        findings_data = json.load(f)

    findings = [
        TierAFinding(
            id=f["id"],
            pattern_id=f["pattern_id"],
            function_id=f.get("function_id", ""),
            description=f["description"],
            severity=f.get("severity", "medium"),
            confidence=f.get("confidence", 0.5),
        )
        for f in findings_data
    ]

    # Create workflow
    # Note: In real use, graph would be loaded from graph_path
    graph = None  # Placeholder

    config = BatchConfig(
        max_concurrent=max_concurrent,
        timeout_seconds=timeout,
        output_file=output_file,
    )

    workflow = TierBWorkflow()
    runner = NoninteractiveLLMRunner(workflow=workflow, config=config)

    # Run
    result = asyncio.run(runner.analyze_batch(findings, graph))

    # Print summary
    print(f"\n{'=' * 40}")
    print("VKG Noninteractive Analysis Complete")
    print(f"{'=' * 40}")
    print(f"Total findings:  {result.total_findings}")
    print(f"Processed:       {result.processed}")
    print(f"Succeeded:       {result.succeeded}")
    print(f"Failed:          {result.failed}")
    print(f"Duration:        {result.duration_seconds:.1f}s")
    print(f"Exit code:       {result.exit_code}")
    print(f"{'=' * 40}\n")

    return result.exit_code


# CLI entry point
if __name__ == "__main__":
    import argparse

    parser = argparse.ArgumentParser(description="VKG Noninteractive Mode")
    parser.add_argument("findings_file", help="JSON file with findings")
    parser.add_argument("-o", "--output", required=True, help="Output file")
    parser.add_argument("-g", "--graph", help="Knowledge graph path")
    parser.add_argument("-c", "--concurrent", type=int, default=5)
    parser.add_argument("-t", "--timeout", type=float, default=120.0)

    args = parser.parse_args()

    exit_code = run_noninteractive(
        findings_file=args.findings_file,
        output_file=args.output,
        graph_path=args.graph,
        max_concurrent=args.concurrent,
        timeout=args.timeout,
    )

    sys.exit(exit_code)
```

```yaml
  push:
    paths:
      - 'contracts/**'

      - uses: actions/checkout@v4

          python-version: '3.11'

      - name: Run Tier B Verification
        run: |
          vkg analyze-batch findings.json \
            -o results.json \
            -g graph.json \
            --concurrent 3 \
            --timeout 60

      - name: Check Results
        run: |
          # Fail if critical vulnerabilities found
          python -c "
          import json
          with open('results.json') as f:
              data = json.load(f)
          critical = [r for r in data['results'] if r['verdict'] == 'VULNERABLE']
          if critical:
              print(f'Found {len(critical)} vulnerabilities!')
              exit(1)
          "

      - name: Upload Report
        uses: actions/upload-artifact@v4

          name: vkg-report
```
