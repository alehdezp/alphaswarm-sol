## Objective

Enable task-appropriate model selection across tiers: cheap (fast), default (balanced), best (quality).

---

## Context

### Why Tiers Matter

Different tasks have different requirements:

| Task | Priority | Best Tier |
|------|----------|-----------|
| Quick confidence check | Speed | CHEAP |
| Standard verification | Balance | DEFAULT |
| Complex business logic | Quality | BEST |

### Cost Implications

| Tier | Model | Cost/1M tokens | Use Case |
|------|-------|----------------|----------|
| CHEAP | Haiku/GPT-4o-mini | $0.25-0.60 | Triage, quick checks |
| DEFAULT | Sonnet/GPT-4o | $3-15 | Standard verification |
| BEST | Opus/o1 | $15-75 | Complex analysis |

---

## Design

### Tier Selection Logic

```python
def select_tier(finding: TierAFinding, config: TierConfig) -> ModelTier:
    """Select appropriate model tier for a finding."""

    # High-value targets get best model
    if finding.severity == "critical":
        return ModelTier.BEST

    # Low-confidence needs better model
    if finding.confidence < 0.4:
        return ModelTier.BEST

    # Complex patterns need better model
    if finding.pattern_id in COMPLEX_PATTERNS:
        return ModelTier.BEST

    # Quick triage for low-severity
    if finding.severity in ("low", "info"):
        return ModelTier.CHEAP

    # Default for everything else
    return ModelTier.DEFAULT
```

```python
"""
Enhanced LLM Client with Multi-Tier Support

Provides automatic tier selection based on task complexity.
"""

from enum import Enum
from dataclasses import dataclass
from typing import Dict, Optional, List, Any


class ModelTier(Enum):
    """Model tier for cost/quality tradeoff."""
    CHEAP = "cheap"      # Fast, low-cost (Haiku, GPT-4o-mini)
    DEFAULT = "default"  # Balanced (Sonnet, GPT-4o)
    BEST = "best"        # Highest quality (Opus, o1)


# Model mappings per provider and tier
TIER_MODELS: Dict[str, Dict[ModelTier, str]] = {
    "anthropic": {
        ModelTier.CHEAP: "claude-3-haiku-20240307",
        ModelTier.DEFAULT: "claude-sonnet-4-20250514",
        ModelTier.BEST: "claude-opus-4-5-20251101",
    },
    "openai": {
        ModelTier.CHEAP: "gpt-4o-mini",
        ModelTier.DEFAULT: "gpt-4o",
        ModelTier.BEST: "o1",
    },
    "google": {
        ModelTier.CHEAP: "gemini-2.0-flash",
        ModelTier.DEFAULT: "gemini-pro",
        ModelTier.BEST: "gemini-pro",  # No higher tier available
    },
}

# Patterns that require BEST tier
COMPLEX_PATTERNS = {
    "business-logic",
    "economic-exploit",
    "governance-attack",
    "cross-contract",
    "state-machine",
}


@dataclass
class TierConfig:
    """Configuration for tier selection."""
    default_tier: ModelTier = ModelTier.DEFAULT
    critical_tier: ModelTier = ModelTier.BEST
    low_severity_tier: ModelTier = ModelTier.CHEAP
    low_confidence_threshold: float = 0.4
    enable_auto_tier: bool = True
    cost_optimization: bool = False  # Prefer cheaper when possible


class TierSelector:
    """Selects appropriate model tier for tasks."""

    def __init__(self, config: TierConfig = None):
        self.config = config or TierConfig()

    def select(
        self,
        severity: str = "medium",
        confidence: float = 0.5,
        pattern_id: str = None,
        context_tokens: int = 0,
        explicit_tier: ModelTier = None,
    ) -> ModelTier:
        """
        Select model tier for a task.

        Args:
            severity: Finding severity
            confidence: Tier A confidence
            pattern_id: Pattern that triggered finding
            context_tokens: Token count for context
            explicit_tier: Explicit tier override

        Returns:
            Selected ModelTier
        """
        # Explicit override
        if explicit_tier is not None:
            return explicit_tier

        # Auto-selection disabled
        if not self.config.enable_auto_tier:
            return self.config.default_tier

        # Critical severity -> best tier
        if severity == "critical":
            return self.config.critical_tier

        # Complex patterns -> best tier
        if pattern_id and pattern_id in COMPLEX_PATTERNS:
            return ModelTier.BEST

        # Low confidence -> needs better model
        if confidence < self.config.low_confidence_threshold:
            return ModelTier.BEST

        # Large context -> needs larger context window (often best tier)
        if context_tokens > 32000:
            return ModelTier.BEST

        # Cost optimization: use cheaper when possible
        if self.config.cost_optimization:
            if severity in ("low", "info"):
                return self.config.low_severity_tier
            if confidence > 0.7:
                return ModelTier.CHEAP  # High confidence, quick check

        # Default
        return self.config.default_tier

    def get_model(self, provider: str, tier: ModelTier) -> str:
        """Get model name for provider and tier."""
        provider_models = TIER_MODELS.get(provider, {})
        return provider_models.get(tier, provider_models.get(ModelTier.DEFAULT, ""))

    def estimate_cost(
        self,
        provider: str,
        tier: ModelTier,
        input_tokens: int,
        output_tokens: int,
    ) -> float:
        """Estimate cost for a request."""
        from .metrics import CostCalculator

        calculator = CostCalculator()
        model = self.get_model(provider, tier)
        return calculator.calculate(input_tokens, output_tokens, provider, model)


class MultiTierLLMClient:
    """
    LLM client with automatic tier selection.

    Extends UnifiedLLMClient with tier-aware routing.
    """

    def __init__(
        self,
        base_client: "UnifiedLLMClient",
        tier_config: TierConfig = None,
    ):
        self.client = base_client
        self.selector = TierSelector(tier_config)
        self._tier_usage: Dict[ModelTier, int] = {t: 0 for t in ModelTier}

    async def analyze(
        self,
        context: str,
        prompt: str,
        severity: str = "medium",
        confidence: float = 0.5,
        pattern_id: str = None,
        explicit_tier: ModelTier = None,
    ) -> "LLMResult":
        """
        Analyze with automatic tier selection.

        Args:
            context: Code context
            prompt: Analysis prompt
            severity: Finding severity
            confidence: Tier A confidence
            pattern_id: Pattern ID
            explicit_tier: Override tier selection

        Returns:
            LLMResult from appropriate tier
        """
        # Estimate tokens
        from .slicer import TokenEstimator
        estimator = TokenEstimator()
        context_tokens = estimator.count(context)

        # Select tier
        tier = self.selector.select(
            severity=severity,
            confidence=confidence,
            pattern_id=pattern_id,
            context_tokens=context_tokens,
            explicit_tier=explicit_tier,
        )

        # Track usage
        self._tier_usage[tier] += 1

        # Call with selected tier
        return await self.client.analyze(
            context=context,
            prompt=prompt,
            tier=tier,
        )

    def get_tier_usage(self) -> Dict[str, int]:
        """Get usage by tier."""
        return {t.value: count for t, count in self._tier_usage.items()}

    def estimate_savings(self, findings_count: int) -> Dict[str, Any]:
        """
        Estimate cost savings from tier selection.

        Compares actual tier usage to always-using-BEST.
        """
        # Rough estimates
        avg_tokens = 3000
        tier_costs = {
            ModelTier.CHEAP: 0.001,    # ~$1/1M tokens
            ModelTier.DEFAULT: 0.01,    # ~$10/1M tokens
            ModelTier.BEST: 0.05,       # ~$50/1M tokens
        }

        actual_cost = sum(
            count * tier_costs[tier] * avg_tokens / 1000
            for tier, count in self._tier_usage.items()
        )

        best_only_cost = findings_count * tier_costs[ModelTier.BEST] * avg_tokens / 1000

        return {
            "actual_cost_usd": round(actual_cost, 4),
            "best_only_cost_usd": round(best_only_cost, 4),
            "savings_usd": round(best_only_cost - actual_cost, 4),
            "savings_pct": round((best_only_cost - actual_cost) / best_only_cost * 100, 1) if best_only_cost > 0 else 0,
        }
```
