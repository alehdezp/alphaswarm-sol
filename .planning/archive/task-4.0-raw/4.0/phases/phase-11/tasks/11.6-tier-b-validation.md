## Objective

Prove that Tier B adds measurable value: precision improvement >= 10%, recall preserved, cost < $0.50/audit.

---

## Context

### Why This Matters

Tier B is only valuable if it improves outcomes:
- If it doesn't reduce FP, it's waste
- If it dismisses TP, it's harmful
- If it costs too much, it's impractical

This task creates a rigorous validation framework.

---

## Design

### Validation Protocol

```
1. Prepare Ground Truth Dataset
   - 50 findings with known labels (TP/FP)
   - Mix of patterns and severities
   - Include edge cases

2. Run Tier A Only
   - Record precision (TP / (TP + FP))
   - Record recall (TP / (TP + FN))
   - Record time

3. Run Tier A + Tier B
   - Same findings
   - Record precision improvement
   - Record recall preservation
   - Record cost

4. Compare and Report
   - Precision delta >= 10%?
   - Recall delta <= 0%?
   - Cost < $0.50?
```

### Metrics Definitions

| Metric | Definition | Target |
|--------|------------|--------|
| Precision | TP / (TP + FP) | Improve >= 10% |
| Recall | TP / (TP + FN) | No degradation |
| FP Reduction | (FP_before - FP_after) / FP_before | >= 20% |
| Cost | Total USD spent | < $0.50 |

"""
Tier B Validation Test Suite

Validates that Tier B improves precision while preserving recall.
"""

import pytest
from dataclasses import dataclass
from typing import List, Dict, Any
import json
import os

from true_vkg.llm.workflow import TierBWorkflow, TierAFinding, TierBConfig
from true_vkg.llm.fp_filter import FalsePositiveFilter, apply_fp_filter
from true_vkg.llm.client import UnifiedLLMClient


@dataclass
class LabeledFinding:
    """Finding with ground truth label."""
    finding: TierAFinding
    ground_truth: str  # "TP" or "FP"
    notes: str = ""


@dataclass
class ValidationResult:
    """Result of validation run."""
    mode: str  # "tier_a" or "tier_ab"
    total_findings: int
    true_positives: int
    false_positives: int
    false_negatives: int
    precision: float
    recall: float
    f1_score: float
    cost_usd: float
    time_seconds: float

    def to_dict(self) -> dict:
        return {
            "mode": self.mode,
            "total_findings": self.total_findings,
            "true_positives": self.true_positives,
            "false_positives": self.false_positives,
            "false_negatives": self.false_negatives,
            "precision": round(self.precision, 4),
            "recall": round(self.recall, 4),
            "f1_score": round(self.f1_score, 4),
            "cost_usd": round(self.cost_usd, 4),
            "time_seconds": round(self.time_seconds, 2),
        }


class TierBValidator:
    """Validates Tier B effectiveness."""

    def __init__(
        self,
        workflow: TierBWorkflow = None,
        llm_client: UnifiedLLMClient = None,
    ):
        self.workflow = workflow
        self.llm_client = llm_client

    def run_tier_a_only(
        self,
        labeled_findings: List[LabeledFinding],
    ) -> ValidationResult:
        """
        Run Tier A only (no LLM) validation.

        Assumes all Tier A findings are reported as-is.
        """
        import time
        start = time.time()

        # In Tier A only mode, all findings are reported
        # We compare against ground truth
        tp = sum(1 for lf in labeled_findings if lf.ground_truth == "TP")
        fp = sum(1 for lf in labeled_findings if lf.ground_truth == "FP")
        fn = 0  # Tier A doesn't miss findings

        precision = tp / (tp + fp) if (tp + fp) > 0 else 0
        recall = tp / (tp + fn) if (tp + fn) > 0 else 1.0
        f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

        return ValidationResult(
            mode="tier_a",
            total_findings=len(labeled_findings),
            true_positives=tp,
            false_positives=fp,
            false_negatives=fn,
            precision=precision,
            recall=recall,
            f1_score=f1,
            cost_usd=0.0,  # No LLM cost
            time_seconds=time.time() - start,
        )

    async def run_tier_ab(
        self,
        labeled_findings: List[LabeledFinding],
        graph: Any,
    ) -> ValidationResult:
        """
        Run Tier A + Tier B validation.

        Uses LLM to filter findings, then compares to ground truth.
        """
        import time

        if self.workflow is None:
            raise ValueError("Workflow required for Tier A+B validation")

        start = time.time()

        # Run Tier B on all findings
        results = await self.workflow.analyze_batch(
            [lf.finding for lf in labeled_findings],
            graph,
        )

        # Determine what gets reported
        # AUTO_CONFIRM and VULNERABLE verdicts are reported
        # AUTO_DISMISS and SAFE verdicts are filtered
        reported = set()
        for lf, result in zip(labeled_findings, results):
            if result.final_verdict in ("VULNERABLE", "UNCERTAIN"):
                reported.add(lf.finding.id)

        # Calculate metrics
        tp = 0
        fp = 0
        fn = 0

        for lf in labeled_findings:
            is_reported = lf.finding.id in reported
            is_actual_vuln = lf.ground_truth == "TP"

            if is_reported and is_actual_vuln:
                tp += 1
            elif is_reported and not is_actual_vuln:
                fp += 1
            elif not is_reported and is_actual_vuln:
                fn += 1
            # Not reported and not actual vuln = correct dismissal (not counted)

        precision = tp / (tp + fp) if (tp + fp) > 0 else 0
        recall = tp / (tp + fn) if (tp + fn) > 0 else 1.0
        f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

        # Get cost from workflow stats
        stats = self.workflow.get_stats()
        cost = stats.get("llm_usage", {}).get("usage", {}).get("cost_usd", 0)

        return ValidationResult(
            mode="tier_ab",
            total_findings=len(labeled_findings),
            true_positives=tp,
            false_positives=fp,
            false_negatives=fn,
            precision=precision,
            recall=recall,
            f1_score=f1,
            cost_usd=cost,
            time_seconds=time.time() - start,
        )

    def compare(
        self,
        tier_a: ValidationResult,
        tier_ab: ValidationResult,
    ) -> Dict[str, Any]:
        """Compare Tier A and Tier A+B results."""
        precision_delta = tier_ab.precision - tier_a.precision
        recall_delta = tier_ab.recall - tier_a.recall
        f1_delta = tier_ab.f1_score - tier_a.f1_score
        fp_reduction = (
            (tier_a.false_positives - tier_ab.false_positives) / tier_a.false_positives
            if tier_a.false_positives > 0 else 0
        )

        return {
            "tier_a": tier_a.to_dict(),
            "tier_ab": tier_ab.to_dict(),
            "comparison": {
                "precision_delta": round(precision_delta, 4),
                "recall_delta": round(recall_delta, 4),
                "f1_delta": round(f1_delta, 4),
                "fp_reduction_pct": round(fp_reduction * 100, 2),
                "cost_per_finding": round(tier_ab.cost_usd / tier_ab.total_findings, 4) if tier_ab.total_findings > 0 else 0,
            },
            "pass_criteria": {
                "precision_improved": precision_delta >= 0.10,
                "recall_preserved": recall_delta >= 0,
                "cost_acceptable": tier_ab.cost_usd < 0.50,
                "fp_reduced": fp_reduction >= 0.20,
            },
            "overall_pass": (
                precision_delta >= 0.10 and
                recall_delta >= 0 and
                tier_ab.cost_usd < 0.50
            ),
        }
