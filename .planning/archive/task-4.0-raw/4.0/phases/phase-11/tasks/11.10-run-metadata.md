## Objective

## Context

### Why Track Metadata?

1. **Reproducibility**: Same inputs should produce similar outputs
2. **Debugging**: Understand why a verdict was given
3. **Auditing**: Track who ran what, when
4. **Cost Analysis**: Historical cost tracking
5. **Regression Detection**: Compare runs over time

```python
"""
LLM Run Metadata Tracking

Tracks all LLM runs with complete metadata for reproducibility.
"""

import json
import os
import platform
import sys
import uuid
from dataclasses import dataclass, field, asdict
from datetime import datetime
from pathlib import Path
from typing import Dict, Any, List, Optional
import logging

logger = logging.getLogger(__name__)

# Default storage location
DEFAULT_METADATA_DIR = ".vrs/llm_runs"


@dataclass
class RunConfig:
    """Configuration used for the run."""
    auto_confirm_threshold: float = 0.9
    auto_dismiss_threshold: float = 0.3
    token_budget: int = 4000
    provider: str = "anthropic"
    model: str = "claude-sonnet-4-20250514"
    model_tier: str = "default"


@dataclass
class InputSummary:
    """Summary of input to the run."""
    findings_count: int = 0
    patterns: List[str] = field(default_factory=list)
    source_files: List[str] = field(default_factory=list)
    severities: Dict[str, int] = field(default_factory=dict)


@dataclass
class ResultsSummary:
    """Summary of run results."""
    vulnerable: int = 0
    safe: int = 0
    uncertain: int = 0
    auto_confirmed: int = 0
    auto_dismissed: int = 0
    errors: int = 0


@dataclass
class UsageSummary:
    """LLM usage summary."""
    input_tokens: int = 0
    output_tokens: int = 0
    total_tokens: int = 0
    cost_usd: float = 0.0
    requests: int = 0
    duration_seconds: float = 0.0


@dataclass
class Environment:
    """Execution environment info."""
    python_version: str = ""
    os: str = ""
    os_version: str = ""
    machine: str = ""
    vkg_version: str = ""

    @classmethod
    def capture(cls) -> "Environment":
        """Capture current environment."""
        return cls(
            python_version=sys.version.split()[0],
            os=platform.system(),
            os_version=platform.release(),
            machine=platform.machine(),
            vkg_version=_get_vkg_version(),
        )


@dataclass
class RunMetadata:
    """Complete metadata for an LLM run."""
    run_id: str
    timestamp: str
    config: RunConfig
    input_summary: InputSummary
    results_summary: ResultsSummary
    usage: UsageSummary
    environment: Environment
    notes: str = ""
    tags: List[str] = field(default_factory=list)

    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary."""
        return {
            "run_id": self.run_id,
            "timestamp": self.timestamp,
            "config": asdict(self.config),
            "input_summary": asdict(self.input_summary),
            "results_summary": asdict(self.results_summary),
            "usage": asdict(self.usage),
            "environment": asdict(self.environment),
            "notes": self.notes,
            "tags": self.tags,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "RunMetadata":
        """Create from dictionary."""
        return cls(
            run_id=data["run_id"],
            timestamp=data["timestamp"],
            config=RunConfig(**data.get("config", {})),
            input_summary=InputSummary(**data.get("input_summary", {})),
            results_summary=ResultsSummary(**data.get("results_summary", {})),
            usage=UsageSummary(**data.get("usage", {})),
            environment=Environment(**data.get("environment", {})),
            notes=data.get("notes", ""),
            tags=data.get("tags", []),
        )


class MetadataTracker:
    """
    Tracks LLM run metadata.

    Features:
    - Automatic run ID generation
    - Environment capture
    - Persistent storage
    - History querying
    """

    def __init__(self, storage_dir: str = None):
        self.storage_dir = Path(storage_dir or DEFAULT_METADATA_DIR)
        self.current_run: Optional[RunMetadata] = None
        self._ensure_storage_dir()

    def _ensure_storage_dir(self):
        """Ensure storage directory exists."""
        self.storage_dir.mkdir(parents=True, exist_ok=True)

    def start_run(
        self,
        config: RunConfig = None,
        notes: str = "",
        tags: List[str] = None,
    ) -> RunMetadata:
        """
        Start a new run.

        Args:
            config: Run configuration
            notes: Optional notes
            tags: Optional tags

        Returns:
            RunMetadata for the new run
        """
        run_id = self._generate_run_id()

        self.current_run = RunMetadata(
            run_id=run_id,
            timestamp=datetime.utcnow().isoformat() + "Z",
            config=config or RunConfig(),
            input_summary=InputSummary(),
            results_summary=ResultsSummary(),
            usage=UsageSummary(),
            environment=Environment.capture(),
            notes=notes,
            tags=tags or [],
        )

        logger.info(f"Started LLM run: {run_id}")
        return self.current_run

    def update_input(
        self,
        findings_count: int = None,
        patterns: List[str] = None,
        source_files: List[str] = None,
        severities: Dict[str, int] = None,
    ):
        """Update input summary."""
        if self.current_run is None:
            raise RuntimeError("No active run")

        if findings_count is not None:
            self.current_run.input_summary.findings_count = findings_count
        if patterns is not None:
            self.current_run.input_summary.patterns = patterns
        if source_files is not None:
            self.current_run.input_summary.source_files = source_files
        if severities is not None:
            self.current_run.input_summary.severities = severities

    def update_results(
        self,
        vulnerable: int = None,
        safe: int = None,
        uncertain: int = None,
        auto_confirmed: int = None,
        auto_dismissed: int = None,
        errors: int = None,
    ):
        """Update results summary."""
        if self.current_run is None:
            raise RuntimeError("No active run")

        summary = self.current_run.results_summary
        if vulnerable is not None:
            summary.vulnerable = vulnerable
        if safe is not None:
            summary.safe = safe
        if uncertain is not None:
            summary.uncertain = uncertain
        if auto_confirmed is not None:
            summary.auto_confirmed = auto_confirmed
        if auto_dismissed is not None:
            summary.auto_dismissed = auto_dismissed
        if errors is not None:
            summary.errors = errors

    def update_usage(
        self,
        input_tokens: int = None,
        output_tokens: int = None,
        cost_usd: float = None,
        requests: int = None,
        duration_seconds: float = None,
    ):
        """Update usage summary."""
        if self.current_run is None:
            raise RuntimeError("No active run")

        usage = self.current_run.usage
        if input_tokens is not None:
            usage.input_tokens = input_tokens
        if output_tokens is not None:
            usage.output_tokens = output_tokens
        if input_tokens is not None or output_tokens is not None:
            usage.total_tokens = usage.input_tokens + usage.output_tokens
        if cost_usd is not None:
            usage.cost_usd = cost_usd
        if requests is not None:
            usage.requests = requests
        if duration_seconds is not None:
            usage.duration_seconds = duration_seconds

    def end_run(self) -> RunMetadata:
        """
        End current run and save metadata.

        Returns:
            Completed RunMetadata
        """
        if self.current_run is None:
            raise RuntimeError("No active run")

        # Save to file
        self._save_run(self.current_run)

        result = self.current_run
        self.current_run = None

        logger.info(f"Ended LLM run: {result.run_id}")
        return result

    def _save_run(self, metadata: RunMetadata):
        """Save run metadata to file."""
        filename = f"{metadata.run_id}.json"
        filepath = self.storage_dir / filename

        with open(filepath, 'w') as f:
            json.dump(metadata.to_dict(), f, indent=2)

        logger.debug(f"Saved run metadata to {filepath}")

    def _generate_run_id(self) -> str:
        """Generate unique run ID."""
        date_part = datetime.utcnow().strftime("%Y%m%d-%H%M%S")
        unique_part = uuid.uuid4().hex[:6]
        return f"llm-{date_part}-{unique_part}"

    def get_run(self, run_id: str) -> Optional[RunMetadata]:
        """Load run by ID."""
        filepath = self.storage_dir / f"{run_id}.json"
        if not filepath.exists():
            return None

        with open(filepath) as f:
            data = json.load(f)
        return RunMetadata.from_dict(data)

    def list_runs(self, limit: int = 20) -> List[RunMetadata]:
        """List recent runs."""
        runs = []
        files = sorted(self.storage_dir.glob("llm-*.json"), reverse=True)

        for filepath in files[:limit]:
            try:
                with open(filepath) as f:
                    data = json.load(f)
                runs.append(RunMetadata.from_dict(data))
            except Exception as e:
                logger.warning(f"Failed to load {filepath}: {e}")

        return runs

    def get_history_summary(self) -> Dict[str, Any]:
        """Get summary of run history."""
        runs = self.list_runs(limit=100)

        if not runs:
            return {"runs": 0}

        total_cost = sum(r.usage.cost_usd for r in runs)
        total_tokens = sum(r.usage.total_tokens for r in runs)
        total_findings = sum(r.input_summary.findings_count for r in runs)

        return {
            "runs": len(runs),
            "total_cost_usd": round(total_cost, 2),
            "total_tokens": total_tokens,
            "total_findings_analyzed": total_findings,
            "avg_cost_per_run": round(total_cost / len(runs), 4),
            "date_range": {
                "earliest": runs[-1].timestamp,
                "latest": runs[0].timestamp,
            },
        }


def _get_vkg_version() -> str:
    """Get BSKG version."""
    try:
        from true_vkg import __version__
        return __version__
    except ImportError:
        return "unknown"


# Global tracker instance
_tracker: Optional[MetadataTracker] = None


def get_metadata_tracker() -> MetadataTracker:
    """Get global metadata tracker."""
    global _tracker
    if _tracker is None:
        _tracker = MetadataTracker()
    return _tracker
```

```python
"""Tests for LLM run metadata tracking."""

import pytest
import tempfile
import json
from pathlib import Path

from true_vkg.llm.metadata import (
    MetadataTracker, RunMetadata, RunConfig, InputSummary,
    ResultsSummary, UsageSummary, Environment
)


class TestRunMetadata:
    """Test metadata dataclass."""

    def test_to_dict(self):
        """Metadata converts to dict."""
        metadata = RunMetadata(
            run_id="test-123",
            timestamp="2026-01-07T12:00:00Z",
            config=RunConfig(),
            input_summary=InputSummary(findings_count=10),
            results_summary=ResultsSummary(vulnerable=3),
            usage=UsageSummary(cost_usd=0.15),
            environment=Environment(python_version="3.11"),
        )

        d = metadata.to_dict()
        assert d["run_id"] == "test-123"
        assert d["input_summary"]["findings_count"] == 10

    def test_from_dict(self):
        """Metadata loads from dict."""
        data = {
            "run_id": "test-123",
            "timestamp": "2026-01-07T12:00:00Z",
            "config": {},
            "input_summary": {"findings_count": 5},
            "results_summary": {},
            "usage": {},
            "environment": {},
        }

        metadata = RunMetadata.from_dict(data)
        assert metadata.run_id == "test-123"
        assert metadata.input_summary.findings_count == 5


class TestMetadataTracker:
    """Test metadata tracker."""

    @pytest.fixture
    def tracker(self, tmp_path):
        """Create tracker with temp storage."""
        return MetadataTracker(storage_dir=str(tmp_path))

    def test_start_run(self, tracker):
        """Start run creates metadata."""
        metadata = tracker.start_run()

        assert metadata.run_id.startswith("llm-")
        assert tracker.current_run is not None

    def test_update_input(self, tracker):
        """Update input summary."""
        tracker.start_run()
        tracker.update_input(
            findings_count=15,
            patterns=["reentrancy", "access"],
        )

        assert tracker.current_run.input_summary.findings_count == 15
        assert "reentrancy" in tracker.current_run.input_summary.patterns

    def test_update_results(self, tracker):
        """Update results summary."""
        tracker.start_run()
        tracker.update_results(
            vulnerable=5,
            safe=10,
        )

        assert tracker.current_run.results_summary.vulnerable == 5
        assert tracker.current_run.results_summary.safe == 10

    def test_update_usage(self, tracker):
        """Update usage summary."""
        tracker.start_run()
        tracker.update_usage(
            input_tokens=1000,
            output_tokens=500,
            cost_usd=0.05,
        )

        assert tracker.current_run.usage.total_tokens == 1500
        assert tracker.current_run.usage.cost_usd == 0.05

    def test_end_run(self, tracker):
        """End run saves metadata."""
        tracker.start_run()
        run_id = tracker.current_run.run_id
        tracker.update_input(findings_count=5)

        result = tracker.end_run()

        assert result.run_id == run_id
        assert tracker.current_run is None

        # File should exist
        saved_file = tracker.storage_dir / f"{run_id}.json"
        assert saved_file.exists()

    def test_get_run(self, tracker):
        """Load run by ID."""
        tracker.start_run()
        run_id = tracker.current_run.run_id
        tracker.update_input(findings_count=7)
        tracker.end_run()

        loaded = tracker.get_run(run_id)

        assert loaded is not None
        assert loaded.run_id == run_id
        assert loaded.input_summary.findings_count == 7

    def test_list_runs(self, tracker):
        """List recent runs."""
        # Create multiple runs
        for i in range(3):
            tracker.start_run(notes=f"Run {i}")
            tracker.end_run()

        runs = tracker.list_runs(limit=10)

        assert len(runs) == 3

    def test_history_summary(self, tracker):
        """Get history summary."""
        tracker.start_run()
        tracker.update_usage(cost_usd=0.10, input_tokens=500, output_tokens=200)
        tracker.update_input(findings_count=5)
        tracker.end_run()

        tracker.start_run()
        tracker.update_usage(cost_usd=0.20, input_tokens=1000, output_tokens=400)
        tracker.update_input(findings_count=10)
        tracker.end_run()

        summary = tracker.get_history_summary()

        assert summary["runs"] == 2
        assert summary["total_cost_usd"] == 0.30
        assert summary["total_findings_analyzed"] == 15
```
