"""
LLM Metrics and Cost Tracking

Tracks token usage, costs, and provides budget enforcement.
"""

import time
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Any
from datetime import datetime
import logging
import json

logger = logging.getLogger(__name__)


# Cost per 1M tokens (USD)
PROVIDER_COSTS: Dict[str, Dict[str, Dict[str, float]]] = {
    "anthropic": {
        "claude-sonnet-4-20250514": {"input": 3.00, "output": 15.00},
        "claude-opus-4-5-20251101": {"input": 15.00, "output": 75.00},
        "claude-3-haiku-20240307": {"input": 0.25, "output": 1.25},
    },
    "openai": {
        "gpt-4o": {"input": 5.00, "output": 15.00},
        "gpt-4o-mini": {"input": 0.15, "output": 0.60},
        "o1": {"input": 15.00, "output": 60.00},
    },
    "google": {
        "gemini-pro": {"input": 1.25, "output": 5.00},
        "gemini-2.0-flash": {"input": 0.075, "output": 0.30},
    },
    "ollama": {
        "default": {"input": 0.0, "output": 0.0},
    },
}


@dataclass
class TokenUsage:
    """Token usage for a single request."""
    input_tokens: int
    output_tokens: int
    provider: str
    model: str
    timestamp: float = field(default_factory=time.time)

    @property
    def total_tokens(self) -> int:
        return self.input_tokens + self.output_tokens


@dataclass
class CostEntry:
    """Cost entry for a single request."""
    input_cost: float
    output_cost: float
    total_cost: float
    provider: str
    model: str
    operation: str
    timestamp: float


@dataclass
class UsageSummary:
    """Summary of LLM usage."""
    total_input_tokens: int
    total_output_tokens: int
    total_tokens: int
    total_cost_usd: float
    request_count: int
    by_provider: Dict[str, Dict[str, Any]]
    by_operation: Dict[str, Dict[str, Any]]

    def to_dict(self) -> dict:
        return {
            "total_input_tokens": self.total_input_tokens,
            "total_output_tokens": self.total_output_tokens,
            "total_tokens": self.total_tokens,
            "total_cost_usd": round(self.total_cost_usd, 4),
            "request_count": self.request_count,
            "by_provider": self.by_provider,
            "by_operation": self.by_operation,
        }


class CostCalculator:
    """Calculates costs based on provider pricing."""

    def __init__(self, custom_costs: Dict = None):
        self.costs = custom_costs or PROVIDER_COSTS

    def calculate(
        self,
        input_tokens: int,
        output_tokens: int,
        provider: str,
        model: str,
    ) -> float:
        """
        Calculate cost for token usage.

        Args:
            input_tokens: Number of input tokens
            output_tokens: Number of output tokens
            provider: Provider name (anthropic, openai, etc.)
            model: Model name

        Returns:
            Cost in USD
        """
        provider_costs = self.costs.get(provider, {})
        model_costs = provider_costs.get(model)

        if model_costs is None:
            # Try default for provider
            model_costs = provider_costs.get("default", {"input": 0.0, "output": 0.0})
            if model_costs["input"] == 0 and model_costs["output"] == 0:
                logger.warning(f"Unknown model {provider}/{model}, using zero cost")

        input_cost = (input_tokens / 1_000_000) * model_costs["input"]
        output_cost = (output_tokens / 1_000_000) * model_costs["output"]

        return input_cost + output_cost

    def get_breakdown(
        self,
        input_tokens: int,
        output_tokens: int,
        provider: str,
        model: str,
    ) -> Dict[str, float]:
        """Get detailed cost breakdown."""
        provider_costs = self.costs.get(provider, {})
        model_costs = provider_costs.get(model, provider_costs.get("default", {"input": 0.0, "output": 0.0}))

        input_cost = (input_tokens / 1_000_000) * model_costs["input"]
        output_cost = (output_tokens / 1_000_000) * model_costs["output"]

        return {
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "input_cost_usd": round(input_cost, 6),
            "output_cost_usd": round(output_cost, 6),
            "total_cost_usd": round(input_cost + output_cost, 6),
            "input_rate_per_1m": model_costs["input"],
            "output_rate_per_1m": model_costs["output"],
        }


class LLMMetricsTracker:
    """
    Tracks LLM usage metrics and costs.

    Features:
    - Per-request tracking
    - Per-provider aggregation
    - Per-operation breakdown
    - Real-time cost estimation
    """

    def __init__(self):
        self.usages: List[TokenUsage] = []
        self.costs: List[CostEntry] = []
        self.calculator = CostCalculator()
        self._start_time = time.time()

    def record(
        self,
        input_tokens: int,
        output_tokens: int,
        provider: str,
        model: str,
        operation: str = "verify",
    ):
        """
        Record token usage.

        Args:
            input_tokens: Input token count
            output_tokens: Output token count
            provider: Provider name
            model: Model name
            operation: Operation type (verify, filter, etc.)
        """
        usage = TokenUsage(
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            provider=provider,
            model=model,
        )
        self.usages.append(usage)

        # Calculate cost
        cost = self.calculator.calculate(input_tokens, output_tokens, provider, model)
        breakdown = self.calculator.get_breakdown(input_tokens, output_tokens, provider, model)

        self.costs.append(CostEntry(
            input_cost=breakdown["input_cost_usd"],
            output_cost=breakdown["output_cost_usd"],
            total_cost=breakdown["total_cost_usd"],
            provider=provider,
            model=model,
            operation=operation,
            timestamp=time.time(),
        ))

        logger.debug(
            f"Recorded: {input_tokens}+{output_tokens} tokens, ${cost:.4f} "
            f"({provider}/{model}/{operation})"
        )

    def get_summary(self) -> UsageSummary:
        """Get usage summary."""
        total_input = sum(u.input_tokens for u in self.usages)
        total_output = sum(u.output_tokens for u in self.usages)
        total_cost = sum(c.total_cost for c in self.costs)

        # By provider
        by_provider: Dict[str, Dict[str, Any]] = {}
        for usage in self.usages:
            key = usage.provider
            if key not in by_provider:
                by_provider[key] = {
                    "input_tokens": 0,
                    "output_tokens": 0,
                    "cost_usd": 0.0,
                    "requests": 0,
                }
            by_provider[key]["input_tokens"] += usage.input_tokens
            by_provider[key]["output_tokens"] += usage.output_tokens
            by_provider[key]["requests"] += 1

        for cost in self.costs:
            if cost.provider in by_provider:
                by_provider[cost.provider]["cost_usd"] += cost.total_cost

        # By operation
        by_operation: Dict[str, Dict[str, Any]] = {}
        for cost in self.costs:
            key = cost.operation
            if key not in by_operation:
                by_operation[key] = {
                    "cost_usd": 0.0,
                    "requests": 0,
                }
            by_operation[key]["cost_usd"] += cost.total_cost
            by_operation[key]["requests"] += 1

        return UsageSummary(
            total_input_tokens=total_input,
            total_output_tokens=total_output,
            total_tokens=total_input + total_output,
            total_cost_usd=total_cost,
            request_count=len(self.usages),
            by_provider=by_provider,
            by_operation=by_operation,
        )

    def get_current_cost(self) -> float:
        """Get current total cost."""
        return sum(c.total_cost for c in self.costs)

    def format_progress(self, current: int, total: int) -> str:
        """
        Format progress string for display.

        Example: "[12/100] Tokens: 23,456 | Cost: $0.12 | Rate: 8 req/min"
        """
        summary = self.get_summary()
        elapsed = time.time() - self._start_time
        rate = len(self.usages) / elapsed * 60 if elapsed > 0 else 0

        return (
            f"[{current}/{total}] "
            f"Tokens: {summary.total_tokens:,} | "
            f"Cost: ${summary.total_cost_usd:.2f} | "
            f"Rate: {rate:.1f} req/min"
        )

    def format_summary(self) -> str:
        """Format final usage summary."""
        summary = self.get_summary()
        elapsed = time.time() - self._start_time

        lines = [
            "",
            "LLM Usage Summary",
            "=" * 40,
            f"Total requests:     {summary.request_count}",
            f"Total tokens:       {summary.total_tokens:,}",
            f"  Input:            {summary.total_input_tokens:,}",
            f"  Output:           {summary.total_output_tokens:,}",
            f"Total cost:         ${summary.total_cost_usd:.4f}",
            f"Time elapsed:       {elapsed:.1f}s",
            "",
            "By Provider:",
        ]

        for provider, data in summary.by_provider.items():
            lines.append(
                f"  {provider}: {data['requests']} req, "
                f"{data['input_tokens'] + data['output_tokens']:,} tokens, "
                f"${data['cost_usd']:.4f}"
            )

        lines.append("")
        lines.append("By Operation:")
        for op, data in summary.by_operation.items():
            lines.append(f"  {op}: {data['requests']} req, ${data['cost_usd']:.4f}")

        return "\n".join(lines)

    def save_to_file(self, path: str):
        """Save metrics to JSON file."""
        data = {
            "summary": self.get_summary().to_dict(),
            "usages": [
                {
                    "input_tokens": u.input_tokens,
                    "output_tokens": u.output_tokens,
                    "provider": u.provider,
                    "model": u.model,
                    "timestamp": u.timestamp,
                }
                for u in self.usages
            ],
            "costs": [
                {
                    "input_cost": c.input_cost,
                    "output_cost": c.output_cost,
                    "total_cost": c.total_cost,
                    "provider": c.provider,
                    "model": c.model,
                    "operation": c.operation,
                    "timestamp": c.timestamp,
                }
                for c in self.costs
            ],
            "generated_at": datetime.now().isoformat(),
        }

        with open(path, 'w') as f:
            json.dump(data, f, indent=2)


# Global metrics tracker
_metrics_tracker: Optional[LLMMetricsTracker] = None


def get_metrics_tracker() -> LLMMetricsTracker:
    """Get global metrics tracker."""
    global _metrics_tracker
    if _metrics_tracker is None:
        _metrics_tracker = LLMMetricsTracker()
    return _metrics_tracker


def reset_metrics_tracker():
    """Reset global metrics tracker (for testing)."""
    global _metrics_tracker
    _metrics_tracker = None

            input_tokens=1000,
