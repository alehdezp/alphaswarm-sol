```python
"""Edge weight calculation for PPR.

Edge weights determine how probability flows through the graph.
Security-relevant edges get higher weights.
"""
from typing import Dict, Any
from enum import Enum


class EdgeType(Enum):
    """VKG edge types with security relevance."""
    CALLS = "calls"
    CALLS_EXTERNAL = "calls_external"
    READS_STATE = "reads_state"
    WRITES_STATE = "writes_state"
    HAS_MODIFIER = "has_modifier"
    INHERITS = "inherits"
    USES = "uses"


# Base weights per edge type (from R9.1 research)
BASE_WEIGHTS: Dict[EdgeType, float] = {
    EdgeType.CALLS: 1.0,
    EdgeType.CALLS_EXTERNAL: 1.5,  # Security-relevant
    EdgeType.READS_STATE: 0.8,
    EdgeType.WRITES_STATE: 1.2,    # State changes important
    EdgeType.HAS_MODIFIER: 0.5,    # Guards reduce weight
    EdgeType.INHERITS: 0.3,        # Structural, not attack path
    EdgeType.USES: 0.5,
}


def calculate_edge_weight(edge: Dict[str, Any]) -> float:
    """Calculate PPR weight for an edge.

    Args:
        edge: Edge dictionary with type, risk_score, guards_at_source

    Returns:
        Weight value (unnormalized)
    """
    # Get edge type
    edge_type_str = edge.get("type", "uses")
    try:
        edge_type = EdgeType(edge_type_str)
    except ValueError:
        edge_type = EdgeType.USES

    # Base weight
    weight = BASE_WEIGHTS.get(edge_type, 0.5)

    # Risk adjustment: increase weight for risky edges
    risk_score = edge.get("risk_score", 0.0)
    if risk_score > 0:
        weight *= (1.0 + min(risk_score, 1.0))  # Cap at 2x

    # Taint adjustment: increase weight for tainted data flow
    if edge.get("taint_source"):
        weight *= 1.3

    # Guard penalty: decrease weight for guarded edges
    if edge.get("guards_at_source"):
        weight *= 0.6

    return weight


def normalize_weights(
    weights: Dict[str, float],
    edges_from_node: Dict[str, list]
) -> Dict[str, float]:
    """Normalize weights so outgoing weights sum to 1 per node.

    This is CRITICAL for PPR correctness.
    """
    normalized = {}

    for node_id, edges in edges_from_node.items():
        total = sum(weights.get(e["id"], 1.0) for e in edges)
        if total > 0:
            for e in edges:
                edge_id = e["id"]
                normalized[edge_id] = weights.get(edge_id, 1.0) / total

    return normalized
```

### Step 2: Create PPR Algorithm

**File:** `src/true_vkg/kg/ppr.py`

```python
"""Personalized PageRank for BSKG security graph traversal.

This module implements PPR adapted for vulnerability detection:
- Seeds are vulnerability-related nodes
- Edge weights reflect security relevance
- Result is relevance score per node for context extraction
"""
from typing import Dict, List, Set, Optional, Any
from dataclasses import dataclass
import math
from .ppr_weights import calculate_edge_weight, normalize_weights


@dataclass
class PPRConfig:
    """PPR algorithm configuration."""
    alpha: float = 0.15      # Teleport probability
    max_iter: int = 50       # Maximum iterations
    epsilon: float = 1e-4    # Convergence threshold

    # Preset configs for context modes
    @classmethod
    def strict(cls) -> "PPRConfig":
        """High teleport - stays close to seeds."""
        return cls(alpha=0.25, max_iter=30, epsilon=1e-3)

    @classmethod
    def standard(cls) -> "PPRConfig":
        """Balanced exploration."""
        return cls(alpha=0.15, max_iter=50, epsilon=1e-4)

    @classmethod
    def relaxed(cls) -> "PPRConfig":
        """Wide exploration."""
        return cls(alpha=0.10, max_iter=100, epsilon=1e-5)


class VKGPPR:
    """Personalized PageRank for BSKG graphs.

    Usage:
        ppr = VKGPPR(graph)
        scores = ppr.run(seed_nodes=["withdraw", "balances"])
        # scores is Dict[str, float] mapping node_id to relevance
    """

    def __init__(self, graph: Dict[str, Any]):
        """Initialize with BSKG graph.

        Args:
            graph: BSKG graph dictionary with nodes and edges
        """
        self.graph = graph
        self.nodes = self._extract_nodes()
        self.edges = self._extract_edges()
        self.in_edges = self._build_in_edges()
        self.out_edges = self._build_out_edges()
        self.edge_weights = self._calculate_weights()

    def _extract_nodes(self) -> Set[str]:
        """Extract all node IDs from graph."""
        nodes = set()
        for node in self.graph.get("nodes", []):
            nodes.add(node.get("id", node.get("name")))
        return nodes

    def _extract_edges(self) -> List[Dict[str, Any]]:
        """Extract all edges from graph."""
        return self.graph.get("edges", [])

    def _build_in_edges(self) -> Dict[str, List[Dict]]:
        """Build mapping of node -> incoming edges."""
        in_edges: Dict[str, List[Dict]] = {n: [] for n in self.nodes}
        for edge in self.edges:
            target = edge.get("target", edge.get("to"))
            if target in in_edges:
                in_edges[target].append(edge)
        return in_edges

    def _build_out_edges(self) -> Dict[str, List[Dict]]:
        """Build mapping of node -> outgoing edges."""
        out_edges: Dict[str, List[Dict]] = {n: [] for n in self.nodes}
        for edge in self.edges:
            source = edge.get("source", edge.get("from"))
            if source in out_edges:
                out_edges[source].append(edge)
        return out_edges

    def _calculate_weights(self) -> Dict[str, float]:
        """Calculate and normalize edge weights."""
        # Raw weights
        raw = {}
        for edge in self.edges:
            edge_id = edge.get("id", f"{edge.get('source')}->{edge.get('target')}")
            raw[edge_id] = calculate_edge_weight(edge)

        # Normalize per source node
        return normalize_weights(raw, self.out_edges)

    def run(
        self,
        seeds: List[str],
        config: Optional[PPRConfig] = None
    ) -> Dict[str, float]:
        """Run Personalized PageRank from seed nodes.

        Args:
            seeds: List of seed node IDs
            config: PPR configuration (default: standard)

        Returns:
            Dictionary mapping node_id to relevance score
        """
        if config is None:
            config = PPRConfig.standard()

        # Filter seeds to valid nodes
        valid_seeds = [s for s in seeds if s in self.nodes]
        if not valid_seeds:
            # No valid seeds - return uniform distribution
            return {n: 1.0 / len(self.nodes) for n in self.nodes}

        # Initialize scores
        scores = {n: 0.0 for n in self.nodes}
        seed_value = 1.0 / len(valid_seeds)
        for seed in valid_seeds:
            scores[seed] = seed_value

        # Iterate until convergence
        for iteration in range(config.max_iter):
            new_scores = self._iterate(scores, valid_seeds, config.alpha)

            # Check convergence
            if self._converged(scores, new_scores, config.epsilon):
                break

            scores = new_scores

        return scores

    def _iterate(
        self,
        scores: Dict[str, float],
        seeds: List[str],
        alpha: float
    ) -> Dict[str, float]:
        """Perform one PPR iteration.

        PPR formula:
        PPR(v) = alpha * (1/|seeds| if v in seeds else 0)
               + (1-alpha) * Σ(PPR(u) * weight(u->v))
        """
        new_scores = {}
        seed_set = set(seeds)
        seed_teleport = alpha / len(seeds)

        for node in self.nodes:
            # Teleport component
            if node in seed_set:
                teleport = seed_teleport
            else:
                teleport = 0.0

            # Walk component
            walk = 0.0
            for edge in self.in_edges[node]:
                source = edge.get("source", edge.get("from"))
                edge_id = edge.get("id", f"{source}->{node}")
                weight = self.edge_weights.get(edge_id, 0.0)
                walk += scores.get(source, 0.0) * weight

            new_scores[node] = teleport + (1 - alpha) * walk

        return new_scores

    def _converged(
        self,
        old: Dict[str, float],
        new: Dict[str, float],
        epsilon: float
    ) -> bool:
        """Check if scores have converged."""
        max_diff = max(abs(new[n] - old[n]) for n in self.nodes)
        return max_diff < epsilon

    def get_top_nodes(
        self,
        scores: Dict[str, float],
        k: int = 10
    ) -> List[tuple]:
        """Get top k nodes by score.

        Returns:
            List of (node_id, score) tuples, sorted descending
        """
        sorted_nodes = sorted(scores.items(), key=lambda x: x[1], reverse=True)
        return sorted_nodes[:k]


def run_ppr(
    graph: Dict[str, Any],
    seeds: List[str],
    config: Optional[PPRConfig] = None
) -> Dict[str, float]:
    """Convenience function to run PPR on a graph.

    Args:
        graph: BSKG graph dictionary
        seeds: Seed node IDs
        config: PPR configuration

    Returns:
        Node relevance scores
    """
    ppr = VKGPPR(graph)
    return ppr.run(seeds, config)
```

"""Tests for PPR algorithm."""
import pytest
from true_vkg.kg.ppr import VKGPPR, PPRConfig, run_ppr


class TestPPRAlgorithm:
    """Test PPR implementation."""

    def _create_simple_graph(self):
        """Create simple test graph."""
        return {
            "nodes": [
                {"id": "A"},
                {"id": "B"},
                {"id": "C"},
            ],
            "edges": [
                {"id": "e1", "source": "A", "target": "B", "type": "calls"},
                {"id": "e2", "source": "B", "target": "C", "type": "writes_state"},
            ]
        }

    def test_seeds_get_highest_scores(self):
        """Seed nodes should have highest PPR scores."""
        graph = self._create_simple_graph()
        ppr = VKGPPR(graph)
        scores = ppr.run(seeds=["A"])

        top = ppr.get_top_nodes(scores, k=3)
        # A should be first (it's the seed)
        assert top[0][0] == "A"
        assert top[0][1] > top[1][1]  # A higher than B

    def test_convergence(self):
        """Algorithm should converge."""
        graph = self._create_simple_graph()
        ppr = VKGPPR(graph)

        # Should not raise or infinite loop
        scores = ppr.run(seeds=["A"], config=PPRConfig(max_iter=100))

        # All scores should be non-negative
        assert all(s >= 0 for s in scores.values())

    def test_weights_normalized(self):
        """Edge weights should be normalized per source."""
        graph = {
            "nodes": [{"id": "A"}, {"id": "B"}, {"id": "C"}],
            "edges": [
                {"id": "e1", "source": "A", "target": "B", "type": "calls"},
                {"id": "e2", "source": "A", "target": "C", "type": "calls"},
            ]
        }
        ppr = VKGPPR(graph)

        # Weights from A should sum to 1
        w1 = ppr.edge_weights.get("e1", 0)
        w2 = ppr.edge_weights.get("e2", 0)
        assert abs(w1 + w2 - 1.0) < 0.001

    def test_risk_score_increases_weight(self):
        """Risky edges should get higher weight."""
        graph = {
            "nodes": [{"id": "A"}, {"id": "B"}, {"id": "C"}],
            "edges": [
                {"id": "e1", "source": "A", "target": "B", "type": "calls", "risk_score": 0.0},
                {"id": "e2", "source": "A", "target": "C", "type": "calls", "risk_score": 0.8},
            ]
        }
        ppr = VKGPPR(graph)

        # Normalized, but e2 should be higher proportion
        w1 = ppr.edge_weights.get("e1", 0)
        w2 = ppr.edge_weights.get("e2", 0)
        assert w2 > w1

    def test_guard_penalty_decreases_weight(self):
        """Guarded edges should get lower weight."""
        graph = {
            "nodes": [{"id": "A"}, {"id": "B"}, {"id": "C"}],
            "edges": [
                {"id": "e1", "source": "A", "target": "B", "type": "calls", "guards_at_source": False},
                {"id": "e2", "source": "A", "target": "C", "type": "calls", "guards_at_source": True},
            ]
        }
        ppr = VKGPPR(graph)

        w1 = ppr.edge_weights.get("e1", 0)
        w2 = ppr.edge_weights.get("e2", 0)
        assert w1 > w2  # Guarded edge has lower weight

    def test_config_presets(self):
        """Config presets have appropriate values."""
        strict = PPRConfig.strict()
        standard = PPRConfig.standard()
        relaxed = PPRConfig.relaxed()

        # Strict has highest alpha (more teleport)
        assert strict.alpha > standard.alpha
        assert standard.alpha > relaxed.alpha


class TestPPRIntegration:
    """Integration tests with realistic graphs."""

    def test_reentrancy_pattern(self):
        """PPR on reentrancy-like graph identifies relevant nodes."""
        graph = {
            "nodes": [
                {"id": "withdraw"},
                {"id": "external_call"},
                {"id": "balance_update"},
                {"id": "unrelated_func"},
            ],
            "edges": [
                {"id": "e1", "source": "withdraw", "target": "external_call",
                 "type": "calls_external", "risk_score": 0.9},
                {"id": "e2", "source": "withdraw", "target": "balance_update",
                 "type": "writes_state"},
                {"id": "e3", "source": "unrelated_func", "target": "balance_update",
                 "type": "reads_state"},
            ]
        }

        scores = run_ppr(graph, seeds=["withdraw"])

        # withdraw should be highest (seed)
        # external_call and balance_update should be next (connected)
        # unrelated_func should be lowest
        assert scores["withdraw"] > scores["external_call"]
        assert scores["external_call"] > scores["unrelated_func"]
