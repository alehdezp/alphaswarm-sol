## Critical Context

Sending unnecessary code to LLM APIs creates risks:
1. **Prompt injection** - Unused code could contain malicious comments
2. **Data leakage** - Business logic exposed unnecessarily
3. **Attack surface** - More code = more opportunities for LLM mistakes
4. **Cost waste** - Paying for tokens that don't help analysis

### Step 1: Create Context Policy Module

**File:** `src/true_vkg/llm/context_policy.py`

```python
"""Context policy for data minimization.

This module enforces the principle of least privilege for LLM context.
The LLM should NEVER see more code than necessary for analysis.
"""
from dataclasses import dataclass
from enum import Enum
from typing import Set, List, Any
import hashlib
import json
from datetime import datetime


class ContextPolicyLevel(Enum):
    """Context policy levels.

    STRICT: Only code directly referenced by finding
    STANDARD: Finding + 1-hop dependencies (DEFAULT)
    RELAXED: Full contract context (requires explicit opt-in)
    """
    STRICT = "strict"
    STANDARD = "standard"
    RELAXED = "relaxed"


@dataclass
class ContextAuditEntry:
    """Audit log entry for LLM context submission."""
    timestamp: datetime
    finding_id: str
    policy_level: ContextPolicyLevel
    bytes_sent: int
    items_included: List[str]
    items_filtered: List[str]

    def to_dict(self) -> dict:
        return {
            "timestamp": self.timestamp.isoformat(),
            "finding_id": self.finding_id,
            "policy_level": self.policy_level.value,
            "bytes_sent": self.bytes_sent,
            "items_included": self.items_included,
            "items_filtered": self.items_filtered,
        }


class ContextPolicy:
    """Enforce minimum context principle for LLM calls.

    Usage:
        policy = ContextPolicy(level=ContextPolicyLevel.STANDARD)
        filtered = policy.filter_context(full_context, finding)
        # filtered contains only necessary items
    """

    def __init__(
        self,
        level: ContextPolicyLevel = ContextPolicyLevel.STANDARD,
        enable_audit: bool = True
    ):
        self.level = level
        self.enable_audit = enable_audit
        self.audit_log: List[ContextAuditEntry] = []

    def filter_context(
        self,
        context: "Context",
        finding: "Finding"
    ) -> "Context":
        """Filter context to minimum necessary for finding.

        Args:
            context: Full context object
            finding: The finding being analyzed

        Returns:
            Filtered context with only necessary items
        """
        if self.level == ContextPolicyLevel.STRICT:
            filtered = self._filter_strict(context, finding)
        elif self.level == ContextPolicyLevel.STANDARD:
            filtered = self._filter_standard(context, finding)
        else:
            filtered = context  # RELAXED = no filtering

        if self.enable_audit:
            self._log_context(context, filtered, finding)

        return filtered

    def _filter_strict(
        self,
        context: "Context",
        finding: "Finding"
    ) -> "Context":
        """Only include code directly referenced by finding.

        Includes:
        - The function containing the finding
        - State variables it reads/writes
        - External call targets (interface only)
        """
        relevant_ids = set()

        # The function with the finding
        relevant_ids.add(finding.function_id)

        # State variables it reads
        for var_id in finding.state_reads:
            relevant_ids.add(var_id)

        # State variables it writes
        for var_id in finding.state_writes:
            relevant_ids.add(var_id)

        # External call targets (just the signature)
        for call in finding.external_calls:
            relevant_ids.add(call.target_id)

        return context.filter_to_ids(relevant_ids)

    def _filter_standard(
        self,
        context: "Context",
        finding: "Finding"
    ) -> "Context":
        """Include finding + 1-hop dependencies.

        Includes everything from STRICT plus:
        - Callers of the vulnerable function (max 3)
        - Callees of the vulnerable function (max 3)
        - Related modifiers
        """
        # Start with strict
        relevant_ids = set()
        relevant_ids.add(finding.function_id)
        for var_id in finding.state_reads + finding.state_writes:
            relevant_ids.add(var_id)
        for call in finding.external_calls:
            relevant_ids.add(call.target_id)

        # Add 1-hop callers (max 3)
        func = context.get_function(finding.function_id)
        if func and hasattr(func, 'callers'):
            for caller_id in func.callers[:3]:
                relevant_ids.add(caller_id)

        # Add 1-hop callees (max 3)
        if func and hasattr(func, 'callees'):
            for callee_id in func.callees[:3]:
                relevant_ids.add(callee_id)

        # Add modifiers
        if func and hasattr(func, 'modifiers'):
            for mod_id in func.modifiers:
                relevant_ids.add(mod_id)

        return context.filter_to_ids(relevant_ids)

    def _log_context(
        self,
        full_context: "Context",
        filtered_context: "Context",
        finding: "Finding"
    ):
        """Log what was sent to LLM for audit."""
        full_ids = set(full_context.get_all_ids())
        filtered_ids = set(filtered_context.get_all_ids())
        excluded_ids = full_ids - filtered_ids

        entry = ContextAuditEntry(
            timestamp=datetime.now(),
            finding_id=finding.id,
            policy_level=self.level,
            bytes_sent=filtered_context.size_bytes(),
            items_included=list(filtered_ids),
            items_filtered=list(excluded_ids)
        )

        self.audit_log.append(entry)

    def get_audit_log(self) -> List[ContextAuditEntry]:
        """Get audit log of context submissions."""
        return self.audit_log

    def validate_no_secrets(self, context: "Context") -> List[str]:
        """Check context for potential secrets.

        Returns list of warnings if suspicious patterns found.
        """
        warnings = []

        content = context.to_string()

        # Check for common secret patterns
        patterns = [
            ("PRIVATE_KEY", "private key"),
            ("API_KEY", "API key"),
            ("SECRET", "secret"),
            ("PASSWORD", "password"),
            ("0x[a-fA-F0-9]{64}", "hex key pattern"),
        ]

        import re
        for pattern, description in patterns:
            if re.search(pattern, content, re.IGNORECASE):
                warnings.append(f"Potential {description} detected in context")

        return warnings


def require_explicit_relaxed(func):
    """Decorator to require explicit opt-in for RELAXED policy.

    Usage:
        @require_explicit_relaxed
        def send_to_llm(context, policy):
            ...
    """
    def wrapper(*args, **kwargs):
        policy = kwargs.get('policy') or (args[1] if len(args) > 1 else None)
        if isinstance(policy, ContextPolicy):
            if policy.level == ContextPolicyLevel.RELAXED:
                understand_risk = kwargs.get('i_understand_risk', False)
                if not understand_risk:
                    raise ValueError(
                        "RELAXED context policy requires explicit "
                        "'i_understand_risk=True' parameter. "
                        "This sends full contract context to LLM."
                    )
        return func(*args, **kwargs)
    return wrapper
```

### Step 2: Create Context Data Structure

**File:** `src/true_vkg/llm/context.py`

```python
"""Context data structure for LLM consumption."""
from dataclasses import dataclass, field
from typing import Dict, Set, List, Any, Optional


@dataclass
class ContextItem:
    """A single item in the context (function, variable, etc)."""
    id: str
    type: str  # "function", "state_variable", "modifier", etc
    name: str
    content: str  # Source code or representation
    metadata: Dict[str, Any] = field(default_factory=dict)

    def size_bytes(self) -> int:
        return len(self.content.encode('utf-8'))


@dataclass
class Context:
    """Collection of context items for LLM analysis."""
    items: Dict[str, ContextItem] = field(default_factory=dict)

    def add(self, item: ContextItem):
        """Add item to context."""
        self.items[item.id] = item

    def get_all_ids(self) -> List[str]:
        """Get all item IDs."""
        return list(self.items.keys())

    def get_function(self, func_id: str) -> Optional[ContextItem]:
        """Get function by ID."""
        return self.items.get(func_id)

    def filter_to_ids(self, ids: Set[str]) -> "Context":
        """Create new context with only specified IDs."""
        new_context = Context()
        for id in ids:
            if id in self.items:
                new_context.add(self.items[id])
        return new_context

    def size_bytes(self) -> int:
        """Total size in bytes."""
        return sum(item.size_bytes() for item in self.items.values())

    def to_string(self) -> str:
        """Serialize to string for LLM."""
        parts = []
        for item in self.items.values():
            parts.append(f"// {item.type}: {item.name}\n{item.content}")
        return "\n\n".join(parts)


@dataclass
class Finding:
    """Stub finding for context policy."""
    id: str
    function_id: str
    state_reads: List[str] = field(default_factory=list)
    state_writes: List[str] = field(default_factory=list)
    external_calls: List[Any] = field(default_factory=list)
```

```python
"""Tests for context policy security."""
import pytest
from true_vkg.llm.context_policy import (
    ContextPolicy,
    ContextPolicyLevel,
    require_explicit_relaxed
)
from true_vkg.llm.context import Context, ContextItem, Finding


class TestContextPolicy:
    """Test context policy enforcement."""

    def _create_test_context(self) -> Context:
        """Create test context with multiple items."""
        ctx = Context()
        ctx.add(ContextItem(id="func1", type="function", name="withdraw", content="function withdraw() {...}"))
        ctx.add(ContextItem(id="func2", type="function", name="deposit", content="function deposit() {...}"))
        ctx.add(ContextItem(id="func3", type="function", name="helper", content="function helper() {...}"))
        ctx.add(ContextItem(id="var1", type="state_variable", name="balances", content="mapping(address => uint) balances"))
        return ctx

    def _create_test_finding(self) -> Finding:
        """Create test finding."""
        return Finding(
            id="VKG-001",
            function_id="func1",
            state_reads=["var1"],
            state_writes=["var1"]
        )

    def test_strict_filters_to_finding_only(self):
        """STRICT policy only includes finding-related code."""
        ctx = self._create_test_context()
        finding = self._create_test_finding()

        policy = ContextPolicy(level=ContextPolicyLevel.STRICT)
        filtered = policy.filter_context(ctx, finding)

        # Should only include func1 and var1
        assert "func1" in filtered.get_all_ids()
        assert "var1" in filtered.get_all_ids()
        assert "func2" not in filtered.get_all_ids()
        assert "func3" not in filtered.get_all_ids()

    def test_standard_is_default(self):
        """STANDARD is the default policy level."""
        policy = ContextPolicy()
        assert policy.level == ContextPolicyLevel.STANDARD

    def test_relaxed_requires_explicit_opt_in(self):
        """RELAXED policy requires explicit flag."""
        @require_explicit_relaxed
        def send_context(context, policy, i_understand_risk=False):
            return True

        ctx = self._create_test_context()
        policy = ContextPolicy(level=ContextPolicyLevel.RELAXED)

        # Should raise without flag
        with pytest.raises(ValueError, match="i_understand_risk"):
            send_context(ctx, policy)

        # Should work with flag
        result = send_context(ctx, policy, i_understand_risk=True)
        assert result is True

    def test_audit_log_created(self):
        """Audit log captures context submissions."""
        ctx = self._create_test_context()
        finding = self._create_test_finding()

        policy = ContextPolicy(level=ContextPolicyLevel.STRICT, enable_audit=True)
        policy.filter_context(ctx, finding)

        audit = policy.get_audit_log()
        assert len(audit) == 1
        assert audit[0].finding_id == "VKG-001"
        assert audit[0].policy_level == ContextPolicyLevel.STRICT
        assert len(audit[0].items_filtered) > 0  # Some items were filtered

    def test_secret_detection_warns(self):
        """Secret detection warns about potential leaks."""
        ctx = Context()
        ctx.add(ContextItem(
            id="func1",
            type="function",
            name="test",
            content="string PRIVATE_KEY = 'abc123'"
        ))

        policy = ContextPolicy()
        warnings = policy.validate_no_secrets(ctx)

        assert len(warnings) > 0
        assert "private key" in warnings[0].lower()
```
