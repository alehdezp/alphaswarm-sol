```python
"""Key abbreviations for compact output.

These abbreviations reduce token count while remaining readable.
"""

# Full key -> abbreviated key
KEY_ABBREVIATIONS = {
    "contract": "c",
    "function": "fn",
    "severity": "sev",
    "confidence": "conf",
    "line_number": "ln",
    "pattern_id": "pid",
    "finding_id": "fid",
    "description": "desc",
    "recommendation": "rec",
    "evidence": "ev",
    "external_call": "xcall",
    "state_variable": "svar",
    "modifier": "mod",
    "visibility": "vis",
    "properties": "props",
}

# Abbreviated -> full key (reverse mapping)
KEY_EXPANSIONS = {v: k for k, v in KEY_ABBREVIATIONS.items()}

# Common value abbreviations
VALUE_ABBREVIATIONS = {
    "critical": "CRIT",
    "high": "HIGH",
    "medium": "MED",
    "low": "LOW",
    "informational": "INFO",
    "public": "pub",
    "external": "ext",
    "internal": "int",
    "private": "priv",
}


def abbreviate_keys(data: dict) -> dict:
    """Recursively abbreviate keys in a dictionary."""
    if isinstance(data, dict):
        return {
            KEY_ABBREVIATIONS.get(k, k): abbreviate_keys(v)
            for k, v in data.items()
        }
    elif isinstance(data, list):
        return [abbreviate_keys(item) for item in data]
    elif isinstance(data, str):
        return VALUE_ABBREVIATIONS.get(data, data)
    else:
        return data


def expand_keys(data: dict) -> dict:
    """Recursively expand abbreviated keys."""
    if isinstance(data, dict):
        return {
            KEY_EXPANSIONS.get(k, k): expand_keys(v)
            for k, v in data.items()
        }
    elif isinstance(data, list):
        return [expand_keys(item) for item in data]
    else:
        return data
```

### Step 2: Create Compact Encoder

**File:** `src/true_vkg/output/compact.py`

```python
"""Compact output format for LLM consumption.

Provides 30-40% token reduction vs JSON while remaining readable.
"""
import yaml
import json
from typing import Any, Dict, Optional
from enum import Enum
from .abbreviations import abbreviate_keys, expand_keys


class DetailLevel(Enum):
    """Output detail levels."""
    SUMMARY = "summary"    # Minimal: id, severity, location
    DETAILED = "detailed"  # Standard: + description, evidence
    FULL = "full"          # Everything


class CompactEncoder:
    """Encode BSKG data in compact YAML format.

    Usage:
        encoder = CompactEncoder(detail=DetailLevel.DETAILED)
        compact = encoder.encode(finding_dict)
        # compact is a YAML string

    Token Reduction:
        - Key abbreviations: ~15% reduction
        - YAML vs JSON: ~10% reduction
        - Detail filtering: ~10-20% reduction
        - Total: ~30-40% reduction
    """

    def __init__(self, detail: DetailLevel = DetailLevel.DETAILED):
        self.detail = detail

    def encode(self, data: Dict[str, Any]) -> str:
        """Encode data as compact YAML.

        Args:
            data: Dictionary to encode

        Returns:
            Compact YAML string
        """
        # Filter by detail level
        filtered = self._filter_detail(data)

        # Abbreviate keys
        abbreviated = abbreviate_keys(filtered)

        # Convert to YAML
        return yaml.dump(
            abbreviated,
            default_flow_style=False,
            allow_unicode=True,
            sort_keys=False,
            width=120  # Longer lines = fewer tokens
        )

    def _filter_detail(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Filter data based on detail level."""
        if self.detail == DetailLevel.FULL:
            return data

        if self.detail == DetailLevel.SUMMARY:
            return self._extract_summary(data)

        # DETAILED: remove verbose fields
        return self._extract_detailed(data)

    def _extract_summary(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Extract summary fields only."""
        summary_keys = {
            "finding_id", "pattern_id", "severity", "confidence",
            "contract", "function", "line_number"
        }

        if isinstance(data, dict):
            return {
                k: self._extract_summary(v)
                for k, v in data.items()
                if k in summary_keys or k == "findings"
            }
        elif isinstance(data, list):
            return [self._extract_summary(item) for item in data]
        else:
            return data

    def _extract_detailed(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Extract detailed fields (exclude verbose)."""
        verbose_keys = {
            "raw_source", "ast_dump", "slither_output",
            "full_trace", "debug_info"
        }

        if isinstance(data, dict):
            return {
                k: self._extract_detailed(v)
                for k, v in data.items()
                if k not in verbose_keys
            }
        elif isinstance(data, list):
            return [self._extract_detailed(item) for item in data]
        else:
            return data


class CompactDecoder:
    """Decode compact format back to standard dicts."""

    def decode(self, compact_str: str) -> Dict[str, Any]:
        """Decode compact YAML to dictionary.

        Args:
            compact_str: Compact YAML string

        Returns:
            Dictionary with expanded keys
        """
        data = yaml.safe_load(compact_str)
        return expand_keys(data)


def encode_finding(finding: Dict[str, Any], detail: DetailLevel = DetailLevel.DETAILED) -> str:
    """Convenience function to encode a finding."""
    return CompactEncoder(detail=detail).encode(finding)


def decode_finding(compact_str: str) -> Dict[str, Any]:
    """Convenience function to decode a finding."""
    return CompactDecoder().decode(compact_str)


def compare_token_counts(data: Dict[str, Any]) -> Dict[str, int]:
    """Compare token counts between formats.

    Returns dict with token counts for each format.
    Note: Uses approximate token counting (chars/4).
    """
    json_str = json.dumps(data, indent=2)
    compact_str = CompactEncoder(detail=DetailLevel.DETAILED).encode(data)

    return {
        "json_chars": len(json_str),
        "compact_chars": len(compact_str),
        "json_tokens_approx": len(json_str) // 4,
        "compact_tokens_approx": len(compact_str) // 4,
        "reduction_percent": round(
            (1 - len(compact_str) / len(json_str)) * 100, 1
        )
    }
```

```python
"""Tests for compact output format."""
import pytest
from true_vkg.output.compact import (
    CompactEncoder,
    CompactDecoder,
    DetailLevel,
    compare_token_counts,
)


class TestCompactOutput:
    """Test compact output encoding."""

    def _sample_finding(self):
        return {
            "finding_id": "VKG-001",
            "pattern_id": "vm-001",
            "severity": "critical",
            "confidence": 0.95,
            "contract": "Test",
            "function": "withdraw",
            "line_number": 42,
            "description": "Reentrancy vulnerability",
        }

    def test_encode_produces_yaml(self):
        """Encoder produces valid YAML."""
        import yaml

        finding = self._sample_finding()
        encoder = CompactEncoder()
        compact = encoder.encode(finding)

        # Should be valid YAML
        parsed = yaml.safe_load(compact)
        assert parsed is not None

    def test_keys_abbreviated(self):
        """Keys are abbreviated in output."""
        finding = self._sample_finding()
        encoder = CompactEncoder()
        compact = encoder.encode(finding)

        assert "fid:" in compact  # finding_id -> fid
        assert "sev:" in compact  # severity -> sev
        assert "finding_id:" not in compact

    def test_round_trip(self):
        """Encode then decode preserves data."""
        finding = self._sample_finding()
        encoder = CompactEncoder()
        decoder = CompactDecoder()

        compact = encoder.encode(finding)
        recovered = decoder.decode(compact)

        assert recovered["finding_id"] == finding["finding_id"]
        assert recovered["severity"] == "critical"

    def test_token_reduction(self):
        """Compact format uses fewer tokens."""
        finding = self._sample_finding()
        comparison = compare_token_counts(finding)

        assert comparison["compact_chars"] < comparison["json_chars"]
        assert comparison["reduction_percent"] >= 25  # At least 25% reduction

    def test_summary_detail_level(self):
        """Summary level excludes description."""
        finding = self._sample_finding()
        encoder = CompactEncoder(detail=DetailLevel.SUMMARY)
        compact = encoder.encode(finding)

        import yaml
        parsed = yaml.safe_load(compact)

        assert "fid" in parsed
        assert "desc" not in parsed  # description excluded


class TestDetailLevels:
    """Test detail level filtering."""

    def test_full_includes_everything(self):
        """Full level includes all fields."""
        data = {
            "finding_id": "VKG-001",
            "raw_source": "long source code...",
            "debug_info": "verbose debug...",
        }

        encoder = CompactEncoder(detail=DetailLevel.FULL)
        compact = encoder.encode(data)

        import yaml
        parsed = yaml.safe_load(compact)

        # All fields present
        assert "raw_source" in parsed or "fid" in parsed

    def test_detailed_excludes_verbose(self):
        """Detailed level excludes verbose fields."""
        data = {
            "finding_id": "VKG-001",
            "raw_source": "long source code...",
        }

        encoder = CompactEncoder(detail=DetailLevel.DETAILED)
        compact = encoder.encode(data)

        import yaml
        parsed = yaml.safe_load(compact)

        # Verbose field excluded
        assert "raw_source" not in parsed
```
