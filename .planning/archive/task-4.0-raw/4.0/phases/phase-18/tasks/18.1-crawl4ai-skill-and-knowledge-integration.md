### 1.4 Crawl Configuration

```python
@dataclass
class CrawlConfig:
    """Configuration for crawling a source."""
    source_id: str
    url: str
    category: str
    subcategory: Optional[str]

    # Crawl settings
    depth: int = 2                    # How deep to follow links
    same_domain_only: bool = True     # Stay on same domain
    content_selectors: List[str] = field(default_factory=list)  # CSS selectors
    exclude_patterns: List[str] = field(default_factory=list)   # URL patterns to skip

    # Rate limiting
    delay_ms: int = 1000              # Delay between requests
    max_pages: int = 100              # Max pages per source

    # Content extraction
    extract_code_blocks: bool = True  # Extract Solidity code
    extract_tables: bool = True       # Extract vulnerability tables
    llm_summarize: bool = False       # Use LLM to summarize content
```

```python
@dataclass

    mitigation_guide: str            # How to fix

    code_examples: List[CodeExample] # Vulnerable + fixed code

    # Metadata
    severity_range: Tuple[str, str]  # (min, max) severity
    prevalence: str                  # How common (rare/common/very_common)
    difficulty_to_exploit: str       # easy/medium/hard/expert

    # LLM optimization
    token_count: int                 # Estimated tokens for this doc
    retrieval_keywords: List[str]    # Keywords for similarity search
```

## Part 3: Pattern Linkage System

### 3.1 Linkage Types

```python
class PatternLinkageType(Enum):
    """How vulnerability knowledge links to BSKG detection patterns."""

    EXACT_MATCH = "exact_match"
    # Pattern directly detects this vulnerability
    # Example: reentrancy-001 pattern detects classic reentrancy
    # Confidence: HIGH
    # Action: Show pattern detection + knowledge docs

    PARTIAL_MATCH = "partial_match"
    # Pattern covers subset of vulnerability variants
    # Example: access-control-001 catches some but not all access issues
    # Confidence: MEDIUM
    # Action: Show pattern detection + warn about uncovered variants

    THEORETICAL = "theoretical"
    # No automated pattern exists, knowledge-only
    # Example: Complex business logic flaws
    # Confidence: N/A (no detection)
    # Action: Surface knowledge docs during manual review

    REQUIRES_LLM = "requires_llm"
    # Detection requires Tier-B LLM reasoning
    # Example: Intent deviation, semantic mismatches
    # Confidence: MEDIUM (depends on LLM)
    # Action: Trigger LLM analysis with knowledge context

    COMPOSITE = "composite"
    # Multiple patterns needed together
    # Example: Flash loan + oracle = attack
    # Confidence: HIGH when all patterns match
    # Action: Check for pattern combination
```

```python
    """Links vulnerability knowledge to detection patterns."""

    coverage_percentage: float       # What % of variants are covered
    false_positive_notes: str        # Known FP scenarios
    false_negative_notes: str        # Known FN scenarios

    # For THEORETICAL
    why_no_pattern: str              # Why automated detection is hard
    manual_review_hints: List[str]   # What to look for manually

    # For REQUIRES_LLM
    llm_prompt_template: str         # Prompt for LLM analysis
    required_context: List[str]      # What context LLM needs

    # For COMPOSITE
    pattern_combination: List[str]   # Patterns that must co-occur
    combination_logic: str           # AND/OR/SEQUENCE logic

    # Quality metrics
    detection_confidence: float      # 0-1 confidence in detection
    knowledge_completeness: float    # 0-1 how complete our docs are
    last_validated: datetime         # When linkage was verified
```
