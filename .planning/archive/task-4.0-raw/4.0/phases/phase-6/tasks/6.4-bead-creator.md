"""Bead creation from pattern matches."""
from dataclasses import dataclass
from typing import List, Optional, Dict, Any
import hashlib
from datetime import datetime

from ..kg.schema import KnowledgeGraph, FunctionNode, ContractNode
from ..queries.patterns import PatternMatch, PatternResult
from .schema import (
    VulnerabilityBead, PatternContext, InvestigationGuide, TestContext
)
from .types import (
    Finding, CodeSnippet, ExploitReference, Severity, BeadStatus
)
from .templates.loader import load_template
from .exploits.loader import find_exploits_by_pattern, find_exploits_by_class


@dataclass
class BeadCreationConfig:
    """Configuration for bead creation."""
    max_related_code: int = 5           # Max related functions to include
    include_full_contract: bool = False  # Include full contract source
    max_contract_lines: int = 500        # Skip full contract if larger
    context_lines_before: int = 5        # Lines before vulnerable code
    context_lines_after: int = 5         # Lines after vulnerable code


class BeadCreator:
    """Creates VulnerabilityBeads from pattern matches.

    This is the core integration between VKG's pattern engine and the
    beads system. It extracts all necessary context to create a
    self-contained investigation package.
    """

    def __init__(
        self,
        graph: KnowledgeGraph,
        config: Optional[BeadCreationConfig] = None
    ):
        self.graph = graph
        self.config = config or BeadCreationConfig()
        self._bead_counter = 0

    def create_bead(self, match: PatternMatch) -> VulnerabilityBead:
        """Create a bead from a pattern match.

        Args:
            match: Pattern match result from the pattern engine

        Returns:
            Complete VulnerabilityBead ready for investigation
        """
        # Generate unique ID
        bead_id = self._generate_bead_id(match)

        # Extract code context
        func_node = self._get_function_node(match)
        vulnerable_code = self._extract_code_snippet(func_node, match)
        related_code = self._extract_related_code(func_node)
        full_contract = self._get_full_contract(func_node)
        inheritance = self._get_inheritance_chain(func_node)

        # Build pattern context
        pattern_context = self._build_pattern_context(match)

        # Load investigation template
        vuln_class = self._determine_vulnerability_class(match)
        investigation_guide = load_template(vuln_class)
        if investigation_guide is None:
            investigation_guide = self._fallback_investigation_guide()

        # Build test context
        test_context = self._build_test_context(match, func_node)

        # Find similar exploits
        exploits = self._find_similar_exploits(match, vuln_class)

        # Create the bead
        return VulnerabilityBead(
            id=bead_id,
            vulnerability_class=vuln_class,
            pattern_id=match.pattern_id,
            severity=self._determine_severity(match),
            confidence=match.confidence,
            vulnerable_code=vulnerable_code,
            related_code=related_code,
            full_contract=full_contract,
            inheritance_chain=inheritance,
            pattern_context=pattern_context,
            investigation_guide=investigation_guide,
            test_context=test_context,
            similar_exploits=exploits,
            fix_recommendations=self._get_fix_recommendations(match),
            status=BeadStatus.PENDING,
        )

    def create_beads_from_results(
        self,
        results: List[PatternResult]
    ) -> List[VulnerabilityBead]:
        """Create beads from multiple pattern results.

        Args:
            results: List of pattern matching results

        Returns:
            List of VulnerabilityBeads, one per finding
        """
        beads = []
        for result in results:
            for match in result.matches:
                try:
                    bead = self.create_bead(match)
                    beads.append(bead)
                except Exception as e:
                    # Log error but continue with other beads
                    # TODO: Add proper logging
                    print(f"Error creating bead for {match}: {e}")
        return beads

    # === Private methods ===

    def _generate_bead_id(self, match: PatternMatch) -> str:
        """Generate unique bead ID."""
        self._bead_counter += 1
        # Include hash of match for uniqueness across runs
        match_hash = hashlib.md5(
            f"{match.pattern_id}:{match.function_id}:{match.line}".encode()
        ).hexdigest()[:6]
        return f"VKG-{self._bead_counter:04d}-{match_hash}"

    def _get_function_node(self, match: PatternMatch) -> FunctionNode:
        """Get the function node from the graph."""
        # Adapt this to actual graph API
        node = self.graph.get_node(match.function_id)
        if node is None:
            raise ValueError(f"Function not found: {match.function_id}")
        return node

    def _extract_code_snippet(
        self,
        func_node: FunctionNode,
        match: PatternMatch
    ) -> CodeSnippet:
        """Extract the vulnerable code snippet."""
        # Get source code - adapt to actual API
        source = self._get_source_code(func_node)

        return CodeSnippet(
            source=source,
            file_path=func_node.file_path,
            start_line=func_node.start_line,
            end_line=func_node.end_line,
            function_name=func_node.name,
            contract_name=func_node.contract_name,
        )

    def _extract_related_code(
        self,
        func_node: FunctionNode
    ) -> List[CodeSnippet]:
        """Extract related code (called functions, modifiers)."""
        related = []

        # Get called functions
        for call in self._get_function_calls(func_node):
            if len(related) >= self.config.max_related_code:
                break
            called_func = self.graph.get_node(call.target_id)
            if called_func:
                related.append(CodeSnippet(
                    source=self._get_source_code(called_func),
                    file_path=called_func.file_path,
                    start_line=called_func.start_line,
                    end_line=called_func.end_line,
                    function_name=called_func.name,
                    contract_name=called_func.contract_name,
                ))

        # Get modifiers
        for modifier in self._get_modifiers(func_node):
            if len(related) >= self.config.max_related_code:
                break
            related.append(CodeSnippet(
                source=self._get_source_code(modifier),
                file_path=modifier.file_path,
                start_line=modifier.start_line,
                end_line=modifier.end_line,
                function_name=modifier.name,
                contract_name=modifier.contract_name,
            ))

        return related

    def _get_full_contract(self, func_node: FunctionNode) -> Optional[str]:
        """Get full contract source if configured and small enough."""
        if not self.config.include_full_contract:
            return None

        contract = self.graph.get_contract(func_node.contract_name)
        if contract is None:
            return None

        # Check size
        line_count = contract.end_line - contract.start_line
        if line_count > self.config.max_contract_lines:
            return None

        return self._get_contract_source(contract)

    def _get_inheritance_chain(self, func_node: FunctionNode) -> List[str]:
        """Get list of parent contracts."""
        contract = self.graph.get_contract(func_node.contract_name)
        if contract is None:
            return []
        return contract.inheritance or []

    def _build_pattern_context(self, match: PatternMatch) -> PatternContext:
        """Build pattern context from match."""
        return PatternContext(
            pattern_name=match.pattern_name or match.pattern_id,
            pattern_description=match.description or "",
            why_flagged=self._build_why_flagged(match),
            matched_properties=match.matched_properties or [],
            evidence_lines=match.evidence_lines or [match.line],
        )

    def _build_why_flagged(self, match: PatternMatch) -> str:
        """Build human-readable explanation of why code was flagged."""
        parts = []
        if match.matched_properties:
            parts.append(f"Matched properties: {', '.join(match.matched_properties)}")
        if match.evidence:
            parts.append(f"Evidence: {match.evidence}")
        if match.line:
            parts.append(f"Primary evidence at line {match.line}")
        return " | ".join(parts) if parts else "Pattern matched"

    def _determine_vulnerability_class(self, match: PatternMatch) -> str:
        """Determine vulnerability class from pattern."""
        # Map pattern IDs to vulnerability classes
        pattern_id = match.pattern_id.lower()

        class_mappings = {
            "reentrancy": ["reentrancy", "state-write-after", "vm-001"],
            "access_control": ["access", "auth", "owner", "permission"],
            "oracle": ["oracle", "price", "staleness", "chainlink"],
            "dos": ["dos", "unbounded", "loop", "gas"],
            "mev": ["mev", "slippage", "deadline", "sandwich"],
            "token": ["token", "erc20", "transfer", "approval"],
            "upgrade": ["upgrade", "proxy", "initializer", "storage"],
        }

        for vuln_class, keywords in class_mappings.items():
            if any(kw in pattern_id for kw in keywords):
                return vuln_class

        # Default to generic
        return "unknown"

    def _determine_severity(self, match: PatternMatch) -> Severity:
        """Determine severity from match."""
        if hasattr(match, 'severity'):
            severity_map = {
                "critical": Severity.CRITICAL,
                "high": Severity.HIGH,
                "medium": Severity.MEDIUM,
                "low": Severity.LOW,
                "info": Severity.INFO,
            }
            return severity_map.get(match.severity.lower(), Severity.MEDIUM)

        # Infer from vulnerability class
        vuln_class = self._determine_vulnerability_class(match)
        default_severities = {
            "reentrancy": Severity.CRITICAL,
            "access_control": Severity.HIGH,
            "oracle": Severity.HIGH,
            "dos": Severity.MEDIUM,
            "mev": Severity.MEDIUM,
            "token": Severity.MEDIUM,
            "upgrade": Severity.HIGH,
        }
        return default_severities.get(vuln_class, Severity.MEDIUM)

    def _build_test_context(
        self,
        match: PatternMatch,
        func_node: FunctionNode
    ) -> TestContext:
        """Build test context for exploit verification."""
        vuln_class = self._determine_vulnerability_class(match)

        # Generate scaffold based on vulnerability class
        scaffold = self._generate_test_scaffold(vuln_class, func_node)
        scenario = self._generate_attack_scenario(vuln_class, match)

        return TestContext(
            scaffold_code=scaffold,
            attack_scenario=scenario,
            setup_requirements=self._get_setup_requirements(vuln_class),
            expected_outcome=self._get_expected_outcome(vuln_class),
        )

    def _find_similar_exploits(
        self,

        """Find similar real-world exploits."""
        # Try by pattern ID first
        exploits = find_exploits_by_pattern(match.pattern_id)
        if exploits:
            return exploits[:3]  # Max 3

        # Fall back to class
        exploits = find_exploits_by_class(vuln_class)
        return exploits[:3]
