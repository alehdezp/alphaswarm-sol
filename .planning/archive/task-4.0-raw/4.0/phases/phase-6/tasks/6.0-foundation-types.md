```python
"""Foundation types for the Beads system."""
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import List, Optional, Dict, Any
import json

class Severity(Enum):
    """Vulnerability severity levels."""
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"
    INFO = "info"

class BeadStatus(Enum):
    """Status of a vulnerability bead."""
    PENDING = "pending"
    INVESTIGATING = "investigating"
    CONFIRMED = "confirmed"
    REJECTED = "rejected"
    NEEDS_INFO = "needs_info"

class VerdictType(Enum):
    """Type of verdict on a finding."""
    TRUE_POSITIVE = "true_positive"
    FALSE_POSITIVE = "false_positive"
    INCONCLUSIVE = "inconclusive"

@dataclass
class CodeSnippet:
    """A snippet of code with metadata."""
    source: str                    # The actual code
    file_path: str                # Absolute path to file
    start_line: int               # 1-indexed
    end_line: int                 # 1-indexed
    function_name: Optional[str] = None
    contract_name: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "source": self.source,
            "file_path": self.file_path,
            "start_line": self.start_line,
            "end_line": self.end_line,
            "function_name": self.function_name,
            "contract_name": self.contract_name,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "CodeSnippet":
        return cls(**data)

@dataclass
class InvestigationStep:
    """A single step in an investigation guide."""
    step_number: int
    action: str                   # What to do
    look_for: str                 # What evidence to find
    evidence_needed: str          # What confirms this step
    red_flag: Optional[str] = None  # Pattern indicating vulnerability
    safe_if: Optional[str] = None   # Pattern indicating safe code

    def to_dict(self) -> Dict[str, Any]:
        return {
            "step_number": self.step_number,
            "action": self.action,
            "look_for": self.look_for,
            "evidence_needed": self.evidence_needed,
            "red_flag": self.red_flag,
            "safe_if": self.safe_if,
        }

@dataclass
class ExploitReference:
    """Reference to a real-world exploit."""
    id: str                       # e.g., "dao-hack-2016"
    name: str                     # "The DAO Hack"
    date: str                     # ISO format
    loss: str                     # "$60M"
    pattern_id: str               # Linked pattern
    vulnerable_code: str          # Relevant code snippet
    exploit_summary: str          # What happened
    fix: str                      # How to prevent
    source_url: str               # Reference link

    def to_dict(self) -> Dict[str, Any]:
        return {
            "id": self.id,
            "name": self.name,
            "date": self.date,
            "loss": self.loss,
            "pattern_id": self.pattern_id,
            "vulnerable_code": self.vulnerable_code,
            "exploit_summary": self.exploit_summary,
            "fix": self.fix,
            "source_url": self.source_url,
        }

@dataclass
class Verdict:
    """A verdict on a finding."""
    type: VerdictType
    reason: str
    confidence: float             # 0.0-1.0
    evidence: List[str]           # Supporting evidence
    timestamp: datetime = field(default_factory=datetime.now)
    auditor_id: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "type": self.type.value,
            "reason": self.reason,
            "confidence": self.confidence,
            "evidence": self.evidence,
            "timestamp": self.timestamp.isoformat(),
            "auditor_id": self.auditor_id,
        }

@dataclass
class Finding:
    """A potential vulnerability finding from pattern matching.

    This is the INPUT to the Beads system - what the pattern engine produces.
    """
    id: str                       # Unique identifier, e.g., "VKG-001"
    pattern_id: str               # Which pattern matched, e.g., "vm-001"
    function_id: str              # Function node ID in the graph
    contract_name: str
    function_name: str
    file_path: str
    line_number: int
    severity: Severity
    confidence: float             # 0.0-1.0 from pattern
    vulnerability_class: str      # e.g., "reentrancy", "access-control"
    description: str              # Human-readable description
    evidence: List[str] = field(default_factory=list)

    def to_dict(self) -> Dict[str, Any]:
        return {
            "id": self.id,
            "pattern_id": self.pattern_id,
            "function_id": self.function_id,
            "contract_name": self.contract_name,
            "function_name": self.function_name,
            "file_path": self.file_path,
            "line_number": self.line_number,
            "severity": self.severity.value,
            "confidence": self.confidence,
            "vulnerability_class": self.vulnerability_class,
            "description": self.description,
            "evidence": self.evidence,
        }
```

```python
"""Tests for beads foundation types."""
import pytest
import json
from datetime import datetime
from true_vkg.beads import (
    Severity, BeadStatus, VerdictType,
    CodeSnippet, InvestigationStep, ExploitReference,
    Verdict, Finding
)

class TestCodeSnippet:
    def test_to_dict_round_trip(self):
        snippet = CodeSnippet(
            source="function withdraw() {...}",
            file_path="/path/to/file.sol",
            start_line=10,
            end_line=20,
            function_name="withdraw",
            contract_name="Vault"
        )
        data = snippet.to_dict()
        restored = CodeSnippet.from_dict(data)
        assert restored.source == snippet.source
        assert restored.start_line == snippet.start_line

    def test_json_serializable(self):
        snippet = CodeSnippet(
            source="code",
            file_path="/path",
            start_line=1,
            end_line=5
        )
        # Should not raise
        json.dumps(snippet.to_dict())

class TestFinding:
    def test_severity_enum(self):
        finding = Finding(
            id="VKG-001",
            pattern_id="vm-001",
            function_id="func_123",
            contract_name="Vault",
            function_name="withdraw",
            file_path="/path/to/Vault.sol",
            line_number=45,
            severity=Severity.CRITICAL,
            confidence=0.95,
            vulnerability_class="reentrancy",
            description="State update after external call"
        )
        assert finding.severity == Severity.CRITICAL
        data = finding.to_dict()
        assert data["severity"] == "critical"

class TestVerdict:
    def test_timestamp_serialization(self):
        verdict = Verdict(
            type=VerdictType.TRUE_POSITIVE,
            reason="Confirmed via PoC",
            confidence=0.99,
            evidence=["test_exploit.sol executed successfully"]
        )
        data = verdict.to_dict()
        assert "timestamp" in data
        # Should be ISO format
        datetime.fromisoformat(data["timestamp"])
```
