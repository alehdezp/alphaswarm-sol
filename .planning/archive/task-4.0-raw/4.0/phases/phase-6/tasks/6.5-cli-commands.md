DEFAULT_BEAD_DIR = Path(".vrs/beads")


def get_storage(path: Optional[Path] = None) -> "BeadStorage":
    """Get bead storage instance."""
    return BeadStorage(path or DEFAULT_BEAD_DIR)


@click.group()
def beads():

@beads.command("list")
@click.option("--status", type=click.Choice(["pending", "investigating", "confirmed", "rejected"]),
              help="Filter by status")
@click.option("--class", "vuln_class", help="Filter by vulnerability class")
@click.option("--severity", type=click.Choice(["critical", "high", "medium", "low"]),
              help="Filter by severity")
@click.option("--format", "output_format", type=click.Choice(["table", "json", "compact"]),
              default="table", help="Output format")
def list_beads(status: Optional[str], vuln_class: Optional[str],
               severity: Optional[str], output_format: str):
    """List all beads, optionally filtered.

    Examples:
        vkg beads list
        vkg beads list --status pending
        vkg beads list --class reentrancy --severity critical
        vkg beads list --format json
    """
    storage = get_storage()
    beads = storage.list_beads()

    # Apply filters
    if status:
        beads = [b for b in beads if b.status.value == status]
    if vuln_class:
        beads = [b for b in beads if b.vulnerability_class == vuln_class]
    if severity:
        beads = [b for b in beads if b.severity.value == severity]

    if not beads:
        click.echo("No beads found matching criteria.")
        return

    if output_format == "json":
        click.echo(json.dumps([b.to_dict() for b in beads], indent=2))
    elif output_format == "compact":
        for bead in beads:
            click.echo(f"{bead.id} | {bead.vulnerability_class} | {bead.severity.value} | {bead.status.value}")
    else:  # table
        click.echo(f"{'ID':<15} {'Class':<15} {'Severity':<10} {'Status':<12} {'Confidence':<10}")
        click.echo("-" * 70)
        for bead in beads:
            click.echo(f"{bead.id:<15} {bead.vulnerability_class:<15} "
                      f"{bead.severity.value:<10} {bead.status.value:<12} {bead.confidence:.0%}")

    click.echo(f"\nTotal: {len(beads)} beads")


@beads.command("show")
@click.argument("bead_id")
@click.option("--format", "output_format", type=click.Choice(["full", "json", "llm"]),

def show_bead(bead_id: str, output_format: str):
    """Show details of a specific bead.

    Examples:
        vkg beads show VKG-0001
        vkg beads show VKG-0001 --format json
        vkg beads show VKG-0001 --format llm  # LLM investigation prompt

    storage = get_storage()
    bead = storage.get_bead(bead_id)

    if output_format == "json":
        click.echo(bead.to_json())
    elif output_format == "llm":
        click.echo(bead.get_llm_prompt())
    else:  # full
        _print_bead_full(bead)


def _print_bead_full(bead: VulnerabilityBead):
    """Print full bead details in human-readable format."""
    click.echo(f"\n{'='*70}")
    click.echo(f"BEAD: {bead.id}")
    click.echo(f"{'='*70}")

    click.echo(f"\n## SUMMARY")
    click.echo(f"  Vulnerability Class: {bead.vulnerability_class}")
    click.echo(f"  Pattern: {bead.pattern_id}")
    click.echo(f"  Severity: {bead.severity.value.upper()}")
    click.echo(f"  Confidence: {bead.confidence:.0%}")
    click.echo(f"  Status: {bead.status.value}")

    click.echo(f"\n## VULNERABLE CODE")
    click.echo(f"  File: {bead.vulnerable_code.file_path}")
    click.echo(f"  Function: {bead.vulnerable_code.function_name}")
    click.echo(f"  Lines: {bead.vulnerable_code.start_line}-{bead.vulnerable_code.end_line}")
    click.echo(f"\n```solidity")
    click.echo(bead.vulnerable_code.source)
    click.echo(f"```")

    click.echo(f"\n## WHY FLAGGED")
    click.echo(f"  {bead.pattern_context.why_flagged}")

    if bead.pattern_context.matched_properties:
        click.echo(f"\n  Matched Properties:")
        for prop in bead.pattern_context.matched_properties:
            click.echo(f"    - {prop}")

    click.echo(f"\n## INVESTIGATION STEPS")
    for step in bead.investigation_guide.steps:
        click.echo(f"\n  Step {step.step_number}: {step.action}")
        click.echo(f"    Look for: {step.look_for}")

        if step.safe_if:
            click.echo(f"    Safe if: {step.safe_if}")

    click.echo(f"\n## QUESTIONS TO ANSWER")
    for q in bead.investigation_guide.questions_to_answer:
        click.echo(f"  - {q}")

    if bead.similar_exploits:

            click.echo(f"  - {exploit.name} ({exploit.date}): {exploit.loss}")

@beads.command("next")
@click.option("--severity", type=click.Choice(["critical", "high", "medium", "low"]),
              help="Minimum severity to show")
def next_bead(severity: Optional[str]):
    """Get the next priority bead for investigation.

    Priority order:
    1. Critical severity, pending status
    2. High severity, pending status
    3. Medium severity, pending status
    4. Low severity, pending status

    Example:
        vkg beads next
        vkg beads next --severity high
    """
    storage = get_storage()
    beads = storage.list_beads()

    # Filter to pending only
    pending = [b for b in beads if b.status == BeadStatus.PENDING]

    if not pending:
        click.echo("No pending beads to investigate.")
        return

    # Sort by severity
    severity_order = {"critical": 0, "high": 1, "medium": 2, "low": 3, "info": 4}
    pending.sort(key=lambda b: severity_order.get(b.severity.value, 5))

    # Filter by minimum severity if specified
    if severity:
        min_order = severity_order[severity]
        pending = [b for b in pending if severity_order.get(b.severity.value, 5) <= min_order]

    if not pending:
        click.echo(f"No pending beads at {severity} severity or higher.")
        return

    # Get next (highest priority pending)
    next_bead = pending[0]

    click.echo(f"\n## NEXT BEAD TO INVESTIGATE")
    click.echo(f"Total pending: {len(pending)}")
    click.echo()

    # Show full bead
    _print_bead_full(next_bead)

    # Mark as investigating
    next_bead.status = BeadStatus.INVESTIGATING
    storage.save_bead(next_bead)
    click.echo(f"Status updated to 'investigating'")


@beads.command("verdict")
@click.argument("bead_id")
@click.option("--confirmed", "verdict_type", flag_value="confirmed",
              help="Mark as confirmed true positive")
@click.option("--rejected", "verdict_type", flag_value="rejected",
              help="Mark as rejected false positive")
@click.option("--inconclusive", "verdict_type", flag_value="inconclusive",
              help="Mark as inconclusive")
@click.option("--reason", required=True, help="Reason for verdict")
@click.option("--confidence", type=float, default=0.8,
              help="Confidence in verdict (0-1)")
def set_verdict(bead_id: str, verdict_type: str, reason: str, confidence: float):
    """Set verdict on a bead.

    Examples:
        vkg beads verdict VKG-0001 --confirmed --reason "Exploit confirmed via PoC"
        vkg beads verdict VKG-0001 --rejected --reason "Has nonReentrant modifier"
        vkg beads verdict VKG-0001 --inconclusive --reason "Need more context"
    """
    if verdict_type is None:
        click.echo("Error: Must specify --confirmed, --rejected, or --inconclusive", err=True)
        sys.exit(1)

    storage = get_storage()
    bead = storage.get_bead(bead_id)

    if bead is None:
        click.echo(f"Error: Bead {bead_id} not found", err=True)
        sys.exit(1)

    # Map to verdict type
    type_map = {
        "confirmed": VerdictType.TRUE_POSITIVE,
        "rejected": VerdictType.FALSE_POSITIVE,
        "inconclusive": VerdictType.INCONCLUSIVE,
    }

    verdict = Verdict(
        type=type_map[verdict_type],
        reason=reason,
        confidence=confidence,
        evidence=[reason],  # Could be expanded
    )

    bead.set_verdict(verdict)
    storage.save_bead(bead)

    click.echo(f"Verdict set for {bead_id}:")
    click.echo(f"  Type: {verdict_type}")
    click.echo(f"  Reason: {reason}")
    click.echo(f"  Status: {bead.status.value}")

@beads.command("context")
@click.option("--format", "output_format", type=click.Choice(["summary", "full", "json"]),
              default="summary", help="Output format")
def get_context(output_format: str):
    """Get session context - summary of all beads.

    Useful for LLM context window - shows what has been investigated.

    Example:
        vkg beads context
        vkg beads context --format json
    """
    storage = get_storage()
    beads = storage.list_beads()

    if not beads:
        click.echo("No beads in session.")
        return

    # Group by status
    by_status = {}
    for bead in beads:
        status = bead.status.value
        if status not in by_status:
            by_status[status] = []
        by_status[status].append(bead)

    if output_format == "json":
        click.echo(json.dumps({
            "total": len(beads),
            "by_status": {k: len(v) for k, v in by_status.items()},
            "beads": [b.to_dict() for b in beads]
        }, indent=2))
        return

    click.echo(f"\n## SESSION CONTEXT")
    click.echo(f"Total Beads: {len(beads)}")
    click.echo()

    for status, status_beads in by_status.items():
        click.echo(f"### {status.upper()} ({len(status_beads)})")
        for bead in status_beads:
            if output_format == "full":
                click.echo(f"  {bead.id}: {bead.vulnerability_class} in {bead.vulnerable_code.function_name}")
                if bead.verdict:
                    click.echo(f"    Verdict: {bead.verdict.type.value} - {bead.verdict.reason}")
            else:  # summary
                click.echo(f"  - {bead.id} ({bead.severity.value})")
        click.echo()


@beads.command("generate")
@click.argument("project_path", type=click.Path(exists=True))
@click.option("--pattern", help="Only generate for specific pattern ID")
def generate_beads(project_path: str, pattern: Optional[str]):
    """Generate beads from pattern analysis.

    Example:
        vkg beads generate ./contracts
        vkg beads generate ./contracts --pattern reentrancy-basic
    """
    from ..kg.builder import build_graph
    from ..queries.patterns import PatternMatcher

    click.echo(f"Building knowledge graph for {project_path}...")

    try:
        graph = build_graph(project_path)
        click.echo(f"Graph built: {len(graph.functions)} functions")
    except Exception as e:
        click.echo(f"Error building graph: {e}", err=True)
        sys.exit(1)

    click.echo("Running pattern analysis...")

    try:
        matcher = PatternMatcher(graph)
        if pattern:
            results = matcher.match_pattern(pattern)
        else:
            results = matcher.match_all()
        click.echo(f"Found {sum(len(r.matches) for r in results)} matches")
    except Exception as e:
        click.echo(f"Error in pattern analysis: {e}", err=True)
        sys.exit(1)

    click.echo("Creating beads...")

    try:
        beads = create_beads(results, graph)
        click.echo(f"Created {len(beads)} beads")
    except Exception as e:
        click.echo(f"Error creating beads: {e}", err=True)
        sys.exit(1)

    # Save beads
    storage = get_storage()
    for bead in beads:
        storage.save_bead(bead)

    click.echo(f"\nSaved {len(beads)} beads to {storage.path}")
    click.echo(f"\nRun 'vkg beads list' to see all beads")
    click.echo(f"Run 'vkg beads next' to start investigating")

```python
"""Persistent storage for beads."""
import json
from pathlib import Path
from typing import List, Optional

from .schema import VulnerabilityBead


class BeadStorage:
    """File-based storage for vulnerability beads."""

    def __init__(self, path: Path):
        self.path = Path(path)
        self.path.mkdir(parents=True, exist_ok=True)

    def save_bead(self, bead: VulnerabilityBead) -> None:
        """Save a bead to storage."""
        bead_path = self.path / f"{bead.id}.json"
        with open(bead_path, "w") as f:
            json.dump(bead.to_dict(), f, indent=2)

    def get_bead(self, bead_id: str) -> Optional[VulnerabilityBead]:
        """Load a bead by ID."""
        bead_path = self.path / f"{bead_id}.json"
        if not bead_path.exists():
            return None

        with open(bead_path, "r") as f:
            data = json.load(f)

        return VulnerabilityBead.from_dict(data)

    def list_beads(self) -> List[VulnerabilityBead]:
        """List all beads in storage."""
        beads = []
        for path in self.path.glob("*.json"):
            with open(path, "r") as f:
                data = json.load(f)
            beads.append(VulnerabilityBead.from_dict(data))
        return beads

    def delete_bead(self, bead_id: str) -> bool:
        """Delete a bead."""
        bead_path = self.path / f"{bead_id}.json"
        if bead_path.exists():
            bead_path.unlink()
            return True
        return False

    def clear(self) -> int:
        """Clear all beads. Returns count deleted."""
        count = 0
        for path in self.path.glob("*.json"):
            path.unlink()
            count += 1
        return count
```
