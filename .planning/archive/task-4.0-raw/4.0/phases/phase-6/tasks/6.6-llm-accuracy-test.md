```python
"""LLM verdict accuracy testing for beads."""
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Tuple
from enum import Enum
import json
from datetime import datetime
from pathlib import Path

from ..beads.schema import VulnerabilityBead
from ..beads.types import VerdictType
from ..llm import LLMClient


class ExpectedVerdict(Enum):
    """Expected verdict for a test case."""
    TRUE_POSITIVE = "true_positive"
    FALSE_POSITIVE = "false_positive"


@dataclass
class TestCase:
    """A test case with bead and expected verdict."""
    bead: VulnerabilityBead
    expected: ExpectedVerdict
    rationale: str  # Why this is the expected verdict
    source: str     # Where ground truth came from


@dataclass
class LLMVerdict:
    """An LLM's verdict on a bead."""
    verdict: VerdictType
    confidence: float
    reasoning: str
    raw_response: str


@dataclass
class TestResult:
    """Result of testing one bead."""
    test_case: TestCase
    llm_verdicts: List[LLMVerdict]
    final_verdict: VerdictType  # After voting
    is_correct: bool
    notes: str = ""


@dataclass
class AccuracyReport:
    """Full accuracy test report."""
    timestamp: datetime
    model: str
    total_cases: int
    correct_cases: int
    accuracy: float
    results: List[TestResult]
    config: Dict

    def to_dict(self) -> Dict:
        return {
            "timestamp": self.timestamp.isoformat(),
            "model": self.model,
            "total_cases": self.total_cases,
            "correct_cases": self.correct_cases,
            "accuracy": self.accuracy,
            "results": [
                {
                    "bead_id": r.test_case.bead.id,
                    "expected": r.test_case.expected.value,
                    "predicted": r.final_verdict.value,
                    "is_correct": r.is_correct,
                    "verdicts": [
                        {"verdict": v.verdict.value, "confidence": v.confidence}
                        for v in r.llm_verdicts
                    ]
                }
                for r in self.results
            ],
            "config": self.config,
        }

    def summary(self) -> str:
        """Generate summary report."""
        lines = [
            f"# LLM Verdict Accuracy Report",
            f"",
            f"**Model:** {self.model}",
            f"**Date:** {self.timestamp.isoformat()}",
            f"",
            f"## Results",
            f"- Total Cases: {self.total_cases}",
            f"- Correct: {self.correct_cases}",
            f"- **Accuracy: {self.accuracy:.1%}**",
            f"",
            f"## Breakdown",
        ]

        # Breakdown by expected verdict
        tp_cases = [r for r in self.results if r.test_case.expected == ExpectedVerdict.TRUE_POSITIVE]
        fp_cases = [r for r in self.results if r.test_case.expected == ExpectedVerdict.FALSE_POSITIVE]

        tp_correct = sum(1 for r in tp_cases if r.is_correct)
        fp_correct = sum(1 for r in fp_cases if r.is_correct)

        lines.extend([
            f"- True Positives: {tp_correct}/{len(tp_cases)} correct ({tp_correct/len(tp_cases):.1%})" if tp_cases else "- True Positives: N/A",
            f"- False Positives: {fp_correct}/{len(fp_cases)} correct ({fp_correct/len(fp_cases):.1%})" if fp_cases else "- False Positives: N/A",
            f"",
            f"## Failures",
        ])

        failures = [r for r in self.results if not r.is_correct]
        for failure in failures:
            lines.append(f"- {failure.test_case.bead.id}: Expected {failure.test_case.expected.value}, Got {failure.final_verdict.value}")

        return "\n".join(lines)


class BeadAccuracyTester:
    """Test LLM verdict accuracy on beads."""

    VERDICT_PROMPT_SUFFIX = """

Based on the above information, provide your verdict in the following JSON format:
{
    "verdict": "true_positive" or "false_positive",
    "confidence": 0.0 to 1.0,
    "reasoning": "Your detailed reasoning"
}

Respond ONLY with the JSON, no additional text.
"""

    def __init__(
        self,
        llm_client: LLMClient,
        model: str = "claude-3-opus",
        runs_per_bead: int = 3,
        temperature: float = 0.0,
    ):
        self.client = llm_client
        self.model = model
        self.runs_per_bead = runs_per_bead
        self.temperature = temperature

    def run_accuracy_test(
        self,
        test_cases: List[TestCase]
    ) -> AccuracyReport:
        """Run accuracy test on a set of test cases.

        Args:
            test_cases: List of beads with expected verdicts

        Returns:
            AccuracyReport with results
        """
        results = []

        for i, case in enumerate(test_cases):
            print(f"Testing {i+1}/{len(test_cases)}: {case.bead.id}")

            # Get multiple verdicts
            verdicts = []
            for run in range(self.runs_per_bead):
                verdict = self._get_llm_verdict(case.bead)
                verdicts.append(verdict)

            # Determine final verdict (majority vote)
            final_verdict = self._vote(verdicts)

            # Check correctness
            expected_verdict_type = VerdictType.TRUE_POSITIVE if case.expected == ExpectedVerdict.TRUE_POSITIVE else VerdictType.FALSE_POSITIVE
            is_correct = final_verdict == expected_verdict_type

            results.append(TestResult(
                test_case=case,
                llm_verdicts=verdicts,
                final_verdict=final_verdict,
                is_correct=is_correct,
            ))

        # Calculate accuracy
        correct = sum(1 for r in results if r.is_correct)
        accuracy = correct / len(results) if results else 0

        return AccuracyReport(
            timestamp=datetime.now(),
            model=self.model,
            total_cases=len(results),
            correct_cases=correct,
            accuracy=accuracy,
            results=results,
            config={
                "runs_per_bead": self.runs_per_bead,
                "temperature": self.temperature,
            }
        )

    def _get_llm_verdict(self, bead: VulnerabilityBead) -> LLMVerdict:
        """Get LLM verdict for a single bead."""
        # Build prompt
        prompt = bead.get_llm_prompt() + self.VERDICT_PROMPT_SUFFIX

        # Call LLM
        response = self.client.complete(
            prompt=prompt,
            model=self.model,
            temperature=self.temperature,
            max_tokens=1000,
        )

        # Parse response
        try:
            data = json.loads(response)
            return LLMVerdict(
                verdict=VerdictType(data["verdict"]),
                confidence=data["confidence"],
                reasoning=data["reasoning"],
                raw_response=response,
            )
        except (json.JSONDecodeError, KeyError, ValueError) as e:
            # Failed to parse - treat as inconclusive
            return LLMVerdict(
                verdict=VerdictType.INCONCLUSIVE,
                confidence=0.0,
                reasoning=f"Parse error: {e}",
                raw_response=response,
            )

    def _vote(self, verdicts: List[LLMVerdict]) -> VerdictType:
        """Majority vote from multiple verdicts."""
        votes = {}
        for v in verdicts:
            if v.verdict not in votes:
                votes[v.verdict] = 0
            votes[v.verdict] += 1

        # Return most common (excluding INCONCLUSIVE if possible)
        valid_votes = {k: v for k, v in votes.items() if k != VerdictType.INCONCLUSIVE}
        if valid_votes:
            return max(valid_votes, key=valid_votes.get)

        # All inconclusive
        return VerdictType.INCONCLUSIVE


def load_test_cases(path: Path) -> List[TestCase]:
    """Load test cases from JSON file.

    Expected format:
    [
        {
            "bead_path": "path/to/bead.json",
            "expected": "true_positive" or "false_positive",
            "rationale": "Why this is expected",
            "source": "audit report / manual review / etc"
        }
    ]
    """
    with open(path, "r") as f:
        data = json.load(f)

    cases = []
    for item in data:
        bead_path = Path(item["bead_path"])
        with open(bead_path, "r") as f:
            bead_data = json.load(f)
        bead = VulnerabilityBead.from_dict(bead_data)

        cases.append(TestCase(
            bead=bead,
            expected=ExpectedVerdict(item["expected"]),
            rationale=item["rationale"],
            source=item["source"],
        ))

    return cases
```

```python
def analyze_failures(report: AccuracyReport):
    """Analyze what went wrong for failed cases."""
    failures = [r for r in report.results if not r.is_correct]

    print(f"\n## Failure Analysis ({len(failures)} failures)\n")

    for failure in failures:
        print(f"### {failure.test_case.bead.id}")
        print(f"Expected: {failure.test_case.expected.value}")
        print(f"Got: {failure.final_verdict.value}")
        print(f"Rationale: {failure.test_case.rationale}")
        print(f"\nLLM Reasoning:")
        for i, v in enumerate(failure.llm_verdicts):
            print(f"  Run {i+1}: {v.verdict.value} ({v.confidence:.0%})")
            print(f"    {v.reasoning[:200]}...")
        print()

    # Categorize failures
    failure_types = {}
    for f in failures:
        if f.test_case.expected == ExpectedVerdict.TRUE_POSITIVE:
            failure_types.setdefault("missed_tp", []).append(f)
        else:
            failure_types.setdefault("false_fp", []).append(f)

    print(f"\nFailure Types:")
    print(f"  Missed TPs: {len(failure_types.get('missed_tp', []))}")
    print(f"  False FPs: {len(failure_types.get('false_fp', []))}")
```

```python
from pathlib import Path

pytestmark = pytest.mark.llm


@pytest.fixture
def test_cases():
    """Load ground truth test cases."""
    return load_test_cases(Path("tests/fixtures/bead_ground_truth.json"))

def tester():
    """Create accuracy tester."""
    client = LLMClient(...)
    return BeadAccuracyTester(client, runs_per_bead=1)  # 1 run for speed


class TestLLMAccuracy:
    def test_minimum_accuracy(self, tester, test_cases):
        """LLM must achieve minimum accuracy."""
        report = tester.run_accuracy_test(test_cases)
        assert report.accuracy >= 0.70, f"Accuracy {report.accuracy:.0%} below 70% minimum"

    def test_target_accuracy(self, tester, test_cases):
        """LLM should achieve target accuracy."""
        report = tester.run_accuracy_test(test_cases)
        if report.accuracy < 0.80:
            pytest.xfail(f"Accuracy {report.accuracy:.0%} below 80% target")

    def test_balanced_performance(self, tester, test_cases):
        """Performance should be balanced across TP/FP."""
        report = tester.run_accuracy_test(test_cases)

        tp_results = [r for r in report.results
                      if r.test_case.expected.value == "true_positive"]
        fp_results = [r for r in report.results
                      if r.test_case.expected.value == "false_positive"]

        tp_accuracy = sum(1 for r in tp_results if r.is_correct) / len(tp_results)
        fp_accuracy = sum(1 for r in fp_results if r.is_correct) / len(fp_results)

        # Neither should be dramatically worse
        assert tp_accuracy >= 0.60, f"TP accuracy {tp_accuracy:.0%} too low"
        assert fp_accuracy >= 0.60, f"FP accuracy {fp_accuracy:.0%} too low"
```
