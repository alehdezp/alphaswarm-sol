## Objective

Create an orchestrator mode that runs multiple tools (VKG, Slither, Aderyn) and deduplicates findings intelligently.

## CRITICAL CONTEXT

### What This Is NOT

This is NOT a full pipeline orchestrator. It's a simple "run tools, merge results" feature.

### Why This Matters

Users want to run multiple tools and get a unified view. Currently they must:
1. Run each tool separately
2. Manually compare outputs
3. Deduplicate by hand

Orchestrator provides:
1. Single command to run all tools
2. Automated deduplication
3. Flagged disagreements for review

### Realistic Scope

MVP orchestrator:
- Runs BSKG (always)
- Runs Slither (if installed)
- Runs Aderyn (if installed)
- Merges by file + line location
- Outputs combined report

NOT in scope:
- Parallel execution (nice to have, not MVP)
- Custom tool plugins
- Advanced deduplication algorithms

# src/true_vkg/orchestration/runner.py

from __future__ import annotations

import json
import subprocess
import shutil
from dataclasses import dataclass
from pathlib import Path
from typing import Dict, List, Optional
from enum import Enum

class ToolStatus(str, Enum):
    SUCCESS = "success"
    NOT_INSTALLED = "not_installed"
    ERROR = "error"
    SKIPPED = "skipped"

@dataclass
class ToolResult:
    tool: str
    status: ToolStatus
    findings: List[dict]
    error: Optional[str] = None
    execution_time: float = 0.0

class ToolRunner:
    """Run analysis tools on a project."""

    SUPPORTED_TOOLS = ["vkg", "slither", "aderyn"]

    def __init__(self, project_path: Path):
        self.project_path = Path(project_path).resolve()

    def check_tool_installed(self, tool: str) -> bool:
        """Check if a tool is installed."""
        if tool == "vkg":
            return True  # Always available
        elif tool == "slither":
            return shutil.which("slither") is not None
        elif tool == "aderyn":
            return shutil.which("aderyn") is not None
        return False

    def run_tool(self, tool: str) -> ToolResult:
        """Run a single tool and return results."""
        import time
        start = time.time()

        if not self.check_tool_installed(tool):
            return ToolResult(
                tool=tool,
                status=ToolStatus.NOT_INSTALLED,
                findings=[],
                error=f"{tool} not found in PATH",
            )

        try:
            if tool == "vkg":
                findings = self._run_vkg()
            elif tool == "slither":
                findings = self._run_slither()
            elif tool == "aderyn":
                findings = self._run_aderyn()
            else:
                return ToolResult(
                    tool=tool,
                    status=ToolStatus.ERROR,
                    findings=[],
                    error=f"Unknown tool: {tool}",
                )

            return ToolResult(
                tool=tool,
                status=ToolStatus.SUCCESS,
                findings=findings,
                execution_time=time.time() - start,
            )

        except Exception as e:
            return ToolResult(
                tool=tool,
                status=ToolStatus.ERROR,
                findings=[],
                error=str(e),
                execution_time=time.time() - start,
            )

    def _run_vkg(self) -> List[dict]:
        """Run BSKG analysis."""
        from true_vkg.kg.builder import VKGBuilder
        from true_vkg.queries.executor import QueryExecutor
        from true_vkg.queries.patterns import PatternEngine

        builder = VKGBuilder(self.project_path)
        graph = builder.build(self.project_path)

        # Run pattern matching
        engine = PatternEngine()
        findings = engine.run_all_patterns(graph)

        return [self._normalize_vkg_finding(f) for f in findings]

    def _run_slither(self) -> List[dict]:
        """Run Slither analysis."""
        result = subprocess.run(
            ["slither", ".", "--json", "-"],
            cwd=self.project_path,
            capture_output=True,
            text=True,
            timeout=300,  # 5 minute timeout
        )

        if result.returncode != 0 and not result.stdout:
            raise RuntimeError(f"Slither failed: {result.stderr}")

        output = json.loads(result.stdout)
        detectors = output.get("results", {}).get("detectors", [])

        return [self._normalize_slither_finding(d) for d in detectors]

    def _run_aderyn(self) -> List[dict]:
        """Run Aderyn analysis."""
        result = subprocess.run(
            ["aderyn", ".", "--output", "json"],
            cwd=self.project_path,
            capture_output=True,
            text=True,
            timeout=300,
        )

        if result.returncode != 0:
            raise RuntimeError(f"Aderyn failed: {result.stderr}")

        output = json.loads(result.stdout)
        issues = output.get("issues", [])

        return [self._normalize_aderyn_finding(i) for i in issues]

    def _normalize_vkg_finding(self, f: dict) -> dict:
        """Normalize BSKG finding to common format."""
        return {
            "source": "vkg",
            "id": f.get("id", "unknown"),
            "title": f.get("title", ""),
            "severity": f.get("severity", "unknown"),
            "category": f.get("category", "unknown"),
            "file": f.get("location", {}).get("file", ""),
            "line": f.get("location", {}).get("line", 0),
            "function": f.get("location", {}).get("function"),
            "confidence": f.get("confidence", 0.5),
            "raw": f,
        }

    def _normalize_slither_finding(self, d: dict) -> dict:
        """Normalize Slither finding to common format."""
        elements = d.get("elements", [])
        first = elements[0] if elements else {}
        source_mapping = first.get("source_mapping", {})

        return {
            "source": "slither",
            "id": d.get("check", "unknown"),
            "title": d.get("description", "")[:200],
            "severity": d.get("impact", "Low").lower(),
            "category": d.get("check", "unknown"),
            "file": source_mapping.get("filename", ""),
            "line": source_mapping.get("lines", [0])[0] if source_mapping.get("lines") else 0,
            "function": first.get("name"),
            "confidence": {"High": 0.9, "Medium": 0.7, "Low": 0.5}.get(
                d.get("confidence", "Low"), 0.5
            ),
            "raw": d,
        }

    def _normalize_aderyn_finding(self, i: dict) -> dict:
        """Normalize Aderyn finding to common format."""
        return {
            "source": "aderyn",
            "id": i.get("id", "unknown"),
            "title": i.get("title", ""),
            "severity": i.get("severity", "low").lower(),
            "category": i.get("detector", "unknown"),
            "file": i.get("instances", [{}])[0].get("file", ""),
            "line": i.get("instances", [{}])[0].get("line", 0),
            "function": i.get("instances", [{}])[0].get("function"),
            "confidence": 0.7,  # Aderyn doesn't provide confidence
            "raw": i,
        }

### Deduplication

```python
# src/true_vkg/orchestration/dedup.py

from __future__ import annotations

from dataclasses import dataclass
from typing import Dict, List, Set
from pathlib import Path

@dataclass
class DeduplicatedFinding:
    """A finding that may be reported by multiple tools."""
    file: str
    line: int
    function: Optional[str]
    category: str
    severity: str
    sources: List[str]  # ["vkg", "slither", "aderyn"]
    findings: List[dict]  # Original findings from each source
    agreement: bool  # True if all sources agree on category/severity

def deduplicate(
    all_findings: List[dict],
    line_tolerance: int = 5,
) -> List[DeduplicatedFinding]:
    """Deduplicate findings from multiple tools.

    Findings are considered duplicates if:
    1. Same file
    2. Lines within tolerance
    3. Similar category (using category mapping)

    Args:
        all_findings: Normalized findings from all tools
        line_tolerance: Lines within this range are considered same location

    Returns:
        List of deduplicated findings
    """
    # Group by file
    by_file: Dict[str, List[dict]] = {}
    for f in all_findings:
        file_key = Path(f["file"]).name  # Use filename only for matching
        if file_key not in by_file:
            by_file[file_key] = []
        by_file[file_key].append(f)

    results: List[DeduplicatedFinding] = []

    for file_key, file_findings in by_file.items():
        # Sort by line number
        file_findings.sort(key=lambda x: x["line"])

        # Cluster by proximity
        clusters: List[List[dict]] = []
        current_cluster: List[dict] = []

        for f in file_findings:
            if not current_cluster:
                current_cluster.append(f)
            elif f["line"] - current_cluster[-1]["line"] <= line_tolerance:
                current_cluster.append(f)
            else:
                if current_cluster:
                    clusters.append(current_cluster)
                current_cluster = [f]

        if current_cluster:
            clusters.append(current_cluster)

        # Convert clusters to deduplicated findings
        for cluster in clusters:
            sources = list(set(f["source"] for f in cluster))
            severities = set(f["severity"] for f in cluster)
            categories = set(f["category"] for f in cluster)

            # Determine overall severity (most severe)
            severity_order = ["critical", "high", "medium", "low", "info"]
            overall_severity = "info"
            for sev in severity_order:
                if sev in severities:
                    overall_severity = sev
                    break

            # Check agreement
            agreement = len(severities) == 1 and len(categories) <= 2

            results.append(DeduplicatedFinding(
                file=cluster[0]["file"],
                line=cluster[0]["line"],
                function=cluster[0].get("function"),
                category=list(categories)[0] if len(categories) == 1 else ", ".join(categories),
                severity=overall_severity,
                sources=sources,
                findings=cluster,
                agreement=agreement,
            ))

    return results
```

# src/true_vkg/orchestration/output.py

from __future__ import annotations

import json
from dataclasses import dataclass
from typing import List
from datetime import datetime

@dataclass
class OrchestratorReport:
    """Combined report from orchestrator run."""
    project: str
    timestamp: str
    tools_run: List[str]
    tools_skipped: List[str]
    total_findings: int
    deduplicated_findings: int
    findings_by_tool: dict
    disagreements: int
    findings: List[dict]

def generate_report(
    project_path: str,
    tool_results: List[ToolResult],
    deduped: List[DeduplicatedFinding],
) -> OrchestratorReport:
    """Generate orchestrator report."""
    tools_run = [r.tool for r in tool_results if r.status == ToolStatus.SUCCESS]
    tools_skipped = [r.tool for r in tool_results if r.status != ToolStatus.SUCCESS]

    findings_by_tool = {}
    for r in tool_results:
        findings_by_tool[r.tool] = {
            "status": r.status.value,
            "count": len(r.findings),
            "time": r.execution_time,
            "error": r.error,
        }

    disagreements = sum(1 for d in deduped if not d.agreement)

    return OrchestratorReport(
        project=project_path,
        timestamp=datetime.now().isoformat(),
        tools_run=tools_run,
        tools_skipped=tools_skipped,
        total_findings=sum(len(r.findings) for r in tool_results if r.status == ToolStatus.SUCCESS),
        deduplicated_findings=len(deduped),
        findings_by_tool=findings_by_tool,
        disagreements=disagreements,
        findings=[_serialize_deduped(d) for d in deduped],
    )

def _serialize_deduped(d: DeduplicatedFinding) -> dict:
    return {
        "file": d.file,
        "line": d.line,
        "function": d.function,
        "category": d.category,
        "severity": d.severity,
        "sources": d.sources,
        "agreement": d.agreement,
        "source_count": len(d.sources),
    }

def test_deduplicate_same_location():
    """Findings at same location are deduplicated."""
    findings = [
        {"source": "vkg", "file": "Vault.sol", "line": 45, "category": "reentrancy", "severity": "high"},
        {"source": "slither", "file": "Vault.sol", "line": 47, "category": "reentrancy", "severity": "high"},
    ]
    deduped = deduplicate(findings, line_tolerance=5)
    assert len(deduped) == 1
    assert set(deduped[0].sources) == {"vkg", "slither"}

def test_deduplicate_different_files():
    """Findings in different files are not deduplicated."""
    findings = [
        {"source": "vkg", "file": "Vault.sol", "line": 45, "category": "reentrancy", "severity": "high"},
        {"source": "slither", "file": "Token.sol", "line": 45, "category": "reentrancy", "severity": "high"},
    ]
    deduped = deduplicate(findings, line_tolerance=5)
    assert len(deduped) == 2

def test_disagreement_flagged():
    """Disagreements on severity/category are flagged."""
    findings = [
        {"source": "vkg", "file": "Vault.sol", "line": 45, "category": "reentrancy", "severity": "critical"},
        {"source": "slither", "file": "Vault.sol", "line": 45, "category": "reentrancy", "severity": "low"},
    ]
    deduped = deduplicate(findings, line_tolerance=5)
    assert len(deduped) == 1
    assert not deduped[0].agreement  # Disagreement on severity
