## Objective

Run Slither and Aderyn on the same 3 projects and compare findings to identify VKG's unique value proposition.

## CRITICAL CONTEXT

### Why This Matters

VKG must provide value BEYOND existing tools. If Slither catches everything BSKG catches, BSKG has no reason to exist.

Expected differentiation:
- VKG: Behavioral signatures, semantic operations, LLM-friendly output
- Slither: Comprehensive static analysis, symbolic execution
- Aderyn: Rust-based, fast, newer detector set

### Realistic Expectations

Slither will likely:
- Find MORE issues (it's more mature)
- Have MORE false positives
- Not provide semantic context

VKG should:
- Find some things Slither misses (behavioral patterns)
- Have FEWER false positives (more precise patterns)
- Provide better context for LLM consumption

### Step 2: Normalize Findings

```python
# scripts/normalize_findings.py

from dataclasses import dataclass
from typing import List, Optional
from enum import Enum

class Tool(str, Enum):
    BSKG = "vkg"
    SLITHER = "slither"
    ADERYN = "aderyn"

@dataclass
class NormalizedFinding:
    tool: Tool
    id: str
    title: str
    severity: str
    category: str
    file: str
    line: int
    function: Optional[str]
    confidence: float

def normalize_vkg(findings: list) -> List[NormalizedFinding]:
    """Normalize BSKG findings."""
    return [
        NormalizedFinding(
            tool=Tool.VKG,
            id=f["id"],
            title=f["title"],
            severity=f["severity"],
            category=f["category"],
            file=f["location"]["file"],
            line=f["location"]["line"],
            function=f["location"].get("function"),
            confidence=f["confidence"],
        )
        for f in findings
    ]

def normalize_slither(output: dict) -> List[NormalizedFinding]:
    """Normalize Slither JSON output."""
    findings = []
    for detector in output.get("results", {}).get("detectors", []):
        # Slither structure varies by detector
        elements = detector.get("elements", [])
        first_elem = elements[0] if elements else {}

        findings.append(NormalizedFinding(
            tool=Tool.SLITHER,
            id=detector.get("check", "unknown"),
            title=detector.get("description", "")[:100],
            severity=_map_slither_severity(detector.get("impact", "Low")),
            category=detector.get("check", "unknown"),
            file=first_elem.get("source_mapping", {}).get("filename", ""),
            line=first_elem.get("source_mapping", {}).get("start", 0),
            function=first_elem.get("name", None),
            confidence=_map_slither_confidence(detector.get("confidence", "Low")),
        ))
    return findings

def normalize_aderyn(output: dict) -> List[NormalizedFinding]:
    """Normalize Aderyn JSON output."""
    findings = []
    for issue in output.get("issues", []):
        findings.append(NormalizedFinding(
            tool=Tool.ADERYN,
            id=issue.get("id", "unknown"),
            title=issue.get("title", ""),
            severity=issue.get("severity", "low").lower(),
            category=issue.get("category", "unknown"),
            file=issue.get("location", {}).get("file", ""),
            line=issue.get("location", {}).get("line", 0),
            function=issue.get("location", {}).get("function"),
            confidence=issue.get("confidence", 0.5),
        ))
    return findings

def _map_slither_severity(impact: str) -> str:
    return {
        "High": "high",
        "Medium": "medium",
        "Low": "low",
        "Informational": "info",
    }.get(impact, "low")

def _map_slither_confidence(conf: str) -> float:
    return {
        "High": 0.9,
        "Medium": 0.7,
        "Low": 0.5,
    }.get(conf, 0.5)
```

### Step 3: Deduplicate Across Tools

```python
def deduplicate_findings(
    all_findings: List[NormalizedFinding]
) -> dict:
    """Group findings by location to find overlaps.

    Returns:
        {
            "vkg_only": [...],      # Only BSKG found
            "slither_only": [...],  # Only Slither found
            "aderyn_only": [...],   # Only Aderyn found
            "vkg_slither": [...],   # BSKG + Slither agree
            "vkg_aderyn": [...],    # BSKG + Aderyn agree
            "all_three": [...],     # All tools agree
            "slither_aderyn": [...] # Slither + Aderyn, not VKG
        }
    """
    # Group by (file, line_bucket)
    # Use line buckets because line numbers may differ slightly
    def location_key(f: NormalizedFinding) -> str:
        line_bucket = f.line // 10 * 10  # Group by 10-line blocks
        return f"{f.file}:{line_bucket}"

    groups = {}
    for finding in all_findings:
        key = location_key(finding)
        if key not in groups:
            groups[key] = []
        groups[key].append(finding)

    result = {
        "vkg_only": [],
        "slither_only": [],
        "aderyn_only": [],
        "vkg_slither": [],
        "vkg_aderyn": [],
        "all_three": [],
        "slither_aderyn": [],
    }

    for key, findings in groups.items():
        tools = {f.tool for f in findings}

        if tools == {Tool.VKG}:
            result["vkg_only"].extend(findings)
        elif tools == {Tool.SLITHER}:
            result["slither_only"].extend(findings)
        elif tools == {Tool.ADERYN}:
            result["aderyn_only"].extend(findings)
        elif tools == {Tool.VKG, Tool.SLITHER}:
            result["vkg_slither"].extend(findings)
        elif tools == {Tool.VKG, Tool.ADERYN}:
            result["vkg_aderyn"].extend(findings)
        elif tools == {Tool.SLITHER, Tool.ADERYN}:
            result["slither_aderyn"].extend(findings)
        else:  # All three
            result["all_three"].extend(findings)

    return result
```

def calculate_comparison(dedup: dict) -> dict:
    """Calculate comparison statistics."""
    vkg_total = (
        len(dedup["vkg_only"]) +
        len(dedup["vkg_slither"]) +
        len(dedup["vkg_aderyn"]) +
        len(dedup["all_three"])
    )
    slither_total = (
        len(dedup["slither_only"]) +
        len(dedup["vkg_slither"]) +
        len(dedup["slither_aderyn"]) +
        len(dedup["all_three"])
    )
    aderyn_total = (
        len(dedup["aderyn_only"]) +
        len(dedup["vkg_aderyn"]) +
        len(dedup["slither_aderyn"]) +
        len(dedup["all_three"])
    )

    return {
        "total_findings": {
            "vkg": vkg_total,
            "slither": slither_total,
            "aderyn": aderyn_total,
        },
        "unique_findings": {
            "vkg_only": len(dedup["vkg_only"]),
            "slither_only": len(dedup["slither_only"]),
            "aderyn_only": len(dedup["aderyn_only"]),
        },
        "agreement": {
            "all_three": len(dedup["all_three"]),
            "vkg_slither": len(dedup["vkg_slither"]),
            "vkg_aderyn": len(dedup["vkg_aderyn"]),
            "slither_aderyn": len(dedup["slither_aderyn"]),
        },
        "vkg_unique_value": len(dedup["vkg_only"]),
        "vkg_missed_by_others": len(dedup["slither_aderyn"]),
    }
