## Tier Definitions

### TIER 0: Core (Must Always Work)

These MUST work for BSKG to function at all. Failure = BSKG cannot run.

| Component | Purpose | Failure = Fatal? |
|-----------|---------|------------------|
| Python runtime | Execute BSKG | Yes |
| Slither | Parse Solidity, build AST | Yes |
| VKGBuilder | Build knowledge graph | Yes |
| Pattern Engine | Match patterns | Yes |
| Local file system | Read/write .vkg | Yes |

**Commands guaranteed at Tier 0:**
- `vkg build-kg` (requires Slither)
- `vkg query` (pattern matching only)
- `vkg lens-report` (pattern matching only)
- `vkg doctor` (diagnostics)

### TIER 1: Enhancement (Adds Value, Not Required)

These enhance analysis but BSKG continues without them.

| Component | Purpose | Failure Impact |
|-----------|---------|----------------|
| Aderyn | Additional static analysis | Fewer findings |
| Medusa | Fuzz testing | No fuzz coverage |
| Foundry | Test scaffold execution | Cannot run tests |
| LLM providers | Tier B analysis | No false positive filtering |

**Commands at Tier 1:**
- `vkg analyze --tier-b` (needs LLM)
- `vkg scaffold --run` (needs Foundry)

### TIER 2: Nice-to-Have (Optional Features)

These are optional features that enhance UX.

| Component | Purpose | Failure Impact |
|-----------|---------|----------------|
| MCP integrations | External tool calls | No MCP features |
| Real-time monitoring | Live contract watching | No streaming |
| Learning system | Pattern improvement | No auto-learning |
| Telemetry | Usage metrics | No analytics |

"""
Dependency Tier System

Defines VKG's dependency hierarchy for graceful degradation.
"""

from enum import IntEnum, auto
from dataclasses import dataclass
from typing import Set, Optional, Callable
import subprocess
import shutil


class Tier(IntEnum):
    """Dependency tiers. Lower = more critical."""
    CORE = 0       # Must work or BSKG fails
    ENHANCEMENT = 1  # Adds value, not required
    OPTIONAL = 2    # Nice-to-have


@dataclass
class Dependency:
    """A BSKG dependency with tier classification."""
    name: str
    tier: Tier
    description: str
    check_cmd: Optional[str] = None  # Command to verify availability
    install_hint: Optional[str] = None

    def is_available(self) -> bool:
        """Check if dependency is available."""
        if self.check_cmd is None:
            return True

        try:
            result = subprocess.run(
                self.check_cmd.split(),
                capture_output=True,
                timeout=5
            )
            return result.returncode == 0
        except Exception:
            return False

    def get_version(self) -> Optional[str]:
        """Get dependency version if available."""
        if self.check_cmd is None:
            return None
        try:
            result = subprocess.run(
                self.check_cmd.split(),
                capture_output=True,
                timeout=5,
                text=True
            )
            return result.stdout.strip().split('\n')[0]
        except Exception:
            return None


# Dependency registry
DEPENDENCIES = {
    # TIER 0: Core
    "python": Dependency(
        name="python",
        tier=Tier.CORE,
        description="Python runtime",
        check_cmd="python --version",
    ),
    "slither": Dependency(
        name="slither",
        tier=Tier.CORE,
        description="Solidity static analyzer",
        check_cmd="slither --version",
        install_hint="pip install slither-analyzer",
    ),

    # TIER 1: Enhancement
    "aderyn": Dependency(
        name="aderyn",
        tier=Tier.ENHANCEMENT,
        description="Rust-based Solidity analyzer",
        check_cmd="aderyn --version",
        install_hint="cargo install aderyn",
    ),
    "foundry": Dependency(
        name="foundry",
        tier=Tier.ENHANCEMENT,
        description="Ethereum development toolkit",
        check_cmd="forge --version",
        install_hint="curl -L https://foundry.paradigm.xyz | bash",
    ),
    "medusa": Dependency(
        name="medusa",
        tier=Tier.ENHANCEMENT,
        description="Fuzzing tool",
        check_cmd="medusa --version",
        install_hint="See https://github.com/crytic/medusa",
    ),

    # TIER 1: LLM Providers (check via env var)
    "llm_anthropic": Dependency(
        name="llm_anthropic",
        tier=Tier.ENHANCEMENT,
        description="Anthropic Claude API",
    ),
    "llm_openai": Dependency(
        name="llm_openai",
        tier=Tier.ENHANCEMENT,
        description="OpenAI API",
    ),

    # TIER 2: Optional
    "mcp": Dependency(
        name="mcp",
        tier=Tier.OPTIONAL,
        description="Model Context Protocol",
    ),
}


def get_available_tiers() -> Set[Tier]:
    """Get set of tiers that are fully available."""
    available = set()

    for tier in Tier:
        tier_deps = [d for d in DEPENDENCIES.values() if d.tier == tier]
        if all(d.is_available() for d in tier_deps):
            available.add(tier)

    return available


def get_degradation_message(missing_tier: Tier) -> str:
    """Get user message for degraded operation."""
    messages = {
        Tier.CORE: "CRITICAL: Core dependencies missing. BSKG cannot run.",
        Tier.ENHANCEMENT: "Running with core features only. Some analysis unavailable.",
        Tier.OPTIONAL: "Running without optional features.",
    }
    return messages.get(missing_tier, "Unknown degradation level")

```python
"""
Availability Checker

Runtime checks for BSKG component availability.
"""

from dataclasses import dataclass
from typing import List, Dict, Any
import os

from .tiers import DEPENDENCIES, Tier, Dependency


@dataclass
class AvailabilityReport:
    """Report of component availability."""
    tier: Tier
    available: List[str]
    unavailable: List[str]
    degraded: bool
    message: str

    def to_dict(self) -> Dict[str, Any]:
        return {
            "tier": self.tier.name,
            "available": self.available,
            "unavailable": self.unavailable,
            "degraded": self.degraded,
            "message": self.message,
        }


class AvailabilityChecker:
    """Checks BSKG component availability."""

    def __init__(self):
        self._cache: Dict[str, bool] = {}

    def check_all(self) -> List[AvailabilityReport]:
        """Check all dependencies by tier."""
        reports = []

        for tier in Tier:
            tier_deps = [d for d in DEPENDENCIES.values() if d.tier == tier]
            available = []
            unavailable = []

            for dep in tier_deps:
                if self._check_dependency(dep):
                    available.append(dep.name)
                else:
                    unavailable.append(dep.name)

            degraded = len(unavailable) > 0
            if tier == Tier.CORE and degraded:
                message = "FATAL: Core dependencies missing"
            elif degraded:
                message = f"Degraded: Missing {', '.join(unavailable)}"
            else:
                message = "All available"

            reports.append(AvailabilityReport(
                tier=tier,
                available=available,
                unavailable=unavailable,
                degraded=degraded,
                message=message,
            ))

        return reports

    def _check_dependency(self, dep: Dependency) -> bool:
        """Check single dependency with caching."""
        if dep.name in self._cache:
            return self._cache[dep.name]

        # Special handling for LLM providers (check env vars)
        if dep.name.startswith("llm_"):
            provider = dep.name.replace("llm_", "").upper()
            env_var = f"{provider}_API_KEY"
            result = bool(os.environ.get(env_var))
        else:
            result = dep.is_available()

        self._cache[dep.name] = result
        return result

    def get_effective_tier(self) -> Tier:
        """Get the effective operation tier based on availability."""
        reports = self.check_all()

        # If core is degraded, we can't run
        core_report = next(r for r in reports if r.tier == Tier.CORE)
        if core_report.degraded:
            raise RuntimeError(core_report.message)

        # Return highest available tier
        for report in sorted(reports, key=lambda r: r.tier.value, reverse=True):
            if not report.degraded:
                return report.tier

        return Tier.CORE

    def clear_cache(self):
        """Clear availability cache (useful for testing)."""
        self._cache.clear()
```
