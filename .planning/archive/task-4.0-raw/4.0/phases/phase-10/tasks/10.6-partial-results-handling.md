## Objective

When a tool partially fails, clearly indicate which results are complete and which are missing. Never hide failures.

---

## Scenario

```
User runs: vkg analyze

Expected: Slither + Aderyn + BSKG patterns

Reality:
- Slither: OK (15 findings)
- Aderyn: TIMEOUT after 60s (0 findings)
- VKG: OK (23 findings)

Output should show:
- Total: 38 findings
- Note: Aderyn timed out, results incomplete
- Suggestion: Retry with longer timeout
```

## Design

### PartialResult Data Structure

```python
@dataclass
class PartialResult:
    """Result that may be incomplete."""
    source: str           # e.g., "slither", "aderyn"
    complete: bool        # True if fully completed
    findings: List[Any]   # Whatever was returned
    error: Optional[str]  # Error if incomplete
    retry_command: Optional[str]  # How to retry

@dataclass
class AggregatedResults:
    """Combined results from multiple sources."""
    total_findings: int
    complete: bool            # True if all sources complete
    sources: List[PartialResult]

    @property
    def incomplete_sources(self) -> List[str]:
        return [s.source for s in self.sources if not s.complete]
```

```python
"""
Result Aggregation

Combines results from multiple analysis sources, handling partial failures.
"""

from dataclasses import dataclass, field
from typing import List, Optional, Any, Dict
from datetime import datetime


@dataclass
class SourceResult:
    """Result from a single analysis source."""
    source: str
    complete: bool
    findings: List[Dict[str, Any]] = field(default_factory=list)
    error: Optional[str] = None
    runtime_ms: int = 0
    retry_command: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "source": self.source,
            "complete": self.complete,
            "finding_count": len(self.findings),
            "error": self.error,
            "runtime_ms": self.runtime_ms,
            "retry_command": self.retry_command,
        }


@dataclass
class AggregatedResults:
    """Combined results from multiple sources."""
    sources: List[SourceResult] = field(default_factory=list)
    timestamp: datetime = field(default_factory=datetime.now)

    @property
    def total_findings(self) -> int:
        return sum(len(s.findings) for s in self.sources)

    @property
    def complete(self) -> bool:
        return all(s.complete for s in self.sources)

    @property
    def incomplete_sources(self) -> List[str]:
        return [s.source for s in self.sources if not s.complete]

    @property
    def successful_sources(self) -> List[str]:
        return [s.source for s in self.sources if s.complete]

    def add_result(self, result: SourceResult):
        """Add a source result."""
        self.sources.append(result)

    def get_all_findings(self) -> List[Dict[str, Any]]:
        """Get all findings from all sources."""
        findings = []
        for source in self.sources:
            for finding in source.findings:
                finding["_source"] = source.source
                findings.append(finding)
        return findings

    def to_dict(self) -> Dict[str, Any]:
        return {
            "total_findings": self.total_findings,
            "complete": self.complete,
            "incomplete_sources": self.incomplete_sources,
            "sources": [s.to_dict() for s in self.sources],
            "timestamp": self.timestamp.isoformat(),
        }


class ResultAggregator:
    """Aggregates results from multiple analysis sources."""

    def __init__(self):
        self.results = AggregatedResults()

    def add_success(
        self,
        source: str,
        findings: List[Dict[str, Any]],
        runtime_ms: int = 0,
    ):
        """Add successful result."""
        self.results.add_result(SourceResult(
            source=source,
            complete=True,
            findings=findings,
            runtime_ms=runtime_ms,
        ))

    def add_failure(
        self,
        source: str,
        error: str,
        partial_findings: List[Dict[str, Any]] = None,
        retry_command: str = None,
    ):
        """Add failed result."""
        self.results.add_result(SourceResult(
            source=source,
            complete=False,
            findings=partial_findings or [],
            error=error,
            retry_command=retry_command,
        ))

    def add_timeout(
        self,
        source: str,
        timeout_seconds: int,
        partial_findings: List[Dict[str, Any]] = None,
    ):
        """Add timeout result."""
        self.results.add_result(SourceResult(
            source=source,
            complete=False,
            findings=partial_findings or [],
            error=f"Timed out after {timeout_seconds}s",
            retry_command=f"vkg tools run {source} --timeout {timeout_seconds * 2}",
        ))

    def get_results(self) -> AggregatedResults:
        """Get aggregated results."""
        return self.results
```

### 2. Create: `src/true_vkg/analysis/partial.py`

```python
"""
Partial Result Handling

Utilities for working with incomplete analysis results.
"""

from typing import List, Dict, Any
from rich.console import Console
from rich.table import Table

from .results import AggregatedResults, SourceResult


def format_partial_results(results: AggregatedResults) -> str:
    """Format results with partial indicators."""
    lines = []

    if results.complete:
        lines.append("Analysis Results (Complete)")
    else:
        incomplete = ", ".join(results.incomplete_sources)
        lines.append(f"Analysis Results (Partial - {len(results.incomplete_sources)} source(s) failed)")

    lines.append("")

    for source in results.sources:
        status = "[OK]" if source.complete else "[!]"
        finding_count = len(source.findings)
        status_text = f"Complete ({finding_count} findings)" if source.complete else source.error

        lines.append(f"  {source.source}: {status} {status_text}")

    lines.append("")
    lines.append(f"Total findings: {results.total_findings}")

    if not results.complete:
        lines.append("")
        lines.append("Note: Some sources failed. Results may be incomplete.")
        for source in results.sources:
            if not source.complete and source.retry_command:
                lines.append(f"  Retry {source.source}: {source.retry_command}")

    return "\n".join(lines)


def render_partial_results_rich(results: AggregatedResults, console: Console):
    """Render results with rich formatting."""
    # Status table
    table = Table(title="Analysis Results")
    table.add_column("Source", style="cyan")
    table.add_column("Status")
    table.add_column("Findings", justify="right")
    table.add_column("Details", style="dim")

    for source in results.sources:
        if source.complete:
            status = "[green]OK[/green]"
        else:
            status = "[yellow]FAILED[/yellow]"

        table.add_row(
            source.source,
            status,
            str(len(source.findings)),
            source.error or "",
        )

    console.print(table)

    # Summary
    if results.complete:
        console.print(f"\n[green]Complete:[/green] {results.total_findings} findings")
    else:
        console.print(f"\n[yellow]Partial:[/yellow] {results.total_findings} findings")
        console.print("[yellow]Some sources failed. See retry commands above.[/yellow]")

        # Retry hints
        for source in results.sources:
            if not source.complete and source.retry_command:
                console.print(f"  [dim]{source.retry_command}[/dim]")


def merge_partial_results(
    existing: AggregatedResults,
    new: AggregatedResults,
) -> AggregatedResults:
    """
    Merge new results into existing, replacing failed sources.

    Use case: User retries a failed source, merge new results.
    """
    merged = AggregatedResults()

    # Track which sources we've seen in new results
    new_sources = {s.source for s in new.sources}

    # Add all new results
    for source in new.sources:
        merged.add_result(source)

    # Add existing results for sources not in new
    for source in existing.sources:
        if source.source not in new_sources:
            merged.add_result(source)

    return merged


class PartialResultHandler:
    """Handles partial result scenarios."""

    def __init__(self, console: Console = None):
        self.console = console or Console()

    def handle(self, results: AggregatedResults) -> bool:
        """
        Handle partial results.

        Returns True if all results complete, False if partial.
        """
        render_partial_results_rich(results, self.console)

        if not results.complete:
            self._offer_retry(results)

        return results.complete

    def _offer_retry(self, results: AggregatedResults):
        """Offer retry options for failed sources."""
        failed = [s for s in results.sources if not s.complete]

        if not failed:
            return

        self.console.print("\n[bold]Retry options:[/bold]")
        for source in failed:
            if source.retry_command:
                self.console.print(f"  {source.source}: {source.retry_command}")
            else:
                self.console.print(f"  {source.source}: vkg tools run {source.source}")
```

```python
    handler.handle(results)
```

```python
"""Tests for partial result handling."""

import pytest
from true_vkg.analysis.results import (
    SourceResult, AggregatedResults, ResultAggregator
)
from true_vkg.analysis.partial import (
    format_partial_results, merge_partial_results
)


class TestSourceResult:
    """Test SourceResult dataclass."""

    def test_complete_result(self):
        """Complete result has complete=True."""
        result = SourceResult(
            source="slither",
            complete=True,
            findings=[{"id": 1}],
        )
        assert result.complete is True
        assert len(result.findings) == 1

    def test_failed_result(self):
        """Failed result has error."""
        result = SourceResult(
            source="aderyn",
            complete=False,
            error="Timeout",
        )
        assert result.complete is False
        assert result.error == "Timeout"


class TestAggregatedResults:
    """Test AggregatedResults."""

    def test_total_findings(self):
        """Total findings sums all sources."""
        results = AggregatedResults()
        results.add_result(SourceResult("a", True, [{"id": 1}]))
        results.add_result(SourceResult("b", True, [{"id": 2}, {"id": 3}]))

        assert results.total_findings == 3

    def test_complete_when_all_sources_complete(self):
        """Complete only when all sources complete."""
        results = AggregatedResults()
        results.add_result(SourceResult("a", True, []))
        results.add_result(SourceResult("b", True, []))

        assert results.complete is True

    def test_incomplete_when_any_source_fails(self):
        """Incomplete when any source fails."""
        results = AggregatedResults()
        results.add_result(SourceResult("a", True, []))
        results.add_result(SourceResult("b", False, error="Failed"))

        assert results.complete is False
        assert "b" in results.incomplete_sources


class TestResultAggregator:
    """Test ResultAggregator."""

    def test_add_success(self):
        """Add success correctly."""
        agg = ResultAggregator()
        agg.add_success("slither", [{"id": 1}])

        results = agg.get_results()
        assert results.total_findings == 1
        assert results.complete is True

    def test_add_timeout(self):
        """Add timeout with retry command."""
        agg = ResultAggregator()
        agg.add_timeout("aderyn", 60)

        results = agg.get_results()
        assert results.complete is False
        assert "Timed out" in results.sources[0].error
        assert results.sources[0].retry_command is not None


class TestFormatPartialResults:
    """Test result formatting."""

    def test_complete_results(self):
        """Complete results show 'Complete'."""
        results = AggregatedResults()
        results.add_result(SourceResult("slither", True, [{"id": 1}]))

        output = format_partial_results(results)
        assert "Complete" in output
        assert "Partial" not in output

    def test_partial_results(self):
        """Partial results show warning."""
        results = AggregatedResults()
        results.add_result(SourceResult("slither", True, []))
        results.add_result(SourceResult("aderyn", False, error="Timeout"))

        output = format_partial_results(results)
        assert "Partial" in output
        assert "incomplete" in output.lower()


class TestMergePartialResults:
    """Test result merging."""

    def test_merge_replaces_failed(self):
        """Merge replaces failed source with new success."""
        existing = AggregatedResults()
        existing.add_result(SourceResult("a", True, [{"id": 1}]))
        existing.add_result(SourceResult("b", False, error="Failed"))

        new = AggregatedResults()
        new.add_result(SourceResult("b", True, [{"id": 2}]))

        merged = merge_partial_results(existing, new)

        assert merged.complete is True
        assert merged.total_findings == 2

    def test_merge_preserves_existing_success(self):
        """Merge preserves existing success not in new."""
        existing = AggregatedResults()
        existing.add_result(SourceResult("a", True, [{"id": 1}]))

        new = AggregatedResults()
        new.add_result(SourceResult("b", True, [{"id": 2}]))

        merged = merge_partial_results(existing, new)

        sources = {s.source for s in merged.sources}
        assert sources == {"a", "b"}
```
