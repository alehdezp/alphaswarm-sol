"""
Graph Version Tracking

Tracks graph versions for staleness detection and reproducibility.
"""

import hashlib
import json
from dataclasses import dataclass, field
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Any, Optional


@dataclass
class GraphVersion:
    """Version identifier for a knowledge graph."""
    version_id: str
    fingerprint: str
    code_hash: str
    created_at: datetime
    source_files: List[str] = field(default_factory=list)
    slither_version: Optional[str] = None
    vkg_version: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "version_id": self.version_id,
            "fingerprint": self.fingerprint,
            "code_hash": self.code_hash,
            "created_at": self.created_at.isoformat(),
            "source_files": self.source_files,
            "slither_version": self.slither_version,
            "vkg_version": self.vkg_version,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "GraphVersion":
        return cls(
            version_id=data["version_id"],
            fingerprint=data["fingerprint"],
            code_hash=data["code_hash"],
            created_at=datetime.fromisoformat(data["created_at"]),
            source_files=data.get("source_files", []),
            slither_version=data.get("slither_version"),
            vkg_version=data.get("vkg_version"),
        )


class VersionGenerator:
    """Generates versions for knowledge graphs."""

    def __init__(self):
        self._counter = 0

    def generate(
        self,
        graph: Any,
        source_files: List[Path],
        slither_version: Optional[str] = None,
    ) -> GraphVersion:
        """
        Generate version for a graph.

        Args:
            graph: Knowledge graph object
            source_files: List of source file paths
            slither_version: Slither version used

        Returns:
            GraphVersion with deterministic identifiers
        """
        # Graph fingerprint (deterministic hash of graph structure)
        fingerprint = self._hash_graph(graph)

        # Code hash (hash of source files)
        code_hash = self._hash_files(source_files)

        # Version ID (short, human-readable)
        self._counter += 1
        short_fp = fingerprint[:8]
        version_id = f"v{self._counter}-{short_fp}"

        return GraphVersion(
            version_id=version_id,
            fingerprint=fingerprint,
            code_hash=code_hash,
            created_at=datetime.now(),
            source_files=[str(f) for f in source_files],
            slither_version=slither_version,
            vkg_version=self._get_vkg_version(),
        )

    def _hash_graph(self, graph: Any) -> str:
        """Generate deterministic hash of graph."""
        # Normalize graph to dict
        if hasattr(graph, 'to_dict'):
            data = graph.to_dict()
        elif hasattr(graph, 'nodes') and hasattr(graph, 'edges'):
            data = {
                "nodes": sorted([n.to_dict() if hasattr(n, 'to_dict') else str(n) for n in graph.nodes]),
                "edges": sorted([str(e) for e in graph.edges]),
            }
        else:
            data = str(graph)

        # Canonical JSON
        canonical = json.dumps(data, sort_keys=True, default=str)
        return hashlib.sha256(canonical.encode()).hexdigest()

    def _hash_files(self, files: List[Path]) -> str:
        """Generate hash of source files."""
        hasher = hashlib.sha256()

        for f in sorted(files):
            if f.exists():
                hasher.update(f.read_bytes())
                hasher.update(str(f).encode())

        return hasher.hexdigest()

    def _get_vkg_version(self) -> str:
        """Get current BSKG version."""
        try:
            from true_vkg import __version__
            return __version__
        except ImportError:
            return "unknown"


class VersionStore:
    """Stores and retrieves graph versions."""

    def __init__(self, vkg_dir: Path):
        self.versions_dir = vkg_dir / "versions"
        self.versions_dir.mkdir(parents=True, exist_ok=True)
        self.current_file = vkg_dir / "current_version.json"

    def save(self, version: GraphVersion) -> Path:
        """Save version to store."""
        path = self.versions_dir / f"{version.version_id}.json"
        path.write_text(json.dumps(version.to_dict(), indent=2))

        # Update current version pointer
        self.current_file.write_text(json.dumps({
            "current": version.version_id,
            "path": str(path),
        }))

        return path

    def load(self, version_id: str) -> Optional[GraphVersion]:
        """Load version by ID."""
        path = self.versions_dir / f"{version_id}.json"
        if not path.exists():
            return None

        data = json.loads(path.read_text())
        return GraphVersion.from_dict(data)

    def get_current(self) -> Optional[GraphVersion]:
        """Get current version."""
        if not self.current_file.exists():
            return None

        pointer = json.loads(self.current_file.read_text())
        return self.load(pointer["current"])

    def list_versions(self) -> List[GraphVersion]:
        """List all versions."""
        versions = []
        for path in self.versions_dir.glob("v*.json"):
            data = json.loads(path.read_text())
            versions.append(GraphVersion.from_dict(data))
        return sorted(versions, key=lambda v: v.created_at, reverse=True)
```

"""
Staleness Detection

Detects when findings reference outdated graph versions.
"""

from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import Optional, List, Any
import logging

from .versioning import GraphVersion

logger = logging.getLogger(__name__)


@dataclass
class StalenessResult:
    """Result of staleness check."""
    stale: bool
    finding_version: Optional[str] = None
    current_version: Optional[str] = None
    age: Optional[timedelta] = None
    code_changed: bool = False
    recommendation: Optional[str] = None

    @property
    def severity(self) -> str:
        """Get staleness severity."""
        if not self.stale:
            return "ok"
        if self.code_changed:
            return "critical"  # Code changed, findings may be invalid
        if self.age and self.age > timedelta(days=7):
            return "warning"  # Old but code unchanged
        return "info"

    def to_dict(self) -> dict:
        return {
            "stale": self.stale,
            "finding_version": self.finding_version,
            "current_version": self.current_version,
            "age_seconds": self.age.total_seconds() if self.age else None,
            "code_changed": self.code_changed,
            "severity": self.severity,
            "recommendation": self.recommendation,
        }


class StalenessChecker:
    """Checks findings for staleness."""

    def __init__(self, current_version: GraphVersion):
        self.current_version = current_version

    def check(
        self,
        finding_version: Optional[str],
        finding_code_hash: Optional[str] = None,
    ) -> StalenessResult:
        """
        Check if finding is stale.

        Args:
            finding_version: Version ID from finding
            finding_code_hash: Code hash from finding

        Returns:
            StalenessResult
        """
        if finding_version is None:
            return StalenessResult(
                stale=True,
                current_version=self.current_version.version_id,
                recommendation="Finding has no version. Run: vkg findings refresh",
            )

        if finding_version == self.current_version.version_id:
            return StalenessResult(stale=False)

        # Version mismatch - check if code changed
        code_changed = (
            finding_code_hash is not None and
            finding_code_hash != self.current_version.code_hash
        )

        return StalenessResult(
            stale=True,
            finding_version=finding_version,
            current_version=self.current_version.version_id,
            code_changed=code_changed,
            recommendation=self._get_recommendation(code_changed),
        )

    def _get_recommendation(self, code_changed: bool) -> str:
        if code_changed:
            return (
                "Source code has changed. Findings may be invalid.\n"
                "Options:\n"
                "  vkg findings refresh   # Re-validate against current graph\n"
                "  vkg findings reset     # Regenerate all findings"
            )
        return (
            "Graph rebuilt but source unchanged. Findings likely still valid.\n"
            "Run: vkg findings refresh"
        )

    def check_all(self, findings: List[Any]) -> List[StalenessResult]:
        """Check multiple findings."""
        results = []
        for finding in findings:
            version = getattr(finding, 'graph_version', None)
            code_hash = getattr(finding, 'code_hash', None)
            results.append(self.check(version, code_hash))
        return results


def format_staleness_warning(result: StalenessResult) -> str:
    """Format staleness result as user warning."""
    if not result.stale:
        return ""

    icon = {
        "critical": "[!] CRITICAL",
        "warning": "[!] WARNING",
        "info": "[i] INFO",
    }.get(result.severity, "[?]")

    lines = [
        f"{icon}: Graph has changed since findings were created",
        f"  Graph version: {result.current_version}",
        f"  Findings reference: {result.finding_version}",
    ]

    if result.code_changed:
        lines.append("  Source code: CHANGED (findings may be invalid)")
    else:
        lines.append("  Source code: unchanged")

    if result.recommendation:
        lines.append("")
        lines.append("  " + result.recommendation.replace("\n", "\n  "))

    return "\n".join(lines)
```

# Add imports
from true_vkg.state.versioning import VersionGenerator, VersionStore, GraphVersion

class GraphStore:
    def __init__(self, output_dir: Path):
        self.output_dir = output_dir
        self.version_store = VersionStore(output_dir.parent)
        self.version_generator = VersionGenerator()

    def save(self, graph: Any, source_files: List[Path] = None, **kwargs) -> Path:
        """Save graph with version tracking."""
        # Generate version
        version = self.version_generator.generate(
            graph,
            source_files or [],
            slither_version=graph.metadata.get("slither_version"),
        )

        # Attach version to graph
        if hasattr(graph, 'metadata'):
            graph.metadata["version"] = version.to_dict()

        # Save graph (existing logic)
        path = self._save_graph_file(graph, **kwargs)

        # Save version
        self.version_store.save(version)

        return path

    def get_current_version(self) -> Optional[GraphVersion]:
        """Get current graph version."""
        return self.version_store.get_current()

"""Tests for state versioning."""

import pytest
from datetime import datetime, timedelta
from pathlib import Path
import tempfile

from true_vkg.state.versioning import (
    GraphVersion, VersionGenerator, VersionStore
)
from true_vkg.state.staleness import (
    StalenessResult, StalenessChecker, format_staleness_warning

class TestGraphVersion:
    """Test GraphVersion dataclass."""

    def test_to_dict_roundtrip(self):
        """Version survives dict roundtrip."""
        version = GraphVersion(
            version_id="v1-abc123",
            fingerprint="hash1",
            code_hash="hash2",
            created_at=datetime.now(),
            source_files=["a.sol", "b.sol"],
        )

        data = version.to_dict()
        restored = GraphVersion.from_dict(data)

        assert restored.version_id == version.version_id
        assert restored.fingerprint == version.fingerprint


class TestVersionGenerator:
    """Test version generation."""

    def test_generates_unique_ids(self):
        """Generator creates unique IDs."""
        gen = VersionGenerator()

        v1 = gen.generate({}, [])
        v2 = gen.generate({}, [])

        assert v1.version_id != v2.version_id

    def test_same_graph_same_fingerprint(self):
        """Same graph produces same fingerprint."""
        gen = VersionGenerator()

        graph = {"nodes": [1, 2, 3], "edges": [(1, 2)]}

        v1 = gen.generate(graph, [])
        v2 = gen.generate(graph, [])

        assert v1.fingerprint == v2.fingerprint

    def test_different_graph_different_fingerprint(self):
        """Different graphs have different fingerprints."""
        gen = VersionGenerator()

        v1 = gen.generate({"nodes": [1]}, [])
        v2 = gen.generate({"nodes": [2]}, [])

        assert v1.fingerprint != v2.fingerprint


class TestVersionStore:
    """Test version persistence."""

    def test_save_and_load(self):
        """Versions can be saved and loaded."""
        with tempfile.TemporaryDirectory() as tmpdir:
            store = VersionStore(Path(tmpdir))
            gen = VersionGenerator()

            version = gen.generate({"test": True}, [])
            store.save(version)

            loaded = store.load(version.version_id)
            assert loaded is not None
            assert loaded.version_id == version.version_id

    def test_get_current(self):
        """Current version is tracked."""
        with tempfile.TemporaryDirectory() as tmpdir:
            store = VersionStore(Path(tmpdir))
            gen = VersionGenerator()

            v1 = gen.generate({"v": 1}, [])
            store.save(v1)

            v2 = gen.generate({"v": 2}, [])
            store.save(v2)

            current = store.get_current()
            assert current.version_id == v2.version_id


class TestStalenessChecker:
