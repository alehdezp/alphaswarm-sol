## Test Scenarios

### Scenario 1: Kill Slither Mid-Run

**Setup:**
1. Start large project analysis
2. Wait 5 seconds
3. Kill slither process

**Expected:**
- BSKG detects termination
- Reports partial results or clear error
- Does NOT crash
- Recovery command provided

### Scenario 2: Remove .vkg Directory Mid-Operation

**Setup:**
1. Start analysis that writes to .vkg
2. Delete .vkg directory externally
3. Continue operation

**Expected:**
- BSKG detects missing directory
- Reports clear error
- Does NOT crash
- Suggests `vkg build-kg` to recreate

### Scenario 3: Corrupt graph.json

**Setup:**
1. Build valid graph
2. Corrupt graph.json (invalid JSON)
3. Run query

**Expected:**
- BSKG detects corruption
- Reports clear error
- Does NOT crash
- Suggests `vkg repair` or rebuild

### Scenario 4: Timeout External Tools

**Setup:**
1. Set very short timeout (1s)
2. Run analysis on large project
3. Tools time out

**Expected:**
- Timeouts detected
- Partial results returned
- Does NOT hang
- Retry commands provided

### Scenario 5: Fill Disk During Write

**Setup:**
1. Start operation that writes large output
2. Simulate disk full (or use tmpfs with limit)
3. Write fails

**Expected:**
- Write failure detected
- Atomic writes prevent corruption
- Clear error message
- Does NOT corrupt existing state

"""
Chaos Injection Utilities

Tools for injecting failures during tests.
"""

import os
import signal
import subprocess
import tempfile
import shutil
from pathlib import Path
from typing import Optional, Callable
import threading
import time


class ChaosInjector:
    """Injects various failure conditions."""

    def __init__(self):
        self._cleanup_tasks = []

    def cleanup(self):
        """Run all cleanup tasks."""
        for task in self._cleanup_tasks:
            try:
                task()
            except Exception:
                pass
        self._cleanup_tasks = []

    def kill_process_after(
        self,
        process_name: str,
        delay_seconds: float,
    ) -> threading.Thread:
        """
        Kill a process after delay.

        Args:
            process_name: Process name to kill (e.g., "slither")
            delay_seconds: Delay before killing

        Returns:
            Thread that will perform the kill
        """
        def killer():
            time.sleep(delay_seconds)
            try:
                subprocess.run(["pkill", "-f", process_name], check=False)
            except Exception:
                pass

        thread = threading.Thread(target=killer, daemon=True)
        thread.start()
        return thread

    def delete_path_after(
        self,
        path: Path,
        delay_seconds: float,
    ) -> threading.Thread:
        """
        Delete a path after delay.

        Args:
            path: Path to delete
            delay_seconds: Delay before deletion

        Returns:
            Thread that will perform the deletion
        """
        def deleter():
            time.sleep(delay_seconds)
            try:
                if path.is_dir():
                    shutil.rmtree(path)
                else:
                    path.unlink()
            except Exception:
                pass

        thread = threading.Thread(target=deleter, daemon=True)
        thread.start()
        return thread

    def corrupt_file(self, path: Path) -> None:
        """
        Corrupt a file by writing invalid content.

        Args:
            path: File to corrupt
        """
        original = path.read_text() if path.exists() else ""
        self._cleanup_tasks.append(lambda: path.write_text(original))

        path.write_text("{{{{invalid json not json at all")

    def create_limited_tmpfs(self, size_mb: int = 1) -> Path:
        """
        Create a tmpfs with limited size (Linux only).

        Args:
            size_mb: Size limit in MB

        Returns:
            Path to mounted tmpfs
        """
        tmpdir = tempfile.mkdtemp(prefix="vkg_stress_")

        try:
            subprocess.run([
                "mount", "-t", "tmpfs", "-o", f"size={size_mb}M",
                "tmpfs", tmpdir
            ], check=True)

            self._cleanup_tasks.append(lambda: subprocess.run(
                ["umount", tmpdir], check=False
            ))
            self._cleanup_tasks.append(lambda: os.rmdir(tmpdir))

            return Path(tmpdir)
        except subprocess.CalledProcessError:
            # Fall back to regular tmpdir
            os.rmdir(tmpdir)
            return Path(tempfile.mkdtemp(prefix="vkg_stress_"))


class TimeoutSimulator:
    """Simulates tool timeouts."""

    @staticmethod
    def create_slow_command(duration_seconds: int) -> list:
        """Create a command that takes long time."""
        return ["python", "-c", f"import time; time.sleep({duration_seconds})"]


class DiskFullSimulator:
    """Simulates disk full conditions."""

    def __init__(self, tmpdir: Path, limit_bytes: int):
        self.tmpdir = tmpdir
        self.limit_bytes = limit_bytes
        self._filler: Optional[Path] = None

    def fill(self):
        """Fill the disk to limit."""
        self._filler = self.tmpdir / ".filler"
        with open(self._filler, "wb") as f:
            f.write(b"x" * self.limit_bytes)

    def release(self):
        """Release filled space."""
        if self._filler and self._filler.exists():
            self._filler.unlink()

"""
Stress Tests for BSKG Robustness

Tests BSKG behavior under various failure conditions.
Each scenario must:
1. Not crash VKG
2. Provide clear error
3. Offer recovery path
"""

import pytest
import subprocess
import tempfile
import json
import time
from pathlib import Path
from unittest.mock import patch, MagicMock

from tests.stress.chaos import ChaosInjector, TimeoutSimulator


class TestScenario1_KillSlitherMidRun:
    """Test: Kill Slither process during analysis."""

    @pytest.fixture
    def chaos(self):
        injector = ChaosInjector()
        yield injector
        injector.cleanup()

    def test_slither_killed_does_not_crash_vkg(self, chaos, tmp_path):
        """VKG handles slither being killed."""
        # This test requires slither to be installed and a valid contract

        # Skip if slither not available
        try:
            subprocess.run(["slither", "--version"], check=True, capture_output=True)
        except (subprocess.CalledProcessError, FileNotFoundError):
            pytest.skip("Slither not installed")

        # Create minimal contract
        contract = tmp_path / "Test.sol"
        contract.write_text("// SPDX-License-Identifier: MIT\npragma solidity ^0.8.0;\ncontract Test {}")

        # Kill slither after 2 seconds
        chaos.kill_process_after("slither", 2.0)

        # Run BSKG (this should not crash)
        from typer.testing import CliRunner
        from true_vkg.cli import app

        runner = CliRunner()
        result = runner.invoke(app, ["build-kg", str(contract)])

        # BSKG should not crash (exit code 0 or 1, not 255 or signal)
        assert result.exit_code in (0, 1), f"VKG crashed: {result.output}"


class TestScenario2_RemoveVkgDirectory:
    """Test: .vkg directory removed during operation."""

    def test_vkg_dir_removed_does_not_crash(self, tmp_path):
        """VKG handles missing .vkg directory gracefully."""
        # Create .vkg dir
        vkg_dir = tmp_path / ".true_vkg"
        vkg_dir.mkdir()
        graph_dir = vkg_dir / "graphs"

        assert "not found" in result.output.lower() or "error" in result.output.lower()

class TestScenario4_ToolTimeout:
    """Test: External tools timeout."""

    def test_timeout_returns_partial_results(self):
        """Tool timeout returns partial results, doesn't hang."""
        from true_vkg.tools.runner import ToolRunner

        # Use 1 second timeout with slow command
        runner = ToolRunner(timeout=1)

        start = time.time()
        result = runner.run(["python", "-c", "import time; time.sleep(10)"])
        elapsed = time.time() - start

        # Should return within 2 seconds (1s timeout + overhead)
        assert elapsed < 5, "Tool runner hung past timeout"

        # Should have timeout error
        assert result.success is False
        assert "timeout" in result.error.lower()

        # Should have recovery command
        assert result.recovery is not None

    def test_vkg_continues_after_tool_timeout(self):
        """VKG can continue running after tool times out."""
        from true_vkg.tools.runner import ToolRunner

        runner = ToolRunner(timeout=1)

        # First tool times out
        result1 = runner.run(["python", "-c", "import time; time.sleep(10)"])
        assert result1.success is False

        # BSKG can still run other commands
        result2 = runner.run(["python", "-c", "print('ok')"])
        assert result2.success is True
        assert "ok" in result2.output


class TestScenario5_DiskFull:
    """Test: Disk full during write."""

    def test_disk_full_does_not_corrupt(self, tmp_path):
        """Write failure doesn't corrupt existing files."""
        # Create existing file
        existing = tmp_path / "existing.json"
        existing.write_text('{"valid": true}')

        # Create "full" disk simulation by making read-only
        # (actual tmpfs limits require root)
        output = tmp_path / "output.json"

        # Make directory read-only to simulate disk full
        tmp_path.chmod(0o555)

        try:
            # Try to write - should fail
            try:
                output.write_text('{"new": true}')
                # If this succeeds, permissions didn't work
                pytest.skip("Could not simulate disk full")
            except PermissionError:
                pass

            # Existing file should still be valid
            data = json.loads(existing.read_text())
            assert data["valid"] is True

        finally:
            # Restore permissions
            tmp_path.chmod(0o755)


class TestCrossScenarioRecovery:
