## Design

### Safe Tool Runner

```python
@dataclass
class ToolResult:
    """Result of running a tool."""
    success: bool
    output: Optional[str] = None
    error: Optional[str] = None
    exit_code: int = -1
    runtime_ms: int = 0
    recovery: Optional[str] = None  # What user can do

    def to_dict(self) -> Dict[str, Any]:
        return {
            "success": self.success,
            "output": self.output,
            "error": self.error,
            "exit_code": self.exit_code,
            "runtime_ms": self.runtime_ms,
            "recovery": self.recovery,
        }
```

### Error Categories

| Error | Detection | Recovery |
|-------|-----------|----------|
| Timeout | Timer exceeded | Suggest larger timeout |
| Crash | Non-zero exit | Suggest checking tool installation |
| OOM | MemoryError / killed | Suggest reducing batch size |
| Missing | FileNotFoundError | Suggest installation |
| Permission | PermissionError | Suggest chmod/sudo |

```python
"""
Isolated Tool Runner

Runs external tools with timeout, isolation, and error recovery.
"""

import subprocess
import signal
import time
from dataclasses import dataclass
from typing import Optional, List, Dict, Any
from pathlib import Path
import logging

logger = logging.getLogger(__name__)


@dataclass
class ToolResult:
    """Result of running a tool."""
    success: bool
    output: Optional[str] = None
    error: Optional[str] = None
    exit_code: int = -1
    runtime_ms: int = 0
    recovery: Optional[str] = None
    partial: bool = False  # True if we got partial results

    def to_dict(self) -> Dict[str, Any]:
        return {
            "success": self.success,
            "output": self.output,
            "error": self.error,
            "exit_code": self.exit_code,
            "runtime_ms": self.runtime_ms,
            "recovery": self.recovery,
            "partial": self.partial,
        }


class ToolRunner:
    """Runs external tools with isolation and error handling."""

    DEFAULT_TIMEOUT = 60  # seconds
    MAX_OUTPUT_SIZE = 10 * 1024 * 1024  # 10 MB

    def __init__(
        self,
        timeout: int = DEFAULT_TIMEOUT,
        max_retries: int = 0,
    ):
        self.timeout = timeout
        self.max_retries = max_retries

    def run(
        self,
        command: List[str],
        cwd: Optional[Path] = None,
        env: Optional[Dict[str, str]] = None,
        timeout: Optional[int] = None,
    ) -> ToolResult:
        """
        Run a command with isolation.

        Args:
            command: Command and arguments as list
            cwd: Working directory
            env: Environment variables (merged with current env)
            timeout: Override default timeout

        Returns:
            ToolResult with success status and output/error
        """
        effective_timeout = timeout or self.timeout
        start_time = time.monotonic()

        # Merge environment
        import os
        full_env = os.environ.copy()
        if env:
            full_env.update(env)

        try:
            logger.debug(f"Running: {' '.join(command)}")

            result = subprocess.run(
                command,
                cwd=cwd,
                env=full_env,
                capture_output=True,
                timeout=effective_timeout,
                text=True,
            )

            runtime_ms = int((time.monotonic() - start_time) * 1000)

            if result.returncode == 0:
                return ToolResult(
                    success=True,
                    output=self._truncate_output(result.stdout),
                    exit_code=0,
                    runtime_ms=runtime_ms,
                )
            else:
                return ToolResult(
                    success=False,
                    output=self._truncate_output(result.stdout),
                    error=result.stderr[:1000] if result.stderr else None,
                    exit_code=result.returncode,
                    runtime_ms=runtime_ms,
                    recovery=self._get_recovery_hint(command[0], result.returncode, result.stderr),
                )

        except subprocess.TimeoutExpired as e:
            runtime_ms = int((time.monotonic() - start_time) * 1000)
            return ToolResult(
                success=False,
                error=f"Command timed out after {effective_timeout}s",
                runtime_ms=runtime_ms,
                recovery=f"Try increasing timeout: --timeout {effective_timeout * 2}",
                partial=bool(e.stdout),
                output=e.stdout[:1000] if e.stdout else None,
            )

        except FileNotFoundError:
            return ToolResult(
                success=False,
                error=f"Command not found: {command[0]}",
                recovery=self._get_install_hint(command[0]),
            )

        except PermissionError:
            return ToolResult(
                success=False,
                error=f"Permission denied: {command[0]}",
                recovery=f"Check file permissions or run: chmod +x {command[0]}",
            )

        except MemoryError:
            return ToolResult(
                success=False,
                error="Out of memory",
                recovery="Try reducing input size or increasing system memory",
            )

        except Exception as e:
            logger.exception(f"Unexpected error running {command[0]}")
            return ToolResult(
                success=False,
                error=f"Unexpected error: {type(e).__name__}: {str(e)[:200]}",
                recovery="Check logs for details: vkg doctor --verbose",
            )

    def run_with_retry(
        self,
        command: List[str],
        **kwargs,
    ) -> ToolResult:
        """Run command with retries on transient failures."""
        last_result = None

        for attempt in range(self.max_retries + 1):
            result = self.run(command, **kwargs)

            if result.success:
                return result

            # Check if error is retryable
            if not self._is_retryable(result):
                return result

            last_result = result
            logger.info(f"Retry {attempt + 1}/{self.max_retries} for {command[0]}")
            time.sleep(2 ** attempt)  # Exponential backoff

        return last_result or ToolResult(success=False, error="All retries failed")

    def _truncate_output(self, output: str) -> str:
        """Truncate output if too large."""
        if len(output) > self.MAX_OUTPUT_SIZE:
            return output[:self.MAX_OUTPUT_SIZE] + "\n... (truncated)"
        return output

    def _is_retryable(self, result: ToolResult) -> bool:
        """Check if error is transient and worth retrying."""
        if result.error:
            retryable_patterns = [
                "timed out",
                "connection",
                "temporary",
                "try again",
            ]
            return any(p in result.error.lower() for p in retryable_patterns)
        return False

    def _get_recovery_hint(self, tool: str, exit_code: int, stderr: Optional[str]) -> str:
        """Generate recovery hint based on error."""
        hints = {
            "slither": {
                1: "Check Solidity syntax or try: slither --help",
                127: "Slither not found. Install: pip install slither-analyzer",
            },
            "aderyn": {
                1: "Check contract path or try: aderyn --help",
                127: "Aderyn not found. Install: cargo install aderyn",
            },
            "forge": {
                1: "Check Foundry project or try: forge --help",
                127: "Foundry not found. Install: curl -L https://foundry.paradigm.xyz | bash",
            },
        }

        tool_hints = hints.get(tool, {})
        hint = tool_hints.get(exit_code)

        if hint:
            return hint

        if stderr and "not found" in stderr.lower():
            return self._get_install_hint(tool)

        return f"Check {tool} logs or run: {tool} --help"

    def _get_install_hint(self, tool: str) -> str:
        """Get installation hint for tool."""
        from true_vkg.core.tiers import DEPENDENCIES

        if tool in DEPENDENCIES and DEPENDENCIES[tool].install_hint:
            return DEPENDENCIES[tool].install_hint

        return f"Install {tool} and ensure it's in PATH"


# Convenience function for safe execution
def run_tool_safely(
    command: List[str],
    timeout: int = 60,
    **kwargs,
) -> ToolResult:
    """Run a tool with default safety settings."""
    runner = ToolRunner(timeout=timeout)
    return runner.run(command, **kwargs)
```

### 2. Create: `src/true_vkg/tools/timeout.py`

```python
"""
Timeout handling utilities.

Provides context managers and decorators for timeout enforcement.
"""

import signal
import functools
from typing import Callable, TypeVar, Any
from contextlib import contextmanager

T = TypeVar('T')


class TimeoutError(Exception):
    """Raised when operation times out."""
    pass


@contextmanager
def timeout(seconds: int, message: str = "Operation timed out"):
    """
    Context manager for timeout enforcement.

    Usage:
        with timeout(30, "Analysis took too long"):
            slow_operation()

    Note: Only works on Unix systems (uses SIGALRM).
    """
    def handler(signum, frame):
        raise TimeoutError(message)

    # Set handler
    old_handler = signal.signal(signal.SIGALRM, handler)
    signal.alarm(seconds)

    try:
        yield
    finally:
        signal.alarm(0)
        signal.signal(signal.SIGALRM, old_handler)


def with_timeout(seconds: int):
    """
    Decorator for timeout enforcement.

    Usage:
        @with_timeout(30)
        def slow_function():
            ...
    """
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @functools.wraps(func)
        def wrapper(*args, **kwargs) -> T:
            with timeout(seconds, f"{func.__name__} timed out after {seconds}s"):
                return func(*args, **kwargs)
        return wrapper
    return decorator


class TimeoutManager:
    """Manages timeouts with nested support and Windows compatibility."""

    def __init__(self, default_timeout: int = 60):
        self.default_timeout = default_timeout
        self._active_timeouts = []

    def run_with_timeout(
        self,
        func: Callable[..., T],
        args: tuple = (),
        kwargs: dict = None,
        timeout_seconds: int = None,
    ) -> T:
        """
        Run function with timeout using threading (Windows compatible).

        Args:
            func: Function to run
            args: Positional arguments
            kwargs: Keyword arguments
            timeout_seconds: Timeout in seconds

        Returns:
            Function result

        Raises:
            TimeoutError: If function doesn't complete in time
        """
        import threading

        kwargs = kwargs or {}
        timeout_seconds = timeout_seconds or self.default_timeout
        result = [None]
        exception = [None]

        def target():
            try:
                result[0] = func(*args, **kwargs)
            except Exception as e:
                exception[0] = e

        thread = threading.Thread(target=target)
        thread.start()
        thread.join(timeout=timeout_seconds)

        if thread.is_alive():
            # Thread still running = timeout
            raise TimeoutError(f"Operation timed out after {timeout_seconds}s")

        if exception[0]:
            raise exception[0]

        return result[0]
```
