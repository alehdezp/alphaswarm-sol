"""
Tools subcommand for BSKG CLI.

Provides visibility into tool availability and system state.
"""

import typer
from rich.console import Console
from rich.table import Table
from rich.panel import Panel
from typing import Optional

from true_vkg.core.tiers import Tier, DEPENDENCIES, Dependency
from true_vkg.core.availability import AvailabilityChecker

tools_app = typer.Typer(help="Tool management commands")
console = Console()


@tools_app.command("status")
def status(
    json_output: bool = typer.Option(False, "--json", help="Output as JSON"),
    verbose: bool = typer.Option(False, "--verbose", "-v", help="Show more details"),
) -> None:
    """Show status of all BSKG tools and dependencies."""
    checker = AvailabilityChecker()
    reports = checker.check_all()

    if json_output:
        _output_json(reports)
    else:
        _output_rich(reports, verbose)


def _output_json(reports) -> None:
    """Output status as JSON."""
    import json
    output = {
        "tiers": [r.to_dict() for r in reports],
        "effective_tier": checker.get_effective_tier().name,
    }
    typer.echo(json.dumps(output, indent=2))


def _output_rich(reports, verbose: bool) -> None:
    """Output status with rich formatting."""
    # Build table
    table = Table(show_header=True, header_style="bold")
    table.add_column("Tool", style="cyan")
    table.add_column("Version", style="dim")
    table.add_column("Status")
    table.add_column("Notes", style="dim")

    missing_tools = []

    for report in sorted(reports, key=lambda r: r.tier.value):
        # Add tier header
        tier_name = _get_tier_display_name(report.tier)
        table.add_row(f"[bold]{tier_name}[/bold]", "", "", "")

        # Get dependencies for this tier
        tier_deps = [d for d in DEPENDENCIES.values() if d.tier == report.tier]

        for dep in tier_deps:
            is_available = dep.name in report.available
            version = dep.get_version() if is_available else "-"
            status_icon = "[green][OK][/green]" if is_available else "[yellow][!][/yellow]"
            status_text = "Ready" if is_available else "Not installed"

            if not is_available and dep.install_hint:
                missing_tools.append((dep.name, dep.install_hint))

            table.add_row(
                f"  {dep.name}",
                version or "-",
                f"{status_icon} {status_text}",
                dep.description if verbose else "",
            )

    # Print table
    console.print(Panel(table, title="VKG Tool Status"))

    # Print effective tier
    checker = AvailabilityChecker()
    try:
        effective = checker.get_effective_tier()
        console.print(f"\n[bold]Effective tier:[/bold] {effective.name}")
    except RuntimeError as e:
        console.print(f"\n[bold red]Error:[/bold red] {e}")

    # Print install hints for missing tools
    if missing_tools:
        console.print("\n[bold]To install missing tools:[/bold]")
        for name, hint in missing_tools:
            console.print(f"  {name}: [dim]{hint}[/dim]")


def _get_tier_display_name(tier: Tier) -> str:
    """Get human-readable tier name."""
    return {
        Tier.CORE: "Core (Required)",
        Tier.ENHANCEMENT: "Enhancement (Optional)",
        Tier.OPTIONAL: "Nice-to-Have",
    }.get(tier, tier.name)


@tools_app.command("check")
def check(
    tool: str = typer.Argument(..., help="Tool name to check"),
) -> None:
    """Check if a specific tool is available."""
    if tool not in DEPENDENCIES:
        typer.echo(f"Unknown tool: {tool}", err=True)
        typer.echo(f"Available tools: {', '.join(DEPENDENCIES.keys())}")
        raise typer.Exit(code=1)

    dep = DEPENDENCIES[tool]
    is_available = dep.is_available()

    if is_available:
        version = dep.get_version()
        typer.echo(f"{tool}: OK (version: {version})")
    else:
        typer.echo(f"{tool}: NOT AVAILABLE", err=True)
        if dep.install_hint:
            typer.echo(f"Install with: {dep.install_hint}")
        raise typer.Exit(code=1)


@tools_app.command("list")
def list_tools() -> None:
    """List all known tools."""
    for tier in Tier:
        typer.echo(f"\n{_get_tier_display_name(tier)}:")
        tier_deps = [d for d in DEPENDENCIES.values() if d.tier == tier]
        for dep in tier_deps:
            typer.echo(f"  - {dep.name}: {dep.description}")
```

"""
Tool Registry

Extended tool detection with version parsing and health checks.
"""

from dataclasses import dataclass
from typing import Optional, Dict, Any
import subprocess
import re


@dataclass
class ToolInfo:
    """Information about a detected tool."""
    name: str
    available: bool
    version: Optional[str] = None
    path: Optional[str] = None
    health: str = "unknown"  # ok, degraded, error
    details: Optional[Dict[str, Any]] = None


class ToolRegistry:
    """Registry for external tool detection."""

    # Version extraction patterns
    VERSION_PATTERNS = {
        "slither": r"(\d+\.\d+\.\d+)",
        "aderyn": r"(\d+\.\d+\.\d+)",
        "forge": r"(\d+\.\d+\.\d+)",
        "medusa": r"(\d+\.\d+\.\d+)",
    }

    def detect_tool(self, name: str, check_cmd: str) -> ToolInfo:
        """Detect a tool and gather information."""
        try:
            result = subprocess.run(
                check_cmd.split(),
                capture_output=True,
                timeout=10,
                text=True,
            )

            if result.returncode != 0:
                return ToolInfo(name=name, available=False, health="error")

            # Extract version
            version = self._extract_version(name, result.stdout + result.stderr)

            # Get path
            import shutil
            path = shutil.which(name)

            return ToolInfo(
                name=name,
                available=True,
                version=version,
                path=path,
                health="ok",
            )

        except subprocess.TimeoutExpired:
            return ToolInfo(name=name, available=False, health="error", details={"error": "timeout"})
        except Exception as e:
            return ToolInfo(name=name, available=False, health="error", details={"error": str(e)})

    def _extract_version(self, name: str, output: str) -> Optional[str]:
        """Extract version from command output."""
        pattern = self.VERSION_PATTERNS.get(name, r"(\d+\.\d+\.\d+)")
        match = re.search(pattern, output)
        return match.group(1) if match else None

    def detect_all(self) -> Dict[str, ToolInfo]:
        """Detect all registered tools."""
        from true_vkg.core.tiers import DEPENDENCIES

        results = {}
        for name, dep in DEPENDENCIES.items():
            if dep.check_cmd:
                results[name] = self.detect_tool(name, dep.check_cmd)
            else:
                # No check command = assume available
                results[name] = ToolInfo(name=name, available=True, health="ok")

        return results
```
