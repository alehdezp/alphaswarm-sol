```python
"""
Offline Mode Support

Tier A (graph building and pattern matching) works fully offline.
Tier B (LLM analysis) requires network.

Usage:
    VKG_OFFLINE=1 vkg build-kg ...  # Works
    VKG_OFFLINE=1 vkg analyze --tier-b  # Fails gracefully
"""

import os
import socket
from typing import Optional
from enum import Enum
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)


class NetworkMode(Enum):
    """Network mode for BSKG operation."""
    ONLINE = "online"           # Full capabilities
    OFFLINE = "offline"         # Core only
    AUTO = "auto"               # Detect automatically
    FORCED_OFFLINE = "forced"   # User forced offline


@dataclass
class NetworkStatus:
    """Current network status."""
    mode: NetworkMode
    network_available: bool
    reason: str

    @property
    def can_use_tier_b(self) -> bool:
        return self.mode == NetworkMode.ONLINE


# Tier capabilities
TIER_A_CAPABILITIES = [
    "build-kg",
    "query",
    "lens-report",
    "pattern matching",
    "graph fingerprinting",
    "findings (local)",
]

TIER_B_CAPABILITIES = [
    "LLM analysis",
    "intent annotation",
    "false positive filtering",
    "natural language queries",
    "test scaffold generation",
]


def is_offline_mode() -> bool:
    """Check if BSKG is running in offline mode."""
    env = os.environ.get("VKG_OFFLINE", "").lower()
    if env in ("1", "true", "yes"):
        return True
    if env in ("0", "false", "no"):
        return False
    # Auto-detect
    return not _check_network()


def _check_network(timeout: float = 3.0) -> bool:
    """
    Check if network is available.

    Uses DNS to 8.8.8.8 as reliable indicator.
    """
    try:
        socket.create_connection(("8.8.8.8", 53), timeout=timeout)
        return True
    except OSError:
        return False


def get_network_status() -> NetworkStatus:
    """Get current network status."""
    env = os.environ.get("VKG_OFFLINE", "").lower()
    network_available = _check_network()

    if env in ("1", "true", "yes"):
        return NetworkStatus(
            mode=NetworkMode.FORCED_OFFLINE,
            network_available=network_available,
            reason="User set VKG_OFFLINE=1",
        )

    if env in ("0", "false", "no"):
        if not network_available:
            logger.warning("VKG_OFFLINE=0 but network unavailable")
        return NetworkStatus(
            mode=NetworkMode.ONLINE if network_available else NetworkMode.OFFLINE,
            network_available=network_available,
            reason="User set VKG_OFFLINE=0" if network_available else "Network unavailable despite VKG_OFFLINE=0",
        )

    # Auto mode
    if network_available:
        return NetworkStatus(
            mode=NetworkMode.ONLINE,
            network_available=True,
            reason="Network detected (auto)",
        )
    else:
        return NetworkStatus(
            mode=NetworkMode.OFFLINE,
            network_available=False,
            reason="No network detected (auto)",
        )


def require_network(operation: str) -> None:
    """
    Raise error if network is required but unavailable.

    Args:
        operation: Name of the operation requiring network

    Raises:
        RuntimeError: If offline mode is enabled
    """
    status = get_network_status()
    if not status.can_use_tier_b:
        raise RuntimeError(
            f"Operation '{operation}' requires network access.\n"
            f"Current mode: {status.mode.value} ({status.reason})\n\n"
            f"Options:\n"
            f"  - Check network connection\n"
            f"  - Use --tier-a-only flag\n"
            f"  - Set VKG_OFFLINE=0 to force online mode\n"
        )


def offline_capable(func):
    """
    Decorator marking a function as offline-capable.

    This is documentation - the function works without network.
    """
    func._offline_capable = True
    return func


def requires_network(func):
    """
    Decorator marking a function as requiring network.

    Will raise error if called in offline mode.
    """
    import functools

    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        require_network(func.__name__)
        return func(*args, **kwargs)

    wrapper._requires_network = True
    return wrapper


class OfflineContext:
    """Context manager for offline-aware operations."""

    def __init__(self, operation: str):
        self.operation = operation
        self.status = get_network_status()

    def __enter__(self):
        if self.status.mode == NetworkMode.OFFLINE:
            logger.info(f"Running {self.operation} in offline mode")
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        pass

    @property
    def is_offline(self) -> bool:
        return self.status.mode in (NetworkMode.OFFLINE, NetworkMode.FORCED_OFFLINE)

    def require_tier_b(self):
        """Raise if Tier B is not available."""
        if not self.status.can_use_tier_b:
            raise RuntimeError(
                f"Tier B analysis requires network.\n"
                f"Current: {self.status.mode.value}\n"
                f"Use --tier-a-only or check network."
            )


def format_offline_notice() -> str:
    """Format notice for offline mode."""
    status = get_network_status()

    if status.mode == NetworkMode.ONLINE:
        return ""

    lines = [
        "Note: Running in offline mode",
        f"  Reason: {status.reason}",
        "",
        "Available (Tier A):",
    ]
    for cap in TIER_A_CAPABILITIES:
        lines.append(f"  [OK] {cap}")

    lines.append("")
    lines.append("Unavailable (Tier B - requires network):")
    for cap in TIER_B_CAPABILITIES:
        lines.append(f"  [!] {cap}")

    lines.append("")
    lines.append("Use: VKG_OFFLINE=0 vkg ... (to force online)")

    return "\n".join(lines)
```

```python
"""
Network Detection

Reliable network availability checking.
"""

import socket
import urllib.request
from typing import Optional, Tuple
import logging

logger = logging.getLogger(__name__)


class NetworkDetector:
    """Detects network availability."""

    # Endpoints to try (in order)
    CHECK_ENDPOINTS = [
        ("8.8.8.8", 53),      # Google DNS
        ("1.1.1.1", 53),      # Cloudflare DNS
        ("208.67.222.222", 53),  # OpenDNS
    ]

    def __init__(self, timeout: float = 3.0):
        self.timeout = timeout
        self._cache: Optional[bool] = None

    def is_available(self, use_cache: bool = True) -> bool:
        """
        Check if network is available.

        Args:
            use_cache: Use cached result if available

        Returns:
            True if network is available
        """
        if use_cache and self._cache is not None:
            return self._cache

        result = self._check()
        self._cache = result
        return result

    def _check(self) -> bool:
        """Perform network check."""
        for host, port in self.CHECK_ENDPOINTS:
            try:
                socket.create_connection((host, port), timeout=self.timeout)
                logger.debug(f"Network available (reached {host}:{port})")
                return True
            except OSError:
                continue

        logger.debug("Network unavailable (all endpoints failed)")
        return False

    def check_llm_api(self, provider: str) -> Tuple[bool, str]:
        """
        Check if LLM API is reachable.

        Args:
            provider: Provider name (anthropic, openai, google)

        Returns:
            Tuple of (available, message)
        """
        api_urls = {
            "anthropic": "https://api.anthropic.com/",
            "openai": "https://api.openai.com/",
            "google": "https://generativelanguage.googleapis.com/",
        }

        url = api_urls.get(provider)
        if not url:
            return False, f"Unknown provider: {provider}"

        try:
            req = urllib.request.Request(url, method="HEAD")
            urllib.request.urlopen(req, timeout=self.timeout)
            return True, "API reachable"
        except urllib.request.URLError as e:
            return False, str(e)
        except Exception as e:
            return False, f"Check failed: {e}"

    def clear_cache(self):
        """Clear cached network status."""
        self._cache = None


# Global detector instance
_detector = NetworkDetector()


def is_network_available() -> bool:
    """Check if network is available."""
    return _detector.is_available()


def check_api_connectivity(provider: str) -> Tuple[bool, str]:
    """Check API connectivity for provider."""
    return _detector.check_llm_api(provider)
```
