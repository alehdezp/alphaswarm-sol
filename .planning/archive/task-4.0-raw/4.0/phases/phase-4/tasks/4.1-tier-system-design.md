### Tier Definitions

```python
# File: src/true_vkg/testing/tiers.py

from enum import Enum
from dataclasses import dataclass
from typing import Optional, List

class TestTier(Enum):
    """Test generation quality tiers with realistic expectations."""

    TIER_1_TEMPLATE = 1   # Always works, needs manual completion
    TIER_2_SMART = 2      # Attempts compilation, 30-40% success (REALISTIC)
    TIER_3_COMPLETE = 3   # Full test (aspirational, <10% success)

@dataclass
class TierDefinition:
    """Definition of what each tier provides."""
    tier: TestTier
    description: str
    success_rate_target: float  # Realistic target
    success_rate_minimum: float  # Minimum acceptable
    provides: List[str]
    does_not_provide: List[str]

TIER_DEFINITIONS = {
    TestTier.TIER_1_TEMPLATE: TierDefinition(
        tier=TestTier.TIER_1_TEMPLATE,
        description="Template with TODO comments - always generates",
        success_rate_target=1.0,  # 100% - always produces output
        success_rate_minimum=1.0,
        provides=[
            "Test file structure",
            "Vulnerability description in comments",
            "Attack vector explanation",
            "TODO markers for manual completion",
            "Suggested assertion patterns",
        ],
        does_not_provide=[
            "Correct import paths",
            "Compilable code",
            "Working test logic",
        ]
    ),
    TestTier.TIER_2_SMART: TierDefinition(
        tier=TestTier.TIER_2_SMART,
        description="Smart template with resolved imports - 30-40% compile rate",
        success_rate_target=0.40,  # 40% - REALISTIC, not 60%
        success_rate_minimum=0.25,  # 25% minimum
        provides=[
            "Everything from Tier 1",
            "Attempted import resolution",
            "Pragma version matching",
            "Basic setup() structure",
        ],
        does_not_provide=[
            "Guaranteed compilation",
            "Correct test logic",
            "Full attack implementation",
        ]
    ),
    TestTier.TIER_3_COMPLETE: TierDefinition(
        tier=TestTier.TIER_3_COMPLETE,
        description="Complete test - aspirational, <10% success",
        success_rate_target=0.10,  # 10% - highly aspirational
        success_rate_minimum=0.05,
        provides=[
            "Everything from Tier 2",
            "Attack implementation attempt",
            "Assertion logic",
        ],
        does_not_provide=[
            "Guaranteed correctness",
            "Coverage of complex scenarios",
        ]
    ),
}
```

### Fallback Strategy

```python
# Fallback is CRITICAL - Tier 2 failure MUST produce Tier 1

def generate_test_scaffold(finding: Finding, target_tier: TestTier) -> TestScaffold:
    """Generate test scaffold with automatic fallback."""

    if target_tier == TestTier.TIER_3_COMPLETE:
        try:
            return _generate_tier3(finding)
        except Exception as e:
            logger.warning(f"Tier 3 failed: {e}, falling back to Tier 2")
            target_tier = TestTier.TIER_2_SMART

    if target_tier == TestTier.TIER_2_SMART:
        try:
            return _generate_tier2(finding)
        except Exception as e:
            logger.warning(f"Tier 2 failed: {e}, falling back to Tier 1")
            target_tier = TestTier.TIER_1_TEMPLATE

    # Tier 1 MUST NOT FAIL
    return _generate_tier1(finding)
```
