def generate_tier1_scaffold(finding: Finding) -> TestScaffold:
    """Generate Tier 1 test scaffold.

    MUST NOT FAIL. Always produces output.

    Args:
        finding: The finding to generate test for

    Returns:
        TestScaffold with template content
    """
    # Safe extraction - handle missing data gracefully
    finding_id = finding.id or "UNKNOWN"
    finding_id_safe = _make_safe_identifier(finding_id)
    category_safe = _make_safe_identifier(finding.category or "unknown")
    test_contract_name = f"Test_{finding_id_safe}_{category_safe}"

    # Extract location info
    location = finding.location
    contract_path = location.file if location else "unknown"
    function_name = location.function if location else "unknownFunction"
    line_number = location.line if location else 0

    # Build attack steps from category
    attack_steps = _generate_attack_steps(finding)

    # Generate content
    content = TIER1_TEMPLATE.format(
        timestamp=datetime.now().isoformat(),
        finding_id=finding_id,
        finding_title=finding.title or "Unknown Vulnerability",
        severity=finding.severity.value if finding.severity else "unknown",
        category=finding.category or "unknown",
        category_safe=category_safe,
        location=f"{contract_path}:{function_name}:{line_number}",
        description=_indent_text(finding.description or "No description provided", 4),
        recommendation=_indent_text(finding.recommendation or "No recommendation provided", 4),
        contract_path=contract_path,
        test_contract_name=test_contract_name,
        target_contract_type=_extract_contract_name(contract_path),
        function_name=function_name,
        finding_id_safe=finding_id_safe,
        attack_steps=attack_steps,
    )

    # Generate filename
    filename = f"{test_contract_name}.t.sol"

    return TestScaffold(
        finding_id=finding_id,
        tier=1,
        content=content,
        filename=filename,
    )

def _make_safe_identifier(s: str) -> str:
    """Convert string to safe Solidity identifier."""
    # Replace non-alphanumeric with underscore
    safe = "".join(c if c.isalnum() else "_" for c in s)
    # Remove consecutive underscores
    while "__" in safe:
        safe = safe.replace("__", "_")
    # Remove leading/trailing underscores
    safe = safe.strip("_")
    # Ensure doesn't start with number
    if safe and safe[0].isdigit():
        safe = "X" + safe
    return safe or "Unknown"

def _extract_contract_name(path: str) -> str:
    """Extract contract name from file path."""
    if not path:
        return "TargetContract"
    # Get filename without extension
    name = Path(path).stem
    return name if name else "TargetContract"

def _indent_text(text: str, spaces: int) -> str:
    """Indent multi-line text."""
    prefix = " " * spaces
    lines = text.split("\n")
    return f"\n{prefix}// ".join(lines)

def _generate_attack_steps(finding: Finding) -> str:
    """Generate attack step comments based on category."""
    category = (finding.category or "").lower()

    attack_patterns = {
        "reentrancy": '''// Attack Pattern: Reentrancy
        // 1. Attacker deploys contract with malicious receive()/fallback()
        // 2. Attacker calls vulnerable withdraw() function
        // 3. Before state is updated, attacker re-enters via callback
        // 4. Attacker drains funds by repeated re-entry''',

        "access_control": '''// Attack Pattern: Missing Access Control
        // 1. Identify privileged function without access check
        // 2. Call function as non-privileged user
        // 3. Verify state was modified (should have been rejected)''',

        "oracle": '''// Attack Pattern: Oracle Manipulation
        // 1. Manipulate oracle price (flash loan, sandwich, etc.)
        // 2. Call function that relies on manipulated price
        // 3. Profit from price discrepancy''',

        "mev": '''// Attack Pattern: MEV Vulnerability
        // 1. Monitor mempool for target transaction
        // 2. Front-run with price manipulation
        // 3. Execute original transaction at worse price
        // 4. Back-run to extract profit''',

        "dos": '''// Attack Pattern: Denial of Service
        // 1. Identify resource consumption vector (loops, storage, external calls)
        // 2. Trigger resource exhaustion
        // 3. Verify function becomes unusable''',
    }

    # Find matching pattern or use generic
    for key, pattern in attack_patterns.items():
        if key in category:
            return pattern

    return '''// Attack Pattern: Generic
        // 1. Identify preconditions for vulnerability
        // 2. Execute attack transaction
        // 3. Verify vulnerability was exploited'''
