```python
"""Project structure detection for test scaffold generation."""

from __future__ import annotations

import json
from dataclasses import dataclass, field
from enum import Enum
from pathlib import Path
from typing import Dict, List, Optional, Tuple
import tomllib  # Python 3.11+ built-in

class ProjectType(Enum):
    FOUNDRY = "foundry"
    HARDHAT = "hardhat"
    BROWNIE = "brownie"
    UNKNOWN = "unknown"

@dataclass
class ProjectConfig:
    """Detected project configuration."""
    project_type: ProjectType
    root: Path
    src_dir: Path  # Usually src/ or contracts/
    test_dir: Path  # Usually test/ or tests/
    remappings: Dict[str, str] = field(default_factory=dict)
    solc_version: Optional[str] = None
    dependencies: List[str] = field(default_factory=list)

def detect_project_structure(root: Path) -> ProjectConfig:
    """Detect project type and configuration.

    Priority order:
    1. foundry.toml -> FOUNDRY
    2. hardhat.config.* -> HARDHAT
    3. brownie-config.yaml -> BROWNIE
    4. package.json with hardhat dep -> HARDHAT
    5. UNKNOWN

    Args:
        root: Project root directory

    Returns:
        ProjectConfig with detected settings
    """
    root = Path(root).resolve()

    # 1. Check for Foundry
    foundry_toml = root / "foundry.toml"
    if foundry_toml.exists():
        return _parse_foundry_config(root, foundry_toml)

    # 2. Check for Hardhat
    for config_name in ["hardhat.config.js", "hardhat.config.ts"]:
        if (root / config_name).exists():
            return _parse_hardhat_config(root)

    # 3. Check for Brownie
    brownie_config = root / "brownie-config.yaml"
    if brownie_config.exists():
        return _parse_brownie_config(root, brownie_config)

    # 4. Check package.json for hardhat
    pkg_json = root / "package.json"
    if pkg_json.exists():
        try:
            pkg = json.loads(pkg_json.read_text())
            all_deps = {
                **pkg.get("dependencies", {}),
                **pkg.get("devDependencies", {})
            }
            if "hardhat" in all_deps:
                return _parse_hardhat_config(root)
        except (json.JSONDecodeError, KeyError):
            pass

    # 5. Unknown - return best-effort defaults
    return _unknown_project(root)

def _parse_foundry_config(root: Path, config_path: Path) -> ProjectConfig:
    """Parse Foundry project configuration."""
    remappings: Dict[str, str] = {}

    try:
        config = tomllib.loads(config_path.read_text())
        profile = config.get("profile", {}).get("default", {})

        # Get remappings from foundry.toml
        raw_remappings = profile.get("remappings", [])
        for mapping in raw_remappings:
            if "=" in mapping:
                prefix, target = mapping.split("=", 1)
                remappings[prefix] = target

        # Get source and test directories
        src_dir = root / profile.get("src", "src")
        test_dir = root / profile.get("test", "test")
        solc_version = profile.get("solc_version")

    except Exception:
        src_dir = root / "src"
        test_dir = root / "test"
        solc_version = None

    # Also check remappings.txt
    remappings_txt = root / "remappings.txt"
    if remappings_txt.exists():
        for line in remappings_txt.read_text().splitlines():
            line = line.strip()
            if "=" in line and not line.startswith("#"):
                prefix, target = line.split("=", 1)
                remappings[prefix] = target

    # Detect dependencies from lib/
    dependencies = []
    lib_dir = root / "lib"
    if lib_dir.exists():
        dependencies = [d.name for d in lib_dir.iterdir() if d.is_dir()]

    return ProjectConfig(
        project_type=ProjectType.FOUNDRY,
        root=root,
        src_dir=src_dir,
        test_dir=test_dir,
        remappings=remappings,
        solc_version=solc_version,
        dependencies=dependencies,
    )

def _parse_hardhat_config(root: Path) -> ProjectConfig:
    """Parse Hardhat project configuration."""
    # Hardhat uses node_modules for imports
    remappings = {
        "@openzeppelin/": "node_modules/@openzeppelin/",
    }

    # Standard Hardhat directories
    contracts_dir = root / "contracts"
    if not contracts_dir.exists():
        contracts_dir = root / "src"

    test_dir = root / "test"
    if not test_dir.exists():
        test_dir = root / "tests"

    return ProjectConfig(
        project_type=ProjectType.HARDHAT,
        root=root,
        src_dir=contracts_dir,
        test_dir=test_dir,
        remappings=remappings,
        dependencies=["@openzeppelin/contracts"],  # Common default
    )

def _parse_brownie_config(root: Path, config_path: Path) -> ProjectConfig:
    """Parse Brownie project configuration."""
    # Brownie has different structure
    return ProjectConfig(
        project_type=ProjectType.BROWNIE,
        root=root,
        src_dir=root / "contracts",
        test_dir=root / "tests",
        remappings={},
    )

def _unknown_project(root: Path) -> ProjectConfig:
    """Return best-effort config for unknown project."""
    # Try to find contracts
    src_dir = root
    for candidate in ["src", "contracts", "."]:
        if (root / candidate).exists():
            src_dir = root / candidate
            break

    test_dir = root / "test"
    if not test_dir.exists():
        test_dir = root / "tests"

    return ProjectConfig(
        project_type=ProjectType.UNKNOWN,
        root=root,
        src_dir=src_dir,
        test_dir=test_dir,
        remappings={},
    )
```

```python
# tests/test_testing_detection.py

import tempfile
from pathlib import Path
from true_vkg.testing.detection import detect_project_structure, ProjectType

def test_detect_foundry_project():
    """Detect Foundry project from foundry.toml."""
    with tempfile.TemporaryDirectory() as tmpdir:
        root = Path(tmpdir)
        (root / "foundry.toml").write_text('[profile.default]\nsrc = "src"\n')
        (root / "src").mkdir()

        config = detect_project_structure(root)
        assert config.project_type == ProjectType.FOUNDRY
        assert config.src_dir == root / "src"

def test_detect_hardhat_project():
    """Detect Hardhat project from config file."""
    with tempfile.TemporaryDirectory() as tmpdir:
        root = Path(tmpdir)
        (root / "hardhat.config.js").write_text("module.exports = {}")
        (root / "contracts").mkdir()

        config = detect_project_structure(root)
        assert config.project_type == ProjectType.HARDHAT

def test_detect_hardhat_from_package_json():
    """Detect Hardhat from package.json dependency."""
    with tempfile.TemporaryDirectory() as tmpdir:
        root = Path(tmpdir)
        (root / "package.json").write_text('{"devDependencies": {"hardhat": "^2.0"}}')

        config = detect_project_structure(root)
        assert config.project_type == ProjectType.HARDHAT

def test_detect_unknown_project():
    """Return UNKNOWN for unrecognized project."""
    with tempfile.TemporaryDirectory() as tmpdir:
        root = Path(tmpdir)
        (root / "Contract.sol").write_text("// solidity")

        config = detect_project_structure(root)
        assert config.project_type == ProjectType.UNKNOWN

def test_foundry_remappings_parsed():
    """Parse remappings from foundry.toml."""
    with tempfile.TemporaryDirectory() as tmpdir:
        root = Path(tmpdir)
        (root / "foundry.toml").write_text(
            '[profile.default]\nremappings = ["@oz/=lib/openzeppelin/"]\n'
        )

        config = detect_project_structure(root)
        assert "@oz/" in config.remappings
        assert config.remappings["@oz/"] == "lib/openzeppelin/"
```
