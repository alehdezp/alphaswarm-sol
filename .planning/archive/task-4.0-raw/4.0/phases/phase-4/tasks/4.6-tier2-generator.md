## Objective

Generate "smart" test templates that attempt import resolution and pragma matching. **REALISTIC TARGET: 30-40% compile rate**, not 60%.

## CRITICAL CONTEXT

### Why 30-40% Not 60%

The original TRACKER claimed 60% compile rate. This is unrealistic because:
1. Import resolution is fragile (depends on project setup)
2. Pragma conflicts are common
3. Missing dependencies cannot be auto-installed
4. Complex inheritance chains break easily

**Honest assessment from original doc's own reflection:**
> "20% of tests will compile AND test the right thing"
> "30% will compile but test wrong thing"
> "50% won't compile at all"

So 30-40% compile rate is already optimistic.

### Tier 2 vs Tier 1

Tier 2 adds:
- Attempted import resolution
- Pragma version matching
- Basic `setUp()` structure
- forge-std import (for Foundry)

Tier 2 STILL requires:
- Manual verification of imports
- Manual attack implementation
- Human judgment on test correctness

"""Tier 2 Smart Template Generator - extension to generator.py"""

from true_vkg.testing.detection import detect_project_structure, ProjectType
from true_vkg.testing.remappings import ImportResolver, extract_imports_from_source

TIER2_TEMPLATE = '''// SPDX-License-Identifier: MIT

@dataclass
class Tier2Result:
    """Result of Tier 2 generation attempt."""
    success: bool
    scaffold: TestScaffold
    warnings: List[str]
    compile_confidence: int  # 0-100

def generate_tier2_scaffold(
    finding: Finding,
    project_root: Path,
) -> Tier2Result:
    """Generate Tier 2 smart scaffold.

    May fail - caller should fall back to Tier 1.

    Args:
        finding: The finding to generate test for
        project_root: Root of the Solidity project

    Returns:
        Tier2Result with scaffold and metadata
    """
    warnings: List[str] = []
    compile_confidence = 100  # Start optimistic, reduce for issues

    # Detect project structure
    try:
        config = detect_project_structure(project_root)
        project_type = config.project_type.value
    except Exception as e:
        warnings.append(f"Project detection failed: {e}")
        config = None
        project_type = "unknown"
        compile_confidence -= 30

    # Determine pragma version
    pragma_version = _detect_pragma_version(finding, config)
    if pragma_version.startswith("^0.8"):
        pass  # Good, common version
    else:
        warnings.append(f"Unusual pragma version: {pragma_version}")
        compile_confidence -= 10

    # Build imports section
    imports_section, import_warnings = _build_imports_section(finding, config)
    warnings.extend(import_warnings)
    compile_confidence -= len(import_warnings) * 10

    # Safe identifiers
    finding_id = finding.id or "UNKNOWN"
    finding_id_safe = _make_safe_identifier(finding_id)
    category_safe = _make_safe_identifier(finding.category or "unknown")
    test_contract_name = f"Test_{finding_id_safe}_{category_safe}"

    # Location info
    location = finding.location
    contract_path = location.file if location else "unknown"
    function_name = location.function if location else "unknownFunction"

    # Clamp confidence
    compile_confidence = max(0, min(100, compile_confidence))

    # Generate content
    content = TIER2_TEMPLATE.format(
        timestamp=datetime.now().isoformat(),
        finding_id=finding_id,
        finding_title=finding.title or "Unknown",
        severity=finding.severity.value if finding.severity else "unknown",
        category=finding.category or "unknown",
        category_safe=category_safe,
        location=f"{contract_path}:{function_name}",
        pragma_version=pragma_version,
        project_type=project_type,
        compile_confidence=compile_confidence,
        imports_section=imports_section,
        test_contract_name=test_contract_name,
        target_contract_type=_extract_contract_name(contract_path),
        function_name=function_name,
        finding_id_safe=finding_id_safe,
        attack_steps=_generate_attack_steps(finding),
    )

    scaffold = TestScaffold(
        finding_id=finding_id,
        tier=2,
        content=content,
        filename=f"{test_contract_name}.t.sol",
    )

    return Tier2Result(
        success=compile_confidence >= 40,  # Consider success if >40% confidence
        scaffold=scaffold,
        warnings=warnings,
        compile_confidence=compile_confidence,
    )

def _detect_pragma_version(
    finding: Finding,
    config: Optional[ProjectConfig],
) -> str:
    """Detect appropriate pragma version."""
    # Try to get from finding's source file
    if finding.location and finding.location.file:
        try:
            source_path = Path(finding.location.file)
            if source_path.exists():
                content = source_path.read_text()
                # Extract pragma from source
                import re
                match = re.search(r'pragma solidity\s+([^;]+);', content)
                if match:
                    return match.group(1).strip()
        except Exception:
            pass

    # Try from project config
    if config and config.solc_version:
        return f"^{config.solc_version}"

    # Default to safe version
    return "^0.8.20"

def _build_imports_section(
    finding: Finding,
    config: Optional[ProjectConfig],
) -> Tuple[str, List[str]]:
    """Build imports section with resolution attempts."""
    warnings: List[str] = []
    imports: List[str] = []

    # Always need forge-std for Foundry
    if config and config.project_type == ProjectType.FOUNDRY:
        imports.append('import "forge-std/Test.sol";')
    else:
        imports.append('// TODO: Add test framework import')
        imports.append('// import "forge-std/Test.sol";')
        warnings.append("Non-Foundry project - test framework import unclear")

    # Try to resolve target contract import
    if finding.location and finding.location.file:
        target_file = finding.location.file

        if config:
            resolver = ImportResolver(config)

            # Calculate relative path from test dir to source
            try:
                source_path = Path(target_file)
                test_dir = config.test_dir
                rel_path = _calculate_relative_import(source_path, test_dir)
                imports.append(f'import "{rel_path}";')
            except Exception as e:
                imports.append(f'// TODO: Cannot resolve import for {target_file}')
                imports.append(f'// import "{target_file}";')
                warnings.append(f"Import resolution failed: {e}")
        else:
            imports.append(f'// TODO: Adjust import path')
            imports.append(f'// import "{target_file}";')
            warnings.append("No project config - cannot resolve imports")

    return "\n".join(imports), warnings

def _calculate_relative_import(source: Path, test_dir: Path) -> str:
    """Calculate relative import path from test dir to source."""
    # This is a simplified version - real implementation would be more robust
    try:
        # Get relative path from test_dir to source
        rel = os.path.relpath(source, test_dir)
        # Ensure it starts with ../ or ./
        if not rel.startswith("."):
            rel = "./" + rel
        return rel
    except ValueError:
        # Different drives on Windows, etc.
        return str(source)
