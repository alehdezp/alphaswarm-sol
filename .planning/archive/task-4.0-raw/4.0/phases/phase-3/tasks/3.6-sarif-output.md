## CRITICAL CONTEXT

**Why this task matters:**
SARIF (Static Analysis Results Interchange Format) is THE industry standard for static analysis output. GitHub, GitLab, VS Code, and many other tools consume SARIF. Without SARIF output, BSKG findings cannot integrate into existing security workflows.

**What success looks like:**
1. User runs `vkg report --format sarif`
2. Output uploads to GitHub Security tab
3. GitHub shows BSKG findings in code review
4. Developers see inline annotations in PRs

**What failure looks like:**
- BSKG produces custom JSON nobody can use
- Manual copying of findings to issues
- No IDE integration
- Security workflow friction

## 5. IMPLEMENTATION

### Step 1: Create SARIF Generator Module (2h)

```python
# src/true_vkg/report/sarif.py

from __future__ import annotations

import json
from datetime import datetime
from pathlib import Path
from typing import Any

from true_vkg.findings.model import Finding, Severity
from true_vkg.patterns.loader import load_pattern


# Severity to SARIF level mapping
SEVERITY_TO_LEVEL = {
    Severity.CRITICAL: "error",
    Severity.HIGH: "error",
    Severity.MEDIUM: "warning",
    Severity.LOW: "note",
    Severity.INFO: "none",
}


class SarifGenerator:
    """Generate SARIF 2.1.0 compliant reports."""

    SCHEMA_URI = "https://raw.githubusercontent.com/oasis-tcs/sarif-spec/master/Schemata/sarif-schema-2.1.0.json"
    VERSION = "2.1.0"

    def __init__(
        self,
        tool_name: str = "VKG",
        tool_version: str = "4.0.0",
        info_uri: str = "https://github.com/your-org/vkg",
    ):
        self.tool_name = tool_name
        self.tool_version = tool_version
        self.info_uri = info_uri
        self._rules: dict[str, dict] = {}
        self._rule_indices: dict[str, int] = {}

    def generate(
        self,
        findings: list[Finding],
        source_root: Path,
        start_time: datetime | None = None,
        end_time: datetime | None = None,
    ) -> dict[str, Any]:
        """Generate SARIF document from findings."""

        # Build rules from patterns referenced in findings
        rules = []
        for finding in findings:
            if finding.pattern_id not in self._rule_indices:
                rule = self._build_rule(finding.pattern_id)
                self._rule_indices[finding.pattern_id] = len(rules)
                rules.append(rule)

        # Build results
        results = [self._build_result(f, source_root) for f in findings]

        # Build invocation
        invocation = {
            "executionSuccessful": True,
        }
        if start_time:
            invocation["startTimeUtc"] = start_time.isoformat() + "Z"
        if end_time:
            invocation["endTimeUtc"] = end_time.isoformat() + "Z"

        return {
            "$schema": self.SCHEMA_URI,
            "version": self.VERSION,
            "runs": [
                {
                    "tool": {
                        "driver": {
                            "name": self.tool_name,
                            "version": self.tool_version,
                            "semanticVersion": self.tool_version,
                            "informationUri": self.info_uri,
                            "rules": rules,
                        }
                    },
                    "results": results,
                    "invocations": [invocation],
                    "originalUriBaseIds": {
                        "%SRCROOT%": {
                            "uri": source_root.as_uri() + "/"
                        }
                    },
                }
            ],
        }

    def _build_rule(self, pattern_id: str) -> dict[str, Any]:
        """Build SARIF rule from pattern definition."""
        try:
            pattern = load_pattern(pattern_id)
            return {
                "id": pattern_id,
                "name": self._to_pascal_case(pattern_id),
                "shortDescription": {
                    "text": pattern.get("name", pattern_id)
                },
                "fullDescription": {
                    "text": pattern.get("description", "")
                },
                "helpUri": f"https://vkg.dev/rules/{pattern_id}",
                "defaultConfiguration": {
                    "level": SEVERITY_TO_LEVEL.get(
                        Severity(pattern.get("severity", "medium")),
                        "warning"
                    )
                },
                "properties": {
                    "tags": pattern.get("tags", ["security"]),
                    "precision": "high" if pattern.get("confidence", 0) > 0.8 else "medium",
                }
            }
        except Exception:
            # Fallback for unknown patterns
            return {
                "id": pattern_id,
                "name": self._to_pascal_case(pattern_id),
                "shortDescription": {"text": pattern_id},
                "defaultConfiguration": {"level": "warning"},
            }

    def _build_result(self, finding: Finding, source_root: Path) -> dict[str, Any]:
        """Build SARIF result from finding."""
        return {
            "ruleId": finding.pattern_id,
            "ruleIndex": self._rule_indices.get(finding.pattern_id, 0),
            "level": SEVERITY_TO_LEVEL.get(finding.severity, "warning"),
            "message": {
                "text": finding.description or f"{finding.pattern_id} detected in {finding.location.function}"
            },
            "locations": [
                {
                    "physicalLocation": {
                        "artifactLocation": {
                            "uri": finding.location.file,
                            "uriBaseId": "%SRCROOT%"
                        },
                        "region": {
                            "startLine": finding.location.line,
                            "startColumn": finding.location.column or 1,
                        }
                    }
                }
            ],
            "fingerprints": {
                "vkg/v1": finding.graph_fingerprint or ""
            },
            "properties": {
                "vkg_finding_id": finding.id,
                "behavioral_signature": finding.behavioral_signature,
                "confidence": finding.confidence,
                "tier": finding.tier.value,
            }
        }

    @staticmethod
    def _to_pascal_case(s: str) -> str:
        """Convert kebab-case to PascalCase."""
        return "".join(word.capitalize() for word in s.split("-"))


def generate_sarif(
    findings: list[Finding],
    source_root: Path,
    output_path: Path | None = None,
) -> str:
    """Convenience function to generate SARIF."""
    generator = SarifGenerator()
    sarif = generator.generate(findings, source_root)
    sarif_json = json.dumps(sarif, indent=2)

    if output_path:
        output_path.write_text(sarif_json)

    return sarif_json
```

### Step 3: Validate SARIF Output (1h)

```python
# src/true_vkg/report/validator.py

import json
import jsonschema
from pathlib import Path


SARIF_SCHEMA_URL = "https://raw.githubusercontent.com/oasis-tcs/sarif-spec/master/Schemata/sarif-schema-2.1.0.json"


def validate_sarif(sarif_path: Path) -> tuple[bool, list[str]]:
    """Validate SARIF file against schema."""
    import requests

    # Load SARIF
    sarif = json.loads(sarif_path.read_text())

    # Fetch schema (cache in production)
    schema = requests.get(SARIF_SCHEMA_URL).json()

    errors = []
    try:
        jsonschema.validate(sarif, schema)
        return True, []
    except jsonschema.ValidationError as e:
        errors.append(str(e.message))
        return False, errors


# CLI command
@click.command()
@click.argument("sarif_path", type=click.Path(exists=True))
def validate_output(sarif_path: str):
    """Validate SARIF output file."""
    valid, errors = validate_sarif(Path(sarif_path))
    if valid:
        click.echo(click.style("Valid SARIF 2.1.0", fg="green"))
    else:
        click.echo(click.style("Invalid SARIF", fg="red"))
        for err in errors:
            click.echo(f"  {err}")
        raise SystemExit(1)
```
