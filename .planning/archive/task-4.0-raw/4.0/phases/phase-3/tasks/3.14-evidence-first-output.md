## CRITICAL CONTEXT

**Why this task matters:**
A finding without evidence is useless. "Reentrancy vulnerability in withdraw" tells the auditor NOTHING about WHY it's a vulnerability. Evidence-first output means every finding includes:
1. Behavioral signature (operation sequence)
2. Code snippets with line references
3. Properties that triggered the detection
4. Minimal reproduction steps

**What success looks like:**
```
Finding VKG-001: Reentrancy in Vault.withdraw

Behavioral Signature: R:bal -> X:out -> W:bal
  (Read balance, External call, Write balance)

Evidence:
  1. [code] Vault.sol:45-52 - withdraw function
  2. [property] state_write_after_external_call = true
  3. [call_graph] withdraw -> transfer (external)

Why Vulnerable:
  External call at line 48 transfers ETH before balance
  is decremented at line 51. Attacker can re-enter withdraw
  and drain funds.

Minimal Reproduction:
  1. Deploy attacker contract with receive() that calls withdraw
  2. Deposit 1 ETH to Vault
  3. Call withdraw(1 ETH)
  4. receive() triggers, calls withdraw again
  5. Balance not yet decremented, second withdraw succeeds
  6. Attacker extracts 2 ETH with 1 ETH deposit
```

**What failure looks like:**
- Finding says "reentrancy detected" with no context
- Auditor has to manually trace code to understand
- LLM cannot verify finding without reading whole contract
- 50% of time spent on investigation that should be instant

---

## 1. OBJECTIVE

Implement evidence-first output format that provides complete context for every finding, enabling immediate verification without additional code reading.

### Step 1: Update Evidence Model (1h)

```python
# src/true_vkg/findings/evidence.py

from __future__ import annotations

from dataclasses import dataclass
from enum import Enum
from typing import Any


class EvidenceType(Enum):
    CODE = "code"
    PROPERTY = "property"
    CALL_GRAPH = "call_graph"
    OPERATION = "operation"
    GUARD = "guard"
    PATH = "path"


@dataclass
class Evidence:
    """Evidence supporting a finding."""
    id: str           # EVD-001
    type: EvidenceType
    summary: str      # Human-readable summary
    data: dict[str, Any]  # Type-specific data

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "type": self.type.value,
            "summary": self.summary,
            **self.data,
        }

    @classmethod
    def code_snippet(
        cls,
        id: str,
        file: str,
        lines: tuple[int, int],
        snippet: str,
    ) -> Evidence:
        return cls(
            id=id,
            type=EvidenceType.CODE,
            summary=f"{file}:{lines[0]}-{lines[1]}",
            data={
                "file": file,
                "lines": list(lines),
                "snippet": snippet,
            }
        )

    @classmethod
    def property(
        cls,
        id: str,
        name: str,
        value: Any,
        node_id: str,
    ) -> Evidence:
        return cls(
            id=id,
            type=EvidenceType.PROPERTY,
            summary=f"{name} = {value}",
            data={
                "name": name,
                "value": value,
                "node_id": node_id,
            }
        )

    @classmethod
    def call_graph(
        cls,
        id: str,
        from_func: str,
        to_func: str,
        edge_type: str,
        risk: str = "medium",
    ) -> Evidence:
        return cls(
            id=id,
            type=EvidenceType.CALL_GRAPH,
            summary=f"{from_func} -> {to_func} ({edge_type})",
            data={
                "from": from_func,
                "to": to_func,
                "edge_type": edge_type,
                "risk": risk,
            }
        )

    @classmethod
    def operation(
        cls,
        id: str,
        operation: str,
        node_id: str,
        sequence_position: int,
    ) -> Evidence:
        return cls(
            id=id,
            type=EvidenceType.OPERATION,
            summary=f"{operation} at position {sequence_position}",
            data={
                "operation": operation,
                "node_id": node_id,
                "sequence_position": sequence_position,
            }
        )


class EvidenceBuilder:
    """Builds evidence from graph data."""

    def __init__(self):
        self._next_id = 1

    def _gen_id(self) -> str:
        id = f"EVD-{self._next_id:03d}"
        self._next_id += 1
        return id

    def from_graph_node(
        self,
        node: FunctionNode,
        graph: KnowledgeGraph,
    ) -> list[Evidence]:
        """Extract evidence from graph node."""
        evidence = []

        # 1. Code snippet
        if node.source_code:
            evidence.append(Evidence.code_snippet(
                self._gen_id(),
                node.file,
                (node.start_line, node.end_line),
                node.source_code,
            ))

        # 2. Relevant properties
        for prop_name, prop_value in node.properties.items():
            if self._is_relevant_property(prop_name, prop_value):
                evidence.append(Evidence.property(
                    self._gen_id(),
                    prop_name,
                    prop_value,
                    node.id,
                ))

        # 3. External calls
        for edge in graph.get_edges_from(node.id):
            if edge.type == "external_call":
                evidence.append(Evidence.call_graph(
                    self._gen_id(),
                    node.name,
                    edge.target,
                    edge.type,
                    edge.risk,
                ))

        # 4. Operations
        if node.operations:
            for i, op in enumerate(node.operations):
                evidence.append(Evidence.operation(
                    self._gen_id(),
                    op.name,
                    node.id,
                    i,
                ))

        return evidence

    def _is_relevant_property(self, name: str, value: Any) -> bool:
        """Filter to security-relevant properties."""
        relevant = {
            "state_write_after_external_call",
            "has_reentrancy_guard",
            "has_access_gate",
            "reads_oracle_price",
            "has_staleness_check",
            "visibility",
            "transfers_value",
        }
        return name in relevant and value
```

### Step 2: Generate Behavioral Signature (1h)

```python
# src/true_vkg/findings/signature.py

from true_vkg.kg.sequencing import OperationSequence


OPERATION_SHORTHAND = {
    "READS_USER_BALANCE": "R:bal",
    "WRITES_USER_BALANCE": "W:bal",
    "TRANSFERS_VALUE_OUT": "X:out",
    "CALLS_EXTERNAL": "X:call",
    "CHECKS_PERMISSION": "C:perm",
    "MODIFIES_OWNER": "W:own",
    "READS_ORACLE": "R:orc",
}


def generate_behavioral_signature(operations: list[str]) -> str:
    """Generate compact behavioral signature from operations.

    Example: ["READS_USER_BALANCE", "TRANSFERS_VALUE_OUT", "WRITES_USER_BALANCE"]
    Returns: "R:bal->X:out->W:bal"
    """
    shorthand = [
        OPERATION_SHORTHAND.get(op, op[:3])
        for op in operations
    ]
    return "->".join(shorthand)


def explain_signature(signature: str) -> str:
    """Provide human-readable explanation of signature.

    Example: "R:bal->X:out->W:bal"
    Returns: "Read balance, External call out, Write balance"
    """
    explanations = {
        "R:bal": "Read balance",
        "W:bal": "Write balance",
        "X:out": "External call (value out)",
        "X:call": "External call",
        "C:perm": "Check permission",
        "W:own": "Write owner",
        "R:orc": "Read oracle",
    }

    parts = signature.split("->")
    explained = [explanations.get(p, p) for p in parts]
    return ", ".join(explained)
```

# src/true_vkg/findings/generator.py

from true_vkg.findings.model import Finding
from true_vkg.findings.evidence import EvidenceBuilder
from true_vkg.findings.signature import generate_behavioral_signature


class FindingGenerator:
    """Generate findings with evidence from pattern matches."""

    def __init__(self, graph: KnowledgeGraph):
        self.graph = graph
        self.evidence_builder = EvidenceBuilder()

    def generate_finding(
        self,
        pattern_match: PatternMatch,
    ) -> Finding:
        """Generate finding with full evidence."""

        # Get function node
        node = self.graph.get_node(pattern_match.node_id)

        # Build evidence
        evidence = self.evidence_builder.from_graph_node(node, self.graph)

        # Generate behavioral signature
        operations = [op.name for op in node.operations]
        signature = generate_behavioral_signature(operations)

        # Get pattern metadata
        pattern = pattern_match.pattern

        return Finding(
            id="",  # Will be assigned by store
            pattern_id=pattern.id,
            tier=Tier.TIER_A,
            severity=Severity(pattern.severity),
            location=Location(
                file=node.file,
                line=node.start_line,
                column=node.start_column,
                function=node.name,
                contract=node.contract,
            ),
            behavioral_signature=signature,
            description=self._generate_description(pattern, node),
            evidence=evidence,
            confidence=pattern_match.confidence,
            graph_fingerprint=self.graph.fingerprint,
            why_vulnerable=pattern.get("why_vulnerable", ""),
            minimal_repro_steps=pattern.get("minimal_repro", []),
            fix_recommendation=pattern.get("fix", ""),
        )

    def _generate_description(
        self,
        pattern: Pattern,
        node: FunctionNode,
    ) -> str:
        """Generate human-readable description."""
        return (
            f"{pattern.name} detected in {node.contract}.{node.name}. "
            f"{pattern.description}"
        )
```

# In CLI

def format_finding_detail(finding: Finding) -> str:
    """Format finding with evidence for display."""
    lines = []

    lines.append(f"Finding {finding.id}: {finding.pattern_id}")
    lines.append(f"Location: {finding.location.file}:{finding.location.line}")
    lines.append("")

    lines.append(f"Behavioral Signature: {finding.behavioral_signature}")
    lines.append(f"  ({explain_signature(finding.behavioral_signature)})")
    lines.append("")

    lines.append("Evidence:")
    for i, evd in enumerate(finding.evidence, 1):
        lines.append(f"  {i}. [{evd.type.value}] {evd.summary}")
    lines.append("")

    if finding.why_vulnerable:
        lines.append("Why Vulnerable:")
        lines.append(f"  {finding.why_vulnerable}")
        lines.append("")

    if finding.minimal_repro_steps:
        lines.append("Minimal Reproduction:")
        for step in finding.minimal_repro_steps:
            lines.append(f"  {step}")
        lines.append("")

    if finding.fix_recommendation:
        lines.append(f"Fix: {finding.fix_recommendation}")

    return "\n".join(lines)
```
