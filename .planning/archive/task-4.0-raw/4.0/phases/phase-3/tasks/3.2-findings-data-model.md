## CRITICAL CONTEXT

**Why this task matters:**
Findings must persist across sessions. An auditor (or LLM) may:
1. Find 10 vulnerabilities in session 1
2. Close session
3. Resume in session 2
4. Need to see the 10 findings with investigation status

Without persistence, every session starts from scratch. Investigation notes are lost.

**What failure looks like:**
- User runs `vkg analyze`, gets 10 findings
- User investigates 5, marks 3 as confirmed
- User closes terminal
- Next session: All progress lost, starts over

---

## 1. OBJECTIVE

Implement a persistent finding data model that:
1. Survives terminal/session closure
2. Tracks investigation status per finding
3. Links findings to specific graph version
4. Supports CRUD operations

# src/true_vkg/findings/model.py

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any
import json
import hashlib


class Severity(Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"
    INFO = "info"


class Status(Enum):
    PENDING = "pending"
    INVESTIGATING = "investigating"
    CONFIRMED = "confirmed"
    REJECTED = "rejected"
    FIXED = "fixed"


class Tier(Enum):
    TIER_A = "tier_a"  # Deterministic detection
    TIER_B = "tier_b"  # LLM-assisted


@dataclass
class Location:
    """Source code location."""
    file: str
    line: int
    column: int = 0
    function: str = ""
    contract: str = ""

    def to_dict(self) -> dict:
        return {
            "file": self.file,
            "line": self.line,
            "column": self.column,
            "function": self.function,
            "contract": self.contract,
        }

    @classmethod
    def from_dict(cls, d: dict) -> Location:
        return cls(**d)


@dataclass
class Evidence:
    """Evidence supporting a finding."""
    type: str  # code, property, call_graph, operation
    ref: str   # Reference ID for Layer 2
    summary: str = ""

    def to_dict(self) -> dict:
        return {"type": self.type, "ref": self.ref, "summary": self.summary}

    @classmethod
    def from_dict(cls, d: dict) -> Evidence:
        return cls(**d)


@dataclass
class Finding:
    """A detected vulnerability finding."""
    id: str                           # VKG-001
    pattern_id: str                   # reentrancy-classic
    tier: Tier                        # tier_a
    severity: Severity                # critical
    location: Location                # Vault.sol:45:9
    behavioral_signature: str         # R:bal->X:out->W:bal
    description: str                  # What was detected
    evidence: list[Evidence] = field(default_factory=list)
    confidence: float = 0.0           # 0.0-1.0

    # Investigation state
    status: Status = Status.PENDING
    notes: list[str] = field(default_factory=list)
    verdict_reason: str = ""

    # Metadata
    graph_fingerprint: str = ""       # Links to graph version
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)

    # Attack info
    why_vulnerable: str = ""
    minimal_repro_steps: list[str] = field(default_factory=list)
    fix_recommendation: str = ""

    def add_note(self, note: str) -> None:
        self.notes.append(note)
        self.updated_at = datetime.now()

    def set_status(self, status: Status, reason: str = "") -> None:
        self.status = status
        if reason:
            self.verdict_reason = reason
        self.updated_at = datetime.now()

    @property
    def is_stale(self) -> bool:
        """Finding is stale if graph has changed."""
        # Implementation: compare with current graph fingerprint
        return False  # Placeholder

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "pattern_id": self.pattern_id,
            "tier": self.tier.value,
            "severity": self.severity.value,
            "location": self.location.to_dict(),
            "behavioral_signature": self.behavioral_signature,
            "description": self.description,
            "evidence": [e.to_dict() for e in self.evidence],
            "confidence": self.confidence,
            "status": self.status.value,
            "notes": self.notes,
            "verdict_reason": self.verdict_reason,
            "graph_fingerprint": self.graph_fingerprint,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat(),
            "why_vulnerable": self.why_vulnerable,
            "minimal_repro_steps": self.minimal_repro_steps,
            "fix_recommendation": self.fix_recommendation,
        }

    @classmethod
    def from_dict(cls, d: dict) -> Finding:
        return cls(
            id=d["id"],
            pattern_id=d["pattern_id"],
            tier=Tier(d["tier"]),
            severity=Severity(d["severity"]),
            location=Location.from_dict(d["location"]),
            behavioral_signature=d.get("behavioral_signature", ""),
            description=d["description"],
            evidence=[Evidence.from_dict(e) for e in d.get("evidence", [])],
            confidence=d.get("confidence", 0.0),
            status=Status(d.get("status", "pending")),
            notes=d.get("notes", []),
            verdict_reason=d.get("verdict_reason", ""),
            graph_fingerprint=d.get("graph_fingerprint", ""),
            created_at=datetime.fromisoformat(d["created_at"]) if "created_at" in d else datetime.now(),
            updated_at=datetime.fromisoformat(d["updated_at"]) if "updated_at" in d else datetime.now(),
            why_vulnerable=d.get("why_vulnerable", ""),
            minimal_repro_steps=d.get("minimal_repro_steps", []),
            fix_recommendation=d.get("fix_recommendation", ""),
        )

### 4.2 Finding Store

```python
# src/true_vkg/findings/store.py

from __future__ import annotations

import json
from pathlib import Path
from typing import Iterator

from true_vkg.findings.model import Finding, Status


class FindingStore:
    """Persistent storage for findings."""

    def __init__(self, storage_dir: Path | None = None):
        self.storage_dir = storage_dir or Path.cwd() / ".vkg" / "findings"
        self.storage_dir.mkdir(parents=True, exist_ok=True)
        self._index_path = self.storage_dir / "index.json"
        self._next_id = 1
        self._load_index()

    def _load_index(self) -> None:
        """Load index to determine next ID."""
        if self._index_path.exists():
            index = json.loads(self._index_path.read_text())
            self._next_id = index.get("next_id", 1)

    def _save_index(self) -> None:
        """Save index with next ID."""
        index = {
            "next_id": self._next_id,
            "count": len(list(self.list())),
            "updated_at": datetime.now().isoformat(),
        }
        self._index_path.write_text(json.dumps(index, indent=2))

    def _generate_id(self) -> str:
        """Generate next finding ID."""
        id = f"VKG-{self._next_id:03d}"
        self._next_id += 1
        self._save_index()
        return id

    def _finding_path(self, finding_id: str) -> Path:
        """Get path for finding JSON."""
        return self.storage_dir / f"{finding_id}.json"

    # CRUD Operations

    def create(self, finding: Finding) -> Finding:
        """Create new finding."""
        if not finding.id:
            finding.id = self._generate_id()

        path = self._finding_path(finding.id)
        path.write_text(json.dumps(finding.to_dict(), indent=2))

        return finding

    def read(self, finding_id: str) -> Finding | None:
        """Read finding by ID."""
        path = self._finding_path(finding_id)
        if not path.exists():
            return None
        return Finding.from_dict(json.loads(path.read_text()))

    def update(self, finding: Finding) -> Finding:
        """Update existing finding."""
        finding.updated_at = datetime.now()
        path = self._finding_path(finding.id)
        path.write_text(json.dumps(finding.to_dict(), indent=2))
        return finding

    def delete(self, finding_id: str) -> bool:
        """Delete finding."""
        path = self._finding_path(finding_id)
        if path.exists():
            path.unlink()
            return True
        return False

    # Query Operations

    def list(
        self,
        status: Status | None = None,
        severity: Severity | None = None,
    ) -> Iterator[Finding]:
        """List findings with optional filters."""
        for path in self.storage_dir.glob("VKG-*.json"):
            finding = Finding.from_dict(json.loads(path.read_text()))
            if status and finding.status != status:
                continue
            if severity and finding.severity != severity:
                continue
            yield finding

    def get_by_pattern(self, pattern_id: str) -> list[Finding]:
        """Get all findings for a pattern."""
        return [f for f in self.list() if f.pattern_id == pattern_id]

    def get_pending(self) -> list[Finding]:
        """Get findings pending investigation."""
        return list(self.list(status=Status.PENDING))

    def get_next_priority(self) -> Finding | None:
        """Get next finding to investigate (highest severity pending)."""
        pending = self.get_pending()
        if not pending:
            return None

        # Sort by severity (critical first)
        severity_order = {
            Severity.CRITICAL: 0,
            Severity.HIGH: 1,
            Severity.MEDIUM: 2,
            Severity.LOW: 3,
            Severity.INFO: 4,
        }
        pending.sort(key=lambda f: severity_order.get(f.severity, 5))
        return pending[0]

    # Bulk Operations

    def clear_all(self) -> int:
        """Clear all findings. Returns count deleted."""
        count = 0
        for path in self.storage_dir.glob("VKG-*.json"):
            path.unlink()
            count += 1
        self._next_id = 1
        self._save_index()
        return count

    def mark_stale(self, current_fingerprint: str) -> list[str]:
        """Mark findings from different graph version as stale."""
        stale_ids = []
        for finding in self.list():
            if finding.graph_fingerprint != current_fingerprint:
                finding.notes.append(f"[STALE] Graph changed from {finding.graph_fingerprint[:8]} to {current_fingerprint[:8]}")
                self.update(finding)
                stale_ids.append(finding.id)
        return stale_ids
```

import pytest
from pathlib import Path
import tempfile

from true_vkg.findings.model import Finding, Location, Status, Severity, Tier
from true_vkg.findings.store import FindingStore


class TestFinding:
    def test_create_finding(self):
        f = Finding(
            id="VKG-001",
            pattern_id="reentrancy-classic",
            tier=Tier.TIER_A,
            severity=Severity.CRITICAL,

        assert f.id == "VKG-001"
        assert f.status == Status.PENDING

    def test_to_dict_roundtrip(self):
        f = Finding(
            id="VKG-001",
            pattern_id="test",
            tier=Tier.TIER_A,
            severity=Severity.HIGH,
            location=Location("Test.sol", 10),
            behavioral_signature="",
            description="Test finding",
        )
        d = f.to_dict()
        f2 = Finding.from_dict(d)
        assert f.id == f2.id
        assert f.pattern_id == f2.pattern_id

    def test_add_note(self):
        f = Finding(id="VKG-001", ...)
        original_updated = f.updated_at
        f.add_note("Investigated: looks valid")
        assert len(f.notes) == 1
        assert f.updated_at > original_updated


class TestFindingStore:
    def test_create_and_read(self, tmp_path):
        store = FindingStore(tmp_path / "findings")
        f = Finding(
            id="",  # Will be assigned
            pattern_id="test",
            tier=Tier.TIER_A,
            severity=Severity.MEDIUM,
            location=Location("Test.sol", 10),
            behavioral_signature="",
            description="Test",
        )
        created = store.create(f)
        assert created.id == "VKG-001"

        read = store.read("VKG-001")
        assert read is not None
        assert read.pattern_id == "test"

    def test_update_status(self, tmp_path):
        store = FindingStore(tmp_path / "findings")
        f = store.create(Finding(...))
        f.set_status(Status.CONFIRMED, "Verified manually")
        store.update(f)

        read = store.read(f.id)
        assert read.status == Status.CONFIRMED
        assert read.verdict_reason == "Verified manually"

    def test_get_next_priority(self, tmp_path):
        store = FindingStore(tmp_path / "findings")
        store.create(Finding(severity=Severity.LOW, ...))
        store.create(Finding(severity=Severity.CRITICAL, ...))
        store.create(Finding(severity=Severity.MEDIUM, ...))

        f1 = store.create(Finding(severity=Severity.HIGH, ...))
        f2 = store.create(Finding(severity=Severity.LOW, ...))

    def test_findings_survive_restart(self, tmp_path):

        store1.create(Finding(...))
        store1.create(Finding(...))

        # Simulate restart by creating new store instance
        store2 = FindingStore(tmp_path / "findings")
        all_findings = list(store2.list())
        assert len(all_findings) == 2
