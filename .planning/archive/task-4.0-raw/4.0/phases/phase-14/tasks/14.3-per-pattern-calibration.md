## Objective

Train separate calibrators for each pattern that has sufficient data, falling back to global calibrator for patterns with sparse data.

---

## Why Per-Pattern Calibration?

Different patterns have different precision:
- `vm-001` (reentrancy): Might be 95% accurate
- `auth-003` (access control): Might be 70% accurate
- `mev-001` (slippage): Might be 80% accurate

A global calibrator treats all patterns the same, which is wrong. Per-pattern calibration lets us say "this vm-001 finding at 0.8 raw confidence is 92% likely TP" while "this auth-003 finding at 0.8 raw confidence is only 68% likely TP".

```python
from collections import defaultdict
from typing import Dict, List, Optional, Tuple
import numpy as np
from pathlib import Path
import pickle

from .calibrator import ConfidenceCalibrator, CalibrationMetrics
from .dataset import LabeledDataset, LabeledFinding

class PatternCalibrationStats:
    """Statistics for per-pattern calibration."""

    def __init__(self):
        self.per_pattern_samples: Dict[str, int] = {}
        self.per_pattern_brier: Dict[str, float] = {}
        self.patterns_with_calibrator: List[str] = []
        self.patterns_using_global: List[str] = []
        self.global_samples: int = 0
        self.global_brier: float = 0.0

class PatternCalibrator:
    """
    Per-pattern confidence calibration with global fallback.

    For patterns with >= min_samples: Uses pattern-specific calibrator
    For patterns with < min_samples: Falls back to global calibrator
    """

    DEFAULT_MIN_SAMPLES = 10  # Minimum samples to train pattern-specific

    def __init__(self, min_samples: int = DEFAULT_MIN_SAMPLES):
        """
        Args:
            min_samples: Minimum labeled findings to train pattern-specific calibrator
        """
        self.min_samples = min_samples
        self.calibrators: Dict[str, ConfidenceCalibrator] = {}
        self.global_calibrator = ConfidenceCalibrator()
        self._fitted = False
        self._stats: Optional[PatternCalibrationStats] = None

    def fit(self, findings: List[LabeledFinding]) -> PatternCalibrationStats:
        """
        Fit per-pattern and global calibrators.

        Args:
            findings: List of labeled findings

        Returns:
            PatternCalibrationStats with training statistics
        """
        stats = PatternCalibrationStats()

        # Group findings by pattern
        by_pattern: Dict[str, List[LabeledFinding]] = defaultdict(list)
        for finding in findings:
            by_pattern[finding.pattern_id].append(finding)

        # Fit per-pattern calibrators where we have enough data
        for pattern_id, pattern_findings in by_pattern.items():
            stats.per_pattern_samples[pattern_id] = len(pattern_findings)

            if len(pattern_findings) >= self.min_samples:
                try:
                    calibrator = ConfidenceCalibrator()
                    scores = np.array([f.raw_confidence for f in pattern_findings])
                    labels = np.array([1 if f.is_true_positive else 0 for f in pattern_findings])

                    # Need at least 30 for base calibrator, but we check min_samples first
                    if len(pattern_findings) >= 30:
                        metrics = calibrator.fit(scores, labels)
                        self.calibrators[pattern_id] = calibrator
                        stats.patterns_with_calibrator.append(pattern_id)
                        stats.per_pattern_brier[pattern_id] = metrics.brier_score
                    else:
                        stats.patterns_using_global.append(pattern_id)
                except Exception as e:
                    # If fit fails, use global
                    stats.patterns_using_global.append(pattern_id)
            else:
                stats.patterns_using_global.append(pattern_id)

        # Fit global calibrator on all data
        all_scores = np.array([f.raw_confidence for f in findings])
        all_labels = np.array([1 if f.is_true_positive else 0 for f in findings])

        global_metrics = self.global_calibrator.fit(all_scores, all_labels)
        stats.global_samples = global_metrics.n_samples
        stats.global_brier = global_metrics.brier_score

        self._fitted = True
        self._stats = stats
        return stats

    def calibrate(self, pattern_id: str, raw_confidence: float) -> Tuple[float, str]:
        """
        Calibrate confidence for a specific pattern.

        Args:
            pattern_id: The pattern ID (e.g., "vm-001")
            raw_confidence: Raw confidence score

        Returns:
            Tuple of (calibrated_score, calibrator_used) where calibrator_used
            is either the pattern_id or "global"
        """
        if not self._fitted:
            raise RuntimeError("PatternCalibrator not fitted. Call fit() first.")

        if pattern_id in self.calibrators:
            return self.calibrators[pattern_id].calibrate(raw_confidence), pattern_id
        else:
            return self.global_calibrator.calibrate(raw_confidence), "global"

    def calibrate_finding(self, finding: LabeledFinding) -> float:
        """Convenience method to calibrate a finding directly."""
        calibrated, _ = self.calibrate(finding.pattern_id, finding.raw_confidence)
        return calibrated

    def get_pattern_info(self, pattern_id: str) -> dict:
        """Get calibration info for a specific pattern."""
        if not self._fitted:
            raise RuntimeError("PatternCalibrator not fitted.")

        has_own = pattern_id in self.calibrators
        samples = self._stats.per_pattern_samples.get(pattern_id, 0)
        brier = self._stats.per_pattern_brier.get(pattern_id, self._stats.global_brier)

        return {
            "pattern_id": pattern_id,
            "has_own_calibrator": has_own,
            "samples": samples,
            "brier_score": brier,
            "using": pattern_id if has_own else "global"
        }

    def save(self, path: Path) -> None:
        """Serialize pattern calibrator."""
        path = Path(path)
        path.parent.mkdir(parents=True, exist_ok=True)
        with open(path, 'wb') as f:
            pickle.dump({
                'calibrators': self.calibrators,
                'global': self.global_calibrator,
                'min_samples': self.min_samples,
                'fitted': self._fitted,
                'stats': self._stats
            }, f)

    @classmethod
    def load(cls, path: Path) -> "PatternCalibrator":
        """Load pattern calibrator from file."""
        with open(path, 'rb') as f:
            data = pickle.load(f)

        calibrator = cls(min_samples=data['min_samples'])
        calibrator.calibrators = data['calibrators']
        calibrator.global_calibrator = data['global']
        calibrator._fitted = data['fitted']
        calibrator._stats = data['stats']
        return calibrator

    @property
    def is_fitted(self) -> bool:
        return self._fitted

    @property
    def stats(self) -> Optional[PatternCalibrationStats]:
        return self._stats
```

```python
# Add pattern calibrator training
from true_vkg.calibration.pattern_calibrator import PatternCalibrator

def train_pattern_calibrator(args):
    """Train per-pattern calibrator."""
    dataset = LabeledDataset.load(args.dataset)
    train, test = dataset.train_test_split(args.test_ratio)

    calibrator = PatternCalibrator(min_samples=args.min_samples)
    stats = calibrator.fit(train)

    print(f"\n=== Pattern Calibration Stats ===")
    print(f"Patterns with own calibrator: {len(stats.patterns_with_calibrator)}")
    print(f"Patterns using global: {len(stats.patterns_using_global)}")
    print(f"Global Brier score: {stats.global_brier:.4f}")

    for pattern_id in stats.patterns_with_calibrator[:5]:
        print(f"  {pattern_id}: Brier={stats.per_pattern_brier[pattern_id]:.4f} "
              f"(n={stats.per_pattern_samples[pattern_id]})")

    # Save
    output = args.output.parent / "pattern_calibrator_v1.pkl"
    calibrator.save(output)
    print(f"\nSaved to {output}")

    return calibrator

# Add to main()
parser.add_argument("--min-samples", type=int, default=10)
parser.add_argument("--pattern", action="store_true", help="Train per-pattern calibrator")

if args.pattern:
    train_pattern_calibrator(args)
```

```python
"""Calibration reporting and analysis."""

from typing import List
from .pattern_calibrator import PatternCalibrator, PatternCalibrationStats

def generate_calibration_report(stats: PatternCalibrationStats) -> str:
    """Generate human-readable calibration report."""
    lines = [
        "# Calibration Report",
        "",
        "## Summary",
        f"- Patterns with dedicated calibrator: {len(stats.patterns_with_calibrator)}",
        f"- Patterns using global fallback: {len(stats.patterns_using_global)}",
        f"- Global Brier score: {stats.global_brier:.4f}",
        "",
        "## Pattern Details",
        ""
    ]

    # Sort by sample count
    sorted_patterns = sorted(
        stats.per_pattern_samples.items(),
        key=lambda x: x[1],
        reverse=True
    )

    for pattern_id, count in sorted_patterns:
        has_own = pattern_id in stats.patterns_with_calibrator
        brier = stats.per_pattern_brier.get(pattern_id, "N/A")
        status = "OWN" if has_own else "GLOBAL"
        lines.append(f"| {pattern_id} | {count} | {status} | {brier} |")

    return "\n".join(lines)
```

```python
from true_vkg.calibration.pattern_calibrator import PatternCalibrator, PatternCalibrationStats
from true_vkg.calibration.dataset import LabeledFinding

class TestPatternCalibrator:
    """Tests for PatternCalibrator."""

    def _make_findings(self, pattern_id: str, count: int, tp_rate: float = 0.5) -> List[LabeledFinding]:
        """Helper to create test findings."""
        findings = []
        for i in range(count):
            is_tp = i < int(count * tp_rate)
            findings.append(LabeledFinding(
                id=f"{pattern_id}_{i}",
                pattern_id=pattern_id,
                contract="test.sol",
                function="test",
                line_number=i,
                raw_confidence=0.5 + (0.3 if is_tp else -0.2),
                is_true_positive=is_tp,
                source="test"
            ))
        return findings

    def test_fit_creates_per_pattern_calibrators(self):
        """Test that patterns with enough data get own calibrator."""
        # Pattern A has 50 samples, Pattern B has 5
        findings = self._make_findings("pattern-a", 50) + self._make_findings("pattern-b", 5)

        calibrator = PatternCalibrator(min_samples=10)
        stats = calibrator.fit(findings)

        assert "pattern-a" in stats.patterns_with_calibrator
        assert "pattern-b" in stats.patterns_using_global

    def test_fallback_to_global(self):
        """Test fallback to global for unknown pattern."""
        findings = self._make_findings("known-pattern", 50)

        calibrator = PatternCalibrator(min_samples=10)
        calibrator.fit(findings)

        # Known pattern uses own calibrator
        score, used = calibrator.calibrate("known-pattern", 0.8)
        assert used == "known-pattern"

        # Unknown pattern uses global
        score, used = calibrator.calibrate("unknown-pattern", 0.8)
        assert used == "global"

    def test_per_pattern_stats_populated(self):
        """Test that stats are properly populated."""
        findings = (
            self._make_findings("vm-001", 40, tp_rate=0.9) +
            self._make_findings("auth-001", 35, tp_rate=0.6) +
            self._make_findings("rare-001", 5, tp_rate=0.5)
        )

        calibrator = PatternCalibrator(min_samples=10)
        stats = calibrator.fit(findings)

        assert stats.per_pattern_samples["vm-001"] == 40
        assert stats.per_pattern_samples["auth-001"] == 35
        assert stats.per_pattern_samples["rare-001"] == 5
        assert "vm-001" in stats.per_pattern_brier
        assert "rare-001" not in stats.per_pattern_brier

    def test_save_load_with_patterns(self, tmp_path):
        """Test save/load preserves per-pattern calibrators."""
        findings = self._make_findings("pattern-a", 50) + self._make_findings("pattern-b", 50)

        calibrator = PatternCalibrator(min_samples=10)
        calibrator.fit(findings)

        # Save and load
        path = tmp_path / "pattern_calibrator.pkl"
        calibrator.save(path)
        loaded = PatternCalibrator.load(path)

        # Verify
        assert set(loaded.calibrators.keys()) == set(calibrator.calibrators.keys())
        assert loaded.calibrate("pattern-a", 0.7) == calibrator.calibrate("pattern-a", 0.7)
```
