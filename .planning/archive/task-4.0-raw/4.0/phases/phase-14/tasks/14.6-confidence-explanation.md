## Objective

Provide human-readable explanations of how confidence was calculated, showing each factor that contributed to the final score.

---

## Why Explanations Matter

Auditors need to understand:
1. **Why is this 72%?** What factors went into this score?
2. **Can I trust this?** Is it based on historical data or guesswork?
3. **What would change it?** What guards would reduce the risk?

Without explanations, confidence scores are magic numbers.

```python
"""Human-readable confidence explanations."""

from dataclasses import dataclass, field
from typing import List, Optional

@dataclass
class ConfidenceFactor:
    """A single factor contributing to confidence."""
    name: str
    description: str
    impact: float  # Multiplier applied (e.g., 0.7 = reduced by 30%)
    applied: bool

@dataclass
class ConfidenceExplanation:
    """Complete explanation of confidence calculation."""
    finding_id: str
    pattern_id: str

    # Scores at each stage
    raw_confidence: float
    calibration_adjustment: float  # Difference from raw
    context_adjustment: float  # Difference from calibrated
    final_confidence: float

    # Details
    calibrator_used: str
    calibrator_samples: int  # How many samples trained on
    factors: List[ConfidenceFactor] = field(default_factory=list)

    # Interpretation
    confidence_tier: str  # "high", "medium", "low"
    interpretation: str

    def to_text(self) -> str:
        """Generate human-readable explanation."""
        lines = [
            f"Finding: {self.finding_id}",
            f"Pattern: {self.pattern_id}",
            f"Final Confidence: {self.final_confidence:.0%}",
            "",
            "Score Breakdown:",
            f"  Pattern match strength: {self.raw_confidence:.2f}",
        ]

        if abs(self.calibration_adjustment) > 0.01:
            sign = "+" if self.calibration_adjustment > 0 else ""
            lines.append(
                f"  Calibration adjustment: {sign}{self.calibration_adjustment:.2f} "
                f"(based on {self.calibrator_samples} historical findings)"
            )

        if abs(self.context_adjustment) > 0.01:
            sign = "+" if self.context_adjustment > 0 else ""
            lines.append(f"  Context adjustment: {sign}{self.context_adjustment:.2f}")

        # Show applied factors
        applied_factors = [f for f in self.factors if f.applied]
        if applied_factors:
            lines.append("")
            lines.append("Mitigating Factors Detected:")
            for factor in applied_factors:
                impact_pct = (1 - factor.impact) * 100
                lines.append(f"  - {factor.description}: -{impact_pct:.0f}%")

        lines.append("")
        lines.append(f"Interpretation: {self.interpretation}")

        return "\n".join(lines)

    def to_dict(self) -> dict:
        """Convert to dictionary for JSON serialization."""
        return {
            "finding_id": self.finding_id,
            "pattern_id": self.pattern_id,
            "raw_confidence": self.raw_confidence,
            "calibration_adjustment": self.calibration_adjustment,
            "context_adjustment": self.context_adjustment,
            "final_confidence": self.final_confidence,
            "calibrator_used": self.calibrator_used,
            "calibrator_samples": self.calibrator_samples,
            "factors": [
                {"name": f.name, "description": f.description,
                 "impact": f.impact, "applied": f.applied}
                for f in self.factors
            ],
            "confidence_tier": self.confidence_tier,
            "interpretation": self.interpretation
        }

def generate_interpretation(confidence: float, pattern_id: str) -> str:
    """Generate plain-English interpretation of confidence level."""
    if confidence >= 0.85:
        return (
            f"Very High Confidence ({confidence:.0%}): "
            f"Historically, {confidence:.0%} of similar {pattern_id} findings "
            "were confirmed vulnerabilities. Strongly recommend manual review."
        )
    elif confidence >= 0.70:
        return (
            f"High Confidence ({confidence:.0%}): "
            f"About {confidence:.0%} of similar findings were true positives. "
            "Worth investigating during security review."
        )
    elif confidence >= 0.50:
        return (
            f"Medium Confidence ({confidence:.0%}): "
            "Finding could be a true vulnerability but has significant chance "
            "of being a false positive. Review if time permits."
        )
    elif confidence >= 0.30:
        return (
            f"Low Confidence ({confidence:.0%}): "
            "More likely to be a false positive than a real vulnerability. "
            "Investigate only if the pattern is concerning."
        )
    else:
        return (
            f"Very Low Confidence ({confidence:.0%}): "
            "Likely a false positive or mitigated by detected guards. "
            "Can deprioritize unless specifically relevant."
        )

def get_confidence_tier(confidence: float) -> str:
    """Categorize confidence into tiers."""
    if confidence >= 0.85:
        return "very_high"
    elif confidence >= 0.70:
        return "high"
    elif confidence >= 0.50:
        return "medium"
    elif confidence >= 0.30:
        return "low"
    else:
        return "very_low"

class ExplanationGenerator:
    """Generates explanations for calibrated confidence scores."""

    def __init__(self):
        # Factor definitions
        self.factor_definitions = {
            "reentrancy_guard": ConfidenceFactor(
                name="reentrancy_guard",
                description="ReentrancyGuard modifier detected",
                impact=0.3,
                applied=False
            ),
            "access_gate": ConfidenceFactor(
                name="access_gate",
                description="Access control (onlyOwner, etc.) present",
                impact=0.7,
                applied=False
            ),
            "safe_erc20": ConfidenceFactor(
                name="safe_erc20",
                description="SafeERC20 library used for transfers",
                impact=0.4,
                applied=False
            ),
            "cei_pattern": ConfidenceFactor(
                name="cei_pattern",
                description="Checks-Effects-Interactions pattern followed",
                impact=0.2,
                applied=False
            ),
            "trusted_call": ConfidenceFactor(
                name="trusted_call",
                description="External calls to known trusted contracts",
                impact=0.5,
                applied=False
            ),
            "openzeppelin": ConfidenceFactor(
                name="openzeppelin",
                description="OpenZeppelin contracts used",
                impact=0.8,
                applied=False
            ),
        }

    def generate(
        self,
        finding_id: str,
        pattern_id: str,
        raw_confidence: float,
        calibrated_confidence: float,
        context_adjusted_confidence: float,
        calibrator_used: str,
        calibrator_samples: int,
        applied_adjustments: List[str]
    ) -> ConfidenceExplanation:
        """
        Generate a complete confidence explanation.

        Args:
            finding_id: Unique finding identifier
            pattern_id: Pattern that triggered the finding
            raw_confidence: Original pattern match score
            calibrated_confidence: After calibration model
            context_adjusted_confidence: After context factors
            calibrator_used: Which calibrator (pattern ID or "global")
            calibrator_samples: Training samples for calibrator
            applied_adjustments: List of adjustment strings from context adjuster

        Returns:
            ConfidenceExplanation with full breakdown
        """
        # Build factors list
        factors = []
        for name, factor_def in self.factor_definitions.items():
            factor = ConfidenceFactor(
                name=factor_def.name,
                description=factor_def.description,
                impact=factor_def.impact,
                applied=any(name in adj for adj in applied_adjustments)
            )
            factors.append(factor)

        return ConfidenceExplanation(
            finding_id=finding_id,
            pattern_id=pattern_id,
            raw_confidence=raw_confidence,
            calibration_adjustment=calibrated_confidence - raw_confidence,
            context_adjustment=context_adjusted_confidence - calibrated_confidence,
            final_confidence=context_adjusted_confidence,
            calibrator_used=calibrator_used,
            calibrator_samples=calibrator_samples,
            factors=factors,
            confidence_tier=get_confidence_tier(context_adjusted_confidence),
            interpretation=generate_interpretation(context_adjusted_confidence, pattern_id)
        )
```

### Step 2: Integrate with Pipeline (1h)

Modify: `src/true_vkg/calibration/pipeline.py`

```python
# Add to CalibrationPipeline class

def calibrate_with_explanation(
    self,
    finding_id: str,
    raw_confidence: float,
    pattern_id: str,
    context: Optional[ContextFactors] = None
) -> Tuple[CalibrationResult, ConfidenceExplanation]:
    """Run calibration and generate explanation."""
    result = self.calibrate(raw_confidence, pattern_id, context)

    # Get calibrator stats
    if self.calibrator is not None and pattern_id in self.calibrator.calibrators:
        samples = self.calibrator.stats.per_pattern_samples.get(pattern_id, 0)
    else:
        samples = self.calibrator.stats.global_samples if self.calibrator else 0

    explanation = self.explanation_generator.generate(
        finding_id=finding_id,
        pattern_id=pattern_id,
        raw_confidence=result.raw_confidence,
        calibrated_confidence=result.calibrated_confidence,
        context_adjusted_confidence=result.context_adjusted_confidence,
        calibrator_used=result.calibrator_used,
        calibrator_samples=samples,
        applied_adjustments=result.context_adjustments
    )

    return result, explanation
```

from true_vkg.calibration.explanation import (
    ConfidenceExplanation,
    ExplanationGenerator,
    generate_interpretation,
    get_confidence_tier
)

class TestConfidenceExplanation:
    """Tests for confidence explanation."""

    def test_generate_interpretation_high(self):
        """High confidence should recommend review."""
        interp = generate_interpretation(0.85, "vm-001")
        assert "Very High" in interp
        assert "recommend" in interp.lower()

    def test_generate_interpretation_low(self):
        """Low confidence should deprioritize."""
        interp = generate_interpretation(0.20, "vm-001")
        assert "Very Low" in interp or "Low" in interp
        assert "false positive" in interp.lower()

    def test_confidence_tiers(self):
        """Test tier assignments."""
        assert get_confidence_tier(0.90) == "very_high"
        assert get_confidence_tier(0.75) == "high"
        assert get_confidence_tier(0.55) == "medium"
        assert get_confidence_tier(0.35) == "low"
        assert get_confidence_tier(0.15) == "very_low"

    def test_explanation_to_text(self):
        """Test text generation."""
        explanation = ConfidenceExplanation(
            finding_id="F001",
            pattern_id="vm-001",
            raw_confidence=0.85,
            calibration_adjustment=-0.05,
            context_adjustment=-0.10,
            final_confidence=0.70,
            calibrator_used="vm-001",
            calibrator_samples=50,
            factors=[],
            confidence_tier="high",
            interpretation="Test interpretation"
        )

        text = explanation.to_text()

        assert "F001" in text
        assert "vm-001" in text
        assert "70%" in text
        assert "50" in text  # samples

    def test_explanation_generator(self):
        """Test generator creates valid explanation."""
        generator = ExplanationGenerator()

        explanation = generator.generate(
            finding_id="F001",
            pattern_id="vm-001",
            raw_confidence=0.80,
            calibrated_confidence=0.75,
            context_adjusted_confidence=0.60,
            calibrator_used="vm-001",
            calibrator_samples=45,
            applied_adjustments=["reentrancy_guard: *0.3"]
        )

        assert explanation.finding_id == "F001"
        assert explanation.final_confidence == 0.60
        # Reentrancy guard should be marked as applied
        reentrancy_factor = next(
            f for f in explanation.factors if f.name == "reentrancy_guard"
        )
        assert reentrancy_factor.applied
