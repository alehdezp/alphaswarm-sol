## Objective

Validate that calibration actually works as claimed - that "80% confidence" findings really have ~80% true positive rate.

---

## Why Validation Matters

Without validation, we cannot trust our confidence scores. Validation:
1. Proves calibration works on held-out data
2. Identifies systematic biases
3. Sets expectations for users
4. Catches over/under-confidence issues

---

## Prerequisites

1. **Tasks 14.3 and 14.6 complete**
2. Labeled test set (30% held out from training)
3. PatternCalibrator trained and saved

---

## Validation Protocol

### Step 1: Bucket Analysis

1. Take test set (never seen during training)
2. Calibrate all findings
3. Bucket by calibrated confidence:
   - [0.0 - 0.1), [0.1 - 0.2), ..., [0.9 - 1.0]
4. For each bucket, calculate actual TP rate
5. Compare: Does [0.7-0.8) bucket have ~75% TP?

**Success Criterion:** Each bucket within 10% of expected

### Step 2: Brier Score Comparison

Compare Brier scores:
- Before calibration (raw scores)
- After calibration

**Success Criterion:** Brier score improved (lower)

### Step 3: Per-Pattern Validation

For each pattern with dedicated calibrator:
- Validate bucket alignment
- Compare to global calibrator

**Success Criterion:** Per-pattern no worse than global

```python
"""Calibration validation and quality assessment."""

from dataclasses import dataclass, field
from typing import List, Dict, Optional
import numpy as np
from sklearn.metrics import brier_score_loss

@dataclass
class BucketValidation:
    """Validation results for a single bucket."""
    bucket_start: float
    bucket_end: float
    expected_tp_rate: float  # Middle of bucket
    actual_tp_rate: float
    count: int
    deviation: float  # Absolute difference

    @property
    def is_acceptable(self) -> bool:
        """Within 10% of expected."""
        return self.deviation <= 0.10

@dataclass
class ValidationReport:
    """Complete calibration validation report."""
    # Overall metrics
    brier_score_before: float
    brier_score_after: float
    brier_improvement: float

    # Bucket analysis
    buckets: List[BucketValidation]
    buckets_passing: int
    buckets_total: int
    max_deviation: float
    mean_deviation: float

    # Per-pattern breakdown (optional)
    per_pattern_brier: Dict[str, float] = field(default_factory=dict)

    # Verdict
    validation_passed: bool
    failure_reasons: List[str] = field(default_factory=list)

    def to_markdown(self) -> str:
        """Generate markdown validation report."""
        lines = [
            "# Calibration Validation Report",
            "",
            "## Overall Metrics",
            "",
            f"| Metric | Value |",
            f"|--------|-------|",
            f"| Brier Score (Before) | {self.brier_score_before:.4f} |",
            f"| Brier Score (After) | {self.brier_score_after:.4f} |",
            f"| Improvement | {self.brier_improvement:.1f}% |",
            "",
            "## Bucket Analysis",
            "",
            f"Buckets passing: {self.buckets_passing}/{self.buckets_total}",
            f"Max deviation: {self.max_deviation:.2f}",
            f"Mean deviation: {self.mean_deviation:.2f}",
            "",
            "| Bucket | Expected TP | Actual TP | Deviation | Status |",
            "|--------|-------------|-----------|-----------|--------|",
        ]

        for b in self.buckets:
            status = "PASS" if b.is_acceptable else "FAIL"
            lines.append(
                f"| {b.bucket_start:.1f}-{b.bucket_end:.1f} | "
                f"{b.expected_tp_rate:.2f} | {b.actual_tp_rate:.2f} | "
                f"{b.deviation:.2f} | {status} |"
            )

        lines.extend([
            "",
            "## Verdict",
            "",
            f"**{'PASSED' if self.validation_passed else 'FAILED'}**",
        ])

        if self.failure_reasons:
            lines.append("")
            lines.append("Failure reasons:")
            for reason in self.failure_reasons:
                lines.append(f"- {reason}")

        return "\n".join(lines)

class CalibrationValidator:
    """Validates calibration quality."""

    def __init__(
        self,
        max_deviation: float = 0.10,  # 10% max deviation per bucket
        min_bucket_samples: int = 5,  # Minimum samples to evaluate bucket
        n_buckets: int = 10
    ):
        self.max_deviation = max_deviation
        self.min_bucket_samples = min_bucket_samples
        self.n_buckets = n_buckets

    def validate(
        self,
        raw_scores: np.ndarray,
        calibrated_scores: np.ndarray,
        actuals: np.ndarray
    ) -> ValidationReport:
        """
        Run full validation on test set.

        Args:
            raw_scores: Uncalibrated confidence scores
            calibrated_scores: Calibrated scores
            actuals: Binary ground truth labels

        Returns:
            ValidationReport with all metrics
        """
        # Calculate Brier scores
        brier_before = brier_score_loss(actuals, raw_scores)
        brier_after = brier_score_loss(actuals, calibrated_scores)
        improvement = (brier_before - brier_after) / brier_before * 100

        # Bucket analysis
        buckets = self._analyze_buckets(calibrated_scores, actuals)
        passing = sum(1 for b in buckets if b.is_acceptable)
        deviations = [b.deviation for b in buckets]

        # Determine if validation passed
        failure_reasons = []

        if brier_after >= brier_before:
            failure_reasons.append(
                f"Calibration did not improve Brier score: "
                f"{brier_after:.4f} >= {brier_before:.4f}"
            )

        if passing < len(buckets) * 0.7:  # Need 70% of buckets to pass
            failure_reasons.append(
                f"Too many bucket failures: {passing}/{len(buckets)} passing"
            )

        if max(deviations) > 0.15:  # Hard limit on max deviation
            failure_reasons.append(
                f"Excessive maximum deviation: {max(deviations):.2f}"
            )

        return ValidationReport(
            brier_score_before=brier_before,
            brier_score_after=brier_after,
            brier_improvement=improvement,
            buckets=buckets,
            buckets_passing=passing,
            buckets_total=len(buckets),
            max_deviation=max(deviations) if deviations else 0,
            mean_deviation=np.mean(deviations) if deviations else 0,
            validation_passed=len(failure_reasons) == 0,
            failure_reasons=failure_reasons
        )

    def _analyze_buckets(
        self,
        scores: np.ndarray,
        actuals: np.ndarray
    ) -> List[BucketValidation]:
        """Analyze calibration per confidence bucket."""
        buckets = []
        edges = np.linspace(0, 1, self.n_buckets + 1)

        for i in range(self.n_buckets):
            low, high = edges[i], edges[i + 1]
            expected = (low + high) / 2  # Midpoint

            # Find scores in this bucket
            if i == self.n_buckets - 1:
                mask = (scores >= low) & (scores <= high)
            else:
                mask = (scores >= low) & (scores < high)

            bucket_actuals = actuals[mask]

            if len(bucket_actuals) >= self.min_bucket_samples:
                actual_rate = np.mean(bucket_actuals)
                deviation = abs(actual_rate - expected)

                buckets.append(BucketValidation(
                    bucket_start=low,
                    bucket_end=high,
                    expected_tp_rate=expected,
                    actual_tp_rate=actual_rate,
                    count=len(bucket_actuals),
                    deviation=deviation
                ))

        return buckets

def run_validation(
    dataset_path: str,
    calibrator_path: str,
    output_path: Optional[str] = None
) -> ValidationReport:
    """
    Run complete validation workflow.

    Args:
        dataset_path: Path to labeled dataset JSON
        calibrator_path: Path to trained calibrator
        output_path: Optional path to save report

    Returns:
        ValidationReport
    """
    from .dataset import LabeledDataset
    from .pattern_calibrator import PatternCalibrator

    # Load
    dataset = LabeledDataset.load(dataset_path)
    calibrator = PatternCalibrator.load(calibrator_path)

    # Split (use same split as training for test set)
    _, test = dataset.train_test_split(0.3)

    # Extract arrays
    raw = np.array([f.raw_confidence for f in test])
    actuals = np.array([1 if f.is_true_positive else 0 for f in test])
    calibrated = np.array([
        calibrator.calibrate(f.pattern_id, f.raw_confidence)[0]
        for f in test
    ])

    # Validate
    validator = CalibrationValidator()
    report = validator.validate(raw, calibrated, actuals)

    # Save if requested
    if output_path:
        with open(output_path, 'w') as f:
            f.write(report.to_markdown())

    return report
```

### Step 2: Create CLI Script (1h)

Create: `scripts/validate_calibration.py`

```python
#!/usr/bin/env python
"""Validate calibration quality."""

import argparse
from pathlib import Path
from true_vkg.calibration.validation import run_validation

def main():
    parser = argparse.ArgumentParser(description="Validate calibration")
    parser.add_argument("--dataset", type=Path, required=True)
    parser.add_argument("--calibrator", type=Path, required=True)
    parser.add_argument("--output", type=Path, default=Path("reports/calibration/validation.md"))
    args = parser.parse_args()

    print(f"Validating calibration...")
    print(f"  Dataset: {args.dataset}")
    print(f"  Calibrator: {args.calibrator}")

    report = run_validation(
        str(args.dataset),
        str(args.calibrator),
        str(args.output)
    )

    print(f"\n{'='*50}")
    print(f"Brier Score: {report.brier_score_after:.4f} (was {report.brier_score_before:.4f})")
    print(f"Improvement: {report.brier_improvement:.1f}%")
    print(f"Buckets passing: {report.buckets_passing}/{report.buckets_total}")
    print(f"Max deviation: {report.max_deviation:.2f}")

    if report.validation_passed:
        print(f"\n{'='*50}")
        print("VALIDATION PASSED")
    else:
        print(f"\n{'='*50}")
        print("VALIDATION FAILED")
        for reason in report.failure_reasons:
            print(f"  - {reason}")

    print(f"\nReport saved to {args.output}")

if __name__ == "__main__":
    main()
```

import numpy as np
from true_vkg.calibration.validation import (
    CalibrationValidator,
    BucketValidation,
    ValidationReport
)

class TestCalibrationValidation:
    """Tests for calibration validation."""

    def test_perfect_calibration_passes(self):
        """Perfect calibration should pass validation."""
        validator = CalibrationValidator()

        # Perfect calibration: actual rate = predicted
        n = 1000
        scores = np.random.uniform(0, 1, n)
        # Generate actuals that match predicted probability
        actuals = np.array([
            1 if np.random.random() < s else 0
            for s in scores
        ])

        # "Calibrated" scores are same as raw (already perfect)
        report = validator.validate(scores, scores, actuals)

        # Should generally pass (stochastic, but usually works)
        assert report.max_deviation < 0.20  # Allow some variance

    def test_overconfident_fails(self):
        """Overconfident calibration should fail."""
        validator = CalibrationValidator()

        n = 500
        # Predict 0.9 for everything but only 50% are TP
        scores = np.full(n, 0.9)
        actuals = np.concatenate([np.ones(250), np.zeros(250)])
        np.random.shuffle(actuals)

        report = validator.validate(scores, scores, actuals)

        # Should fail - claiming 90% but only 50% TP
        assert not report.validation_passed

    def test_bucket_validation(self):
        """Test bucket analysis."""
        validator = CalibrationValidator(n_buckets=5)

        # Create data with known bucket behavior
        scores = np.array([0.1, 0.15, 0.3, 0.35, 0.5, 0.55, 0.7, 0.75, 0.9, 0.95])
        # Actuals that roughly match buckets
        actuals = np.array([0, 0, 0, 1, 1, 0, 1, 1, 1, 1])

        report = validator.validate(scores, scores, actuals)

        assert len(report.buckets) > 0
        assert all(b.count >= 1 for b in report.buckets)

    def test_brier_improvement_required(self):
        """Validation should fail if calibration makes things worse."""
        validator = CalibrationValidator()

        n = 200
        # Raw scores are well calibrated
        raw = np.random.uniform(0, 1, n)
        actuals = np.array([1 if np.random.random() < s else 0 for s in raw])

        # "Calibrated" scores are random (worse)
        calibrated = np.random.uniform(0, 1, n)

        report = validator.validate(raw, calibrated, actuals)

        # If calibration made Brier worse, should report it
        if report.brier_score_after > report.brier_score_before:
            assert any("Brier" in r for r in report.failure_reasons)

    def test_report_to_markdown(self):
        """Test markdown report generation."""
        report = ValidationReport(
            brier_score_before=0.25,
            brier_score_after=0.15,
            brier_improvement=40.0,
            buckets=[
                BucketValidation(0.5, 0.6, 0.55, 0.52, 20, 0.03),
                BucketValidation(0.6, 0.7, 0.65, 0.70, 15, 0.05),
            ],
            buckets_passing=2,
            buckets_total=2,
            max_deviation=0.05,
            mean_deviation=0.04,
            validation_passed=True
        )

        md = report.to_markdown()

        assert "# Calibration Validation Report" in md
        assert "0.25" in md  # Brier before
        assert "0.15" in md  # Brier after
        assert "PASSED" in md

## Common Pitfalls

1. **Pitfall:** Train/test leakage
   - **Fix:** Use same split seed as training

2. **Pitfall:** Ignoring edge buckets
   - **Fix:** Document min samples per bucket

3. **Pitfall:** Overly strict thresholds
   - **Fix:** Start with 15% deviation, tighten later
