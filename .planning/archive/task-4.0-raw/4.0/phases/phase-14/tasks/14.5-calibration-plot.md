## Objective

Generate reliability diagrams (calibration plots) to visually assess calibration quality and communicate results.

---

## What is a Reliability Diagram?

A reliability diagram plots predicted confidence (x-axis) against actual true positive rate (y-axis).

**Perfect calibration:** Points on diagonal (y=x)
- 0.7 predicted -> 70% actual TP rate
- 0.9 predicted -> 90% actual TP rate

**Overconfident:** Points below diagonal
- 0.8 predicted but only 60% actual TP

**Underconfident:** Points above diagonal
- 0.6 predicted but 80% actual TP

### Step 1: Create Visualization Module (1.5h)

Create: `src/true_vkg/calibration/visualization.py`

```python
"""Calibration visualization tools."""

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.figure import Figure
from pathlib import Path
from typing import List, Optional, Tuple
from dataclasses import dataclass

@dataclass
class CalibrationBucket:
    """Statistics for a confidence bucket."""
    bucket_start: float
    bucket_end: float
    mean_predicted: float
    actual_tp_rate: float
    count: int

def compute_calibration_buckets(
    predictions: np.ndarray,
    actuals: np.ndarray,
    n_buckets: int = 10
) -> List[CalibrationBucket]:
    """
    Compute calibration statistics per confidence bucket.

    Args:
        predictions: Array of predicted probabilities
        actuals: Array of binary labels (0/1)
        n_buckets: Number of buckets

    Returns:
        List of CalibrationBucket with statistics
    """
    buckets = []
    bucket_edges = np.linspace(0, 1, n_buckets + 1)

    for i in range(n_buckets):
        bucket_start = bucket_edges[i]
        bucket_end = bucket_edges[i + 1]

        # Find predictions in this bucket
        mask = (predictions >= bucket_start) & (predictions < bucket_end)
        if i == n_buckets - 1:
            mask = (predictions >= bucket_start) & (predictions <= bucket_end)

        bucket_preds = predictions[mask]
        bucket_actuals = actuals[mask]

        if len(bucket_preds) > 0:
            buckets.append(CalibrationBucket(
                bucket_start=bucket_start,
                bucket_end=bucket_end,
                mean_predicted=np.mean(bucket_preds),
                actual_tp_rate=np.mean(bucket_actuals),
                count=len(bucket_preds)
            ))

    return buckets

def plot_calibration_curve(
    predictions: np.ndarray,
    actuals: np.ndarray,
    title: str = "Calibration Curve",
    n_buckets: int = 10,
    figsize: Tuple[int, int] = (8, 6)
) -> Figure:
    """
    Generate a reliability diagram.

    Args:
        predictions: Predicted probabilities
        actuals: Binary labels
        title: Plot title
        n_buckets: Number of calibration buckets
        figsize: Figure size

    Returns:
        Matplotlib Figure object
    """
    buckets = compute_calibration_buckets(predictions, actuals, n_buckets)

    fig, ax = plt.subplots(figsize=figsize)

    # Perfect calibration line
    ax.plot([0, 1], [0, 1], 'k--', label='Perfect calibration', alpha=0.7)

    # Plot actual calibration
    mean_preds = [b.mean_predicted for b in buckets]
    actual_rates = [b.actual_tp_rate for b in buckets]
    counts = [b.count for b in buckets]

    # Size points by sample count
    sizes = np.array(counts) / max(counts) * 300 + 50

    scatter = ax.scatter(
        mean_preds, actual_rates,
        s=sizes, alpha=0.7,
        c='blue', edgecolors='darkblue',
        label='Calibration points'
    )

    # Connect with line
    ax.plot(mean_preds, actual_rates, 'b-', alpha=0.5)

    # Labels and formatting
    ax.set_xlabel('Mean Predicted Confidence', fontsize=12)
    ax.set_ylabel('Actual True Positive Rate', fontsize=12)
    ax.set_title(title, fontsize=14)
    ax.set_xlim(-0.05, 1.05)
    ax.set_ylim(-0.05, 1.05)
    ax.legend(loc='lower right')
    ax.grid(True, alpha=0.3)

    # Add sample counts as annotations
    for bucket in buckets:
        ax.annotate(
            f'n={bucket.count}',
            (bucket.mean_predicted, bucket.actual_tp_rate),
            textcoords="offset points",
            xytext=(0, 10),
            ha='center',
            fontsize=8
        )

    plt.tight_layout()
    return fig

def plot_before_after_calibration(
    raw_scores: np.ndarray,
    calibrated_scores: np.ndarray,
    actuals: np.ndarray,
    figsize: Tuple[int, int] = (14, 6)
) -> Figure:
    """
    Plot before and after calibration side by side.

    Args:
        raw_scores: Uncalibrated confidence scores
        calibrated_scores: Calibrated scores
        actuals: Binary labels
        figsize: Figure size

    Returns:
        Matplotlib Figure with two subplots
    """
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=figsize)

    # Before calibration
    buckets_before = compute_calibration_buckets(raw_scores, actuals)
    ax1.plot([0, 1], [0, 1], 'k--', alpha=0.7)
    ax1.scatter(
        [b.mean_predicted for b in buckets_before],
        [b.actual_tp_rate for b in buckets_before],
        s=100, c='red', alpha=0.7
    )
    ax1.plot(
        [b.mean_predicted for b in buckets_before],
        [b.actual_tp_rate for b in buckets_before],
        'r-', alpha=0.5
    )
    ax1.set_xlabel('Mean Predicted Confidence')
    ax1.set_ylabel('Actual True Positive Rate')
    ax1.set_title('Before Calibration')
    ax1.set_xlim(-0.05, 1.05)
    ax1.set_ylim(-0.05, 1.05)
    ax1.grid(True, alpha=0.3)

    # After calibration
    buckets_after = compute_calibration_buckets(calibrated_scores, actuals)
    ax2.plot([0, 1], [0, 1], 'k--', alpha=0.7)
    ax2.scatter(
        [b.mean_predicted for b in buckets_after],
        [b.actual_tp_rate for b in buckets_after],
        s=100, c='green', alpha=0.7
    )
    ax2.plot(
        [b.mean_predicted for b in buckets_after],
        [b.actual_tp_rate for b in buckets_after],
        'g-', alpha=0.5
    )
    ax2.set_xlabel('Mean Predicted Confidence')
    ax2.set_ylabel('Actual True Positive Rate')
    ax2.set_title('After Calibration')
    ax2.set_xlim(-0.05, 1.05)
    ax2.set_ylim(-0.05, 1.05)
    ax2.grid(True, alpha=0.3)

    plt.tight_layout()
    return fig

def save_calibration_plot(
    predictions: np.ndarray,
    actuals: np.ndarray,
    output_path: Path,
    title: str = "Calibration Curve"
) -> None:
    """Generate and save calibration plot."""
    fig = plot_calibration_curve(predictions, actuals, title)
    output_path = Path(output_path)
    output_path.parent.mkdir(parents=True, exist_ok=True)
    fig.savefig(output_path, dpi=150, bbox_inches='tight')
    plt.close(fig)
    print(f"Saved calibration plot to {output_path}")
```

### Step 2: Create CLI Command (1h)

Add to: `scripts/train_calibrator.py` or create `scripts/plot_calibration.py`

```python
#!/usr/bin/env python
"""Generate calibration plots."""

import argparse
import numpy as np
from pathlib import Path
from true_vkg.calibration.dataset import LabeledDataset
from true_vkg.calibration.pattern_calibrator import PatternCalibrator
from true_vkg.calibration.visualization import (
    save_calibration_plot,
    plot_before_after_calibration
)
import matplotlib.pyplot as plt

def main():
    parser = argparse.ArgumentParser(description="Generate calibration plots")
    parser.add_argument("--dataset", type=Path, required=True)
    parser.add_argument("--calibrator", type=Path, required=True)
    parser.add_argument("--output-dir", type=Path, default=Path("reports/calibration"))
    args = parser.parse_args()

    # Load data
    dataset = LabeledDataset.load(args.dataset)
    calibrator = PatternCalibrator.load(args.calibrator)

    # Extract arrays
    raw_scores = np.array([f.raw_confidence for f in dataset.findings])
    actuals = np.array([1 if f.is_true_positive else 0 for f in dataset.findings])

    # Calibrate
    calibrated_scores = np.array([
        calibrator.calibrate(f.pattern_id, f.raw_confidence)[0]
        for f in dataset.findings
    ])

    # Create output directory
    args.output_dir.mkdir(parents=True, exist_ok=True)

    # Plot 1: Before/After comparison
    fig = plot_before_after_calibration(raw_scores, calibrated_scores, actuals)
    fig.savefig(args.output_dir / "before_after.png", dpi=150)
    plt.close(fig)

    # Plot 2: Calibrated only
    save_calibration_plot(
        calibrated_scores, actuals,
        args.output_dir / "calibration_curve.png",
        "Calibrated Confidence vs Actual TP Rate"
    )

    print(f"Plots saved to {args.output_dir}")

if __name__ == "__main__":
    main()
```

import numpy as np
from true_vkg.calibration.visualization import (
    compute_calibration_buckets,
    plot_calibration_curve
)

class TestCalibrationVisualization:
    """Tests for calibration visualization."""

    def test_compute_buckets(self):
        """Test bucket computation."""
        predictions = np.array([0.1, 0.2, 0.3, 0.7, 0.8, 0.9])
        actuals = np.array([0, 0, 1, 1, 1, 1])

        buckets = compute_calibration_buckets(predictions, actuals, n_buckets=10)

        assert len(buckets) > 0
        assert all(b.count > 0 for b in buckets)

    def test_plot_generation(self):
        """Test that plot is generated without error."""
        predictions = np.random.uniform(0, 1, 100)
        actuals = (predictions + np.random.uniform(-0.2, 0.2, 100) > 0.5).astype(int)

        fig = plot_calibration_curve(predictions, actuals)

        assert fig is not None
        assert len(fig.axes) == 1
