## Technical Background

### What is Calibration?

Raw confidence scores from pattern matching don't map to probabilities. A score of 0.8 might have 60% TP rate or 95% TP rate depending on the pattern.

Calibration learns a mapping: `raw_score -> actual_probability`

### Isotonic Regression

We use isotonic regression because:
- Non-parametric: No assumption about score distribution
- Monotonic: Higher raw scores -> higher calibrated scores (makes sense)
- Works well for probability calibration

### Brier Score

Measures calibration quality:
```
Brier = mean((predicted - actual)^2)
```
- 0 = perfect
- 0.25 = random guessing
- Lower is better

### Step 1: Create Core Calibrator Class (2h)

Create: `src/true_vkg/calibration/calibrator.py`

```python
from sklearn.isotonic import IsotonicRegression
from sklearn.metrics import brier_score_loss
import numpy as np
import pickle
from pathlib import Path
from typing import List, Optional
from dataclasses import dataclass

@dataclass
class CalibrationMetrics:
    """Metrics for calibration quality."""
    brier_score: float
    brier_score_before: float  # Before calibration
    improvement: float  # Percentage improvement
    n_samples: int

class ConfidenceCalibrator:
    """Calibrates raw confidence scores to meaningful probabilities."""

    def __init__(self):
        self.isotonic = IsotonicRegression(
            y_min=0.01,  # Don't go to exactly 0
            y_max=0.99,  # Don't go to exactly 1
            out_of_bounds="clip"
        )
        self._fitted = False
        self._training_metrics: Optional[CalibrationMetrics] = None

    def fit(self, raw_scores: np.ndarray, actuals: np.ndarray) -> CalibrationMetrics:
        """
        Fit calibration model.

        Args:
            raw_scores: Array of raw confidence scores (0-1)
            actuals: Array of binary labels (0 or 1 for FP/TP)

        Returns:
            CalibrationMetrics with training statistics
        """
        if len(raw_scores) < 30:
            raise ValueError(f"Need at least 30 samples, got {len(raw_scores)}")

        # Calculate Brier score before calibration
        brier_before = brier_score_loss(actuals, raw_scores)

        # Fit isotonic regression
        self.isotonic.fit(raw_scores, actuals)
        self._fitted = True

        # Calculate Brier score after calibration
        calibrated = self.calibrate_batch(raw_scores)
        brier_after = brier_score_loss(actuals, calibrated)

        # Store metrics
        self._training_metrics = CalibrationMetrics(
            brier_score=brier_after,
            brier_score_before=brier_before,
            improvement=(brier_before - brier_after) / brier_before * 100,
            n_samples=len(raw_scores)
        )

        return self._training_metrics

    def calibrate(self, raw_confidence: float) -> float:
        """
        Calibrate a single raw confidence score.

        Args:
            raw_confidence: Raw score between 0 and 1

        Returns:
            Calibrated probability
        """
        if not self._fitted:
            raise RuntimeError("Calibrator not fitted. Call fit() first.")

        return float(self.isotonic.predict([[raw_confidence]])[0])

    def calibrate_batch(self, raw_scores: np.ndarray) -> np.ndarray:
        """Calibrate multiple scores at once."""
        if not self._fitted:
            raise RuntimeError("Calibrator not fitted. Call fit() first.")

        return self.isotonic.predict(raw_scores.reshape(-1, 1)).flatten()

    def save(self, path: Path) -> None:
        """Serialize calibrator to file."""
        path = Path(path)
        path.parent.mkdir(parents=True, exist_ok=True)
        with open(path, 'wb') as f:
            pickle.dump({
                'isotonic': self.isotonic,
                'fitted': self._fitted,
                'metrics': self._training_metrics
            }, f)

    @classmethod
    def load(cls, path: Path) -> "ConfidenceCalibrator":
        """Load calibrator from file."""
        with open(path, 'rb') as f:
            data = pickle.load(f)

        calibrator = cls()
        calibrator.isotonic = data['isotonic']
        calibrator._fitted = data['fitted']
        calibrator._training_metrics = data['metrics']
        return calibrator

    @property
    def is_fitted(self) -> bool:
        return self._fitted

    @property
    def training_metrics(self) -> Optional[CalibrationMetrics]:
        return self._training_metrics
```

### Step 2: Create Training Script (1.5h)

Create: `scripts/train_calibrator.py`

```python
#!/usr/bin/env python
"""Train the confidence calibrator on labeled data."""

import argparse
import json
import numpy as np
from pathlib import Path
from true_vkg.calibration.calibrator import ConfidenceCalibrator
from true_vkg.calibration.dataset import LabeledDataset

def main():
    parser = argparse.ArgumentParser(description="Train confidence calibrator")
    parser.add_argument("--dataset", type=Path, default=Path("data/labeled/ground_truth.json"))
    parser.add_argument("--output", type=Path, default=Path("data/models/calibrator_v1.pkl"))
    parser.add_argument("--test-ratio", type=float, default=0.3)
    args = parser.parse_args()

    # Load dataset
    dataset = LabeledDataset.load(args.dataset)
    train, test = dataset.train_test_split(args.test_ratio)

    # Extract arrays
    train_scores = np.array([f.raw_confidence for f in train])
    train_labels = np.array([1 if f.is_true_positive else 0 for f in train])

    # Train calibrator
    calibrator = ConfidenceCalibrator()
    metrics = calibrator.fit(train_scores, train_labels)

    print(f"Training complete on {metrics.n_samples} samples")
    print(f"Brier score: {metrics.brier_score:.4f} (was {metrics.brier_score_before:.4f})")
    print(f"Improvement: {metrics.improvement:.1f}%")

    # Save model
    calibrator.save(args.output)
    print(f"Model saved to {args.output}")

    # Validate on test set
    test_scores = np.array([f.raw_confidence for f in test])
    test_labels = np.array([1 if f.is_true_positive else 0 for f in test])

    test_calibrated = calibrator.calibrate_batch(test_scores)
    test_brier = brier_score_loss(test_labels, test_calibrated)
    print(f"\nTest set Brier score: {test_brier:.4f}")

if __name__ == "__main__":
    main()
```

from true_vkg.calibration.calibrator import ConfidenceCalibrator
from pathlib import Path

# Add optional calibration in finding generation
class VKGBuilder:
    def __init__(self, ...):
        # Existing init
        self._calibrator: Optional[ConfidenceCalibrator] = None

    def load_calibrator(self, path: Path) -> None:
        """Load calibrator for confidence adjustment."""
        if path.exists():
            self._calibrator = ConfidenceCalibrator.load(path)

    def _get_calibrated_confidence(self, raw: float, pattern_id: str) -> float:
        """Get calibrated confidence if calibrator available."""
        if self._calibrator is None:
            return raw
        return self._calibrator.calibrate(raw)

### Step 4: Write Tests (1.5h)

Add to: `tests/test_calibration.py`

```python
import numpy as np
import pytest
from pathlib import Path
from true_vkg.calibration.calibrator import ConfidenceCalibrator, CalibrationMetrics

class TestConfidenceCalibrator:
    """Tests for ConfidenceCalibrator."""

    def test_fit_basic(self):
        """Test basic fitting."""
        calibrator = ConfidenceCalibrator()

        # Create synthetic data
        raw_scores = np.random.uniform(0, 1, 100)
        # TP probability increases with score
        actuals = (raw_scores + np.random.uniform(-0.2, 0.2, 100) > 0.5).astype(int)

        metrics = calibrator.fit(raw_scores, actuals)

        assert calibrator.is_fitted
        assert isinstance(metrics, CalibrationMetrics)
        assert metrics.n_samples == 100

    def test_calibrator_improves_brier(self):
        """Test that calibration improves Brier score."""
        calibrator = ConfidenceCalibrator()

        # Overconfident scores: all high but many FPs
        raw_scores = np.array([0.9] * 50 + [0.8] * 50)
        actuals = np.array([1] * 30 + [0] * 20 + [1] * 25 + [0] * 25)

        metrics = calibrator.fit(raw_scores, actuals)

        # Calibration should reduce overconfidence
        assert metrics.brier_score < metrics.brier_score_before
        assert metrics.improvement > 0

    def test_calibrate_single(self):
        """Test calibrating single score."""
        calibrator = ConfidenceCalibrator()
        raw = np.random.uniform(0, 1, 100)
        actuals = (raw > 0.5).astype(int)
        calibrator.fit(raw, actuals)

        result = calibrator.calibrate(0.7)

        assert 0 <= result <= 1
        assert isinstance(result, float)

    def test_calibrate_not_fitted_raises(self):
        """Test that calibrating before fit raises error."""
        calibrator = ConfidenceCalibrator()

        with pytest.raises(RuntimeError, match="not fitted"):
            calibrator.calibrate(0.5)

    def test_save_load_roundtrip(self, tmp_path):
        """Test saving and loading calibrator."""
        calibrator = ConfidenceCalibrator()
        raw = np.random.uniform(0, 1, 100)
        actuals = (raw > 0.5).astype(int)
        calibrator.fit(raw, actuals)

        # Save
        path = tmp_path / "calibrator.pkl"
        calibrator.save(path)

        # Load
        loaded = ConfidenceCalibrator.load(path)

        assert loaded.is_fitted
        assert calibrator.calibrate(0.7) == pytest.approx(loaded.calibrate(0.7))

    def test_insufficient_data_raises(self):
        """Test that fitting with too little data raises error."""
        calibrator = ConfidenceCalibrator()

        with pytest.raises(ValueError, match="at least 30"):
            calibrator.fit(np.array([0.5] * 10), np.array([1] * 10))
```

## Common Pitfalls

1. **Pitfall:** Overfitting to training data
   - **Fix:** Always validate on held-out test set

2. **Pitfall:** Calibrator making scores worse
   - **Fix:** Check input data quality, may need more samples

3. **Pitfall:** Edge cases at 0 and 1
   - **Fix:** Use y_min=0.01, y_max=0.99 in isotonic regression

4. **Pitfall:** Not handling missing calibrator gracefully
   - **Fix:** Fallback to raw score if calibrator not loaded
