class TeamManager:
    """Manage Agent Teams lifecycle for test scenarios.

    Wraps Claude Code's native TeamCreate/SendMessage/TeamDelete into a
    Python-friendly API for test harness integration.
    """

    def __init__(self, workspace_path: Path, team_name: str | None = None):
        self.workspace_path = workspace_path
        self.team_name = team_name or f"test-{uuid4().hex[:8]}"
        self.messages: list[dict] = []  # Captured SendMessage content
        self._teammates: dict[str, dict] = {}  # name -> config
        self._active = False

    def create_team(self, description: str = "") -> str:
        """Create team via TeamCreate. Returns team_name.
        Sets CLAUDE_CODE_EXPERIMENTAL_AGENT_TEAMS=1 in env.
        """
        ...

    def spawn_teammate(
        self,
        name: str,
        agent_type: str,  # subagent_type from catalog
        task_description: str,
    ) -> str:
        """Spawn a teammate into the team. Returns agent_id."""
        ...

    def send_message(
        self,
        recipient: str,
        content: str,
        message_type: str = "message",
    ) -> None:
        """Send message to teammate. Captures full content in self.messages."""
        ...

    def broadcast(self, content: str) -> None:
        """Broadcast to all teammates. Captures in self.messages."""
        ...

    def shutdown_teammate(self, name: str) -> None:
        """Send shutdown_request to specific teammate."""
        ...

    def shutdown_all(self) -> None:
        """Send shutdown_request to all teammates, wait for confirmation."""
        ...

    def delete_team(self) -> None:
        """Delete team via TeamDelete. Cleanup team files."""
        ...

    def get_team_observation(self) -> TeamObservation:
        """Build TeamObservation from captured messages and transcript data.
        Reads teammate transcripts from ~/.claude/teams/{team_name}/
        and message history from captured self.messages.
        """
        ...

    def __enter__(self): ...
    def __exit__(self, *args): ...  # Always calls shutdown_all + delete_team
```

def create_workspace(self, name: str) -> Path:
    """Create isolated Jujutsu workspace for a test run.
    Runs: jj workspace add {name} --path {workspace_path}
    Returns path to the new workspace directory.
    Raises RuntimeError if jj is not available or workspace creation fails.
    """
    ...

def forget_workspace(self, name: str) -> None:
    """Clean up a Jujutsu workspace after test run.
    Runs: jj workspace forget {name}
    Idempotent: does not error if workspace already forgotten.
    """
    ...

def rollback(self, op_id: str) -> None:
    """Rollback to a previous Jujutsu operation state.
    Runs: jj op restore {op_id}
    """
    ...

def list_workspaces(self) -> list[str]:
    """List all Jujutsu workspaces for the current repo.
    Runs: jj workspace list
    Returns list of workspace names.
    """
    ...

def snapshot_operation(self) -> str:
    """Capture current Jujutsu operation ID for later rollback.
    Runs: jj op log --limit 1 --no-graph --template 'operation_id'
    Returns operation ID string.
    """
    ...
```

**Error handling:**
- Check `shutil.which("jj")` before any Jujutsu command. Raise `RuntimeError("jj not found")`.
- All subprocess calls use `subprocess.run` with `check=True`, `capture_output=True`, `text=True`.
- Timeout: 30 seconds per Jujutsu command.
