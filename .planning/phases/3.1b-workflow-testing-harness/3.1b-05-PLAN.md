**Part A — Extend ScenarioConfig (Pydantic) with new fields:**

Add these fields to `ScenarioConfig` (all optional to maintain backward compatibility):

```python
class EvaluationConfig(BaseModel):
    """Evaluation configuration — parsed by 3.1b, executed by 3.1c."""
    contract: str | None = None          # Which evaluation contract to apply
    run_gvs: bool = True                 # Run Graph Value Scoring
    run_reasoning: bool = True           # Run reasoning evaluation

class EvaluationGuidanceConfig(BaseModel):
    """Per-scenario evaluation guidance for Claude Code evaluator."""
    reasoning_questions: list[str] = Field(default_factory=list)
    hooks_if_failed: list[str] = Field(default_factory=list)

class GraderConfig(BaseModel):
    """Grader configuration for a scenario."""
    type: Literal["code", "model"] = "code"
    method: str | None = None            # "string_match", "regex", "schema", "ai_judge"
    prompt: str | None = None            # For model grader: evaluation prompt
    schema_path: str | None = None       # For schema grader: JSON Schema path
    pattern: str | None = None           # For regex grader: regex pattern

class StepExpect(BaseModel):
    """Expected outcomes for a scenario step."""
    response_contains: list[str] = Field(default_factory=list)
    tool_was_used: list[str] = Field(default_factory=list)
    response_not_contains: list[str] = Field(default_factory=list)

class ScenarioStep(BaseModel):
    """A single step in a scenario."""
    prompt: str
    expect: StepExpect | None = None
```
