    1. **Semantic ground truth comparison (finding overlap):**
       For each calibration contract, compare agent findings against expected_findings.json:
       ```python
       import json
       from pathlib import Path

       ground_truth_dir = Path(".vrs/ground-truth")
       obs_dir = Path(".vrs/observations/plan12-retry")
       overlap_results = {}

       for cal in ["cal-01", "cal-02", "cal-03", "cal-04"]:
           # Load expected findings
           expected_path = ground_truth_dir / cal / "expected_findings.json"
           if not expected_path.exists():
               overlap_results[cal] = {"status": "no_ground_truth", "overlap": 0.0}
               continue
           expected = json.loads(expected_path.read_text())
           expected_classes = {f["vulnerability_class"] for f in expected.get("expected_findings", [])}

           # Load agent observations for this contract
           obs_files = list(obs_dir.glob(f"{cal}*.json"))
           agent_classes = set()
           for obs_file in obs_files:
               obs = json.loads(obs_file.read_text())
               for finding in obs.get("findings", []):
                   agent_classes.add(finding.get("vulnerability_class", ""))

           # Compute overlap
           if expected_classes:
               found = expected_classes & agent_classes
               missed = expected_classes - agent_classes
               spurious = agent_classes - expected_classes
               overlap = len(found) / len(expected_classes) if expected_classes else 0.0
               overlap_results[cal] = {
                   "status": "computed",
                   "overlap": round(overlap, 3),
                   "expected": sorted(expected_classes),
                   "found": sorted(found),
                   "missed": sorted(missed),
                   "spurious": sorted(spurious),
               }
           else:
               overlap_results[cal] = {"status": "no_expected_findings", "overlap": 0.0}

       print(json.dumps(overlap_results, indent=2))
       ```
       NOTE: This is a DATA QUALITY metric, not a pass/fail gate. Low overlap is informational — it indicates either (a) agents need better prompting, or (b) expected_findings need refinement. Record the results but do not fail the batch on low overlap alone.

    2. **Batch-level quality gate (DEGRADED count):**

       import json
       from pathlib import Path

       obs_dir = Path(".vrs/observations/plan12-retry")
       degraded_count = 0

       for obs_file in sorted(obs_dir.glob("cal-*.json")):

           verdict = obs.get("integrity_verdict", "UNKNOWN")
           contract_id = obs_file.stem.split("_")[0]  # e.g., "cal-01"
           contract_verdicts[contract_id] = verdict
           if verdict == "DEGRADED":
               degraded_count += 1

       print(f"Contract verdicts: {json.dumps(contract_verdicts, indent=2)}")
       print(f"DEGRADED count: {degraded_count}")
