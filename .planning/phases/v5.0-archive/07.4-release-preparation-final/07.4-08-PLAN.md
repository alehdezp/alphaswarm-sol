// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title Challenge 1: Obfuscated Reentrancy
 * @notice Can you find the reentrancy vulnerability?
 * @dev This contract appears to be a "data synchronization" service.
 *      No functions are named withdraw, transfer, or send.
 *      No variables are named balance, amount, or funds.
 *
 * Difficulty: Easy
 * Name-based tools: Will likely MISS this
 * Behavioral signature: Should CATCH this
 *
 * Hint: Look for the TRANSFERS_VALUE_OUT → WRITES_USER_BALANCE pattern
 */
contract DataSyncService {
    // "Sync quotas" - actually user balances
    mapping(address => uint256) private _syncQuotas;

    // "Sync operations" - actually withdrawal tracking
    mapping(address => uint256) private _pendingSyncs;

    event QuotaAllocated(address indexed node, uint256 quota);
    event SyncCompleted(address indexed node, uint256 dataUnits);

    /**
     * @notice Allocate sync quota (actually: deposit ETH)
     */
    function allocateQuota() external payable {
        _syncQuotas[msg.sender] += msg.value;
        emit QuotaAllocated(msg.sender, msg.value);
    }

    /**
     * @notice Execute pending sync operations (actually: VULNERABLE withdraw)
     * @dev The vulnerability: external call before state update
     */
    function executePendingSync() external {
        uint256 quota = _syncQuotas[msg.sender];
        require(quota > 0, "No quota allocated");

        // Mark as "syncing" - but this doesn't prevent reentrancy!
        _pendingSyncs[msg.sender] = quota;

        // VULNERABLE: External call BEFORE quota is cleared
        // This looks like "data transfer" but it's actually ETH transfer
        (bool synced, ) = msg.sender.call{value: quota}("");
        require(synced, "Sync failed");

        // Clear quota AFTER external call - TOO LATE!
        _syncQuotas[msg.sender] = 0;
        delete _pendingSyncs[msg.sender];

        emit SyncCompleted(msg.sender, quota);
    }

    /**
     * @notice Get node's sync quota
     */
    function getNodeQuota(address node) external view returns (uint256) {
        return _syncQuotas[node];
    }

    /**
     * @notice Get service's total capacity
     */
    function getTotalCapacity() external view returns (uint256) {
        return address(this).balance;
    }
}

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title Challenge 2: Split Logic Reentrancy
 * @notice Can you find the reentrancy when logic is split across functions?
 * @dev The vulnerability is split across two functions:
 *      - One function sends ETH
 *      - A different function updates state
 *      Name-based tools looking at single functions will miss this.
 *
 * Difficulty: Medium
 * Name-based tools: Will likely MISS this (pattern split across functions)
 * Behavioral signature: Cross-function analysis should CATCH this
 *
 * Hint: The vulnerability exists because _processOutput() and _clearPending()
 *       are called in the wrong order.
 */
contract RewardDistributor {
    mapping(address => uint256) private _pendingRewards;
    mapping(address => bool) private _isProcessing;

    event RewardAccrued(address indexed user, uint256 amount);
    event RewardClaimed(address indexed user, uint256 amount);

    /**
     * @notice Accrue rewards for user
     */
    function accrueReward() external payable {
        _pendingRewards[msg.sender] += msg.value;
        emit RewardAccrued(msg.sender, msg.value);
    }

    /**
     * @notice Claim all pending rewards
     * @dev The reentrancy is hidden because the logic is split
     */
    function claimRewards() external {
        require(!_isProcessing[msg.sender], "Already processing");
        require(_pendingRewards[msg.sender] > 0, "No rewards");

        _isProcessing[msg.sender] = true;

        // Step 1: Process the output (sends ETH)
        _processOutput(msg.sender);

        // Step 2: Clear pending (updates state)
        // VULNERABLE: This happens AFTER the ETH is sent!
        _clearPending(msg.sender);

        _isProcessing[msg.sender] = false;
    }

    /**
     * @notice Internal: Process output to user
     * @dev This function alone looks safe - it just sends ETH
     */
    function _processOutput(address user) internal {
        uint256 reward = _pendingRewards[user];

        // External call happens here
        (bool success, ) = user.call{value: reward}("");
        require(success, "Output failed");

        emit RewardClaimed(user, reward);
    }

    /**
     * @notice Internal: Clear pending rewards
     * @dev This function alone looks safe - it just clears state
     */
    function _clearPending(address user) internal {
        _pendingRewards[user] = 0;
    }

    /**
     * @notice Get pending rewards for user
     */
    function pendingRewards(address user) external view returns (uint256) {
        return _pendingRewards[user];
    }
}

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title Challenge 3: Callback Disguise
 * @notice Can you find reentrancy hidden in a "callback" pattern?
 * @dev This contract uses a "hook" pattern that's common in DeFi.
 *      The vulnerability is disguised as an intentional callback mechanism.
 *      Name-based tools might flag this as "intentional reentrancy."
 *
 * Difficulty: Hard
 * Name-based tools: May flag as intentional callback (false negative)
 * Behavioral signature: Should identify the vulnerable pattern regardless
 *
 * Hint: The "hook" sends ETH before updating internal accounting.
 */
interface IVaultHook {
    function onWithdrawCallback(address user, uint256 amount) external;
}

contract HookableVault {
    mapping(address => uint256) private _deposits;
    mapping(address => IVaultHook) private _hooks;
    mapping(address => bool) private _hookEnabled;

    event Deposited(address indexed user, uint256 amount);
    event Withdrawn(address indexed user, uint256 amount);
    event HookRegistered(address indexed user, address hook);

    /**
     * @notice Deposit ETH into vault
     */
    function deposit() external payable {
        _deposits[msg.sender] += msg.value;
        emit Deposited(msg.sender, msg.value);
    }

    /**
     * @notice Register a callback hook (common DeFi pattern)
     */
    function registerHook(address hook) external {
        _hooks[msg.sender] = IVaultHook(hook);
        _hookEnabled[msg.sender] = true;
        emit HookRegistered(msg.sender, hook);
    }

    /**
     * @notice Withdraw with optional callback hook
     * @dev VULNERABLE: The hook is called (with ETH) before state update.
     *      This looks like an "intentional callback" but the order is wrong.
     */
    function withdraw(uint256 amount) external {
        require(_deposits[msg.sender] >= amount, "Insufficient deposit");

        // Call hook if enabled - this is where reentrancy can happen
        // The hook receives ETH before _deposits is updated!
        if (_hookEnabled[msg.sender]) {
            // VULNERABLE: Hook called with value BEFORE state update
            // A malicious hook can re-enter withdraw()
            _hooks[msg.sender].onWithdrawCallback{value: amount}(msg.sender, amount);
        } else {
            // Direct transfer also vulnerable
            (bool success, ) = msg.sender.call{value: amount}("");
            require(success, "Transfer failed");
        }

        // State update happens AFTER external call - VULNERABLE!
        _deposits[msg.sender] -= amount;

        emit Withdrawn(msg.sender, amount);
    }

    /**
     * @notice Get user's deposit balance
     */
    function balanceOf(address user) external view returns (uint256) {
        return _deposits[user];
    }
}

/**
 * @title Malicious Hook (for testing)
 * @dev This hook demonstrates the exploit
 */
contract MaliciousHook is IVaultHook {
    HookableVault private _vault;
    address private _owner;
    uint256 private _attackCount;

    constructor(address vault) {
        _vault = HookableVault(vault);
        _owner = msg.sender;
    }

    function onWithdrawCallback(address, uint256 amount) external {
        // Re-enter if we still have "balance" (state not yet updated)
        if (_attackCount < 3 && address(_vault).balance >= amount) {
            _attackCount++;
            _vault.withdraw(amount);
        }
    }

    function attack(uint256 amount) external {
        require(msg.sender == _owner, "Only owner");
        _attackCount = 0;
        _vault.withdraw(amount);
    }

    receive() external payable {}
}
```

# Adversarial Challenge Pack

**Purpose:** Prove that behavioral signatures catch vulnerabilities that name-based heuristics miss.

## The Challenge

Each contract in this pack contains a **reentrancy vulnerability** that is designed to evade name-based detection tools like Slither's basic detectors.

Your mission: Use AlphaSwarm to find the vulnerabilities.

## Challenges

| # | Contract | Difficulty | Evasion Technique |
|---|----------|------------|-------------------|
| 1 | challenge-1-obfuscated-reentrancy.sol | Easy | Misleading names |
| 2 | challenge-2-split-logic.sol | Medium | Logic split across functions |
| 3 | challenge-3-callback-disguise.sol | Hard | Disguised as callback pattern |

## How to Play

```bash
# Build graph for a challenge
alphaswarm build-kg examples/demo/adversarial/challenge-1-obfuscated-reentrancy.sol

# Try to find the vulnerability
alphaswarm query "pattern:reentrancy-classic"

# Or use behavioral signature query
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE"
```

## Why This Matters

Traditional static analysis tools often rely on:
- Function name patterns (withdraw, transfer, send)
- Variable name patterns (balance, amount, funds)
- Single-function analysis

These contracts demonstrate that:
1. Renaming functions defeats name-based detection
2. Splitting logic across functions defeats single-function analysis
3. Disguising vulnerabilities as "features" defeats pattern matching

**Behavioral signatures detect WHAT code does, not what it's called.**

The answers are in `answers.md`, but try to find the vulnerabilities yourself first.

## Verification

After finding a vulnerability, verify it by:
1. Checking the behavioral signature
2. Tracing the operation ordering
3. Confirming external call happens before state update

- Challenge 2: 2 points (requires cross-function analysis)
- Challenge 3: 3 points (requires understanding callback patterns)

**Perfect score: 6 points**

# Adversarial Challenge Answers

**SPOILER WARNING:** Don't read until you've attempted the challenges!

---

## Challenge 1: Obfuscated Reentrancy

**Location:** `DataSyncService.executePendingSync()`

**Vulnerability:** Classic reentrancy with obfuscated naming

**Analysis:**
```
Function: executePendingSync()
Operations: READS_USER_BALANCE → TRANSFERS_VALUE_OUT → WRITES_USER_BALANCE
Signature: R:bal → X:out → W:bal (VULNERABLE)
```

**Why name-based tools miss it:**
- Function is named `executePendingSync()`, not `withdraw()`
- Variable is named `_syncQuotas`, not `balance`
- ETH transfer is disguised as "data sync"

**Why AlphaSwarm catches it:**
- Detects TRANSFERS_VALUE_OUT (the `msg.sender.call{value: quota}`)
- Detects WRITES_USER_BALANCE (the `_syncQuotas[msg.sender] = 0`)
- Identifies vulnerable ordering (transfer before write)

**Detection query:**
```bash
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE"
```

---

## Challenge 2: Split Logic Reentrancy

**Location:** `RewardDistributor.claimRewards()` → `_processOutput()` + `_clearPending()`

**Vulnerability:** Reentrancy with logic split across multiple functions

**Analysis:**
```
Call chain: claimRewards() → _processOutput() → _clearPending()
Operations in _processOutput(): TRANSFERS_VALUE_OUT
Operations in _clearPending(): WRITES_USER_BALANCE
Combined: TRANSFERS_VALUE_OUT → WRITES_USER_BALANCE (VULNERABLE)
```

**Why name-based tools miss it:**
- Each function looks safe in isolation
- `_processOutput()` just sends ETH
- `_clearPending()` just clears state
- The vulnerability only exists in the calling order

**Why AlphaSwarm catches it:**
- Cross-function analysis traces the call chain
- Identifies that `_processOutput()` (with transfer) happens before `_clearPending()` (with write)
- Constructs combined behavioral signature

**Detection query:**
```bash
alphaswarm query "external calls followed by state writes in different functions"
```

---

## Challenge 3: Callback Disguise

**Location:** `HookableVault.withdraw()`

**Vulnerability:** Reentrancy disguised as intentional callback mechanism

**Analysis:**
```
Function: withdraw()
Hook call: _hooks[msg.sender].onWithdrawCallback{value: amount}(...)
State update: _deposits[msg.sender] -= amount
Signature: X:hook_with_value → W:bal (VULNERABLE)
```

**Why name-based tools miss it:**
- Hooks are a legitimate DeFi pattern
- The external call is to a user-registered callback
- Tools might flag this as "intentional reentrancy"
- The pattern looks like ERC-777 or similar callback patterns

**Why AlphaSwarm catches it:**
- Regardless of intent, the behavioral signature is vulnerable
- Hook receives ETH BEFORE state is updated
- A malicious hook can re-enter and drain funds
- Behavioral analysis doesn't care if it "looks intentional"

**Detection query:**
```bash
alphaswarm query "callback patterns with value transfer before state update"
```

**Key insight:** Just because something is a "pattern" doesn't make it safe. The callback pattern is only safe if:
1. State is updated BEFORE the callback, OR
2. A reentrancy guard is used
