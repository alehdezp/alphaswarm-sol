Create a Python script that generates appendices from source documentation.

The script should:
1. Parse `docs/reference/operations.md` to extract operation definitions
2. Scan `vulndocs/` to collect pattern metadata (count by category, tier, severity)
3. Extract behavioral signature examples from pattern files
4. Generate formatted markdown appendices

```python
#!/usr/bin/env python3
"""Generate paper appendices from source documentation.

Usage:
    uv run python scripts/generate_paper_appendices.py

Outputs:
    docs/paper/appendices/A-operations.md
    docs/paper/appendices/B-patterns.md
    docs/paper/appendices/C-signatures.md
"""

import re
import yaml
from pathlib import Path
from collections import defaultdict
from typing import Any

def extract_operations(reference_path: Path) -> list[dict]:
    """Extract operation definitions from reference doc."""
    content = reference_path.read_text()
    operations = []

    # Parse markdown table for operations
    table_pattern = r'\|\s*`?(\w+)`?\s*\|\s*([^|]+)\s*\|\s*([^|]+)\s*\|'
    for match in re.finditer(table_pattern, content):
        name = match.group(1).strip()
        if name.upper() == name and name not in ('OPERATION', 'CATEGORY'):  # Skip headers
            operations.append({
                'name': name,
                'category': match.group(2).strip(),
                'description': match.group(3).strip(),
            })

    return operations

def collect_patterns(vulndocs_dir: Path) -> dict[str, Any]:
    """Collect pattern metadata from VulnDocs."""
    patterns = []
    by_category = defaultdict(list)
    by_tier = defaultdict(int)
    by_severity = defaultdict(int)

    for pattern_file in vulndocs_dir.rglob('patterns/*.yaml'):
        try:
            with open(pattern_file) as f:
                pattern = yaml.safe_load(f)
                if not pattern:
                    continue

                p = {
                    'id': pattern.get('id', 'unknown'),
                    'name': pattern.get('name', 'Unknown'),
                    'severity': pattern.get('severity', 'medium'),
                    'tier': 'A' if 'tier_a' in pattern.get('match', {}) else ('C' if 'tier_c' in pattern.get('match', {}) else 'B'),
                    'category': pattern_file.parent.parent.name,
                }
                patterns.append(p)
                by_category[p['category']].append(p)
                by_tier[p['tier']] += 1
                by_severity[p['severity']] += 1
        except Exception as e:
            print(f"Warning: Could not parse {pattern_file}: {e}")

    return {
        'patterns': patterns,
        'by_category': dict(by_category),
        'by_tier': dict(by_tier),
        'by_severity': dict(by_severity),
        'total': len(patterns),
    }

def extract_signatures(vulndocs_dir: Path) -> list[dict]:
    """Extract behavioral signature examples from patterns."""
    signatures = []

    # Look for patterns with explicit signature information
    signature_patterns = [
        ('reentrancy', 'R:bal → X:out → W:bal', 'R:bal → W:bal → X:out', 'Read balance, external call, write balance vs CEI pattern'),
        ('access-control', 'X:action', 'C:perm → X:action', 'Action without permission check vs guarded action'),
        ('oracle-manipulation', 'R:oracle → X:action', 'R:oracle → V:price → X:action', 'Unvalidated oracle read vs validated price check'),
        ('flash-loan', 'R:bal → L:flash → X:action → W:bal', 'R:bal → L:flash → X:action → V:repay → W:bal', 'Flash loan without repayment check vs validated'),
        ('state-update', 'X:call → W:state', 'W:state → X:call', 'External call before state update vs state-first'),
    ]

    for name, vuln_sig, safe_sig, description in signature_patterns:
        signatures.append({
            'name': name.replace('-', ' ').title(),
            'vulnerable_sig': vuln_sig,
            'safe_sig': safe_sig,
            'description': description,
        })

    return signatures

def format_operations_appendix(operations: list[dict]) -> str:
    """Format Appendix A: Operations table."""
    lines = [
        "# Appendix A: Semantic Operations Reference",
        "",
        "Complete catalog of semantic operations used for behavioral signature detection.",
        "These operations describe code behavior independently of function or variable names.",
        "",
        "## Operations by Category",
        "",
    ]

    # Group by category
    by_cat = defaultdict(list)
    for op in operations:
        by_cat[op['category']].append(op)

    for category in sorted(by_cat.keys()):
        lines.append(f"### {category}")
        lines.append("")
        lines.append("| Operation | Description |")
        lines.append("|-----------|-------------|")
        for op in sorted(by_cat[category], key=lambda x: x['name']):
            lines.append(f"| `{op['name']}` | {op['description']} |")
        lines.append("")

    return '\n'.join(lines)

def format_patterns_appendix(data: dict[str, Any]) -> str:
    """Format Appendix B: Pattern catalog summary."""
    lines = [
        "# Appendix B: Pattern Catalog Summary",
        "",
        f"Summary of {data['total']} vulnerability detection patterns across {len(data['by_category'])} categories.",
        "",
        "## Overview",
        "",
        "| Metric | Count |",
        "|--------|-------|",
        f"| Total Patterns | {data['total']} |",
        f"| Tier A (High Confidence) | {data['by_tier'].get('A', 0)} |",
        f"| Tier B (LLM-Verified) | {data['by_tier'].get('B', 0)} |",
        f"| Tier C (Label-Dependent) | {data['by_tier'].get('C', 0)} |",
        "",
        "## Patterns by Severity",
        "",
        "| Severity | Count |",
        "|----------|-------|",
    ]

    for sev in ['critical', 'high', 'medium', 'low', 'info']:
        if sev in data['by_severity']:
            lines.append(f"| {sev.title()} | {data['by_severity'][sev]} |")

    lines.extend([
        "",
        "## Patterns by Category",
        "",
        "| Category | Count | Example Pattern |",
        "|----------|-------|-----------------|",
    ])

    for cat in sorted(data['by_category'].keys()):
        patterns = data['by_category'][cat]
        example = patterns[0]['name'] if patterns else 'N/A'
        lines.append(f"| {cat} | {len(patterns)} | {example} |")

    lines.extend([
        "",
        "---",
        "",
        "*Full pattern catalog available in `vulndocs/` directory.*",
    ])

    return '\n'.join(lines)

def format_signatures_appendix(signatures: list[dict]) -> str:
    """Format Appendix C: Behavioral signature examples."""
    lines = [
        "# Appendix C: Behavioral Signature Examples",
        "",
        "Behavioral signatures encode the ordering of semantic operations.",
        "Vulnerable patterns often differ from safe patterns only in operation order.",
        "",
        "## Signature Notation",
        "",
        "- `R:` = Read operation",
        "- `W:` = Write operation",
        "- `X:` = External call",
        "- `C:` = Check/validation",
        "- `V:` = Verify/validate",
        "- `L:` = Loan/borrow",
        "- `→` = Followed by (temporal ordering)",
        "",
        "## Examples",
        "",
    ]

    for sig in signatures:
        lines.extend([
            f"### {sig['name']}",
            "",
            f"**Description:** {sig['description']}",
            "",
            "| Pattern | Signature |",
            "|---------|-----------|",
            f"| Vulnerable | `{sig['vulnerable_sig']}` |",
            f"| Safe | `{sig['safe_sig']}` |",
            "",
        ])

    lines.extend([
        "---",
        "",
        "## Key Insight",
        "",
        "The same operations can be safe or vulnerable depending on their order.",
        "Behavioral signatures capture this ordering, enabling detection regardless of",
        "function names, variable names, or code structure.",
    ])

    return '\n'.join(lines)

def main():
    root = Path(__file__).parent.parent
    output_dir = root / "docs" / "paper" / "appendices"
    output_dir.mkdir(parents=True, exist_ok=True)

    print("Generating paper appendices...")

    # Generate Appendix A: Operations
    ops_path = root / "docs" / "reference" / "operations.md"
    if ops_path.exists():
        operations = extract_operations(ops_path)
        appendix_a = format_operations_appendix(operations)
        (output_dir / "A-operations.md").write_text(appendix_a)
        print(f"  ✓ Appendix A: {len(operations)} operations")
    else:
        print(f"  ⚠ Skipping Appendix A: {ops_path} not found")

    # Generate Appendix B: Patterns
    vulndocs_path = root / "vulndocs"
    if vulndocs_path.exists():
        pattern_data = collect_patterns(vulndocs_path)
        appendix_b = format_patterns_appendix(pattern_data)
        (output_dir / "B-patterns.md").write_text(appendix_b)
        print(f"  ✓ Appendix B: {pattern_data['total']} patterns across {len(pattern_data['by_category'])} categories")
    else:
        print(f"  ⚠ Skipping Appendix B: {vulndocs_path} not found")

    # Generate Appendix C: Signatures
    signatures = extract_signatures(vulndocs_path if vulndocs_path.exists() else root)
    appendix_c = format_signatures_appendix(signatures)
    (output_dir / "C-signatures.md").write_text(appendix_c)
    print(f"  ✓ Appendix C: {len(signatures)} signature examples")

    print("\nDone! Appendices written to docs/paper/appendices/")

if __name__ == "__main__":
    main()
```
