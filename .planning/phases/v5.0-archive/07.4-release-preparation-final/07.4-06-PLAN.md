Create `examples/demo/tours/cli-tour.sh`:

```bash
#!/usr/bin/env bash
# AlphaSwarm CLI Tour: "Defend the Vault"
# A guided walkthrough demonstrating behavioral signature detection.
#
# Usage: bash examples/demo/tours/cli-tour.sh
# Time: ~3 minutes
#
# This script:
# 1. Builds graphs for vulnerable and safe contracts
# 2. Runs vulnerability queries
# 3. Shows the key insight: names don't matter, behavior does

set -e  # Exit on error

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo -e "${BLUE}  AlphaSwarm CLI Tour: Defend the Vault${NC}"
echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo ""

# Step 1: Check installation
echo -e "${YELLOW}Step 1: Checking AlphaSwarm installation...${NC}"
if ! command -v alphaswarm &> /dev/null; then
    echo -e "${RED}Error: alphaswarm not found. Install with:${NC}"
    echo "  pipx install dist/alphaswarm_sol-*.whl"
    exit 1
fi
echo -e "${GREEN}✓ AlphaSwarm version: $(alphaswarm --version)${NC}"
echo ""

# Step 2: Analyze vulnerable contract
echo -e "${YELLOW}Step 2: Building graph for VulnerableVault.sol...${NC}"
echo "  This contract has a classic reentrancy vulnerability."
echo ""
alphaswarm build-kg examples/demo/contracts/VulnerableVault.sol
echo -e "${GREEN}✓ Graph built${NC}"
echo ""

# Step 3: Find vulnerabilities
echo -e "${YELLOW}Step 3: Finding vulnerabilities...${NC}"
echo "  Query: 'functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE'"
echo ""
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE"
echo ""
echo -e "${RED}  ^ These functions have the vulnerable pattern!${NC}"
echo ""

# Step 4: Analyze safe contract
echo -e "${YELLOW}Step 4: Building graph for SafeVault.sol...${NC}"
echo "  This contract uses the safe CEI pattern."
echo ""
alphaswarm build-kg examples/demo/contracts/SafeVault.sol
echo -e "${GREEN}✓ Graph built${NC}"
echo ""

# Step 5: Verify safe contract has no findings
echo -e "${YELLOW}Step 5: Running same query on SafeVault...${NC}"
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE" || true
echo ""
echo -e "${GREEN}  ^ No findings! The CEI pattern is safe.${NC}"
echo ""

# Step 6: The key insight - adversarial naming
echo -e "${YELLOW}Step 6: Testing behavioral signatures (the key insight)...${NC}"
echo ""
echo "  AdversarialNaming.sol has the SAME vulnerability as VulnerableVault,"
echo "  but with completely different function names:"
echo "    - withdraw() → processPendingOperations()"
echo "    - withdrawAmount() → syncExternalState()"
echo ""
echo "  Name-based tools would miss this. Let's see if AlphaSwarm catches it..."
echo ""

alphaswarm build-kg examples/demo/contracts/AdversarialNaming.sol
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE"
echo ""
echo -e "${GREEN}  ✓ AlphaSwarm detects the vulnerability regardless of naming!${NC}"
echo ""

# Summary
echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo -e "${BLUE}  Tour Complete${NC}"
echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo ""
echo "Key Takeaways:"
echo "  1. Behavioral signatures detect WHAT code does, not what it's called"
echo "  2. VulnerableVault and AdversarialNaming have identical signatures"
echo "  3. SafeVault uses CEI pattern (state update before external call)"
echo ""
echo "Next Steps:"
echo "  - Read the paper: docs/paper/PAPER.md"
echo "  - Explore patterns: vulndocs/"
echo "  - Run: alphaswarm --help"
echo ""
```

Create `examples/demo/tours/secure-review.sh`:

```bash
#!/usr/bin/env bash
# AlphaSwarm Secure Review Workflow
# A step-by-step secure code review using behavioral signatures.
#
# Usage: bash examples/demo/tours/secure-review.sh <path/to/contracts/>
# Time: ~5 minutes per contract
#
# This script demonstrates the 3-step secure review workflow:
# 1. Build graph
# 2. Run targeted queries
# 3. Export findings pack

set -e

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

# Check arguments
if [ -z "$1" ]; then
    echo "Usage: $0 <path/to/contracts/>"
    echo ""
    echo "Example:"
    echo "  $0 examples/demo/contracts/VulnerableVault.sol"
    echo "  $0 ./my-project/contracts/"
    exit 1
fi

CONTRACT_PATH="$1"
TIMESTAMP=$(date +%Y%m%d-%H%M%S)
OUTPUT_DIR="reports/${TIMESTAMP}"

echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo -e "${BLUE}  AlphaSwarm Secure Review${NC}"
echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo ""
echo "Target: ${CONTRACT_PATH}"
echo "Output: ${OUTPUT_DIR}/"
echo ""

# Create output directory
mkdir -p "${OUTPUT_DIR}"

# Step 1: Build knowledge graph
echo -e "${YELLOW}Step 1/3: Building Knowledge Graph...${NC}"
alphaswarm build-kg "${CONTRACT_PATH}" -o "${OUTPUT_DIR}/graph.toon" 2>&1 | tee "${OUTPUT_DIR}/build.log"
echo -e "${GREEN}✓ Graph built: ${OUTPUT_DIR}/graph.toon${NC}"
echo ""

# Step 2: Run targeted security queries
echo -e "${YELLOW}Step 2/3: Running Security Queries...${NC}"

{
    echo "# Security Findings Report"
    echo "Generated: $(date)"
    echo "Target: ${CONTRACT_PATH}"
    echo ""
    echo "## Query Results"
    echo ""

    echo "### 1. Reentrancy Patterns"
    echo '```'
    alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE" --graph "${OUTPUT_DIR}/graph.toon" 2>&1 || echo "No findings"
    echo '```'
    echo ""

    echo "### 2. Access Control Issues"
    echo '```'
    alphaswarm query "public functions that write state without access control" --graph "${OUTPUT_DIR}/graph.toon" 2>&1 || echo "No findings"
    echo '```'
    echo ""

    echo "### 3. Pattern-Based Detection"
    echo '```'
    alphaswarm query "pattern:reentrancy-classic" --graph "${OUTPUT_DIR}/graph.toon" 2>&1 || echo "No findings"
    echo '```'
    echo ""

    echo "### 4. External Calls Without Guards"
    echo '```'
    alphaswarm query "external calls without reentrancy guard" --graph "${OUTPUT_DIR}/graph.toon" 2>&1 || echo "No findings"
    echo '```'

} > "${OUTPUT_DIR}/findings.md"

echo -e "${GREEN}✓ Findings written: ${OUTPUT_DIR}/findings.md${NC}"
echo ""

# Step 3: Create findings pack
echo -e "${YELLOW}Step 3/3: Creating Findings Pack...${NC}"

# Copy graph as evidence
cp "${OUTPUT_DIR}/graph.toon" "${OUTPUT_DIR}/evidence.toon"

# Create summary
{
    echo "# Secure Review Summary"
    echo ""
    echo "- Target: ${CONTRACT_PATH}"
    echo "- Date: $(date)"
    echo "- Reviewer Tool: AlphaSwarm $(alphaswarm --version 2>/dev/null || echo 'unknown')"
    echo ""
    echo "## Files"
    echo ""
    echo "| File | Description |"
    echo "|------|-------------|"
    echo "| findings.md | Human-readable findings |"
    echo "| evidence.toon | BSKG evidence bundle |"
    echo "| build.log | Graph build log |"
    echo ""
    echo "## Next Steps"
    echo ""
    echo "1. Review findings.md for detected issues"
    echo "2. Verify findings against source code"
    echo "3. Archive this folder for audit records"

} > "${OUTPUT_DIR}/SUMMARY.md"

echo -e "${GREEN}✓ Summary written: ${OUTPUT_DIR}/SUMMARY.md${NC}"
echo ""

# Final output
echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo -e "${BLUE}  Review Complete${NC}"
echo -e "${BLUE}═══════════════════════════════════════════════════════════════${NC}"
echo ""
echo "Findings pack created: ${OUTPUT_DIR}/"
echo ""
ls -la "${OUTPUT_DIR}/"
echo ""
echo "To view findings:"
echo "  cat ${OUTPUT_DIR}/findings.md"
echo ""
```
