```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title VulnerableVault
 * @notice Demo contract with CLASSIC REENTRANCY vulnerability.
 * @dev This contract intentionally contains a reentrancy vulnerability
 *      for demonstration purposes. DO NOT use in production.
 *
 * Vulnerability: External call (transfer) BEFORE state update (balance decrement).
 * Behavioral Signature: R:bal → X:out → W:bal (vulnerable pattern)
 * Safe Pattern Would Be: R:bal → W:bal → X:out (CEI pattern)
 */
contract VulnerableVault {
    mapping(address => uint256) public balances;

    event Deposit(address indexed user, uint256 amount);
    event Withdrawal(address indexed user, uint256 amount);

    /**
     * @notice Deposit ETH into the vault
     */
    function deposit() external payable {
        balances[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    /**
     * @notice Withdraw ETH from the vault
     * @dev VULNERABLE: External call before state update
     *
     * Attack scenario:
     * 1. Attacker deposits 1 ETH
     * 2. Attacker calls withdraw()
     * 3. Contract sends ETH via call (triggers attacker's receive/fallback)
     * 4. Attacker's fallback re-enters withdraw()
     * 5. Balance check passes (not yet updated)
     * 6. Loop until vault drained
     */
    function withdraw() external {
        uint256 balance = balances[msg.sender];
        require(balance > 0, "No balance");

        // VULNERABLE: External call BEFORE state update
        (bool success, ) = msg.sender.call{value: balance}("");
        require(success, "Transfer failed");

        // State update happens AFTER external call - TOO LATE!
        balances[msg.sender] = 0;

        emit Withdrawal(msg.sender, balance);
    }

    /**
     * @notice Withdraw a specific amount
     * @dev Also vulnerable to reentrancy
     */
    function withdrawAmount(uint256 amount) external {
        require(balances[msg.sender] >= amount, "Insufficient balance");

        // VULNERABLE: Same pattern - external call before state update
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");

        balances[msg.sender] -= amount;

        emit Withdrawal(msg.sender, amount);
    }

    /**
     * @notice Get contract balance
     */
    function getContractBalance() external view returns (uint256) {
        return address(this).balance;
    }

    /**
     * @notice Get user balance
     */
    function getUserBalance(address user) external view returns (uint256) {
        return balances[user];
    }
}
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title SafeVault
 * @notice Demo contract with CORRECT CEI pattern (no reentrancy).
 * @dev This contract demonstrates the safe Checks-Effects-Interactions pattern.
 *
 * Safe Pattern: State update BEFORE external call.
 * Behavioral Signature: R:bal → W:bal → X:out (CEI pattern)
 * Vulnerable Pattern Would Be: R:bal → X:out → W:bal
 */
contract SafeVault {
    mapping(address => uint256) public balances;

    event Deposit(address indexed user, uint256 amount);
    event Withdrawal(address indexed user, uint256 amount);

    /**
     * @notice Deposit ETH into the vault
     */
    function deposit() external payable {
        balances[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    /**
     * @notice Withdraw ETH from the vault
     * @dev SAFE: Follows Checks-Effects-Interactions pattern
     *
     * Why this is safe:
     * 1. CHECK: Verify balance > 0
     * 2. EFFECT: Update state (set balance to 0) BEFORE external call
     * 3. INTERACTION: Then send ETH
     *
     * If attacker re-enters, balance is already 0, require fails.
     */
    function withdraw() external {
        uint256 balance = balances[msg.sender];
        require(balance > 0, "No balance");  // CHECK

        // SAFE: State update BEFORE external call
        balances[msg.sender] = 0;            // EFFECT

        // External call happens AFTER state is updated
        (bool success, ) = msg.sender.call{value: balance}("");  // INTERACTION
        require(success, "Transfer failed");

        emit Withdrawal(msg.sender, balance);
    }

    /**
     * @notice Withdraw a specific amount
     * @dev Also follows CEI pattern
     */
    function withdrawAmount(uint256 amount) external {
        require(balances[msg.sender] >= amount, "Insufficient balance");  // CHECK

        // SAFE: State update BEFORE external call
        balances[msg.sender] -= amount;  // EFFECT

        (bool success, ) = msg.sender.call{value: amount}("");  // INTERACTION
        require(success, "Transfer failed");

        emit Withdrawal(msg.sender, amount);
    }

    /**
     * @notice Get contract balance
     */
    function getContractBalance() external view returns (uint256) {
        return address(this).balance;
    }

    /**
     * @notice Get user balance
     */
    function getUserBalance(address user) external view returns (uint256) {
        return balances[user];
    }
}

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title AdversarialNaming
 * @notice Demo contract with OBFUSCATED function names to test behavioral detection.
 * @dev This contract contains the SAME vulnerabilities as VulnerableVault,
 *      but with intentionally misleading function names designed to evade
 *      name-based heuristic detection.
 *
 * Purpose: Prove that behavioral signature detection works regardless of naming.
 * Name-based tools look for: withdraw, transfer, send, etc.
 * AlphaSwarm looks for: TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE
 */
contract AdversarialNaming {
    // Misleading variable name - not obviously a balance
    mapping(address => uint256) private _userMetrics;

    // Events with misleading names
    event MetricUpdated(address indexed subject, uint256 delta);
    event ProcessComplete(address indexed subject, uint256 value);

    /**
     * @notice "Calibrate" user metrics (actually: deposit ETH)
     */
    function calibrateMetrics() external payable {
        _userMetrics[msg.sender] += msg.value;
        emit MetricUpdated(msg.sender, msg.value);
    }

    /**
     * @notice "Process pending operations" (actually: VULNERABLE withdraw)
     * @dev SAME VULNERABILITY as VulnerableVault.withdraw()
     *      but with a name that won't trigger "withdraw" heuristics
     *
     * Behavioral signature is IDENTICAL: R:bal → X:out → W:bal
     */
    function processPendingOperations() external {
        uint256 metric = _userMetrics[msg.sender];
        require(metric > 0, "No pending operations");

        // VULNERABLE: External call BEFORE state update
        // Name-based tools won't flag this as "withdrawal"
        (bool success, ) = msg.sender.call{value: metric}("");
        require(success, "Operation failed");

        // State update AFTER - vulnerable!
        _userMetrics[msg.sender] = 0;

        emit ProcessComplete(msg.sender, metric);
    }

    /**
     * @notice "Sync external state" (actually: VULNERABLE partial withdraw)
     * @dev Another misleading name for the same vulnerable pattern
     */
    function syncExternalState(uint256 delta) external {
        require(_userMetrics[msg.sender] >= delta, "Sync overflow");

        // VULNERABLE: Same pattern
        (bool success, ) = msg.sender.call{value: delta}("");
        require(success, "Sync failed");

        _userMetrics[msg.sender] -= delta;
    }

    /**
     * @notice "Run internal diagnostics" (actually: view balance)
     */
    function runDiagnostics(address subject) external view returns (uint256) {
        return _userMetrics[subject];
    }

    /**
     * @notice "Get system status" (actually: view contract balance)
     */
    function getSystemStatus() external view returns (uint256) {
        return address(this).balance;
    }
}

# Expected Findings: VulnerableVault.sol

## Finding 1: Classic Reentrancy

**Severity:** Critical
**Function:** `withdraw()`
**Pattern:** reentrancy-classic

### Evidence
- **Operation Sequence:** `READS_USER_BALANCE → TRANSFERS_VALUE_OUT → WRITES_USER_BALANCE`
- **Vulnerable Signature:** `R:bal → X:out → W:bal`
- **Safe Signature Would Be:** `R:bal → W:bal → X:out` (CEI pattern)
- **Line Numbers:** 42-52

### Why This Is Vulnerable
1. Function reads `balances[msg.sender]` (line 43)
2. Sends ETH via `msg.sender.call{value: balance}()` (line 46)
3. Updates `balances[msg.sender] = 0` AFTER the call (line 49)

If the receiver is a contract with a `receive()` or `fallback()` function,
it can re-enter `withdraw()` before the balance is zeroed.

### Verification Queries
```bash
# Query 1: Find reentrancy pattern
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE"

# Query 2: Pattern-based detection
alphaswarm query "pattern:reentrancy-classic"

# Query 3: General vulnerability search
alphaswarm query "external calls without reentrancy guard"
```

## Finding 2: Reentrancy in withdrawAmount()

**Severity:** Critical
**Function:** `withdrawAmount(uint256)`
**Pattern:** reentrancy-classic

Same vulnerability pattern in the `withdrawAmount` function (lines 56-66).

---

## What AlphaSwarm Should NOT Find

- SafeVault.sol uses CEI pattern → NO reentrancy findings
- The `deposit()` function has no external calls → safe

# Expected Findings: AdversarialNaming.sol

## Purpose of This Contract

This contract demonstrates that **behavioral signature detection works regardless of naming**.

The contract has the EXACT SAME vulnerabilities as VulnerableVault.sol, but with intentionally misleading names:

| VulnerableVault | AdversarialNaming | Same Vulnerability? |
|-----------------|-------------------|---------------------|
| `withdraw()` | `processPendingOperations()` | YES - both reentrancy |
| `withdrawAmount()` | `syncExternalState()` | YES - both reentrancy |
| `balances` | `_userMetrics` | Same state variable pattern |

## Finding 1: Reentrancy in processPendingOperations()

**Severity:** Critical
**Function:** `processPendingOperations()`
**Pattern:** reentrancy-classic

### Why Name-Based Tools Miss This

Name-based heuristics look for:
- Function names: `withdraw`, `transfer`, `send`
- Variable names: `balance`, `amount`

This contract uses:
- `processPendingOperations()` - not flagged as "withdrawal"
- `_userMetrics` - not flagged as "balance"

### Why AlphaSwarm Catches This

AlphaSwarm detects the **behavioral signature**:
- `READS_USER_BALANCE → TRANSFERS_VALUE_OUT → WRITES_USER_BALANCE`

The signature is IDENTICAL to VulnerableVault.withdraw().
Names don't matter. Behavior does.

### Verification Query
```bash
# This query catches BOTH contracts with the same pattern
alphaswarm query "functions with TRANSFERS_VALUE_OUT before WRITES_USER_BALANCE"
```

## Key Insight

**Names lie. Behavior does not.**

A malicious developer could rename `withdraw()` to `processRefund()` or `handleCallback()`.
Name-based tools would miss it. AlphaSwarm would catch it.
```
