           async def ingest(
               self,
               url: str,
               category: str | None = None,
               dry_run: bool = False,
               quality_threshold: str = "draft"
           ) -> IngestionResult:

               extracted = self.extractor.extract(content)

               # 4. Categorize
               target = self.categorizer.categorize(extracted, category)

               # 5. Check duplicates
               duplicates = self._check_duplicates(extracted, target)

               # 6. Insert if not dry_run
               if not dry_run:
                   self._insert(extracted, target)
                   self._update_provenance(target, url)

               return IngestionResult(...)
