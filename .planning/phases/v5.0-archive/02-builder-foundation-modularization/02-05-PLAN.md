"""Call tracking with confidence scoring for BSKG builder."""
from __future__ import annotations

from typing import Any

from true_vkg.kg.schema import Node, Edge, Evidence
from true_vkg.kg.builder.context import BuildContext
from true_vkg.kg.builder.types import (
    CallInfo, CallbackPattern, UnresolvedTarget,
    CallConfidence, CallType, TargetResolution
)
from true_vkg.kg.builder.helpers import source_location, evidence_from_location


# Known callback patterns
CALLBACK_PATTERNS = {
    # Flash loan callbacks
    "flashLoan": ["onFlashLoan", "executeOperation"],
    "flash": ["uniswapV2Call", "uniswapV3FlashCallback", "pancakeCall"],
    # ERC callbacks
    "safeTransferFrom": ["onERC721Received", "onERC1155Received"],
    "transfer": ["tokensReceived"],  # ERC777
    # DeFi callbacks
    "swap": ["uniswapV2Call", "uniswapV3SwapCallback"],
}


class CallTracker:
    """Track and analyze function calls with confidence scoring."""

    def __init__(self, ctx: BuildContext) -> None:
        self.ctx = ctx

    def track_all(self, fn: Any, fn_node: Node) -> list[Edge]:
        """Track all calls from a function and create edges.

        Returns edges with confidence scores:
        - HIGH: Direct call to known target
        - MEDIUM: Call through interface or inferred target
        - LOW: Cannot resolve target
        """
        edges = []

        # Internal calls
        for internal_call in self._get_internal_calls(fn):
            edge = self._track_internal_call(fn, fn_node, internal_call)
            if edge:
                edges.append(edge)

        # External calls
        for external_call in self._get_external_calls(fn):
            edge = self._track_external_call(fn, fn_node, external_call)
            if edge:
                edges.append(edge)
            # Check for callback patterns
            callback_edges = self._detect_callbacks(fn, fn_node, external_call)
            edges.extend(callback_edges)

        # Low-level calls
        for low_call in self._get_low_level_calls(fn):
            edge = self._track_low_level_call(fn, fn_node, low_call)
            if edge:
                edges.append(edge)

        return edges

    def _track_internal_call(self, fn: Any, fn_node: Node, call: Any) -> Edge | None:
        """Track internal function call - always HIGH confidence."""
        target_fn = getattr(call, "function", None)
        if not target_fn:
            return None

        call_info = CallInfo(
            call_type="internal",
            target_contract=fn_node.properties.get("contract"),
            target_function=getattr(target_fn, "name", None),
            confidence="HIGH",
            resolution="direct",
        )

        return self._create_call_edge(fn_node, target_fn, call_info)

    def _track_external_call(self, fn: Any, fn_node: Node, call: Any) -> Edge | None:
        """Track external call with confidence scoring."""
        # Try to resolve target
        target_contract, target_fn, confidence, resolution = self._resolve_target(call)

        if resolution == "unresolved":
            # Track unresolved target
            self.ctx.add_unresolved(UnresolvedTarget(
                source_function=fn_node.label,
                source_contract=fn_node.properties.get("contract", ""),
                call_expression=str(getattr(call, "expression", "")),
                call_type="external",
                reason="Cannot determine call target",
                file_path=fn_node.evidence[0].file if fn_node.evidence else None,
            ))

        call_info = CallInfo(
            call_type="external",
            target_contract=target_contract,
            target_function=target_fn,
            confidence=confidence,
            resolution=resolution,
            value_sent=self._sends_value(call),
        )

        return self._create_call_edge(fn_node, call, call_info)

    def _detect_callbacks(self, fn: Any, fn_node: Node, call: Any) -> list[Edge]:
        """Detect callback patterns and create bidirectional edges."""
        edges = []
        call_name = self._get_call_name(call)

        for pattern_fn, callbacks in CALLBACK_PATTERNS.items():
            if pattern_fn in call_name.lower():
                for callback in callbacks:
                    # Create edge: target -> callback
                    callback_edge = self._create_callback_edge(
                        fn_node, call, callback,
                        pattern_type=pattern_fn
                    )
                    if callback_edge:
                        edges.append(callback_edge)

        return edges

    def _resolve_target(self, call: Any) -> tuple[str | None, str | None, CallConfidence, TargetResolution]:
        """Resolve call target with confidence scoring.

        Returns:
            (target_contract, target_function, confidence, resolution)
        """
        # Try direct resolution
        target_contract = self._call_contract_name(call)
        target_fn = getattr(call, "function", None)

        if target_contract and target_fn:
            # Check if it's a concrete contract vs interface
            if self._is_concrete_target(call):
                return target_contract, getattr(target_fn, "name", None), "HIGH", "direct"
            else:
                return target_contract, getattr(target_fn, "name", None), "MEDIUM", "interface"

        # Try inference from context
        inferred = self._infer_target(call)
        if inferred:
            return inferred[0], inferred[1], "MEDIUM", "inferred"

        # Cannot resolve
        return None, None, "LOW", "unresolved"

    def _is_concrete_target(self, call: Any) -> bool:

    def _infer_target(self, call: Any) -> tuple[str, str] | None:
        """Try to infer call target from context."""
        ...

    def _create_call_edge(self, fn_node: Node, target: Any, call_info: CallInfo) -> Edge:
        """Create edge with call metadata."""
        ...

    def _create_callback_edge(
        self, fn_node: Node, call: Any, callback: str, pattern_type: str
    ) -> Edge | None:
        """Create callback edge (bidirectional)."""
        ...

    # Helper methods
    def _get_internal_calls(self, fn: Any) -> list[Any]:
        return getattr(fn, "internal_calls", []) or []

    def _get_external_calls(self, fn: Any) -> list[Any]:
        return getattr(fn, "high_level_calls", []) or []

    def _get_low_level_calls(self, fn: Any) -> list[Any]:
        return getattr(fn, "low_level_calls", []) or []

    def _get_call_name(self, call: Any) -> str:
        fn = getattr(call, "function", None)
        return getattr(fn, "name", "") if fn else ""

    def _sends_value(self, call: Any) -> bool:
        return bool(getattr(call, "value", None))


def track_calls(ctx: BuildContext, fn: Any, fn_node: Node) -> list[Edge]:
    """Convenience function to track calls."""
    tracker = CallTracker(ctx)
    return tracker.track_all(fn, fn_node)
