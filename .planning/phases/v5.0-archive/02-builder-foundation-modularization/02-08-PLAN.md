"""Core orchestration for BSKG building."""
from __future__ import annotations

from datetime import datetime, timezone
from pathlib import Path
from typing import Any

import structlog

from true_vkg.kg.schema import KnowledgeGraph, Node, Edge
from true_vkg.kg.builder.context import BuildContext
from true_vkg.kg.builder.contracts import ContractProcessor
from true_vkg.kg.builder.state_vars import StateVarProcessor
from true_vkg.kg.builder.functions import FunctionProcessor
from true_vkg.kg.builder.calls import CallTracker
from true_vkg.kg.builder.proxy import ProxyResolver
from true_vkg.kg.builder.completeness import CompletenessReporter, CompletenessReport
from true_vkg.kg.builder.helpers import source_location
from true_vkg.kg.solc import select_solc_for_file
from true_vkg.kg.fingerprint import graph_fingerprint

# Rich edge and classification imports
from true_vkg.kg.rich_edge import generate_meta_edges
from true_vkg.kg.classification import NodeClassifier
from true_vkg.kg.paths import PathEnumerator

# Slither import with graceful fallback
try:
    from slither import Slither
except Exception as exc:
    Slither = None
    _SLITHER_IMPORT_ERROR = exc
else:
    _SLITHER_IMPORT_ERROR = None


class VKGBuilder:
    """Construct a knowledge graph for a Solidity project.

    This modular builder orchestrates specialized processors:
    - ContractProcessor: Contract nodes and properties
    - StateVarProcessor: State variable nodes
    - FunctionProcessor: Function nodes with 50+ properties
    - CallTracker: Call edges with confidence scoring
    - ProxyResolver: Proxy pattern detection and resolution
    - CompletenessReporter: Build quality metrics

    All processors share state via BuildContext for dependency injection.
    """

    def __init__(
        self,
        project_root: Path,
        *,
        exclude_dependencies: bool = True,
        generate_completeness_report: bool = True,
    ) -> None:
        self.project_root = project_root
        self.exclude_dependencies = exclude_dependencies
        self.generate_completeness_report = generate_completeness_report
        self.logger = structlog.get_logger()
        self._last_report: CompletenessReport | None = None

    @property
    def last_completeness_report(self) -> CompletenessReport | None:
        """Get the completeness report from the last build."""
        return self._last_report

    def build(self, target: Path) -> KnowledgeGraph:
        """Build a knowledge graph from a Solidity target.

        Args:
            target: Path to Solidity file or directory

        Returns:
            Complete KnowledgeGraph with all nodes, edges, and metadata
        """
        if Slither is None:
            raise RuntimeError(f"Slither is not available: {_SLITHER_IMPORT_ERROR}")

        self.logger.info("vkg_build_start", target=str(target))

        # Initialize Slither
        slither = self._init_slither(target)

        # Initialize graph
        graph = self._init_graph(target, slither)

        # Initialize build context
        ctx = BuildContext(
            project_root=self.project_root,
            graph=graph,
            slither=slither,
            exclude_dependencies=self.exclude_dependencies,
            logger=self.logger,
        )

        # Initialize processors
        contract_processor = ContractProcessor(ctx)
        state_var_processor = StateVarProcessor(ctx)
        function_processor = FunctionProcessor(ctx)
        call_tracker = CallTracker(ctx)
        proxy_resolver = ProxyResolver(ctx)

        # Process contracts in sorted order (determinism)
        contracts = sorted(
            getattr(slither, "contracts", []),
            key=lambda c: c.name
        )

        for contract in contracts:
            self._process_contract(
                ctx, graph, contract,
                contract_processor,
                state_var_processor,
                function_processor,
                call_tracker,
                proxy_resolver,
            )

        # Post-processing phases
        self._generate_rich_edges(graph)
        self._generate_meta_edges(graph)
        self._classify_nodes(graph)
        self._analyze_paths(graph)

        # Generate completeness report
        if self.generate_completeness_report:
            reporter = CompletenessReporter(ctx)
            self._last_report = reporter.generate(graph)
            self.logger.info(
                "completeness_report",
                coverage=self._last_report.coverage.function_coverage,
                high_confidence=self._last_report.confidence.high_percentage,
                warnings=len(self._last_report.warnings),
            )

        self.logger.info(
            "vkg_build_complete",
            nodes=len(graph.nodes),
            edges=len(graph.edges),
            fingerprint=graph_fingerprint(graph)[:16],
        )

        return graph

    def _init_slither(self, target: Path) -> Any:
        """Initialize Slither with appropriate solc version."""
        slither_kwargs = {"exclude_dependencies": self.exclude_dependencies}

        if target.is_file():
            solc_selection = select_solc_for_file(target)
            if solc_selection:
                solc_bin, version = solc_selection
                slither_kwargs["solc"] = solc_bin
                self.logger.info("solc_selected", version=version)

        try:
            return Slither(str(target), **slither_kwargs)
        except Exception as exc:
            self.logger.error("slither_init_failed", error=str(exc))
            raise

    def _init_graph(self, target: Path, slither: Any) -> KnowledgeGraph:
        """Initialize empty knowledge graph with metadata."""
        return KnowledgeGraph(
            metadata={
                "root": str(self.project_root),
                "target": str(target),
                "created_at": datetime.now(timezone.utc).isoformat(),
                "builder": "slither",
                "builder_version": "2.0",  # Modular builder version
                "slither_version": getattr(slither, "__version__", None),
                "exclude_dependencies": self.exclude_dependencies,
            }
        )

    def _process_contract(
        self,
        ctx: BuildContext,
        graph: KnowledgeGraph,
        contract: Any,
        contract_processor: ContractProcessor,
        state_var_processor: StateVarProcessor,
        function_processor: FunctionProcessor,
        call_tracker: CallTracker,
        proxy_resolver: ProxyResolver,
    ) -> None:
        """Process a single contract with all processors."""
        # Create contract node
        contract_node = contract_processor.process(contract)
        graph.nodes[contract_node.id] = contract_node

        # Resolve proxy if applicable
        proxy_info = proxy_resolver.resolve(contract)
        if proxy_info.is_proxy:
            contract_node.properties["proxy_info"] = {
                "pattern": proxy_info.pattern.value,
                "confidence": proxy_info.confidence,
                "implementation": proxy_info.implementation_contract,
            }

        # Process inheritance
        self._add_inheritance(graph, contract, contract_node)

        # Process state variables
        state_var_nodes = state_var_processor.process_all(contract, contract_node)
        for node in state_var_nodes:
            graph.nodes[node.id] = node

        # Process modifiers
        self._add_modifiers(graph, contract, contract_node)

        # Process events
        self._add_events(graph, contract, contract_node)

        # Process functions
        function_nodes = function_processor.process_all(contract, contract_node)
        for fn_node in function_nodes:
            graph.nodes[fn_node.id] = fn_node

        # Track calls for each function
        for fn_node in function_nodes:
            fn = self._get_function_by_node(contract, fn_node)
            if fn:
                call_edges = call_tracker.track_all(fn, fn_node)
                for edge in call_edges:
                    graph.edges[edge.id] = edge

        # Annotate cross-function signals
        self._annotate_cross_function_signals(graph, contract)

        # Cache contract for later reference
        ctx.contract_cache[contract.name] = contract

        """Add inheritance edges."""

    def _add_events(self, graph: KnowledgeGraph, contract: Any, contract_node: Node) -> None:
        """Add event nodes and edges."""

    def _get_function_by_node(self, contract: Any, fn_node: Node) -> Any | None:
        """Get Slither function object from node."""
        for fn in getattr(contract, "functions", []):
            if fn.name == fn_node.label:
                return fn
        return None

    def _generate_rich_edges(self, graph: KnowledgeGraph) -> None:
        """Generate rich edges from function analysis."""
        # Delegate to existing rich_edge module

    def _generate_meta_edges(self, graph: KnowledgeGraph) -> None:
        """Generate meta edges (SIMILAR_TO, etc.)."""
        try:
            meta_edges = generate_meta_edges(graph)
            graph.meta_edges.extend(meta_edges)
        except Exception as exc:
            self.logger.warning("meta_edge_generation_failed", error=str(exc))

    def _classify_nodes(self, graph: KnowledgeGraph) -> None:
        """Classify nodes into semantic roles."""
        classifier = NodeClassifier()
        for node in graph.nodes.values():
            if node.type == "function":
                role = classifier.classify_function(node)
                node.properties["semantic_role"] = role.value

    def _analyze_paths(self, graph: KnowledgeGraph) -> None:
        """Analyze execution paths (optional, for attack scenarios)."""
        # Only for graphs with enough complexity
        if len(graph.nodes) > 10:
            try:
                enumerator = PathEnumerator(graph)
                # Store path summary, not full enumeration
                node.properties["path_reachable"] = enumerator.is_reachable(node.id)
            except Exception:
                pass  # Path analysis is optional


def build_graph(
    target: Path,
    *,
    project_root: Path | None = None,
    exclude_dependencies: bool = True,
    generate_completeness_report: bool = True,
) -> KnowledgeGraph:
    """Convenience function to build a knowledge graph.

    Args:
        target: Path to Solidity file or directory
        project_root: Root of the project (defaults to target's parent)
        exclude_dependencies: Whether to exclude dependency contracts
        generate_completeness_report: Whether to generate completeness report

    Returns:
        Complete KnowledgeGraph
    """
    if project_root is None:
        project_root = target.parent if target.is_file() else target

    builder = VKGBuilder(
        project_root,
        exclude_dependencies=exclude_dependencies,
        generate_completeness_report=generate_completeness_report,
    )
    return builder.build(target)

"""Integration tests for modular BSKG builder."""
import pytest
from pathlib import Path

from true_vkg.kg.builder import VKGBuilder, build_graph, BuildContext
from true_vkg.kg.builder.completeness import CompletenessReport
from true_vkg.kg.fingerprint import graph_fingerprint


# Test contracts directory
CONTRACTS_DIR = Path(__file__).parent.parent / "contracts"


class TestBuilderIntegration:
    """Integration tests for BSKG builder."""

    def test_build_simple_contract(self):
        """Test building a simple contract."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        graph = build_graph(contract)

        assert len(graph.nodes) > 0
        assert len(graph.edges) >= 0
        assert "contract" in [n.type for n in graph.nodes.values()]
        assert "function" in [n.type for n in graph.nodes.values()]

    def test_build_determinism(self):
        """Test that builds are deterministic."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        graph1 = build_graph(contract)
        graph2 = build_graph(contract)

        fp1 = graph_fingerprint(graph1)
        fp2 = graph_fingerprint(graph2)

        assert fp1 == fp2, "Graph fingerprints should be identical"

    def test_completeness_report_generated(self):
        """Test that completeness report is generated."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        builder = VKGBuilder(contract.parent, generate_completeness_report=True)
        graph = builder.build(contract)

        report = builder.last_completeness_report
        assert report is not None
        assert isinstance(report, CompletenessReport)
        assert report.node_count == len(graph.nodes)
        assert report.graph_fingerprint == graph_fingerprint(graph)

    def test_completeness_report_yaml(self):
        """Test that completeness report serializes to YAML."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        builder = VKGBuilder(contract.parent, generate_completeness_report=True)
        builder.build(contract)

        report = builder.last_completeness_report
        yaml_output = report.to_yaml()

        assert "build_info:" in yaml_output
        assert "coverage:" in yaml_output
        assert "confidence_breakdown:" in yaml_output

    def test_function_properties_count(self):
        """Test that functions have expected property count."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        graph = build_graph(contract)

        for node in graph.nodes.values():
            if node.type == "function":
                # Should have many security properties
                assert len(node.properties) >= 20, \
                    f"Function {node.label} has too few properties: {len(node.properties)}"

    def test_call_edges_have_confidence(self):
        """Test that call edges have confidence scoring."""
        contract = CONTRACTS_DIR / "AuthReentrancyNoGuard.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        graph = build_graph(contract)

        call_edges = [e for e in graph.edges.values() if e.type == "calls"]
        if call_edges:
            for edge in call_edges:
                assert "confidence" in edge.properties, \
                    f"Edge {edge.id} missing confidence"
                assert edge.properties["confidence"] in ("HIGH", "MEDIUM", "LOW")

    def test_stable_node_ids(self):
        """Test that node IDs are stable and don't include file paths."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        graph = build_graph(contract)

        for node_id in graph.nodes.keys():
            # IDs should be kind:hash format, not include paths
            assert ":" in node_id
            assert "/" not in node_id, f"Node ID contains path: {node_id}"
            assert "\\" not in node_id, f"Node ID contains path: {node_id}"


class TestBuilderVsLegacy:
    """Tests comparing modular builder to legacy builder."""

    def test_node_count_matches(self):
        """Test that modular builder produces same node count as legacy."""
        contract = CONTRACTS_DIR / "AccessGateIfReturn.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        # Build with modular builder
        graph = build_graph(contract)

        # Build with legacy builder
        from true_vkg.kg.builder import VKGBuilder as LegacyBuilder
        legacy = LegacyBuilder(contract.parent)
        legacy_graph = legacy.build(contract)

        # Compare node counts (may differ slightly due to improvements)
        # Allow 10% variance for enhancements
        ratio = len(graph.nodes) / len(legacy_graph.nodes) if legacy_graph.nodes else 1
        assert 0.9 <= ratio <= 1.1, \
            f"Node count mismatch: {len(graph.nodes)} vs {len(legacy_graph.nodes)}"

    def test_key_properties_preserved(self):
        """Test that key security properties are preserved."""
        contract = CONTRACTS_DIR / "AuthReentrancyNoGuard.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        graph = build_graph(contract)

        # Find function nodes and check key properties
        for node in graph.nodes.values():
            if node.type == "function":
                props = node.properties
                # These properties should always be present
                assert "visibility" in props
                assert "has_access_gate" in props or "requires_auth" in props
                assert "has_external_calls" in props or "external_call_count" in props
