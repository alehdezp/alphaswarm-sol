from dataclasses import dataclass, field
from typing import Literal
from enum import Enum

class ProxyPattern(Enum):
    """Supported proxy patterns."""
    TRANSPARENT = "transparent"  # EIP-1967 Transparent
    UUPS = "uups"                # Universal Upgradeable Proxy Standard
    DIAMOND = "diamond"          # EIP-2535 Diamond
    BEACON = "beacon"            # Beacon Proxy
    MINIMAL = "minimal"          # EIP-1167 Minimal Proxy (clone)
    UNKNOWN = "unknown"          # Detected as proxy but pattern unclear
    NONE = "none"                # Not a proxy


@dataclass
class ProxyInfo:
    """Proxy detection and resolution result."""
    is_proxy: bool
    pattern: ProxyPattern
    confidence: CallConfidence

    # Implementation info
    implementation_slot: str | None = None
    implementation_contract: str | None = None
    implementation_address: str | None = None  # If statically known

    # Diamond-specific
    facets: list[str] = field(default_factory=list)
    facet_selectors: dict[str, list[str]] = field(default_factory=dict)

    # Beacon-specific
    beacon_contract: str | None = None

    # Admin info
    admin_slot: str | None = None
    admin_contract: str | None = None

    # Resolution notes
    resolution_notes: list[str] = field(default_factory=list)
    unresolved_reason: str | None = None


# EIP-1967 storage slots (pre-computed)

"""Proxy resolution for all proxy patterns."""
from __future__ import annotations

from typing import Any

from true_vkg.kg.builder.context import BuildContext
from true_vkg.kg.builder.types import (
    ProxyInfo, ProxyPattern, CallConfidence,
    EIP1967_IMPLEMENTATION_SLOT, EIP1967_ADMIN_SLOT, EIP1967_BEACON_SLOT
)


# Known proxy base contracts
TRANSPARENT_PROXY_BASES = {
    "TransparentUpgradeableProxy",
    "ERC1967Proxy",
    "AdminUpgradeabilityProxy",  # OpenZeppelin v3
}

UUPS_BASES = {
    "UUPSUpgradeable",
    "ERC1967Upgrade",
}

DIAMOND_BASES = {
    "Diamond",
    "DiamondCutFacet",
    "DiamondLoupeFacet",
}

BEACON_BASES = {
    "BeaconProxy",
    "UpgradeableBeacon",
}


class ProxyResolver:
    """Detect and resolve proxy patterns.

    Supports:
    - EIP-1967 Transparent Proxy
    - UUPS (Universal Upgradeable Proxy Standard)
    - EIP-2535 Diamond Proxy
    - Beacon Proxy
    - EIP-1167 Minimal Proxy (clone)

    Resolution is best-effort: returns ProxyInfo with confidence
    and notes, never raises on resolution failure.
    """

    def __init__(self, ctx: BuildContext) -> None:
        self.ctx = ctx

    def resolve(self, contract: Any) -> ProxyInfo:
        """Detect proxy pattern and resolve implementation.

        Returns ProxyInfo with:
        - pattern: Detected proxy pattern
        - confidence: HIGH/MEDIUM/LOW
        - implementation_contract: Resolved implementation (if possible)
        - resolution_notes: Details about detection/resolution
        """
        # Step 1: Check Slither's built-in detection
        if self._is_slither_proxy(contract):
            return self._resolve_slither_proxy(contract)

        # Step 2: Check inheritance for known bases
        pattern = self._detect_from_inheritance(contract)
        if pattern != ProxyPattern.NONE:
            return self._resolve_by_pattern(contract, pattern)

        # Step 3: Check for Diamond-specific patterns
        if self._is_diamond(contract):
            return self._resolve_diamond(contract)

        # Step 4: Check for storage slot patterns
        if self._has_eip1967_slots(contract):
            return self._resolve_eip1967(contract)

        # Step 5: Heuristic detection (LOW confidence)
        if self._name_suggests_proxy(contract):
            return self._heuristic_resolution(contract)

        return ProxyInfo(is_proxy=False, pattern=ProxyPattern.NONE, confidence="HIGH")

    def _is_slither_proxy(self, contract: Any) -> bool:
        """Check Slither's built-in proxy detection."""
        return (
            getattr(contract, "is_proxy", False) or
            getattr(contract, "is_upgradeable_proxy", False)
        )

    def _detect_from_inheritance(self, contract: Any) -> ProxyPattern:
        """Detect proxy pattern from inheritance chain."""
        inheritance = self._get_inheritance_names(contract)

        if inheritance & TRANSPARENT_PROXY_BASES:
            return ProxyPattern.TRANSPARENT
        if inheritance & UUPS_BASES:
            return ProxyPattern.UUPS
        if inheritance & DIAMOND_BASES:
            return ProxyPattern.DIAMOND
        if inheritance & BEACON_BASES:
            return ProxyPattern.BEACON

        return ProxyPattern.NONE

    def _is_diamond(self, contract: Any) -> bool:
        """Check for Diamond pattern indicators."""
        fn_names = {fn.name for fn in getattr(contract, "functions", [])}
        # Diamond must have diamondCut or loupe functions
        diamond_fns = {"diamondCut", "facetAddresses", "facetFunctionSelectors"}
        return bool(fn_names & diamond_fns)

    def _has_eip1967_slots(self, contract: Any) -> bool:
        """Check for EIP-1967 storage slot usage."""
        # Check source for slot constants
        source = self._get_source(contract)
        return (
            EIP1967_IMPLEMENTATION_SLOT in source or
            "eip1967.proxy.implementation" in source.lower()
        )

    def _name_suggests_proxy(self, contract: Any) -> bool:
        """Heuristic: name contains proxy-related terms."""
        name = contract.name.lower()
        return any(term in name for term in ("proxy", "upgradeable", "delegator"))

    def _resolve_slither_proxy(self, contract: Any) -> ProxyInfo:
        """Resolve using Slither's detection."""
        # Try to get implementation from _delegates_to
        delegates = getattr(contract, "_delegates_to", None)
        impl_contract = None
        if delegates:
            impl_contract = getattr(delegates, "name", None)

        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.TRANSPARENT,  # Most common
            confidence="MEDIUM",
            implementation_contract=impl_contract,
            resolution_notes=["Detected via Slither is_proxy/is_upgradeable_proxy"],
        )

    def _resolve_by_pattern(self, contract: Any, pattern: ProxyPattern) -> ProxyInfo:
        """Resolve based on detected pattern."""
        if pattern == ProxyPattern.TRANSPARENT:
            return self._resolve_transparent(contract)
        elif pattern == ProxyPattern.UUPS:
            return self._resolve_uups(contract)
        elif pattern == ProxyPattern.BEACON:
            return self._resolve_beacon(contract)
        else:
            return ProxyInfo(
                is_proxy=True,
                pattern=pattern,
                confidence="MEDIUM",
                resolution_notes=[f"Detected {pattern.value} from inheritance"],
            )

    def _resolve_transparent(self, contract: Any) -> ProxyInfo:
        """Resolve EIP-1967 transparent proxy."""
        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.TRANSPARENT,
            confidence="HIGH",
            implementation_slot=EIP1967_IMPLEMENTATION_SLOT,
            admin_slot=EIP1967_ADMIN_SLOT,
            resolution_notes=["EIP-1967 Transparent Proxy detected from inheritance"],
        )

    def _resolve_uups(self, contract: Any) -> ProxyInfo:
        """Resolve UUPS proxy."""
        # Check for _authorizeUpgrade
        has_authorize = any(
            fn.name == "_authorizeUpgrade"
            for fn in getattr(contract, "functions", [])
        )
        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.UUPS,
            confidence="HIGH" if has_authorize else "MEDIUM",
            implementation_slot=EIP1967_IMPLEMENTATION_SLOT,
            resolution_notes=[
                "UUPS pattern detected",
                f"_authorizeUpgrade: {'found' if has_authorize else 'not found'}"
            ],
        )

    def _resolve_diamond(self, contract: Any) -> ProxyInfo:
        """Resolve Diamond proxy and enumerate facets."""
        facets = self._enumerate_facets(contract)
        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.DIAMOND,
            confidence="HIGH" if facets else "MEDIUM",
            facets=facets,
            resolution_notes=[
                "EIP-2535 Diamond pattern detected",
                f"Facets found: {len(facets)}"
            ],
        )

    def _resolve_beacon(self, contract: Any) -> ProxyInfo:
        """Resolve Beacon proxy."""
        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.BEACON,
            confidence="HIGH",
            implementation_slot=EIP1967_BEACON_SLOT,
            resolution_notes=["Beacon proxy pattern detected"],
        )

    def _resolve_eip1967(self, contract: Any) -> ProxyInfo:
        """Resolve from EIP-1967 slot detection."""
        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.TRANSPARENT,
            confidence="MEDIUM",
            implementation_slot=EIP1967_IMPLEMENTATION_SLOT,
            resolution_notes=["Detected EIP-1967 slots in source"],
        )

    def _heuristic_resolution(self, contract: Any) -> ProxyInfo:
        """Heuristic resolution when pattern unclear."""
        self.ctx.add_warning(
            f"Proxy '{contract.name}' detected by name heuristic only - "
            "pattern unclear, resolution may be incomplete"
        )
        return ProxyInfo(
            is_proxy=True,
            pattern=ProxyPattern.UNKNOWN,
            confidence="LOW",
            unresolved_reason="Pattern detected by name heuristic only",
            resolution_notes=["Name suggests proxy but pattern not confirmed"],
        )

    def _enumerate_facets(self, contract: Any) -> list[str]:
        """Enumerate Diamond facets if possible."""
        # This would require analyzing facetAddresses() return
        # For now, return empty - full implementation needs runtime analysis
        return []

    def _get_inheritance_names(self, contract: Any) -> set[str]:
        """Get set of inherited contract names."""
        inheritance = getattr(contract, "inheritance", []) or []
        return {getattr(c, "name", "") for c in inheritance}

    def _get_source(self, contract: Any) -> str:
        """Get contract source code."""
        # Try to get from source mapping
        try:
            source_mapping = getattr(contract, "source_mapping", None)
            if source_mapping:
                return str(source_mapping.content or "")
        except Exception:
            pass
        return ""


def resolve_proxy(ctx: BuildContext, contract: Any) -> ProxyInfo:
    """Convenience function to resolve proxy."""
    resolver = ProxyResolver(ctx)
    return resolver.resolve(contract)
