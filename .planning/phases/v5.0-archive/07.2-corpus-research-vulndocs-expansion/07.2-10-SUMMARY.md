### Evidence Debt Boundaries

@property
def debt_level(self) -> str:
    ratio = self.weak_evidence_ratio
    if ratio == 0:
        return "none"
    elif ratio <= 0.10:
        return "low"
    elif ratio <= 0.25:
        return "moderate"
    else:
        return "high"  # Promotion blocked
