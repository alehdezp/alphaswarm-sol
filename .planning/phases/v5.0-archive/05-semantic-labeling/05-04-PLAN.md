```python
"""Label validation and scoring.

Validates labels against taxonomy and scores label quality.
"""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from enum import Enum
from typing import Dict, List, Optional, Tuple

from true_vkg.labels.schema import FunctionLabel, LabelConfidence, LabelSet
from true_vkg.labels.taxonomy import CORE_TAXONOMY, get_label_by_id, is_valid_label
from true_vkg.labels.overlay import LabelOverlay

logger = logging.getLogger(__name__)


class ValidationStatus(str, Enum):
    """Label validation status."""
    VALID = "valid"
    INVALID_LABEL = "invalid_label"      # Label ID not in taxonomy
    MISSING_REASONING = "missing_reasoning"  # LOW confidence without reasoning
    INVALID_CONFIDENCE = "invalid_confidence"
    CUSTOM_NEEDS_REVIEW = "custom_needs_review"  # Custom label flagged


@dataclass
class ValidationResult:
    """Result of validating a single label."""
    label: FunctionLabel
    status: ValidationStatus
    message: Optional[str] = None


@dataclass
class LabelSetValidation:
    """Result of validating a label set."""
    function_id: str
    results: List[ValidationResult] = field(default_factory=list)
    valid_count: int = 0
    invalid_count: int = 0

    @property
    def all_valid(self) -> bool:
        return self.invalid_count == 0


@dataclass
class QualityScore:
    """Quality score for a label or label set."""
    score: float  # 0.0 to 1.0
    confidence_distribution: Dict[str, int] = field(default_factory=dict)
    coverage_score: float = 0.0  # How many relevant categories are covered
    reasoning_completeness: float = 0.0  # % of LOW labels with reasoning


class LabelValidator:
    """Validates labels against taxonomy and scores quality.

    Usage:
        validator = LabelValidator()

        # Validate a label
        result = validator.validate_label(label)

        # Validate all labels for a function
        validation = validator.validate_label_set(label_set)

        # Score label quality
        score = validator.score_labels(label_set)
    """

    def __init__(self, allow_custom: bool = False):
        """Initialize validator.

        Args:
            allow_custom: Whether to allow custom (non-taxonomy) labels
        """
        self.allow_custom = allow_custom
        self._valid_labels = {label.id for label in CORE_TAXONOMY}

    def validate_label(self, label: FunctionLabel) -> ValidationResult:
        """Validate a single label.

        Args:
            label: FunctionLabel to validate

        Returns:
            ValidationResult with status and message
        """
        # Check if label ID is valid
        if not is_valid_label(label.label_id):
            if self.allow_custom:
                return ValidationResult(
                    label=label,
                    status=ValidationStatus.CUSTOM_NEEDS_REVIEW,
                    message=f"Custom label '{label.label_id}' needs review"
                )
            return ValidationResult(
                label=label,
                status=ValidationStatus.INVALID_LABEL,
                message=f"Label '{label.label_id}' not in taxonomy"
            )

        # Check confidence requirements
        if label.confidence == LabelConfidence.LOW:
            if not label.reasoning:
                return ValidationResult(
                    label=label,
                    status=ValidationStatus.MISSING_REASONING,
                    message="LOW confidence label requires reasoning"
                )

        return ValidationResult(
            label=label,
            status=ValidationStatus.VALID,
        )

    def validate_label_set(self, label_set: LabelSet) -> LabelSetValidation:
        """Validate all labels in a set.

        Args:
            label_set: LabelSet to validate

        Returns:
            LabelSetValidation with per-label results
        """
        validation = LabelSetValidation(function_id=label_set.function_id)

        for label in label_set.labels:
            result = self.validate_label(label)
            validation.results.append(result)

            if result.status == ValidationStatus.VALID:
                validation.valid_count += 1
            else:
                validation.invalid_count += 1

        return validation

    def validate_overlay(self, overlay: LabelOverlay) -> Dict[str, LabelSetValidation]:
        """Validate all labels in an overlay.

        Args:
            overlay: LabelOverlay to validate

        Returns:
            Dict of function_id -> LabelSetValidation
        """
        results = {}
        for function_id, label_set in overlay.labels.items():
            results[function_id] = self.validate_label_set(label_set)
        return results

    def score_labels(self, label_set: LabelSet) -> QualityScore:
        """Score the quality of a label set.

        Higher scores indicate:
        - More HIGH confidence labels
        - Better category coverage
        - Complete reasoning for uncertain labels

        Args:
            label_set: LabelSet to score

        Returns:
            QualityScore with detailed metrics
        """
        if not label_set.labels:
            return QualityScore(score=0.0)

        # Confidence distribution
        conf_dist = {"high": 0, "medium": 0, "low": 0}
        for label in label_set.labels:
            conf_dist[label.confidence.value] += 1

        # Base score from confidence
        total = len(label_set.labels)
        weighted_sum = (
            conf_dist["high"] * 1.0 +
            conf_dist["medium"] * 0.7 +
            conf_dist["low"] * 0.4
        )
        base_score = weighted_sum / total

        # Coverage score (how many categories represented)
        categories = {label.category for label in label_set.labels}
        coverage = len(categories) / 6  # 6 total categories

        # Reasoning completeness for LOW labels
        low_labels = [l for l in label_set.labels if l.confidence == LabelConfidence.LOW]
        if low_labels:
            with_reasoning = sum(1 for l in low_labels if l.reasoning)
            reasoning_completeness = with_reasoning / len(low_labels)
        else:
            reasoning_completeness = 1.0  # No LOW labels = complete

        # Combined score
        final_score = (base_score * 0.5 + coverage * 0.3 + reasoning_completeness * 0.2)

        return QualityScore(
            score=final_score,
            confidence_distribution=conf_dist,
            coverage_score=coverage,
            reasoning_completeness=reasoning_completeness,
        )

    def get_precision_estimate(
        self,
        overlay: LabelOverlay,
        ground_truth: Optional[LabelOverlay] = None,
    ) -> float:
        """Estimate label precision against optional ground truth.

        If no ground truth, uses heuristics (confidence levels, validation).

        Args:
            overlay: Labels to evaluate
            ground_truth: Optional ground truth labels

        Returns:
            Precision estimate (0.0 to 1.0)
        """
        if ground_truth:
            # Compare against ground truth
            correct = 0
            total = 0
            for func_id, label_set in overlay.labels.items():
                gt_set = ground_truth.get_labels(func_id)
                gt_label_ids = {l.label_id for l in gt_set.labels}

                for label in label_set.labels:
                    total += 1
                    if label.label_id in gt_label_ids:
                        correct += 1

            return correct / total if total > 0 else 0.0
        else:
            # Heuristic: assume HIGH confidence = likely correct
            correct_estimate = 0
            total = 0
            for label_set in overlay.labels.values():
                for label in label_set.labels:
                    total += 1
                    if label.confidence == LabelConfidence.HIGH:
                        correct_estimate += 0.9
                    elif label.confidence == LabelConfidence.MEDIUM:
                        correct_estimate += 0.7
                    else:
                        correct_estimate += 0.4

            return correct_estimate / total if total > 0 else 0.0


def validate_labels(overlay: LabelOverlay) -> Tuple[int, int]:
    """Quick validation - returns (valid_count, invalid_count)."""
    validator = LabelValidator()
    valid = 0
    invalid = 0
    for label_set in overlay.labels.values():
        validation = validator.validate_label_set(label_set)
        valid += validation.valid_count
        invalid += validation.invalid_count
    return valid, invalid
```

```python
"""Context-filtered label retrieval.

Filters labels based on analysis context to prevent context pollution.
Only returns labels relevant to the current analysis focus.
"""

from __future__ import annotations

from dataclasses import dataclass
from typing import Dict, List, Optional, Set

from true_vkg.labels.schema import FunctionLabel, LabelConfidence, LabelSet
from true_vkg.labels.overlay import LabelOverlay
from true_vkg.labels.taxonomy import LabelCategory


# Mapping of analysis contexts to relevant label categories
CONTEXT_TO_CATEGORIES: Dict[str, Set[str]] = {
    # Vulnerability-focused contexts
    "reentrancy": {"state_mutation", "external_interaction", "value_handling"},
    "access_control": {"access_control"},
    "oracle_manipulation": {"external_interaction", "invariants"},
    "value_extraction": {"value_handling", "invariants"},
    "dos": {"external_interaction", "state_mutation"},
    "flash_loan": {"value_handling", "external_interaction", "state_mutation"},

    # Feature-focused contexts
    "authorization": {"access_control"},
    "state_integrity": {"state_mutation", "invariants"},
    "external_calls": {"external_interaction"},
    "token_handling": {"value_handling"},
    "timing": {"temporal"},

    # General context - returns all
    "general": set(),  # Empty = all categories
}


@dataclass
class FilteredLabels:
    """Result of filtering labels for a context."""
    function_id: str
    labels: List[FunctionLabel]
    context: str
    categories_included: Set[str]
    labels_filtered_out: int


class LabelFilter:
    """Context-aware label filter.

    Prevents context pollution by only returning labels relevant
    to the current analysis focus.

    Usage:
        filter = LabelFilter(overlay)

        # Get labels for reentrancy analysis
        labels = filter.get_filtered_labels("fn_1", "reentrancy")

        # Only state_mutation, external_interaction, value_handling labels returned
    """

    def __init__(
        self,
        overlay: LabelOverlay,
        min_confidence: LabelConfidence = LabelConfidence.MEDIUM,
    ):
        """Initialize filter.

        Args:
            overlay: Label overlay to filter from
            min_confidence: Minimum confidence level to include
        """
        self.overlay = overlay
        self.min_confidence = min_confidence

    def get_filtered_labels(
        self,
        function_id: str,
        context: str,
        override_categories: Optional[Set[str]] = None,
    ) -> FilteredLabels:
        """Get labels filtered by analysis context.

        Args:
            function_id: Function to get labels for
            context: Analysis context (e.g., "reentrancy", "access_control")
            override_categories: Optional explicit category set

        Returns:
            FilteredLabels with relevant labels only
        """
        label_set = self.overlay.get_labels(function_id)

        # Determine relevant categories
        if override_categories:
            categories = override_categories
        else:
            categories = CONTEXT_TO_CATEGORIES.get(context, set())

        # Empty set means all categories (general context)
        all_categories = not categories

        # Filter labels
        filtered = []
        filtered_out = 0

        for label in label_set.labels:
            # Check confidence threshold
            if self._confidence_value(label.confidence) < self._confidence_value(self.min_confidence):
                filtered_out += 1
                continue

            # Check category relevance
            if all_categories or label.category in categories:
                filtered.append(label)
            else:
                filtered_out += 1

        return FilteredLabels(
            function_id=function_id,
            labels=filtered,
            context=context,
            categories_included=categories if categories else set(LabelCategory.__members__.keys()),
            labels_filtered_out=filtered_out,
        )

    def get_labels_for_pattern(
        self,
        function_id: str,
        pattern_categories: List[str],
    ) -> List[FunctionLabel]:
        """Get labels relevant to a specific pattern's categories.

        Args:
            function_id: Function ID
            pattern_categories: Categories the pattern cares about

        Returns:
            List of relevant labels
        """
        result = self.get_filtered_labels(
            function_id,
            "general",
            override_categories=set(pattern_categories),
        )
        return result.labels

    def format_labels_for_llm(
        self,
        function_id: str,
        context: str,
        max_labels: int = 10,
    ) -> str:
        """Format filtered labels as text for LLM context.

        Args:
            function_id: Function ID
            context: Analysis context
            max_labels: Maximum labels to include

        Returns:
            Formatted string for LLM context
        """
        result = self.get_filtered_labels(function_id, context)

        if not result.labels:
            return f"No relevant labels for {function_id}"

        lines = [f"Labels for {function_id} (context: {context}):"]
        for label in result.labels[:max_labels]:
            conf_marker = {"high": "[H]", "medium": "[M]", "low": "[L]"}[label.confidence.value]
            lines.append(f"  {conf_marker} {label.label_id}")
            if label.reasoning:
                lines.append(f"      Reasoning: {label.reasoning}")

        if len(result.labels) > max_labels:
            lines.append(f"  ... and {len(result.labels) - max_labels} more")

        return "\n".join(lines)

    def get_functions_by_label_in_context(
        self,
        label_id: str,
        context: str,
    ) -> List[str]:
        """Find functions with a specific label that's relevant to context.

        Args:
            label_id: Label to search for
            context: Analysis context

        Returns:
            List of function IDs with the label (in context)
        """
        categories = CONTEXT_TO_CATEGORIES.get(context, set())

        # Check if label's category is relevant to context
        label_category = label_id.split(".")[0] if "." in label_id else label_id
        if categories and label_category not in categories:
            return []  # Label not relevant to this context

        return self.overlay.get_functions_with_label(label_id, self.min_confidence)

    @staticmethod
    def _confidence_value(confidence: LabelConfidence) -> float:
        """Get numeric value for confidence comparison."""
        return {"high": 0.8, "medium": 0.5, "low": 0.0}[confidence.value]


def get_relevant_categories(context: str) -> Set[str]:
    """Get label categories relevant to an analysis context."""
    return CONTEXT_TO_CATEGORIES.get(context, set())


def filter_labels_for_context(
    overlay: LabelOverlay,
    function_ids: List[str],
    context: str,
) -> Dict[str, List[FunctionLabel]]:
    """Convenience function to filter labels for multiple functions.

    Args:
        overlay: Label overlay
        function_ids: Functions to get labels for
        context: Analysis context

    Returns:
        Dict of function_id -> filtered labels
    """
    filter = LabelFilter(overlay)
    return {
        func_id: filter.get_filtered_labels(func_id, context).labels
        for func_id in function_ids
    }
```
