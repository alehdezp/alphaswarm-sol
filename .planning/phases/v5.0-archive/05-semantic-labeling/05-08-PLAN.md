   ```python
   """Ground truth corpus for label evaluation."""

   from pathlib import Path
   from typing import Dict, List, Optional
   import yaml

   from true_vkg.labels.schema import FunctionLabel, LabelConfidence, LabelSource
   from true_vkg.labels.overlay import LabelOverlay

   CORPUS_DIR = Path(__file__).parent


   def load_ground_truth(name: str) -> Dict:
       """Load a ground truth file."""
       path = CORPUS_DIR / f"labeled_{name}.yaml"
       if not path.exists():
           raise FileNotFoundError(f"Ground truth not found: {name}")
       with open(path) as f:
           return yaml.safe_load(f)


   def ground_truth_to_overlay(data: Dict) -> LabelOverlay:
       """Convert ground truth data to LabelOverlay."""
       overlay = LabelOverlay()
       for func_data in data.get("functions", []):
           func_id = func_data["function_id"]
           for label_data in func_data.get("labels", []):
               label = FunctionLabel(
                   label_id=label_data["label_id"],
                   confidence=LabelConfidence(label_data.get("confidence", "medium")),
                   source=LabelSource.USER_OVERRIDE,  # Ground truth is manually verified
                   reasoning=label_data.get("reasoning"),
               )
               overlay.add_label(func_id, label)
       return overlay


   def get_expected_findings(data: Dict) -> List[Dict]:
       """Get expected findings from ground truth."""
       return data.get("expected_findings", [])


   def list_ground_truth_files() -> List[str]:
       """List available ground truth files."""
       return [
           p.stem.replace("labeled_", "")
           for p in CORPUS_DIR.glob("labeled_*.yaml")
       ]


   def load_all_ground_truth() -> LabelOverlay:
       """Load and merge all ground truth into one overlay."""
       overlay = LabelOverlay()
       for name in list_ground_truth_files():
           data = load_ground_truth(name)
           partial = ground_truth_to_overlay(data)
           overlay = overlay.merge(partial)
       return overlay
   ```

```python
"""Label Evaluation Harness.

Measures label precision, detection delta, and token efficiency.
"""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional, Set, TYPE_CHECKING

from true_vkg.labels.schema import FunctionLabel, LabelConfidence, LabelSet
from true_vkg.labels.overlay import LabelOverlay
from true_vkg.labels.validator import LabelValidator, QualityScore

if TYPE_CHECKING:
    from true_vkg.kg.schema import KnowledgeGraph

logger = logging.getLogger(__name__)


@dataclass
class PrecisionMetrics:
    """Precision metrics for label evaluation."""
    true_positives: int = 0
    false_positives: int = 0
    false_negatives: int = 0
    total_predicted: int = 0
    total_ground_truth: int = 0

    @property
    def precision(self) -> float:
        """Calculate precision = TP / (TP + FP)."""
        if self.true_positives + self.false_positives == 0:
            return 0.0
        return self.true_positives / (self.true_positives + self.false_positives)

    @property
    def recall(self) -> float:
        """Calculate recall = TP / (TP + FN)."""
        if self.true_positives + self.false_negatives == 0:
            return 0.0
        return self.true_positives / (self.true_positives + self.false_negatives)

    @property
    def f1_score(self) -> float:
        """Calculate F1 = 2 * (P * R) / (P + R)."""
        p, r = self.precision, self.recall
        if p + r == 0:
            return 0.0
        return 2 * (p * r) / (p + r)


@dataclass
class DetectionMetrics:
    """Detection metrics comparing with/without labels."""
    baseline_findings: int = 0
    label_findings: int = 0
    new_findings: int = 0
    lost_findings: int = 0

    @property
    def detection_delta(self) -> float:
        """Calculate detection improvement percentage."""
        if self.baseline_findings == 0:
            return float('inf') if self.new_findings > 0 else 0.0
        return (self.new_findings - self.lost_findings) / self.baseline_findings * 100


@dataclass
class TokenMetrics:
    """Token usage metrics."""
    total_tokens: int = 0
    total_cost_usd: float = 0.0
    functions_labeled: int = 0
    avg_tokens_per_function: float = 0.0
    max_tokens_single_call: int = 0

    def calculate_averages(self) -> None:
        """Calculate average metrics."""
        if self.functions_labeled > 0:
            self.avg_tokens_per_function = self.total_tokens / self.functions_labeled


@dataclass
class EvaluationReport:
    """Complete evaluation report."""
    precision_metrics: PrecisionMetrics = field(default_factory=PrecisionMetrics)
    detection_metrics: DetectionMetrics = field(default_factory=DetectionMetrics)
    token_metrics: TokenMetrics = field(default_factory=TokenMetrics)
    quality_scores: Dict[str, QualityScore] = field(default_factory=dict)
    passed_exit_gate: bool = False
    exit_gate_details: Dict[str, bool] = field(default_factory=dict)

    def check_exit_gate(
        self,
        min_precision: float = 0.75,
        min_detection_delta: float = 5.0,
        max_tokens_per_call: int = 6000,
    ) -> bool:
        """Check if evaluation passes exit gate criteria.

        Args:
            min_precision: Minimum precision (default 0.75)
            min_detection_delta: Minimum detection improvement % (default 5.0)
            max_tokens_per_call: Maximum tokens per labeling call (default 6000)

        Returns:
            True if all criteria pass
        """
        self.exit_gate_details = {
            "precision_met": self.precision_metrics.precision >= min_precision,
            "detection_delta_met": self.detection_metrics.detection_delta >= min_detection_delta,
            "token_budget_met": self.token_metrics.max_tokens_single_call <= max_tokens_per_call,
        }
        self.passed_exit_gate = all(self.exit_gate_details.values())
        return self.passed_exit_gate

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to dictionary."""
        return {
            "precision": {
                "value": self.precision_metrics.precision,
                "recall": self.precision_metrics.recall,
                "f1": self.precision_metrics.f1_score,
                "true_positives": self.precision_metrics.true_positives,
                "false_positives": self.precision_metrics.false_positives,
            },
            "detection": {
                "delta_percent": self.detection_metrics.detection_delta,
                "baseline": self.detection_metrics.baseline_findings,
                "with_labels": self.detection_metrics.label_findings,
                "new_findings": self.detection_metrics.new_findings,
            },
            "tokens": {
                "total": self.token_metrics.total_tokens,
                "cost_usd": self.token_metrics.total_cost_usd,
                "avg_per_function": self.token_metrics.avg_tokens_per_function,
            },
            "exit_gate": {
                "passed": self.passed_exit_gate,
                "details": self.exit_gate_details,
            },
        }


class LabelEvaluator:
    """Evaluates label quality against ground truth.

    Usage:
        evaluator = LabelEvaluator(ground_truth_overlay)
        report = evaluator.evaluate(predicted_overlay, token_metrics)
        print(f"Precision: {report.precision_metrics.precision:.2%}")
    """

    def __init__(self, ground_truth: LabelOverlay):
        """Initialize evaluator.

        Args:
            ground_truth: Ground truth labels
        """
        self.ground_truth = ground_truth
        self.validator = LabelValidator()

    def evaluate(
        self,
        predicted: LabelOverlay,
        token_metrics: Optional[TokenMetrics] = None,
        baseline_findings: Optional[int] = None,
        label_findings: Optional[int] = None,
    ) -> EvaluationReport:
        """Evaluate predicted labels against ground truth.

        Args:
            predicted: Predicted labels from labeler
            token_metrics: Optional token usage metrics
            baseline_findings: Findings without labels (for delta)
            label_findings: Findings with labels (for delta)

        Returns:
            EvaluationReport with all metrics
        """
        report = EvaluationReport()

        # Calculate precision metrics
        report.precision_metrics = self._calculate_precision(predicted)

        # Calculate detection delta if provided
        if baseline_findings is not None and label_findings is not None:
            report.detection_metrics = self._calculate_detection(
                baseline_findings, label_findings
            )

        # Add token metrics
        if token_metrics:
            report.token_metrics = token_metrics

        # Calculate quality scores per function
        for func_id, label_set in predicted.labels.items():
            report.quality_scores[func_id] = self.validator.score_labels(label_set)

        # Check exit gate
        report.check_exit_gate()

        return report

    def _calculate_precision(self, predicted: LabelOverlay) -> PrecisionMetrics:
        """Calculate precision against ground truth."""
        metrics = PrecisionMetrics()

        # Get all function IDs from both
        all_funcs = set(self.ground_truth.labels.keys()) | set(predicted.labels.keys())

        for func_id in all_funcs:
            gt_labels = self.ground_truth.get_labels(func_id)
            pred_labels = predicted.get_labels(func_id)

            gt_label_ids = {l.label_id for l in gt_labels.labels}
            pred_label_ids = {l.label_id for l in pred_labels.labels}

            # True positives: in both GT and predicted
            tp = gt_label_ids & pred_label_ids
            metrics.true_positives += len(tp)

            # False positives: in predicted but not GT
            fp = pred_label_ids - gt_label_ids
            metrics.false_positives += len(fp)

            # False negatives: in GT but not predicted
            fn = gt_label_ids - pred_label_ids
            metrics.false_negatives += len(fn)

        metrics.total_predicted = sum(len(ls.labels) for ls in predicted.labels.values())
        metrics.total_ground_truth = sum(len(ls.labels) for ls in self.ground_truth.labels.values())

        return metrics

    def _calculate_detection(
        self,
        baseline: int,
        with_labels: int,
    ) -> DetectionMetrics:
        """Calculate detection improvement."""
        metrics = DetectionMetrics()
        metrics.baseline_findings = baseline
        metrics.label_findings = with_labels
        metrics.new_findings = max(0, with_labels - baseline)
        metrics.lost_findings = max(0, baseline - with_labels)
        return metrics


def run_evaluation(
    predicted_overlay: LabelOverlay,
    ground_truth_overlay: LabelOverlay,
    token_metrics: Optional[TokenMetrics] = None,
) -> EvaluationReport:
    """Convenience function to run evaluation.

    Args:
        predicted_overlay: Labels from labeler
        ground_truth_overlay: Known-correct labels
        token_metrics: Optional token usage

    Returns:
        EvaluationReport
    """
    evaluator = LabelEvaluator(ground_truth_overlay)
    return evaluator.evaluate(predicted_overlay, token_metrics)
```

```python
#!/usr/bin/env python3
"""Run label evaluation on real-world corpus.

This script executes the labeling pipeline on a real-world contract corpus
(not just test contracts) and measures the +5% detection delta required
by exit gate LABEL-12.

Usage:
    uv run python scripts/run_label_evaluation.py --corpus .vrs/benchmarks/corpora/
    uv run python scripts/run_label_evaluation.py --corpus examples/damm-vuln-defi/

The script:
1. Runs pattern matching WITHOUT labels (baseline)
2. Runs semantic labeling on the corpus
3. Runs pattern matching WITH labels (including Tier C patterns)
4. Computes detection delta and reports exit gate status
"""

import argparse
import asyncio
import json
import logging
import sys
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple

# Add project root to path
sys.path.insert(0, str(Path(__file__).parent.parent / "src"))

from true_vkg.kg.builder import build_knowledge_graph
from true_vkg.queries.executor import PatternExecutor
from true_vkg.labels import LLMLabeler, LabelingConfig, LabelOverlay
from true_vkg.labels.evaluation import (
    LabelEvaluator,
    EvaluationReport,
    TokenMetrics,
    DetectionMetrics,
)

logging.basicConfig(level=logging.INFO, format="%(levelname)s: %(message)s")
logger = logging.getLogger(__name__)


def find_solidity_files(corpus_path: Path) -> List[Path]:
    """Find all Solidity files in corpus."""
    sol_files = list(corpus_path.rglob("*.sol"))
    # Filter out test files and mocks
    sol_files = [
        f for f in sol_files
        if "test" not in f.name.lower()
        and "mock" not in f.name.lower()
        and "node_modules" not in str(f)
    ]
    return sol_files


def run_baseline_patterns(graph: Any, patterns_dir: Path) -> Tuple[int, List[Dict]]:
    """Run Tier A/B patterns without labels (baseline)."""
    executor = PatternExecutor(patterns_dir)

    # Run only Tier A and B patterns (no labels)
    findings = executor.run_tier_ab(graph)

    return len(findings), findings


def run_labeled_patterns(
    graph: Any,
    overlay: LabelOverlay,
    patterns_dir: Path,
) -> Tuple[int, List[Dict]]:
    """Run all patterns including Tier C with labels."""
    executor = PatternExecutor(patterns_dir, label_overlay=overlay)

    # Run all tiers including C
    findings = executor.run_all_tiers(graph)

    return len(findings), findings


async def run_labeling(
    graph: Any,
    config: LabelingConfig,
) -> Tuple[LabelOverlay, TokenMetrics]:
    """Run semantic labeling on graph."""
    from true_vkg.llm.client import get_client

    provider = await get_client()
    labeler = LLMLabeler(provider, config)

    # Get function node IDs
    function_ids = [
        node.id for node in graph.nodes.values()
        if node.type == "function"
    ]

    logger.info(f"Labeling {len(function_ids)} functions...")

    result = await labeler.label_functions(graph, function_ids)

    token_metrics = TokenMetrics(
        total_tokens=result.total_tokens,
        total_cost_usd=result.total_cost_usd,
        functions_labeled=result.functions_labeled,
        max_tokens_single_call=config.max_tokens_per_call,
    )
    token_metrics.calculate_averages()

    return labeler.get_overlay(), token_metrics


def compare_findings(
    baseline: List[Dict],
    with_labels: List[Dict],
) -> Dict[str, Any]:
    """Compare baseline and labeled findings."""
    baseline_ids = {f["id"] for f in baseline}
    label_ids = {f["id"] for f in with_labels}

    new_findings = label_ids - baseline_ids
    lost_findings = baseline_ids - label_ids

    return {
        "baseline_count": len(baseline),
        "label_count": len(with_labels),
        "new_findings": list(new_findings),
        "lost_findings": list(lost_findings),
        "net_new": len(new_findings) - len(lost_findings),
    }


def print_report(
    report: EvaluationReport,
    comparison: Dict[str, Any],
    corpus_path: Path,
) -> None:
    """Print evaluation report."""
    print("\n" + "=" * 60)
    print("SEMANTIC LABELING EVALUATION REPORT")
    print(f"Corpus: {corpus_path}")
    print("=" * 60)

    print("\n--- Detection Metrics ---")
    print(f"Baseline findings (Tier A/B): {comparison['baseline_count']}")
    print(f"With labels (Tier A/B/C):     {comparison['label_count']}")
    print(f"New findings from labels:     {len(comparison['new_findings'])}")
    print(f"Lost findings:                {len(comparison['lost_findings'])}")
    print(f"Detection Delta:              {report.detection_metrics.detection_delta:+.1f}%")

    print("\n--- Token Usage ---")
    print(f"Total tokens:                 {report.token_metrics.total_tokens:,}")
    print(f"Total cost:                   ${report.token_metrics.total_cost_usd:.4f}")
    print(f"Avg tokens/function:          {report.token_metrics.avg_tokens_per_function:.0f}")

    print("\n--- Exit Gate Status (LABEL-12) ---")
    report.check_exit_gate(
        min_precision=0.75,
        min_detection_delta=5.0,
        max_tokens_per_call=6000,
    )

    for criterion, passed in report.exit_gate_details.items():
        status = "PASS" if passed else "FAIL"
        print(f"  {criterion}: {status}")

    print("\n" + "-" * 60)
    overall = "PASS" if report.passed_exit_gate else "FAIL"
    print(f"OVERALL EXIT GATE: {overall}")
    print("=" * 60 + "\n")


async def main() -> int:
    parser = argparse.ArgumentParser(
        description="Run label evaluation on real-world corpus"
    )
    parser.add_argument(
        "--corpus",
        type=Path,
        required=True,
        help="Path to corpus directory with Solidity files",
    )
    parser.add_argument(
        "--patterns",
        type=Path,
        default=Path("patterns"),
        help="Path to patterns directory",
    )
    parser.add_argument(
        "--output",
        type=Path,
        help="Path to save JSON report",
    )
    parser.add_argument(
        "--max-tokens",
        type=int,
        default=6000,
        help="Max tokens per labeling call",
    )
    args = parser.parse_args()

    # Verify corpus exists
    if not args.corpus.exists():
        logger.error(f"Corpus not found: {args.corpus}")
        return 1

    # Find Solidity files
    sol_files = find_solidity_files(args.corpus)
    if not sol_files:
        logger.error(f"No Solidity files found in {args.corpus}")
        return 1

    logger.info(f"Found {len(sol_files)} Solidity files in corpus")

    # Build knowledge graph
    logger.info("Building knowledge graph...")
    graph = build_knowledge_graph(args.corpus)

    func_count = sum(1 for n in graph.nodes.values() if n.type == "function")
    logger.info(f"Graph has {func_count} functions")

    # Run baseline (no labels)
    logger.info("Running baseline pattern matching (Tier A/B)...")
    baseline_count, baseline_findings = run_baseline_patterns(graph, args.patterns)
    logger.info(f"Baseline: {baseline_count} findings")

    # Run labeling
    logger.info("Running semantic labeling...")
    config = LabelingConfig(max_tokens_per_call=args.max_tokens)
    overlay, token_metrics = await run_labeling(graph, config)

    label_count = sum(len(ls.labels) for ls in overlay.labels.values())
    logger.info(f"Applied {label_count} labels to {len(overlay.labels)} functions")

    # Run with labels (including Tier C)
    logger.info("Running pattern matching with labels (Tier A/B/C)...")
    labeled_count, labeled_findings = run_labeled_patterns(
        graph, overlay, args.patterns
    )
    logger.info(f"With labels: {labeled_count} findings")

    # Compare findings
    comparison = compare_findings(baseline_findings, labeled_findings)

    # Build detection metrics
    detection_metrics = DetectionMetrics(
        baseline_findings=baseline_count,
        label_findings=labeled_count,
        new_findings=len(comparison["new_findings"]),
        lost_findings=len(comparison["lost_findings"]),
    )

    # Build report
    report = EvaluationReport(
        detection_metrics=detection_metrics,
        token_metrics=token_metrics,
    )

    # Print report
    print_report(report, comparison, args.corpus)

    # Save JSON if requested
    if args.output:
        result = {
            "corpus": str(args.corpus),
            "solidity_files": len(sol_files),
            "functions": func_count,
            "evaluation": report.to_dict(),
            "comparison": comparison,
        }
        with open(args.output, "w") as f:
            json.dump(result, f, indent=2)
        logger.info(f"Report saved to {args.output}")

    # Return exit code based on exit gate
    return 0 if report.passed_exit_gate else 1


if __name__ == "__main__":
    sys.exit(asyncio.run(main()))
```
