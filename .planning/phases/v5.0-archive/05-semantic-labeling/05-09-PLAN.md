   ```python
   async def _run_labeling_async(
       graph: KnowledgeGraph,
       output_path: Optional[Path],
       format: str,
   ) -> Optional[LabelOverlay]:
       """Run semantic labeling on graph."""
       from true_vkg.llm.client import get_client
       from true_vkg.labels import LLMLabeler, LabelingConfig

       try:
           # Get LLM provider
           provider = await get_client()  # Or however the client is obtained

           # Configure labeler
           config = LabelingConfig(
               max_tokens_per_call=6000,
               max_functions_per_batch=5,
           )

           labeler = LLMLabeler(provider, config)

           # Get function node IDs
           function_ids = [
               node.id for node in graph.nodes.values()
               if node.type == "function"
           ]

           console.print(f"Labeling {len(function_ids)} functions...")

           # Run labeling
           result = await labeler.label_functions(graph, function_ids)

           console.print(f"  Tokens used: {result.total_tokens}")
           console.print(f"  Cost: ${result.total_cost_usd:.4f}")
           console.print(f"  Labels applied: {result.labels_applied}")

           # Get overlay
           overlay = labeler.get_overlay()

           # Export if path provided
           if output_path:
               if format == "yaml":
                   overlay.export_yaml(output_path)
               else:
                   overlay.export_json(output_path)
               console.print(f"Labels exported to {output_path}")

           return overlay

       except Exception as e:
           console.print(f"[red]Labeling failed: {e}[/red]")
           return None


   def _run_labeling(graph, output_path, format) -> Optional[LabelOverlay]:
       """Sync wrapper for async labeling."""
       import asyncio
       return asyncio.run(_run_labeling_async(graph, output_path, format))


   def _count_labels(overlay: LabelOverlay) -> int:
       """Count total labels in overlay."""
       return sum(len(ls.labels) for ls in overlay.labels.values())
   ```

```python
                    "signature": "function deposit() external payable",
                    "visibility": "external",

            ),
        }
        graph.edges = {}
        return graph

    @pytest.fixture
    def mock_provider(self):
        """Create mock LLM provider."""
        provider = MagicMock()
        provider.generate_with_tools = AsyncMock(return_value=MagicMock(
            content="",
            tool_calls=[{
                "name": "apply_labels_batch",
                "input": {
                    "labels": [
                        {
                            "function_id": "func_1",
                            "label": "value_handling.transfers_value_out",
                            "confidence": "high",
                        },
                        {
                            "function_id": "func_1",

                            "function_id": "func_2",

                }
            }],
            input_tokens=500,
            output_tokens=200,
            cost_usd=0.001,
        ))
        return provider

    @pytest.mark.asyncio
    async def test_labeling_flow(self, mock_graph, mock_provider):
        """Test complete labeling flow."""
        config = LabelingConfig(max_tokens_per_call=6000)
        labeler = LLMLabeler(mock_provider, config)

        # Run labeling
        result = await labeler.label_functions(
            mock_graph,
            ["func_1", "func_2"],
        )

        # Check results
        assert result.functions_labeled == 2
        assert result.labels_applied == 3
        assert result.total_tokens > 0

        # Check overlay
        overlay = labeler.get_overlay()
        assert len(overlay.labels) == 2

        func_1_labels = overlay.get_labels("func_1")
        assert func_1_labels.has_label("value_handling.transfers_value_out")

    @pytest.mark.asyncio
    async def test_tier_c_matching_with_labels(self, mock_graph, mock_provider):
        """Test Tier C pattern matching with labels."""
        # First, label the functions
        labeler = LLMLabeler(mock_provider)
        await labeler.label_functions(mock_graph, ["func_1", "func_2"])
        overlay = labeler.get_overlay()

        # Create Tier C matcher
        matcher = TierCMatcher(overlay)

        # Test has_label condition
        result = matcher.match(
            "func_1",
            conditions_all=[
                TierCCondition(TierCConditionType.HAS_LABEL, "value_handling.transfers_value_out"),

        assert result.matched
        assert "value_handling.transfers_value_out" in result.matched_labels

class TestContextFiltering:
    """Test context-filtered label retrieval."""

    @pytest.fixture
    def labeled_overlay(self):
        """Create overlay with various labels."""
        overlay = LabelOverlay()
        overlay.add_label("func_1", FunctionLabel(
            "access_control.owner_only", LabelConfidence.HIGH, LabelSource.LLM
        ))
        overlay.add_label("func_1", FunctionLabel(
            "value_handling.transfers_value_out", LabelConfidence.MEDIUM, LabelSource.LLM
        ))
        overlay.add_label("func_1", FunctionLabel(
            "state_mutation.writes_critical", LabelConfidence.LOW, LabelSource.LLM
        ))
        return overlay

    def test_reentrancy_context_filters_access_control(self, labeled_overlay):
        """Test that access_control labels are filtered for reentrancy context."""
        filter = LabelFilter(labeled_overlay)
        result = filter.get_filtered_labels("func_1", "reentrancy")

        # Should NOT include access_control labels
        label_ids = [l.label_id for l in result.labels]
        assert "access_control.owner_only" not in label_ids

        # Should include value_handling and state_mutation
        assert "value_handling.transfers_value_out" in label_ids
        assert "state_mutation.writes_critical" in label_ids

    def test_access_control_context(self, labeled_overlay):
        """Test access_control context."""
        filter = LabelFilter(labeled_overlay)
        result = filter.get_filtered_labels("func_1", "access_control")

        label_ids = [l.label_id for l in result.labels]
        assert "access_control.owner_only" in label_ids
        assert "value_handling.transfers_value_out" not in label_ids
```
