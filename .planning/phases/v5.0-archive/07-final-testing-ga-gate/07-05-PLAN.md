   ```python
       """Test detection on obfuscated code."""

       def test_renamed_functions(self):
           """
           PHIL-01 test: Detection works without function name hints.

           Create contracts where:
           - withdraw() renamed to xfer_out_42()
           - balances renamed to _a1b2c3
           - Semantic operations should still detect reentrancy
           """
           # Load obfuscated protocol
           protocols = load_protocols(filter_name="adversarial-obfuscated")
           if not protocols:
               pytest.skip("No obfuscated protocols in corpus")

           for protocol in protocols:
               # Run detection
               from alphaswarm.kg.builder import VKGBuilder
               graph = VKGBuilder().build(str(protocol.contracts_path))

               # Verify semantic operations are detected, not names
               functions = graph.get_functions()
               for func in functions:
                   ops = func.get("semantic_operations", [])
                   # Should have operations even with weird names
                   if "transfer" in str(func).lower() or "value" in str(func).lower():
                       assert ops, f"Expected operations for {func['name']}"

           """Detection works when logic is split across internal functions."""

   @pytest.mark.adversarial
   class TestFalsePositives:
       """Test false positive rate on clean protocols."""

       def test_audited_clean_protocols(self):
           """Verify low FP rate on protocols with no known vulns."""
           # Load protocols marked as clean (no vulnerabilities in ground truth)
           protocols = load_protocols()
           clean_protocols = [p for p in protocols if len(p.vulnerabilities) == 0]

           if not clean_protocols:
               pytest.skip("No clean protocols in corpus")

           total_findings = 0
           for protocol in clean_protocols:
               # Run detection
               from alphaswarm.kg.builder import VKGBuilder
               from alphaswarm.queries.executor import QueryExecutor

               try:
                   graph = VKGBuilder().build(str(protocol.contracts_path))
                   executor = QueryExecutor(graph)
                   findings = executor.run_all_patterns()
                   total_findings += len(findings)
               except Exception:
                   continue

           # Track FP rate
           print(f"Total findings on clean protocols: {total_findings}")
           # No assertion - this is for measurement


   @pytest.mark.adversarial
   class TestKnownHardCases:
       """Test known hard cases from DVDeFi failures."""

       @pytest.mark.parametrize("challenge", KNOWN_HARD_CASES)
       def test_dvdefi_hard_case(self, challenge: str):
           """Test detection on known DVDeFi failures."""
           protocols = load_protocols(filter_name=challenge)

           if not protocols:
               pytest.skip(f"Protocol {challenge} not in corpus")

           protocol = protocols[0]

           # Run detection
           from alphaswarm.kg.builder import VKGBuilder
           from alphaswarm.queries.executor import QueryExecutor

           graph = VKGBuilder().build(str(protocol.contracts_path))
           executor = QueryExecutor(graph)
           findings = executor.run_all_patterns()

           # These are expected to fail (known limitations)
           # Track what we DO detect for improvement
           detected_classes = {f.get("vuln_class") for f in findings}
           expected_classes = {v.vulnerability_class for v in protocol.vulnerabilities}

           # Log for analysis
           print(f"\n{challenge}:")
           print(f"  Expected: {expected_classes}")
           print(f"  Detected: {detected_classes}")
           print(f"  Gap: {expected_classes - detected_classes}")

           # Don't fail - these are known hard cases
           # But track for improvement backlog
   ```
