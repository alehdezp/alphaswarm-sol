   ```python
   """Orchestration tracing and replay for regression testing."""
   import json
   import uuid
   from dataclasses import dataclass, asdict, field
   from datetime import datetime
   from pathlib import Path
   from typing import List, Dict, Any, Optional

   from tests.framework.corpus_db import CorpusDB


   @dataclass
   class AgentTurn:
       """Single agent turn in orchestration."""
       agent_role: str      # "attacker" | "defender" | "verifier" | "arbiter"
       model: str           # "claude-opus-4" | "claude-sonnet-4"
       input_context: str   # Truncated for storage
       output: str          # Agent response
       tool_calls: List[dict] = field(default_factory=list)
       tokens_in: int = 0
       tokens_out: int = 0
       duration_ms: int = 0


   @dataclass
   class OrchestrationTrace:
       """Complete record of an orchestration run."""
       trace_id: str
       timestamp: int
       protocol: str
       finding_id: str
       vuln_class: str

       # Agent interactions
       turns: List[AgentTurn] = field(default_factory=list)

       # Debate outcomes
       attacker_claim: Optional[str] = None
       defender_counter: Optional[str] = None
       verifier_check: Optional[str] = None
       arbiter_verdict: Optional[str] = None

       # Final outcome
       final_verdict: str = "uncertain"  # confirmed/likely/uncertain/rejected
       confidence_bucket: str = "uncertain"
       human_review_required: bool = True

       # Cost tracking
       total_tokens: int = 0
       total_cost_usd: float = 0.0

       def add_turn(self, turn: AgentTurn) -> None:
           """Add an agent turn and update totals."""
           self.turns.append(turn)
           self.total_tokens += turn.tokens_in + turn.tokens_out
           # Rough cost estimate
           self.total_cost_usd += (turn.tokens_in * 0.000003 + turn.tokens_out * 0.000015)


   class OrchestrationRecorder:
       """Record orchestration runs for analysis and replay."""

       def __init__(self, traces_dir: str = ".planning/testing/orchestration-traces"):
           self.traces_dir = Path(traces_dir)
           self.traces_dir.mkdir(parents=True, exist_ok=True)
           self.current_trace: Optional[OrchestrationTrace] = None

       def start_trace(
           self,
           protocol: str,
           finding_id: str,
           vuln_class: str
       ) -> str:
           """Start recording a new orchestration trace."""
           trace_id = f"trace-{datetime.now().strftime('%Y%m%d-%H%M%S')}-{uuid.uuid4().hex[:8]}"
           self.current_trace = OrchestrationTrace(
               trace_id=trace_id,
               timestamp=int(datetime.now().timestamp()),
               protocol=protocol,
               finding_id=finding_id,
               vuln_class=vuln_class
           )
           return trace_id

       def record_turn(
           self,
           agent_role: str,
           model: str,
           input_context: str,
           output: str,
           tool_calls: List[dict] = None,
           tokens_in: int = 0,
           tokens_out: int = 0,
           duration_ms: int = 0
       ) -> None:
           """Record an agent turn."""
           if self.current_trace is None:
               raise ValueError("No active trace - call start_trace first")

           turn = AgentTurn(
               agent_role=agent_role,
               model=model,
               input_context=input_context[:1000],  # Truncate for storage
               output=output[:2000],
               tool_calls=tool_calls or [],
               tokens_in=tokens_in,
               tokens_out=tokens_out,
               duration_ms=duration_ms
           )
           self.current_trace.add_turn(turn)

       def set_debate_outcomes(
           self,
           attacker_claim: str,
           defender_counter: str,
           verifier_check: str,
           arbiter_verdict: str
       ) -> None:
           """Set debate phase outcomes."""
           if self.current_trace:
               self.current_trace.attacker_claim = attacker_claim
               self.current_trace.defender_counter = defender_counter
               self.current_trace.verifier_check = verifier_check
               self.current_trace.arbiter_verdict = arbiter_verdict

       def finish_trace(
           self,
           final_verdict: str,
           human_review_required: bool = True
       ) -> OrchestrationTrace:
           """Finish and save the current trace."""
           if self.current_trace is None:
               raise ValueError("No active trace")

           self.current_trace.final_verdict = final_verdict
           self.current_trace.human_review_required = human_review_required

           # Save to file
           trace_file = self.traces_dir / f"{self.current_trace.trace_id}.json"
           with open(trace_file, "w") as f:
               json.dump(asdict(self.current_trace), f, indent=2)

           trace = self.current_trace
           self.current_trace = None
           return trace

       def load_trace(self, trace_id: str) -> OrchestrationTrace:
           """Load a saved trace for replay."""
           trace_file = self.traces_dir / f"{trace_id}.json"
           with open(trace_file) as f:
               data = json.load(f)

           # Reconstruct dataclass
           turns = [AgentTurn(**t) for t in data.pop("turns", [])]
           return OrchestrationTrace(turns=turns, **data)

       def list_traces(self) -> List[str]:
           """List all saved trace IDs."""
           return [f.stem for f in self.traces_dir.glob("trace-*.json")]


   def compare_traces(
       original: OrchestrationTrace,
       replayed: OrchestrationTrace
   ) -> Dict[str, Any]:
       """Compare original and replayed traces for regression detection."""
       comparison = {
           "verdict_match": original.final_verdict == replayed.final_verdict,
           "original_verdict": original.final_verdict,
           "replayed_verdict": replayed.final_verdict,
           "turn_count_diff": len(replayed.turns) - len(original.turns),
           "cost_diff_usd": replayed.total_cost_usd - original.total_cost_usd,
           "token_diff": replayed.total_tokens - original.total_tokens
       }

       # Check if verdict changed unfavorably
       verdict_order = ["confirmed", "likely", "uncertain", "rejected"]
       orig_idx = verdict_order.index(original.final_verdict) if original.final_verdict in verdict_order else 2
       repl_idx = verdict_order.index(replayed.final_verdict) if replayed.final_verdict in verdict_order else 2

       comparison["verdict_regression"] = repl_idx > orig_idx  # Moving toward uncertain/rejected is regression

       return comparison
   ```
