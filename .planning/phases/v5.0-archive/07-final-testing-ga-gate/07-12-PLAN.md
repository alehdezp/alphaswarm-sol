import pytest

from dataclasses import dataclass


@dataclass
class Finding:
    """Normalized finding for deduplication testing."""
    tool: str
    detector: str
    file: str
    function: str
    line: int
    severity: str
    vuln_class: str
    code_snippet: str


class TestLocationBasedDedup:
    """Test location-based deduplication (file+function+line)."""

    @pytest.mark.dedup
    def test_exact_location_match(self):
        """Findings at same location should dedupe."""
        from true_vkg.dedup.location import LocationDeduplicator

        dedup = LocationDeduplicator()

        findings = [
            Finding("slither", "reentrancy-eth", "Vault.sol", "withdraw", 42, "high", "reentrancy", "..."),
            Finding("aderyn", "reentrancy", "Vault.sol", "withdraw", 42, "high", "reentrancy", "..."),
        ]

        result = dedup.deduplicate(findings)

        # Should merge into single finding
        assert len(result) == 1
        assert "slither" in result[0].tools
        assert "aderyn" in result[0].tools

    @pytest.mark.dedup
    def test_nearby_lines_grouped(self):
        """Findings within threshold lines should consider grouping."""
        from true_vkg.dedup.location import LocationDeduplicator

        dedup = LocationDeduplicator(line_threshold=5)

        findings = [
            Finding("slither", "reentrancy-eth", "Vault.sol", "withdraw", 42, "high", "reentrancy", "..."),
            Finding("mythril", "reentrancy", "Vault.sol", "withdraw", 44, "high", "reentrancy", "..."),
        ]

        result = dedup.deduplicate(findings)

        # Should group if within threshold
        assert len(result) <= 2

    @pytest.mark.dedup
    def test_different_functions_not_grouped(self):
        """Different functions should NOT dedupe on location alone."""
        from true_vkg.dedup.location import LocationDeduplicator

        dedup = LocationDeduplicator()

        findings = [
            Finding("slither", "reentrancy-eth", "Vault.sol", "withdraw", 42, "high", "reentrancy", "..."),
            Finding("slither", "reentrancy-eth", "Vault.sol", "deposit", 42, "high", "reentrancy", "..."),
        ]

        # Different functions = different findings
        assert len(result) == 2
