"""Incremental improvement tracking for VKG.

Enables before/after comparison when improvements are made.
Supports Phase 5.1, 5.2, and future roadmap phases.
"""
import json
import sqlite3
from pathlib import Path
from datetime import datetime
from typing import Dict, Any, List, Optional
from dataclasses import dataclass, asdict
from enum import Enum


class MetricType(Enum):
    """Types of metrics tracked."""
    PRECISION = "precision"
    RECALL = "recall"
    F1 = "f1"
    LATENCY = "latency"
    TOKEN_COST = "token_cost"
    COVERAGE = "coverage"
    FALSE_POSITIVE_RATE = "false_positive_rate"


@dataclass
class Metric:
    """A single metric measurement."""
    name: str
    value: float
    unit: str
    timestamp: str
    phase: Optional[str] = None
    context: Optional[Dict[str, Any]] = None


@dataclass
class ImprovementResult:
    """Result of comparing before/after metrics."""
    metric_name: str
    before: float
    after: float
    delta: float
    percent_change: float
    improved: bool
    regression: bool
    threshold_met: bool


@dataclass
class Baseline:
    """Baseline metrics for a component."""
    component: str
    version: str
    metrics: Dict[str, Metric]
    created_at: str
    phase_requirements: List[str]


class ImprovementTracker:
    """Tracks incremental improvements across BSKG development.

    Supports:
    - Before/after comparison
    - Baseline storage
    - Regression detection
    - Phase-specific tracking (5.1, 5.2, future)
    """

    def __init__(self, db_path: Optional[Path] = None):
        self.db_path = db_path or Path(".vrs/testing/improvements.db")
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self._init_db()

    def _init_db(self):
        """Initialize SQLite database for metric storage."""
        with sqlite3.connect(self.db_path) as conn:
            conn.executescript("""
                CREATE TABLE IF NOT EXISTS baselines (
                    id INTEGER PRIMARY KEY,
                    component TEXT NOT NULL,
                    version TEXT NOT NULL,
                    metrics_json TEXT NOT NULL,
                    created_at TEXT NOT NULL,
                    phase_requirements TEXT,
                    UNIQUE(component, version)
                );

                CREATE TABLE IF NOT EXISTS measurements (
                    id INTEGER PRIMARY KEY,
                    component TEXT NOT NULL,
                    metric_name TEXT NOT NULL,
                    value REAL NOT NULL,
                    unit TEXT,
                    phase TEXT,
                    context_json TEXT,
                    measured_at TEXT NOT NULL
                );

                CREATE TABLE IF NOT EXISTS improvements (
                    id INTEGER PRIMARY KEY,
                    component TEXT NOT NULL,
                    metric_name TEXT NOT NULL,
                    before_value REAL NOT NULL,
                    after_value REAL NOT NULL,
                    delta REAL NOT NULL,
                    percent_change REAL NOT NULL,
                    improved BOOLEAN NOT NULL,
                    regression BOOLEAN NOT NULL,
                    phase TEXT,
                    recorded_at TEXT NOT NULL
                );

                CREATE INDEX IF NOT EXISTS idx_measurements_component
                    ON measurements(component, metric_name);

                CREATE INDEX IF NOT EXISTS idx_improvements_phase
                    ON improvements(phase);
            """)

    def set_baseline(
        self,
        component: str,
        version: str,
        metrics: Dict[str, Metric],
        phase_requirements: Optional[List[str]] = None
    ) -> Baseline:
        """Set baseline metrics for a component."""
        baseline = Baseline(
            component=component,
            version=version,
            metrics=metrics,
            created_at=datetime.now().isoformat(),
            phase_requirements=phase_requirements or []
        )

        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                """INSERT OR REPLACE INTO baselines
                   (component, version, metrics_json, created_at, phase_requirements)
                   VALUES (?, ?, ?, ?, ?)""",
                (
                    component,
                    version,
                    json.dumps({k: asdict(v) for k, v in metrics.items()}),
                    baseline.created_at,
                    json.dumps(phase_requirements)
                )
            )

        return baseline

    def get_baseline(self, component: str, version: Optional[str] = None) -> Optional[Baseline]:
        """Get baseline for a component."""
        with sqlite3.connect(self.db_path) as conn:
            if version:
                row = conn.execute(
                    "SELECT * FROM baselines WHERE component = ? AND version = ?",
                    (component, version)
                ).fetchone()
            else:
                # Get latest
                row = conn.execute(
                    """SELECT * FROM baselines WHERE component = ?
                       ORDER BY created_at DESC LIMIT 1""",
                    (component,)
                ).fetchone()

            if not row:
                return None

            metrics_data = json.loads(row[3])
            metrics = {k: Metric(**v) for k, v in metrics_data.items()}

            return Baseline(
                component=row[1],
                version=row[2],
                metrics=metrics,
                created_at=row[4],
                phase_requirements=json.loads(row[5]) if row[5] else []
            )

    def record_metric(
        self,
        component: str,
        metric_name: str,
        value: float,
        unit: str = "",
        phase: Optional[str] = None,
        context: Optional[Dict[str, Any]] = None
    ) -> Metric:
        """Record a metric measurement."""
        metric = Metric(
            name=metric_name,
            value=value,
            unit=unit,
            timestamp=datetime.now().isoformat(),
            phase=phase,
            context=context
        )

        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                """INSERT INTO measurements
                   (component, metric_name, value, unit, phase, context_json, measured_at)
                   VALUES (?, ?, ?, ?, ?, ?, ?)""",
                (
                    component,
                    metric_name,
                    value,
                    unit,
                    phase,
                    json.dumps(context) if context else None,
                    metric.timestamp
                )
            )

        return metric

    def compare_to_baseline(
        self,
        component: str,
        current_metrics: Dict[str, float],
        threshold: float = 0.0
    ) -> List[ImprovementResult]:
        """Compare current metrics to baseline."""
        baseline = self.get_baseline(component)
        if not baseline:
            return []

        results = []
        for metric_name, current_value in current_metrics.items():
            if metric_name in baseline.metrics:
                baseline_value = baseline.metrics[metric_name].value
                delta = current_value - baseline_value
                percent_change = (delta / baseline_value * 100) if baseline_value != 0 else 0

                # For most metrics, higher is better (precision, recall, etc.)
                # For latency/cost, lower is better
                higher_is_better = metric_name not in ["latency", "token_cost", "false_positive_rate"]

                improved = delta > threshold if higher_is_better else delta < -threshold
                regression = delta < -threshold if higher_is_better else delta > threshold

                result = ImprovementResult(
                    metric_name=metric_name,
                    before=baseline_value,
                    after=current_value,
                    delta=delta,
                    percent_change=percent_change,
                    improved=improved,
                    regression=regression,
                    threshold_met=not regression
                )
                results.append(result)

                # Record improvement
                self._record_improvement(component, result)

        return results

    def _record_improvement(self, component: str, result: ImprovementResult, phase: Optional[str] = None):
        """Record an improvement comparison."""
        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                """INSERT INTO improvements
                   (component, metric_name, before_value, after_value, delta,
                    percent_change, improved, regression, phase, recorded_at)
                   VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)""",
                (
                    component,
                    result.metric_name,
                    result.before,
                    result.after,
                    result.delta,
                    result.percent_change,
                    result.improved,
                    result.regression,
                    phase,
                    datetime.now().isoformat()
                )
            )

    def get_improvement_history(
        self,
        component: Optional[str] = None,
        phase: Optional[str] = None,
        limit: int = 100
    ) -> List[Dict[str, Any]]:
        """Get improvement history."""
        with sqlite3.connect(self.db_path) as conn:
            query = "SELECT * FROM improvements WHERE 1=1"
            params = []

            if component:
                query += " AND component = ?"
                params.append(component)
            if phase:
                query += " AND phase = ?"
                params.append(phase)

            query += " ORDER BY recorded_at DESC LIMIT ?"
            params.append(limit)

            rows = conn.execute(query, params).fetchall()

            return [
                {
                    "component": row[1],
                    "metric_name": row[2],
                    "before": row[3],
                    "after": row[4],
                    "delta": row[5],
                    "percent_change": row[6],
                    "improved": bool(row[7]),
                    "regression": bool(row[8]),
                    "phase": row[9],
                    "recorded_at": row[10]
                }
                for row in rows
            ]

    def detect_regressions(
        self,
        component: str,
        current_metrics: Dict[str, float],
        threshold: float = 0.05  # 5% regression threshold
    ) -> List[ImprovementResult]:
        """Detect regressions from baseline."""
        results = self.compare_to_baseline(component, current_metrics, threshold)
        return [r for r in results if r.regression]

    def get_phase_progress(self, phase: str) -> Dict[str, Any]:
        """Get progress for a specific phase."""
        with sqlite3.connect(self.db_path) as conn:
            # Get all improvements for this phase
            improvements = conn.execute(
                """SELECT metric_name, improved, regression
                   FROM improvements WHERE phase = ?""",
                (phase,)
            ).fetchall()

            # Get baselines with this phase in requirements
            baselines = conn.execute(
                """SELECT component, phase_requirements
                   FROM baselines WHERE phase_requirements LIKE ?""",
                (f'%{phase}%',)
            ).fetchall()

            improved_count = sum(1 for i in improvements if i[1])
            regression_count = sum(1 for i in improvements if i[2])

            return {
                "phase": phase,
                "total_improvements": len(improvements),
                "improved": improved_count,
                "regressions": regression_count,
                "components_tracked": len(baselines)
            }


class PhaseTestTracker:
    """Track test coverage for specific phases (5.1, 5.2, future)."""

    def __init__(self, tracker: ImprovementTracker):
        self.tracker = tracker
        self.phase_requirements = self._load_phase_requirements()

    def _load_phase_requirements(self) -> Dict[str, List[str]]:
        """Load phase requirements from ROADMAP.md."""
        roadmap_path = Path(".planning/ROADMAP.md")
        if not roadmap_path.exists():
            return {}

        requirements = {}
        current_phase = None

        content = roadmap_path.read_text()
        for line in content.split("\n"):
            # Detect phase headers
            if "Phase 5.1" in line:
                current_phase = "5.1"
                requirements[current_phase] = []
            elif "Phase 5.2" in line:
                current_phase = "5.2"
                requirements[current_phase] = []
            elif line.startswith("| TOOL-") or line.startswith("| SDK-"):
                # Extract requirement ID
                parts = line.split("|")
                if len(parts) > 1:
                    req_id = parts[1].strip()
                    if current_phase and req_id:
                        requirements[current_phase].append(req_id)

        return requirements

    def get_phase_requirements(self, phase: str) -> List[str]:
        """Get requirements for a phase."""
        return self.phase_requirements.get(phase, [])

    def record_requirement_test(
        self,
        phase: str,
        requirement_id: str,
        passed: bool,
        metrics: Optional[Dict[str, float]] = None
    ):
        """Record test result for a requirement."""
        component = f"phase-{phase}-{requirement_id}"

        if metrics:
            for name, value in metrics.items():
                self.tracker.record_metric(
                    component=component,
                    metric_name=name,
                    value=value,
                    phase=phase,
                    context={"requirement": requirement_id, "passed": passed}
                )

    def get_phase_coverage(self, phase: str) -> Dict[str, Any]:
        """Get test coverage for a phase."""
        requirements = self.get_phase_requirements(phase)

        with sqlite3.connect(self.tracker.db_path) as conn:
            tested = conn.execute(
                """SELECT DISTINCT component FROM measurements
                   WHERE phase = ?""",
                (phase,)
            ).fetchall()

        tested_reqs = {t[0].split("-")[-1] for t in tested if t[0].startswith(f"phase-{phase}")}

        return {
            "phase": phase,
            "total_requirements": len(requirements),
            "tested": len(tested_reqs),
            "untested": [r for r in requirements if r not in tested_reqs],
            "coverage_percent": (len(tested_reqs) / len(requirements) * 100) if requirements else 0
        }

"""Roadmap-driven test generation for VKG.

Generates test stubs from ROADMAP.md requirements.
Supports incremental testing as phases are built.
"""
import re
from pathlib import Path
from typing import Dict, Any, List, Optional
from dataclasses import dataclass


@dataclass
class Requirement:
    """A requirement from the roadmap."""
    id: str
    description: str
    source: str
    priority: str
    phase: str
    tested: bool = False
    test_file: Optional[str] = None


@dataclass
class PhaseSpec:
    """Specification for a roadmap phase."""
    number: str
    name: str
    requirements: List[Requirement]
    dependencies: List[str]
    status: str


class RoadmapParser:
    """Parse ROADMAP.md to extract testable requirements."""

    def __init__(self, roadmap_path: Optional[Path] = None):
        self.roadmap_path = roadmap_path or Path(".planning/ROADMAP.md")

    def parse(self) -> Dict[str, PhaseSpec]:
        """Parse roadmap into phase specifications."""
        if not self.roadmap_path.exists():
            return {}

        content = self.roadmap_path.read_text()
        phases = {}

        # Parse phase headers and requirements
        current_phase = None
        current_requirements = []

        for line in content.split("\n"):
            # Phase header pattern: ## Phase X.X: Name
            phase_match = re.match(r"##\s+Phase\s+([\d.]+):\s+(.+)", line)
            if phase_match:
                # Save previous phase
                if current_phase:
                    phases[current_phase.number] = PhaseSpec(
                        number=current_phase.number,
                        name=current_phase.name,
                        requirements=current_requirements,
                        dependencies=current_phase.dependencies,
                        status=current_phase.status
                    )

                current_phase = type("Phase", (), {
                    "number": phase_match.group(1),
                    "name": phase_match.group(2).strip(),
                    "dependencies": [],
                    "status": "unknown"
                })()
                current_requirements = []

            # Requirement row: | REQ-ID | Description | ... |
            req_match = re.match(r"\|\s*([A-Z]+-\d+)\s*\|\s*(.+?)\s*\|", line)
            if req_match and current_phase:
                req = Requirement(
                    id=req_match.group(1),
                    description=req_match.group(2).strip(),
                    source="ROADMAP.md",
                    priority="MUST" if "MUST" in line else ("SHOULD" if "SHOULD" in line else "MAY"),
                    phase=current_phase.number
                )
                current_requirements.append(req)

        # Save last phase
        if current_phase:
            phases[current_phase.number] = PhaseSpec(
                number=current_phase.number,
                name=current_phase.name,
                requirements=current_requirements,
                dependencies=[],
                status="unknown"
            )

        return phases

    def get_phase(self, phase_number: str) -> Optional[PhaseSpec]:
        """Get specification for a specific phase."""
        phases = self.parse()
        return phases.get(phase_number)

    def get_untested_requirements(self, phase_number: str) -> List[Requirement]:
        """Get requirements without tests."""
        phase = self.get_phase(phase_number)
        if not phase:
            return []

        return [r for r in phase.requirements if not r.tested]


class TestStubGenerator:
    """Generate test stubs from requirements."""

    def __init__(self):
        self.test_template = '''
@pytest.mark.requirement("{req_id}")
@pytest.mark.phase("{phase}")
def test_{req_id_snake}(self):
    """
    Requirement: {req_id}
    Description: {description}
    Priority: {priority}

    TODO: Implement test for this requirement.
    """
    # Arrange
    # TODO: Set up test fixtures

    # Act
    # TODO: Execute the functionality

    # Assert
    # TODO: Verify the requirement is met
    pytest.skip("Test not implemented")
'''

    def generate_stub(self, req: Requirement) -> str:
        """Generate a test stub for a requirement."""
        req_id_snake = req.id.lower().replace("-", "_")

        return self.test_template.format(
            req_id=req.id,
            req_id_snake=req_id_snake,
            description=req.description,
            priority=req.priority,
            phase=req.phase
        )

    def generate_test_file(self, phase: PhaseSpec) -> str:
        """Generate a complete test file for a phase."""
        phase_snake = phase.number.replace(".", "_")

        header = f'''"""
Auto-generated tests for Phase {phase.number}: {phase.name}

Generated from ROADMAP.md requirements.
Implement each test to validate the requirement.
"""
import pytest


class TestPhase{phase_snake}Requirements:
    """Tests for Phase {phase.number} requirements."""
'''

        stubs = [self.generate_stub(req) for req in phase.requirements]

        return header + "\n".join(stubs)

    def generate_for_phase(self, phase_number: str, output_dir: Optional[Path] = None) -> Path:
        """Generate test file for a phase."""
        parser = RoadmapParser()
        phase = parser.get_phase(phase_number)

        if not phase:
            raise ValueError(f"Phase {phase_number} not found in roadmap")

        content = self.generate_test_file(phase)

        output_dir = output_dir or Path("tests/requirements")
        output_dir.mkdir(parents=True, exist_ok=True)

        output_file = output_dir / f"test_phase_{phase_number.replace('.', '_')}_requirements.py"
        output_file.write_text(content)

        return output_file


class IncrementalTestRunner:
    """Run tests incrementally as phases are built."""

    def __init__(self, tracker: "ImprovementTracker"):
        self.tracker = tracker
        self.parser = RoadmapParser()

    def run_phase_tests(self, phase_number: str) -> Dict[str, Any]:
        """Run all tests for a phase and track results."""
        import subprocess

        phase = self.parser.get_phase(phase_number)
        if not phase:
            return {"error": f"Phase {phase_number} not found"}

        # Run pytest for this phase
        result = subprocess.run(
            [
                "pytest",
                f"tests/requirements/test_phase_{phase_number.replace('.', '_')}_requirements.py",
                f"tests/integration/",
                "-v",
                f"-m", f"phase{phase_number.replace('.', '')}",
                "--json-report",
                "--json-report-file=/tmp/phase_test_report.json"
            ],
            capture_output=True,
            text=True
        )

        # Parse results
        report_path = Path("/tmp/phase_test_report.json")
        if report_path.exists():
            import json
            report = json.loads(report_path.read_text())

            passed = report.get("summary", {}).get("passed", 0)
            failed = report.get("summary", {}).get("failed", 0)
            skipped = report.get("summary", {}).get("skipped", 0)

            # Record metrics
            total = passed + failed + skipped
            pass_rate = passed / total if total > 0 else 0

            self.tracker.record_metric(
                component=f"phase-{phase_number}",
                metric_name="test_pass_rate",
                value=pass_rate,
                phase=phase_number
            )

            return {
                "phase": phase_number,
                "passed": passed,
                "failed": failed,
                "skipped": skipped,
                "pass_rate": pass_rate,
                "requirements_total": len(phase.requirements)
            }

        return {"error": "Test report not generated", "stdout": result.stdout, "stderr": result.stderr}

    def run_incremental(self, phases: List[str]) -> List[Dict[str, Any]]:
        """Run tests for multiple phases incrementally."""
        results = []
        for phase in phases:
            result = self.run_phase_tests(phase)
            results.append(result)

            # Stop if phase fails (dependencies not met)
            if result.get("failed", 0) > 0:
                break

        return results

        return {
