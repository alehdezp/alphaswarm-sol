"""Shadow audit comparison tests - BSKG vs human auditor findings."""
import pytest
from pathlib import Path
from typing import List, Dict, Any, Set, Tuple
from dataclasses import dataclass

from tests.corpus.loader import load_protocols, ProtocolMetadata, Vulnerability


@dataclass
class ShadowAuditResult:
    """Result of shadow audit comparison for one protocol."""
    protocol_name: str
    human_findings: int
    vkg_findings: int
    overlap: int  # Found by both
    human_only: int  # Human found, BSKG missed
    vkg_only: int  # BSKG found, human didn't mention
    recall: float  # BSKG found / human found
    novel_potential: int  # VKG-only that might be real


def compare_findings(
    vkg_findings: List[Dict[str, Any]],
    human_findings: List[Vulnerability]
) -> ShadowAuditResult:
    """
    Compare BSKG findings to human audit findings.

    Matching criteria:
    - Same contract (fuzzy match)
    - Same function (fuzzy match)
    - Same or compatible vulnerability class
    """
    # Build comparison sets
    human_set: Set[Tuple[str, str, str]] = set()
    for v in human_findings:
        # Normalize for matching
        contract = v.contract.lower().replace(".sol", "")
        function = v.function.lower()
        vuln_class = normalize_vuln_class(v.vulnerability_class)
        human_set.add((contract, function, vuln_class))

    vkg_set: Set[Tuple[str, str, str]] = set()
    for f in vkg_findings:
        contract = f.get("contract", "").lower().replace(".sol", "")
        function = f.get("function", "").lower()
        vuln_class = normalize_vuln_class(f.get("vuln_class", ""))
        vkg_set.add((contract, function, vuln_class))

    # Calculate overlap
    overlap = len(human_set & vkg_set)
    human_only = len(human_set - vkg_set)
    vkg_only = len(vkg_set - human_set)

    recall = overlap / len(human_set) if human_set else 1.0

    return ShadowAuditResult(
        protocol_name="",  # Filled by caller
        human_findings=len(human_set),
        vkg_findings=len(vkg_set),
        overlap=overlap,
        human_only=human_only,
        vkg_only=vkg_only,
        recall=recall,
        novel_potential=vkg_only  # VKG-only findings need human review
    )


def normalize_vuln_class(vuln_class: str) -> str:
    """Normalize vulnerability class names for comparison."""
    mapping = {
        "reentrancy": "reentrancy",
        "reentrant": "reentrancy",
        "re-entrancy": "reentrancy",
        "access-control": "access-control",
        "access control": "access-control",
        "authorization": "access-control",
        "oracle": "oracle",
        "price-oracle": "oracle",
        "price manipulation": "oracle",
        # Add more mappings as needed
    }
    normalized = vuln_class.lower().strip()
    return mapping.get(normalized, normalized)


@pytest.mark.shadow
class TestShadowAudit:
    """Shadow audit comparison tests."""

    @pytest.mark.parametrize("protocol", [
        p for p in load_protocols() if len(p.vulnerabilities) > 0
    ][:10], ids=lambda p: p.name)
    def test_shadow_audit_protocol(self, protocol: ProtocolMetadata):
        """Compare BSKG findings to human audit for a protocol."""
        if not protocol.contracts_path.exists():
            pytest.skip(f"Contracts not available for {protocol.name}")

        # Run BSKG detection
        from alphaswarm.kg.builder import VKGBuilder
        from alphaswarm.queries.executor import QueryExecutor

        try:
            graph = VKGBuilder().build(str(protocol.contracts_path))
            executor = QueryExecutor(graph)
            vkg_findings = executor.run_all_patterns()
        except Exception as e:
            pytest.fail(f"VKG failed: {e}")

        # Compare to human findings
        result = compare_findings(vkg_findings, protocol.vulnerabilities)
        result.protocol_name = protocol.name

        # Log results
        print(f"\n=== Shadow Audit: {protocol.name} ===")
        print(f"Human findings: {result.human_findings}")
        print(f"VKG findings: {result.vkg_findings}")
        print(f"Overlap (both found): {result.overlap}")
        print(f"Human only (VKG missed): {result.human_only}")
        print(f"VKG only (potential novel): {result.vkg_only}")
        print(f"Recall (VKG/Human): {result.recall:.1%}")

        # Target: 80% recall per 07-CONTEXT.md
        # Don't hard-fail, but warn
        if result.recall < 0.80:
            print(f"  WARNING: Below 80% recall target")

    def test_aggregate_shadow_results(self):
        """Calculate aggregate shadow audit metrics."""
        protocols = [p for p in load_protocols() if len(p.vulnerabilities) > 0]

        if not protocols:
            pytest.skip("No protocols with ground truth")

        total_human = 0
        total_overlap = 0
        total_vkg_only = 0
        results = []

        for protocol in protocols[:20]:
            if not protocol.contracts_path.exists():
                continue

            try:
                from alphaswarm.kg.builder import VKGBuilder
                from alphaswarm.queries.executor import QueryExecutor

                graph = VKGBuilder().build(str(protocol.contracts_path))
                executor = QueryExecutor(graph)
                vkg_findings = executor.run_all_patterns()

                result = compare_findings(vkg_findings, protocol.vulnerabilities)
                result.protocol_name = protocol.name
                results.append(result)

                total_human += result.human_findings
                total_overlap += result.overlap
                total_vkg_only += result.vkg_only

            except Exception:
                continue

        # Aggregate metrics
        aggregate_recall = total_overlap / total_human if total_human > 0 else 0

        print("\n=== Aggregate Shadow Audit Results ===")
        print(f"Protocols analyzed: {len(results)}")
        print(f"Total human findings: {total_human}")
        print(f"Total overlap: {total_overlap}")
        print(f"Total VKG-only: {total_vkg_only}")
        print(f"Aggregate recall: {aggregate_recall:.1%}")

        # Target assertion
        assert aggregate_recall >= 0.80, (
            f"Aggregate recall {aggregate_recall:.1%} < 80% target"
        )
