   ```python
   """Corpus protocol loading and metadata."""
   from dataclasses import dataclass, field
   from pathlib import Path
   from typing import List, Optional, Dict, Any
   import yaml

   @dataclass
   class Vulnerability:
       """Ground truth vulnerability from audit report."""
       finding_id: str
       severity: str  # "critical" | "high" | "medium" | "low"
       vulnerability_class: str  # e.g., "reentrancy", "access-control"
       contract: str
       function: str
       lines: Optional[List[int]] = None
       description: str = ""
       exploited: bool = False
       exploit_tx: Optional[str] = None

   @dataclass
   class ProtocolMetadata:
       """Metadata for a protocol in the corpus."""
       name: str
       version: str  # git hash or version tag
       source_url: str
       contracts_path: Path
       audit_reports: List[str] = field(default_factory=list)
       complexity_rating: str = "medium"  # "simple" | "medium" | "complex"
       protocol_type: str = "defi"  # "lending" | "dex" | "yield" | "nft" | "dao" | "vault"
       proxy_type: Optional[str] = None  # "EIP-1967" | "UUPS" | "Diamond" | None
       vulnerabilities: List[Vulnerability] = field(default_factory=list)
       # Testing thresholds
       min_recall: float = 0.70
       min_precision: float = 0.80
       # Categorization
       is_adversarial: bool = False
       is_core: bool = False  # Core 15 vs surface 35+

   def load_protocol(protocol_dir: Path) -> ProtocolMetadata:
       """Load single protocol from directory."""
       metadata_path = protocol_dir / "metadata.yaml"
       ground_truth_path = protocol_dir / "ground_truth.yaml"

       if not metadata_path.exists():
           raise FileNotFoundError(f"No metadata.yaml in {protocol_dir}")

       with open(metadata_path) as f:
           meta = yaml.safe_load(f)

       vulnerabilities = []
       if ground_truth_path.exists():
           with open(ground_truth_path) as f:
               gt = yaml.safe_load(f)
               vulnerabilities = [
                   Vulnerability(**v) for v in gt.get("vulnerabilities", [])
               ]

       return ProtocolMetadata(
           name=meta["name"],
           version=meta.get("version", "unknown"),
           source_url=meta.get("source_url", ""),
           contracts_path=protocol_dir / "contracts",
           audit_reports=meta.get("audit_reports", []),
           complexity_rating=meta.get("complexity_rating", "medium"),
           protocol_type=meta.get("protocol_type", "defi"),
           proxy_type=meta.get("proxy_type"),
           vulnerabilities=vulnerabilities,
           min_recall=meta.get("min_recall", 0.70),
           min_precision=meta.get("min_precision", 0.80),
           is_adversarial=meta.get("is_adversarial", False),
           is_core=meta.get("is_core", False)
       )

   def load_protocols(
       corpus_dir: Path = Path("tests/corpus/protocols"),
       filter_core: bool = False,
       filter_adversarial: bool = False,
       filter_type: Optional[str] = None,
       filter_name: Optional[str] = None
   ) -> List[ProtocolMetadata]:
       """Load all protocols from corpus directory."""
       protocols = []

       for protocol_dir in sorted(corpus_dir.iterdir()):
           if not protocol_dir.is_dir():
               continue
           if protocol_dir.name.startswith("."):
               continue

           try:
               protocol = load_protocol(protocol_dir)

               # Apply filters
               if filter_core and not protocol.is_core:
                   continue
               if filter_adversarial and not protocol.is_adversarial:
                   continue
               if filter_type and protocol.protocol_type != filter_type:
                   continue
               if filter_name and filter_name.lower() not in protocol.name.lower():
                   continue

               protocols.append(protocol)
           except Exception as e:
               print(f"Warning: Failed to load {protocol_dir}: {e}")

       return protocols

   def get_corpus_stats(protocols: List[ProtocolMetadata]) -> Dict[str, Any]:
       """Get statistics about loaded corpus."""
       # Implementation: count by type, complexity, vuln class, etc.
   ```

```python
"""Main corpus test runner for BSKG evaluation."""
import pytest
import time
from pathlib import Path
from typing import List, Dict, Any

from tests.corpus.loader import load_protocols, ProtocolMetadata, Vulnerability
from tests.framework.corpus_db import CorpusDB, ProtocolResult
from tests.framework.test_modes import TestConfig


def evaluate_findings(
    findings: List[Dict[str, Any]],
    ground_truth: List[Vulnerability]
) -> Dict[str, Any]:
    """
    Compare BSKG findings to ground truth.

    Returns metrics dict with TP, FP, FN, precision, recall, F1.
    """
    # Build sets for comparison
    gt_set = {(v.contract, v.function, v.vulnerability_class) for v in ground_truth}
    finding_set = {
        (f.get("contract"), f.get("function"), f.get("vuln_class"))
        for f in findings
    }

    tp = len(gt_set & finding_set)
    fp = len(finding_set - gt_set)
    fn = len(gt_set - finding_set)

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    f1 = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0.0

    return {
        "true_positives": tp,
        "false_positives": fp,
        "false_negatives": fn,
        "precision": precision,
        "recall": recall,
        "f1_score": f1,
        "findings": findings,
        "ground_truth": [v.__dict__ for v in ground_truth]
    }


def run_detection_pipeline(
    protocol: ProtocolMetadata,
    config: TestConfig
) -> tuple[List[Dict[str, Any]], int, int]:
    """
    Run BSKG detection pipeline on protocol.

    Returns: (findings, build_time_ms, detection_time_ms)
    """
    # Import here to avoid circular imports
    from alphaswarm.kg.builder import VKGBuilder
    from alphaswarm.queries.executor import QueryExecutor

    # Build graph
    build_start = time.time()
    try:
        graph = VKGBuilder().build(str(protocol.contracts_path))
    except Exception as e:
        return [], 0, 0  # Build failure
    build_time = int((time.time() - build_start) * 1000)

    # Run pattern detection
    detect_start = time.time()
    try:
        executor = QueryExecutor(graph)
        # Run all patterns or specific patterns based on config
        findings = executor.run_all_patterns()
    except Exception as e:
        findings = []
    detect_time = int((time.time() - detect_start) * 1000)

    return findings, build_time, detect_time


# Parametrized test: one test per protocol
@pytest.mark.corpus
@pytest.mark.parametrize("protocol", load_protocols(), ids=lambda p: p.name)
def test_protocol_detection(
    protocol: ProtocolMetadata,
    corpus_db: CorpusDB,
    current_run_id: str,
    test_config: TestConfig
):
    """Run BSKG detection on protocol and verify against ground truth."""

    # Skip if filtered by config
    if test_config.max_protocols > 0:
        # TODO: Track count across tests
        pass

    # Run detection
    findings, build_time, detect_time = run_detection_pipeline(protocol, test_config)

    # Evaluate against ground truth
    metrics = evaluate_findings(findings, protocol.vulnerabilities)

    # Record to database if enabled
    if corpus_db is not None and current_run_id is not None:
        import json
        result = ProtocolResult(
            run_id=current_run_id,
            protocol_name=protocol.name,
            protocol_version=protocol.version,
            true_positives=metrics["true_positives"],
            false_positives=metrics["false_positives"],
            false_negatives=metrics["false_negatives"],
            precision=metrics["precision"],
            recall=metrics["recall"],
            f1_score=metrics["f1_score"],
            build_time_ms=build_time,
            detection_time_ms=detect_time,
            total_time_ms=build_time + detect_time,
            llm_calls=0,  # TODO: Track LLM usage
            total_tokens=0,
            cost_usd=0.0,
            avg_evidence_score=0.0,  # TODO: Grade evidence
            findings_json=json.dumps(metrics["findings"])
        )
        corpus_db.record_result(result)

    # Assert thresholds
    assert metrics["recall"] >= protocol.min_recall, (
        f"{protocol.name}: Recall {metrics['recall']:.1%} < {protocol.min_recall:.1%}"
    )
    assert metrics["precision"] >= protocol.min_precision, (
        f"{protocol.name}: Precision {metrics['precision']:.1%} < {protocol.min_precision:.1%}"
    )


@pytest.mark.corpus
def test_corpus_summary(corpus_db: CorpusDB, current_run_id: str):
    """Generate summary after all protocol tests complete."""
    if corpus_db is None:
        pytest.skip("No database - run with --save-results")

    summary = corpus_db.get_run_summary(current_run_id)

    # Log summary
    print(f"\n=== Corpus Run Summary ===")
    print(f"Protocols tested: {summary.get('total_protocols', 0)}")
    print(f"Avg precision: {summary.get('avg_precision', 0):.1%}")
    print(f"Avg recall: {summary.get('avg_recall', 0):.1%}")
    print(f"Avg F1: {summary.get('avg_f1', 0):.1%}")
```
