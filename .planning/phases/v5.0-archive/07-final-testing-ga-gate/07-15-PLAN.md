```python
class AutoFixer:
    def __init__(self, dry_run: bool = False):
        self.fixes_applied: List[FixResult] = []
        self.dry_run = dry_run

    def attempt_fix(self, error: Exception, context: dict) -> FixResult:
        """Attempt to auto-fix an error. Returns FixResult with success/failure."""
        category = self.categorize_error(error)
        strategy = self.get_strategy(category)

        if self.dry_run:
            return FixResult(category=category, would_fix=True, applied=False)

        result = strategy.apply(error, context)
        self.fixes_applied.append(result)
        return result

    def categorize_error(self, error: Exception) -> FixCategory:
        """Categorize error for appropriate fix strategy."""
        ...

    def get_fixes_for_synthesis(self) -> List[FixResult]:
        """Return all fixes for 0.5.1 synthesis."""
        return self.fixes_applied
```

**Fix strategies per category:**
- IMPORT_ERROR: Try alternative import paths, check __init__.py exports
- DEPENDENCY_MISSING: Run `uv add` for missing packages
- PATH_ERROR: Resolve relative paths, create missing directories
- SYNTAX_ERROR: Basic syntax correction, report for manual review
- CONFIG_ERROR: Create default configs, copy from templates
- PERMISSION_ERROR: chmod, create parent directories
- TIMEOUT_ERROR: Exponential backoff retry

```python
    def recoverable(self, context: str):
        """Context manager for recoverable operations."""
        try:
            yield
        except Exception as e:
            result = self.handle_error(e, context)
```
