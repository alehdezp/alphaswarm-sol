"""Performance measurement tests for VKG."""
import pytest
import time
import statistics
from pathlib import Path
from typing import List, Dict, Any, Tuple
from dataclasses import dataclass

from tests.corpus.loader import load_protocols, ProtocolMetadata
from tests.framework.corpus_db import CorpusDB


@dataclass
class PerformanceMetrics:
    """Performance metrics for a single run."""
    protocol_name: str
    contract_count: int
    function_count: int
    build_time_ms: int
    detection_time_ms: int
    total_time_ms: int
    memory_peak_mb: float
    llm_calls: int
    llm_tokens: int
    llm_cost_usd: float


def measure_build_time(contracts_path: Path) -> Tuple[Any, int, int, int]:
    """
    Measure BSKG build time for a contracts path.

    Returns: (graph, build_time_ms, contract_count, function_count)
    """
    from alphaswarm.kg.builder import VKGBuilder

    start = time.perf_counter()
    graph = VKGBuilder().build(str(contracts_path))
    end = time.perf_counter()

    build_time_ms = int((end - start) * 1000)
    contract_count = len(graph.get_contracts()) if graph else 0
    function_count = len(graph.get_functions()) if graph else 0

    return graph, build_time_ms, contract_count, function_count


def measure_detection_time(graph: Any) -> Tuple[List[Dict], int, int, int, float]:
    """
    Measure pattern detection time.

    Returns: (findings, detect_time_ms, llm_calls, llm_tokens, llm_cost)
    """
    from alphaswarm.queries.executor import QueryExecutor

    start = time.perf_counter()
    executor = QueryExecutor(graph)
    findings = executor.run_all_patterns()
    end = time.perf_counter()

    detect_time_ms = int((end - start) * 1000)

    # TODO: Extract LLM metrics from executor
    llm_calls = 0
    llm_tokens = 0
    llm_cost = 0.0

    return findings, detect_time_ms, llm_calls, llm_tokens, llm_cost


@pytest.mark.performance
class TestBuildPerformance:
    """Test BSKG build performance."""

    @pytest.mark.parametrize("protocol", load_protocols()[:10], ids=lambda p: p.name)
    def test_build_time(self, protocol: ProtocolMetadata, corpus_db: CorpusDB):
        """Measure build time for each protocol."""
        if not protocol.contracts_path.exists():
            pytest.skip(f"Contracts not available for {protocol.name}")

        graph, build_time, contracts, functions = measure_build_time(protocol.contracts_path)

        # Log metrics
        print(f"\n{protocol.name}:")
        print(f"  Contracts: {contracts}, Functions: {functions}")
        print(f"  Build time: {build_time}ms")
        print(f"  Time per contract: {build_time / max(contracts, 1):.1f}ms")

        # No hard assertion - this is measurement
        # But flag outliers
        if build_time > 60000:  # > 60 seconds
            print(f"  WARNING: Build time exceeds 60s threshold")

    def test_build_scalability(self):
        """Test build time scales reasonably with code size."""
        protocols = load_protocols()

        times = []
        sizes = []

        for protocol in protocols[:20]:
            if not protocol.contracts_path.exists():
                continue

            graph, build_time, contracts, functions = measure_build_time(protocol.contracts_path)
            times.append(build_time)
            sizes.append(functions)

        # Simple correlation check
        if len(times) > 5:
            # Build time should correlate with size
            # Linear or slightly superlinear is acceptable
            print(f"\nScalability analysis:")
            print(f"  Min build: {min(times)}ms, Max: {max(times)}ms")
            print(f"  Min functions: {min(sizes)}, Max: {max(sizes)}")

#!/usr/bin/env python
"""Run BSKG performance benchmarks and generate report."""
import argparse
import json
import time
import statistics
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Any
import subprocess


def run_benchmarks(
    protocols_dir: Path,
    output_file: Path,
    max_protocols: int = 50
) -> Dict[str, Any]:
    """Run full benchmark suite."""
    from tests.corpus.loader import load_protocols
    from tests.corpus.test_performance import (
        measure_build_time,
        measure_detection_time
    )

    results = {
        "timestamp": datetime.now().isoformat(),
        "vkg_version": get_vkg_version(),
        "git_commit": get_git_commit(),
        "protocols": [],
        "summary": {}
    }

    protocols = load_protocols(corpus_dir=protocols_dir)[:max_protocols]

    build_times = []
    detect_times = []
    total_times = []
    function_counts = []

    for protocol in protocols:
        print(f"Benchmarking {protocol.name}...")

        if not protocol.contracts_path.exists():
            print(f"  Skipping - contracts not found")
            continue

        try:
            # Measure build
            graph, build_time, contracts, functions = measure_build_time(
                protocol.contracts_path
            )

            if graph is None:
                print(f"  Skipping - build failed")
                continue

            # Measure detection
            findings, detect_time, llm_calls, llm_tokens, llm_cost = measure_detection_time(graph)

            total_time = build_time + detect_time

            # Record
            protocol_result = {
                "name": protocol.name,
                "contracts": contracts,
                "functions": functions,
                "build_time_ms": build_time,
                "detect_time_ms": detect_time,
                "total_time_ms": total_time,
                "findings": len(findings),
                "llm_calls": llm_calls,
                "llm_tokens": llm_tokens,
                "llm_cost_usd": llm_cost
            }
            results["protocols"].append(protocol_result)

            build_times.append(build_time)
            detect_times.append(detect_time)
            total_times.append(total_time)
            function_counts.append(functions)

            print(f"  Build: {build_time}ms, Detect: {detect_time}ms, Total: {total_time}ms")

        except Exception as e:
            print(f"  Error: {e}")

    # Calculate summary
    if build_times:
        results["summary"] = {
            "protocols_tested": len(results["protocols"]),
            "build_time": {
                "min_ms": min(build_times),
                "max_ms": max(build_times),
                "avg_ms": statistics.mean(build_times),
                "median_ms": statistics.median(build_times)
            },
            "detect_time": {
                "min_ms": min(detect_times),
                "max_ms": max(detect_times),
                "avg_ms": statistics.mean(detect_times),
                "median_ms": statistics.median(detect_times)
            },
            "total_time": {
                "min_ms": min(total_times),
                "max_ms": max(total_times),
                "avg_ms": statistics.mean(total_times),
                "median_ms": statistics.median(total_times)
            },
            "functions": {
                "min": min(function_counts),
                "max": max(function_counts),
                "avg": statistics.mean(function_counts)
            }
        }

    # Write results
    with open(output_file, "w") as f:
        json.dump(results, f, indent=2)

    return results


def get_vkg_version() -> str:
    """Get BSKG version."""
    try:
        from alphaswarm import __version__
        return __version__
    except ImportError:
        return "unknown"


def get_git_commit() -> str:
    """Get current git commit."""
    try:
        result = subprocess.check_output(
            ["git", "rev-parse", "HEAD"],
            stderr=subprocess.DEVNULL
        )
        return result.decode().strip()[:8]
    except:
        return "unknown"


def print_summary(results: Dict[str, Any]) -> None:
    """Print benchmark summary."""
    summary = results.get("summary", {})

    print("\n" + "=" * 60)
    print("VKG Performance Benchmark Summary")
    print("=" * 60)
    print(f"Version: {results.get('vkg_version', 'unknown')}")
    print(f"Commit: {results.get('git_commit', 'unknown')}")
    print(f"Protocols tested: {summary.get('protocols_tested', 0)}")
    print()

    if "build_time" in summary:
        bt = summary["build_time"]
        print("Build Time:")
        print(f"  Min: {bt['min_ms']}ms")
        print(f"  Max: {bt['max_ms']}ms")
        print(f"  Avg: {bt['avg_ms']:.1f}ms")
        print(f"  Median: {bt['median_ms']}ms")
        print()

    if "total_time" in summary:
        tt = summary["total_time"]
        print("Total Time (Build + Detection):")
        print(f"  Min: {tt['min_ms']}ms")
        print(f"  Max: {tt['max_ms']}ms")
        print(f"  Avg: {tt['avg_ms']:.1f}ms")
        print(f"  Median: {tt['median_ms']}ms")


def main():
    parser = argparse.ArgumentParser(description="Run BSKG performance benchmarks")
    parser.add_argument(
        "--protocols-dir",
        type=Path,
        default=Path("tests/corpus/protocols"),
        help="Directory containing protocols"
    )
    parser.add_argument(
        "--output",
        type=Path,
        default=Path(".planning/testing/benchmark-results.json"),
        help="Output file for results"
    )
    parser.add_argument(
        "--max-protocols",
        type=int,
        default=50,
        help="Maximum protocols to benchmark"
    )

    args = parser.parse_args()

    args.output.parent.mkdir(parents=True, exist_ok=True)

    results = run_benchmarks(
        protocols_dir=args.protocols_dir,
        output_file=args.output,
        max_protocols=args.max_protocols
    )

    print_summary(results)


if __name__ == "__main__":
    main()
