2. Create `tests/framework/test_modes.py`:
   ```python
   """Test mode configuration for Quick/Standard/Thorough testing."""
   from dataclasses import dataclass
   from enum import Enum
   from typing import List, Optional

   class TestMode(Enum):
       QUICK = "quick"        # Pattern-specific only (< 30s)
       STANDARD = "standard"  # Integration tests (< 5min)
       THOROUGH = "thorough"  # Full corpus (hours)

   @dataclass
   class TestConfig:
       """Configuration for a test mode."""
       mode: TestMode
       patterns: List[str]        # Pattern filters (empty = all)
       protocols: List[str]       # Protocol filters (empty = all)
       max_protocols: int         # Maximum protocols to test
       parallel_workers: int      # xdist workers
       cost_budget_usd: float     # Max LLM cost
       skip_llm: bool             # Use heuristics only
       save_results: bool         # Persist to SQLite

   # Default configurations
   QUICK_CONFIG = TestConfig(
       mode=TestMode.QUICK,
       patterns=[],
       protocols=[],
       max_protocols=0,  # Pattern tests only, no corpus
       parallel_workers=1,
       cost_budget_usd=1.0,
       skip_llm=True,
       save_results=False
   )

   STANDARD_CONFIG = TestConfig(
       mode=TestMode.STANDARD,
       patterns=[],
       protocols=[],
       max_protocols=10,  # Subset of corpus
       parallel_workers=4,
       cost_budget_usd=10.0,
       skip_llm=False,
       save_results=True
   )

   THOROUGH_CONFIG = TestConfig(
       mode=TestMode.THOROUGH,
       patterns=[],
       protocols=[],
       max_protocols=0,  # All protocols
       parallel_workers=10,  # Full parallelism
       cost_budget_usd=200.0,
       skip_llm=False,
       save_results=True
   )

   def get_test_config(mode: str = "standard") -> TestConfig:
       """Get test configuration for mode."""
       configs = {
           "quick": QUICK_CONFIG,
           "standard": STANDARD_CONFIG,
           "thorough": THOROUGH_CONFIG
       }
       return configs.get(mode, STANDARD_CONFIG)

   def pytest_addoption_mode(parser):
       """Add --mode option to pytest."""
       parser.addoption(
           "--mode",
           action="store",
           default="standard",
           choices=["quick", "standard", "thorough"],
           help="Test mode: quick (< 30s), standard (< 5min), thorough (hours)"
       )
   ```

       pytest_addoption_mode(parser)
       parser.addoption(
           "--protocol",
           action="store",
           default=None,
           help="Run tests for specific protocol only"
       )
       parser.addoption(
           "--save-results",
           action="store_true",
           default=False,
           help="Save results to SQLite (default for standard/thorough)"
       )


   @pytest.fixture(scope="session")
   def test_config(request) -> TestConfig:
       """Get test configuration based on --mode."""
       mode = request.config.getoption("--mode")
       config = get_test_config(mode)

       # Override save_results if explicitly set
       if request.config.getoption("--save-results"):
           config.save_results = True

       return config


   @pytest.fixture(scope="session")
   def corpus_db(test_config) -> Generator[CorpusDB, None, None]:
       """SQLite database for test results."""
       if not test_config.save_results:
           yield None
           return

       db = CorpusDB("tests/corpus/results.db")
       yield db


   @pytest.fixture(scope="session")

       """Start a corpus run and yield run_id."""
       if corpus_db is None:
           yield None
           return

       # Get BSKG version and git commit
       try:
           from alphaswarm import __version__
           vkg_version = __version__
       except ImportError:
           vkg_version = "unknown"

       try:
           git_commit = subprocess.check_output(
               ["git", "rev-parse", "HEAD"],
               cwd=Path(__file__).parent.parent.parent
           ).decode().strip()[:8]
       except subprocess.CalledProcessError:
           git_commit = "unknown"

       run_id = corpus_db.start_run(
           mode=test_config.mode.value,
           vkg_version=vkg_version,
           git_commit=git_commit
       )

       yield run_id

       """Filter tests based on mode."""

       if mode == "quick":
