class TestFoundryExecution:
    """SDK-13: Execute tests via Foundry CLI and capture results."""

    @pytest.mark.testgen
    @pytest.mark.slow
    def test_foundry_cli_available(self):
        """Foundry (forge) should be detected."""
        from true_vkg.testgen.foundry import FoundryRunner

        runner = FoundryRunner()

        assert runner.is_available() or runner.installation_hint is not None

    @pytest.mark.testgen
    @pytest.mark.slow
    def test_run_generated_test(self, tmp_path):
        """Should execute generated Foundry test."""
        from true_vkg.testgen.foundry import FoundryRunner, FoundryScaffoldGenerator
        from true_vkg.beads.bead import Bead

        runner = FoundryRunner()

        if not runner.is_available():
            pytest.skip("Foundry not installed")

        # Generate test
        generator = FoundryScaffoldGenerator()
        bead = Bead(
            id="exec-test-1",
            finding_type="reentrancy",
            evidence={"contract": "Vault", "function": "withdraw"}
        )
        scaffold = generator.generate(bead)

        # Write test file
        test_file = tmp_path / "test" / "Exploit.t.sol"
        test_file.parent.mkdir(parents=True, exist_ok=True)
        test_file.write_text(scaffold)

        # Execute
        result = runner.run(tmp_path)

        assert result.executed
        assert result.passed is not None or result.failed is not None

    @pytest.mark.testgen
    def test_capture_test_output(self):
        """Test output should be captured and parsed."""
        from true_vkg.testgen.foundry import FoundryResultParser

        parser = FoundryResultParser()

        # Mock forge output
        forge_output = """
[PASS] testReentrancyExploit() (gas: 123456)
[FAIL] testAccessControl() (gas: 789)
    Reason: Unauthorized
"""

        results = parser.parse(forge_output)

        assert len(results) == 2
        assert results[0].name == "testReentrancyExploit"
        assert results[0].passed
        assert results[1].name == "testAccessControl"
        assert not results[1].passed


class TestConfidenceFromTests:
    """SDK-14: Integrate test pass/fail into confidence elevation."""

    @pytest.mark.testgen
    def test_passing_test_confirms(self):
        """Passing exploit test should confirm vulnerability."""
        from true_vkg.confidence import ConfidenceScorer
        from true_vkg.beads.bead import Bead

        scorer = ConfidenceScorer()

        bead = Bead(
            id="conf-test-1",
            finding_type="reentrancy",
            confidence="likely",
            evidence={}
        )

        # Test passed = exploit confirmed
        bead.test_result = {"passed": True, "test_name": "testReentrancyExploit"}

        scorer.apply_test_result(bead)

        assert bead.confidence == "confirmed"

    @pytest.mark.testgen
    def test_failing_test_reduces(self):
        """Failing exploit test should reduce confidence."""
        from true_vkg.confidence import ConfidenceScorer
        from true_vkg.beads.bead import Bead

        scorer = ConfidenceScorer()

        bead = Bead(
            id="conf-test-2",
            finding_type="reentrancy",
            confidence="likely",
            evidence={}
        )

        # Test failed = exploit did not work
        bead.test_result = {"passed": False, "reason": "ReentrancyGuard"}

        scorer.apply_test_result(bead)

        # Should reduce (not necessarily reject)
        assert bead.confidence in ["uncertain", "rejected"]

    @pytest.mark.testgen
    def test_confidence_string_in_evidence(self):
        """'confirmed via test' should appear in evidence."""
        from true_vkg.confidence import ConfidenceScorer
        from true_vkg.beads.bead import Bead

        scorer = ConfidenceScorer()

        bead = Bead(
            id="conf-test-3",
            finding_type="reentrancy",
            confidence="likely",
            evidence={}
        )

        bead.test_result = {"passed": True}
        scorer.apply_test_result(bead)

        assert "confirmed via test" in str(bead.evidence) or "test_confirmed" in bead.evidence
