"""End-to-end validation tests per vulnerability class."""
import pytest
from pathlib import Path
from typing import List, Dict, Any, Tuple
from dataclasses import dataclass

from tests.corpus.loader import load_protocols, ProtocolMetadata, Vulnerability
from tests.framework.corpus_db import CorpusDB


# Vulnerability classes to validate (from PHILOSOPHY.md)
VULN_CLASSES = [
    "reentrancy",
    "access-control",
    "oracle-manipulation",
    "flash-loan",
    "arithmetic",
    "unchecked-return",
    "front-running",
    "price-manipulation",
    "governance-attack",
    "signature-replay",
]

# Thresholds per 07-CONTEXT.md
TIER_A_PRECISION = 0.95
TIER_B_PRECISION = 0.80
CRITICAL_HIGH_RECALL = 0.90
MEDIUM_LOW_RECALL = 0.70


@dataclass
class ClassValidationResult:
    """Validation result for a vulnerability class."""
    vuln_class: str
    protocols_tested: int
    total_ground_truth: int
    total_detected: int
    true_positives: int
    false_positives: int
    false_negatives: int
    precision: float
    recall: float
    f1: float
    tier_a_findings: int
    tier_b_findings: int
    tier_a_precision: float
    tier_b_precision: float
    critical_high_recall: float
    medium_low_recall: float


def get_protocols_for_class(vuln_class: str) -> List[ProtocolMetadata]:
    """Get protocols with ground truth for this vulnerability class."""
    all_protocols = load_protocols()
    return [
        p for p in all_protocols
        if any(v.vulnerability_class == vuln_class for v in p.vulnerabilities)
    ]


def run_detection_for_class(
    protocols: List[ProtocolMetadata],
    vuln_class: str
) -> Tuple[List[Dict], List[Vulnerability]]:
    """Run BSKG detection for specific vulnerability class."""
    from alphaswarm.kg.builder import VKGBuilder
    from alphaswarm.queries.executor import QueryExecutor

    all_findings = []
    all_ground_truth = []

    for protocol in protocols:
        try:
            # Build graph
            graph = VKGBuilder().build(str(protocol.contracts_path))

            # Run patterns for this class only
            executor = QueryExecutor(graph)
            # Filter to class-specific patterns
            findings = executor.run_patterns_for_class(vuln_class)
            all_findings.extend(findings)

            # Collect ground truth
            gt = [v for v in protocol.vulnerabilities if v.vulnerability_class == vuln_class]
            all_ground_truth.extend(gt)

        except Exception as e:
            print(f"Warning: {protocol.name} failed: {e}")

    return all_findings, all_ground_truth


def calculate_class_metrics(
    findings: List[Dict],
    ground_truth: List[Vulnerability],
    vuln_class: str
) -> ClassValidationResult:
    """Calculate comprehensive metrics for vulnerability class."""
    # Match findings to ground truth
    gt_set = {(v.contract, v.function) for v in ground_truth}
    finding_set = {(f.get("contract"), f.get("function")) for f in findings}

    tp = len(gt_set & finding_set)
    fp = len(finding_set - gt_set)
    fn = len(gt_set - finding_set)

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    f1 = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0.0

    # Tier breakdown (assume tier from finding metadata)
    tier_a = [f for f in findings if f.get("tier") == "A"]
    tier_b = [f for f in findings if f.get("tier") == "B"]

    # Severity breakdown
    critical_high_gt = [v for v in ground_truth if v.severity in ("critical", "high")]
    medium_low_gt = [v for v in ground_truth if v.severity in ("medium", "low")]

    # Calculate tier precision (against all GT for simplicity)
    tier_a_tp = len([f for f in tier_a if (f.get("contract"), f.get("function")) in gt_set])
    tier_a_prec = tier_a_tp / len(tier_a) if tier_a else 1.0

    tier_b_tp = len([f for f in tier_b if (f.get("contract"), f.get("function")) in gt_set])
    tier_b_prec = tier_b_tp / len(tier_b) if tier_b else 1.0

    # Calculate severity recall
    ch_detected = len([v for v in critical_high_gt if (v.contract, v.function) in finding_set])
    ch_recall = ch_detected / len(critical_high_gt) if critical_high_gt else 1.0

    ml_detected = len([v for v in medium_low_gt if (v.contract, v.function) in finding_set])
    ml_recall = ml_detected / len(medium_low_gt) if medium_low_gt else 1.0

    return ClassValidationResult(
        vuln_class=vuln_class,
        protocols_tested=len(set(v.contract for v in ground_truth)),
        total_ground_truth=len(ground_truth),
        total_detected=len(findings),
        true_positives=tp,
        false_positives=fp,
        false_negatives=fn,
        precision=precision,
        recall=recall,
        f1=f1,
        tier_a_findings=len(tier_a),
        tier_b_findings=len(tier_b),
        tier_a_precision=tier_a_prec,
        tier_b_precision=tier_b_prec,
        critical_high_recall=ch_recall,
        medium_low_recall=ml_recall
    )


# Parametrized test: one per vulnerability class
@pytest.mark.corpus
@pytest.mark.e2e
@pytest.mark.parametrize("vuln_class", VULN_CLASSES)
def test_vulnerability_class_validation(vuln_class: str, corpus_db: CorpusDB):
    """Validate detection for a specific vulnerability class."""
    protocols = get_protocols_for_class(vuln_class)

    if not protocols:
        pytest.skip(f"No protocols with {vuln_class} ground truth")

    # Run detection
    findings, ground_truth = run_detection_for_class(protocols, vuln_class)

    if not ground_truth:
        pytest.skip(f"No ground truth for {vuln_class}")

    # Calculate metrics
    result = calculate_class_metrics(findings, ground_truth, vuln_class)

    # Log results
    print(f"\n=== {vuln_class.upper()} Validation ===")
    print(f"Ground truth: {result.total_ground_truth}, Detected: {result.total_detected}")
    print(f"Precision: {result.precision:.1%}, Recall: {result.recall:.1%}, F1: {result.f1:.1%}")
    print(f"Tier A precision: {result.tier_a_precision:.1%}")
    print(f"Tier B precision: {result.tier_b_precision:.1%}")
    print(f"Critical/High recall: {result.critical_high_recall:.1%}")

    # Assert thresholds per 07-CONTEXT.md
    assert result.tier_a_precision >= TIER_A_PRECISION, (
        f"Tier A precision {result.tier_a_precision:.1%} < {TIER_A_PRECISION:.1%}"
    )
    assert result.tier_b_precision >= TIER_B_PRECISION, (
        f"Tier B precision {result.tier_b_precision:.1%} < {TIER_B_PRECISION:.1%}"
    )
    assert result.critical_high_recall >= CRITICAL_HIGH_RECALL, (
        f"Critical/High recall {result.critical_high_recall:.1%} < {CRITICAL_HIGH_RECALL:.1%}"
    )
