import asyncio
from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict, List, Optional, Callable, Any
import logging

from true_vkg.agents.runtime import AgentRuntime, AgentConfig, AgentRole, AgentResponse
from true_vkg.agents.hooks import AgentInbox, WorkClaim
from true_vkg.beads.schema import VulnerabilityBead

logger = logging.getLogger(__name__)

@dataclass
class PropulsionConfig:
    """Configuration for propulsion engine"""
    max_concurrent_per_role: int = 2
    poll_interval_seconds: float = 1.0
    work_timeout_seconds: int = 300
    enable_resume: bool = True
    on_complete: Optional[Callable[[str, AgentResponse], None]] = None
    on_error: Optional[Callable[[str, Exception], None]] = None

@dataclass
class WorkResult:
    """Result of processing one work item"""
    bead_id: str
    agent_role: AgentRole
    success: bool
    response: Optional[AgentResponse] = None
    error: Optional[str] = None
    duration_ms: int = 0
    resumed: bool = False  # Whether this was resumed work

    def to_dict(self) -> Dict:
        return {
            "bead_id": self.bead_id,
            "agent_role": self.agent_role.value,
            "success": self.success,
            "error": self.error,
            "duration_ms": self.duration_ms,
            "resumed": self.resumed,
        }

class PropulsionEngine:
    """Autonomous work-pulling engine for agents.

    Per PHILOSOPHY.md:
    - Agents pull work from their hooks (inboxes)
    - Context-fresh: new agent instance per bead
    - Work state persists so agents can resume

    Per 05.2-CONTEXT.md:
    - Hybrid work claiming: orchestrator assigns, agents can pull more
    - Configurable agents per role
    - Beads distributed round-robin
    """

    def __init__(
        self,
        runtime: AgentRuntime,
        inboxes: Dict[AgentRole, AgentInbox],
        config: Optional[PropulsionConfig] = None,
    ):
        self.runtime = runtime
        self.inboxes = inboxes
        self.config = config or PropulsionConfig()
        self._running = False
        self._active_tasks: Dict[str, asyncio.Task] = {}  # bead_id -> task
        self._results: List[WorkResult] = []

    async def run(self, timeout: Optional[int] = None) -> List[WorkResult]:
        """Run propulsion until all inboxes empty or timeout.

        Args:
            timeout: Optional timeout in seconds

        Returns:
            List of work results
        """
        self._running = True
        self._results = []
        start = datetime.now()

        try:
            while self._running:
                # Check timeout
                if timeout and (datetime.now() - start).total_seconds() > timeout:
                    logger.info("Propulsion timeout reached")
                    break

                # Check if any work remains
                if not self._has_work():
                    # Wait for active tasks to complete
                    if self._active_tasks:
                        await asyncio.sleep(self.config.poll_interval_seconds)
                        continue
                    else:
                        logger.info("All work complete")
                        break

                # Spawn workers for each role
                for role, inbox in self.inboxes.items():
                    await self._spawn_if_needed(role, inbox)

                await asyncio.sleep(self.config.poll_interval_seconds)

        finally:
            self._running = False
            # Wait for remaining tasks
            if self._active_tasks:
                await asyncio.gather(*self._active_tasks.values(), return_exceptions=True)

        return self._results

    async def _spawn_if_needed(self, role: AgentRole, inbox: AgentInbox) -> None:
        """Spawn agent for role if under concurrency limit"""
        # Count active tasks for this role
        active_for_role = sum(
            1 for task_id in self._active_tasks
            if task_id.startswith(role.value)
        )

        if active_for_role >= self.config.max_concurrent_per_role:
            return

        # Claim work
        claim = inbox.claim_work()
        if claim is None:
            return

        # Spawn context-fresh agent
        task_id = f"{role.value}_{claim.bead.id}"
        task = asyncio.create_task(self._process_work(claim))
        self._active_tasks[task_id] = task

        # Add callback for cleanup
        task.add_done_callback(lambda t: self._on_task_done(task_id, claim))

    async def _process_work(self, claim: WorkClaim) -> WorkResult:
        """Process single work item with context-fresh agent.

        Per 05.2-CONTEXT.md: Context-fresh (new instance per bead)
        """
        start = datetime.now()
        bead = claim.bead
        role = claim.agent_role

        # Check for resumable work state
        resumed = False
        work_context = ""
        if self.config.enable_resume and bead.work_state:
            work_context = f"\n\nResuming from previous state:\n{bead.work_state}"
            resumed = True

        # Build agent prompt
        prompt = self._build_agent_prompt(bead, role, work_context)

        # Get agent config for this role
        agent_config = self._get_agent_config(role)

        try:
            # Spawn context-fresh agent (new instance)
            response = await asyncio.wait_for(
                self.runtime.spawn_agent(agent_config, prompt),
                timeout=self.config.work_timeout_seconds,
            )

            # Save work state for resumability
            self._save_work_state(bead, response)

            duration = int((datetime.now() - start).total_seconds() * 1000)
            result = WorkResult(
                bead_id=bead.id,
                agent_role=role,
                success=True,
                response=response,
                duration_ms=duration,
                resumed=resumed,
            )

            if self.config.on_complete:
                self.config.on_complete(bead.id, response)

            return result

        except asyncio.TimeoutError:
            duration = int((datetime.now() - start).total_seconds() * 1000)
            result = WorkResult(
                bead_id=bead.id,
                agent_role=role,
                success=False,
                error="Timeout",
                duration_ms=duration,
                resumed=resumed,
            )
            if self.config.on_error:
                self.config.on_error(bead.id, TimeoutError("Work timeout"))
            return result

        except Exception as e:
            duration = int((datetime.now() - start).total_seconds() * 1000)
            result = WorkResult(
                bead_id=bead.id,
                agent_role=role,
                success=False,
                error=str(e),
                duration_ms=duration,
                resumed=resumed,
            )
            if self.config.on_error:
                self.config.on_error(bead.id, e)
            return result

    def _on_task_done(self, task_id: str, claim: WorkClaim) -> None:
        """Handle task completion"""
        task = self._active_tasks.pop(task_id, None)
        if task is None:
            return

        try:
            result = task.result()
            self._results.append(result)

            inbox = self.inboxes.get(claim.agent_role)
            if inbox:
                if result.success:
                    inbox.complete_work(claim.bead.id)
                else:
                    inbox.fail_work(claim.bead.id)

        except Exception as e:
            logger.error(f"Task {task_id} failed: {e}")
            # Mark as failed
            inbox = self.inboxes.get(claim.agent_role)
            if inbox:
                inbox.fail_work(claim.bead.id)

    def _has_work(self) -> bool:
        """Check if any inbox has pending work"""
        return any(inbox.pending_count > 0 for inbox in self.inboxes.values())

    def _build_agent_prompt(
        self, bead: VulnerabilityBead, role: AgentRole, extra_context: str = ""
    ) -> str:
        """Build role-specific prompt from bead"""
        base_prompt = bead.get_llm_prompt()

        role.meta/instructions = {
            AgentRole.ATTACKER: "Focus on constructing an exploit scenario.",
            AgentRole.DEFENDER: "Focus on finding mitigations and safe patterns.",
            AgentRole.VERIFIER: "Cross-check evidence and synthesize verdict.",
            AgentRole.TEST_BUILDER: "Generate a Foundry test to demonstrate the vulnerability.",
        }

        instruction = role.meta/instructions.get(role, "Analyze this vulnerability.")
        return f"{base_prompt}\n\n## Your Role: {role.value.upper()}\n{instruction}{extra_context}"

    def _get_agent_config(self, role: AgentRole) -> AgentConfig:
        """Get agent config for role"""
        from true_vkg.agents.roles.prompts import (
            ATTACKER_SYSTEM_PROMPT,
            DEFENDER_SYSTEM_PROMPT,
            VERIFIER_SYSTEM_PROMPT,
            TEST_BUILDER_SYSTEM_PROMPT,
        )

        prompts = {
            AgentRole.ATTACKER: ATTACKER_SYSTEM_PROMPT,
            AgentRole.DEFENDER: DEFENDER_SYSTEM_PROMPT,
            AgentRole.VERIFIER: VERIFIER_SYSTEM_PROMPT,
            AgentRole.TEST_BUILDER: TEST_BUILDER_SYSTEM_PROMPT,
        }

        return AgentConfig(
            role=role,
            system_prompt=prompts.get(role, "You are a security auditor."),
            tools=[],
        )

    def _save_work_state(self, bead: VulnerabilityBead, response: AgentResponse) -> None:
        """Save work state for resumability"""
        if bead.work_state is None:
            bead.work_state = {}

        bead.work_state["last_response_summary"] = response.content[:500]
        bead.work_state["tokens_used"] = {
            "input": response.input_tokens,
            "output": response.output_tokens,
        }
        bead.last_updated = datetime.now()

    def stop(self) -> None:
        """Stop propulsion gracefully"""
        self._running = False

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Dict, List, Optional, Set
import asyncio
import logging

from true_vkg.agents.runtime import AgentRuntime, AgentRole
from true_vkg.agents.hooks import AgentInbox, InboxConfig, HookStorage
from true_vkg.agents.infrastructure import SupervisorAgent, IntegratorAgent
from true_vkg.beads.schema import VulnerabilityBead
from true_vkg.orchestration.schemas import Pool
from .engine import PropulsionEngine, PropulsionConfig, WorkResult

logger = logging.getLogger(__name__)

class CoordinatorStatus(str, Enum):
    """Status of coordinator"""
    IDLE = "idle"
    RUNNING = "running"
    PAUSED = "paused"
    COMPLETE = "complete"
    FAILED = "failed"

@dataclass
class CoordinatorConfig:
    """Configuration for agent coordinator"""
    agents_per_role: Dict[str, int] = field(default_factory=lambda: {
        "attacker": 2,
        "defender": 2,
        "verifier": 1,
        "test_builder": 1,
    })
    enable_supervisor: bool = True
    supervisor_check_interval: int = 60
    enable_integration: bool = True
    poll_interval: float = 1.0

@dataclass
class CoordinatorReport:
    """Report from coordinator run"""
    status: CoordinatorStatus
    total_beads: int
    completed_beads: int
    failed_beads: int
    results_by_role: Dict[str, int]
    duration_seconds: float
    stuck_work: List[str]

    def to_dict(self) -> Dict:
        return {
            "status": self.status.value,
            "total_beads": self.total_beads,
            "completed_beads": self.completed_beads,
            "failed_beads": self.failed_beads,
            "results_by_role": self.results_by_role,
            "duration_seconds": self.duration_seconds,
            "stuck_work": self.stuck_work,
        }

class AgentCoordinator:
    """Coordinates multiple agents processing a pool of beads.

    Orchestrates:
    1. Inbox creation and bead distribution
    2. Propulsion engine for each agent role
    3. Supervisor monitoring
    4. Integration of results
    5. **Verifier queue assignment when attacker AND defender complete**

    Per 05.2-CONTEXT.md:
    - Pool config specifies how many agents per role
    - Beads distributed round-robin
    - Verifier processes beads only after both attacker and defender complete
    """

    def __init__(
        self,
        runtime: AgentRuntime,
        config: Optional[CoordinatorConfig] = None,
    ):
        self.runtime = runtime
        self.config = config or CoordinatorConfig()
        self._inboxes: Dict[AgentRole, AgentInbox] = {}
        self._engines: Dict[AgentRole, PropulsionEngine] = {}
        self._supervisor: Optional[SupervisorAgent] = None
        self._integrator: Optional[IntegratorAgent] = None
        self._status = CoordinatorStatus.IDLE
        # Track completion status for verifier assignment
        self._attacker_complete: Set[str] = set()  # bead_ids completed by attacker
        self._defender_complete: Set[str] = set()  # bead_ids completed by defender
        self._bead_registry: Dict[str, VulnerabilityBead] = {}  # bead_id -> bead

    def setup_for_pool(self, pool: Pool, beads: List[VulnerabilityBead]) -> None:
        """Set up coordinator for a pool of beads.

        Creates inboxes for each role and distributes beads.

        Args:
            pool: Pool to process
            beads: Beads to distribute
        """
        # Register beads for later lookup
        for bead in beads:
            self._bead_registry[bead.id] = bead

        # Create inboxes for active roles
        active_roles = [
            AgentRole.ATTACKER,
            AgentRole.DEFENDER,
            AgentRole.VERIFIER,
        ]

        for role in active_roles:
            max_agents = self.config.agents_per_role.get(role.value, 1)
            self._inboxes[role] = AgentInbox(
                role,
                InboxConfig(max_queue_size=len(beads)),
            )

        # Distribute beads to inboxes (round-robin initially to attacker/defender)
        for bead in beads:
            # Both attacker and defender analyze each bead
            self._inboxes[AgentRole.ATTACKER].assign(bead)
            self._inboxes[AgentRole.DEFENDER].assign(bead)

        # Create propulsion engine with completion callback for verifier assignment
        self._engines[AgentRole.ATTACKER] = PropulsionEngine(
            self.runtime,
            {AgentRole.ATTACKER: self._inboxes[AgentRole.ATTACKER]},
            PropulsionConfig(
                max_concurrent_per_role=self.config.agents_per_role.get("attacker", 2),
                poll_interval_seconds=self.config.poll_interval,
                on_complete=self._on_attacker_complete,
            ),
        )

        self._engines[AgentRole.DEFENDER] = PropulsionEngine(
            self.runtime,
            {AgentRole.DEFENDER: self._inboxes[AgentRole.DEFENDER]},
            PropulsionConfig(
                max_concurrent_per_role=self.config.agents_per_role.get("defender", 2),
                poll_interval_seconds=self.config.poll_interval,
                on_complete=self._on_defender_complete,
            ),
        )

        logger.info(f"Coordinator setup complete: {len(beads)} beads across {len(active_roles)} roles")

    def _on_attacker_complete(self, bead_id: str, response) -> None:
        """Handle attacker completing work on a bead.

        If defender has also completed, assign to verifier queue.
        """
        self._attacker_complete.add(bead_id)
        self._check_verifier_assignment(bead_id)

    def _on_defender_complete(self, bead_id: str, response) -> None:
        """Handle defender completing work on a bead.

        If attacker has also completed, assign to verifier queue.
        """
        self._defender_complete.add(bead_id)
        self._check_verifier_assignment(bead_id)

    def _check_verifier_assignment(self, bead_id: str) -> None:
        """Assign bead to verifier if both attacker and defender have completed.

        This is the key link: completed attacker/defender work triggers verifier queue assignment.
        """
        if bead_id in self._attacker_complete and bead_id in self._defender_complete:
            bead = self._bead_registry.get(bead_id)
            if bead and AgentRole.VERIFIER in self._inboxes:
                logger.info(f"Bead {bead_id}: Both attacker and defender complete, assigning to verifier")
                self._inboxes[AgentRole.VERIFIER].assign(bead)

    async def run(self, timeout: Optional[int] = None) -> CoordinatorReport:
        """Run coordinated agent execution.

        1. Start propulsion engines for attacker/defender
        2. Run supervisor checks periodically
        3. Send completed beads to verifier (via completion callbacks)
        4. Integrate final results

        Args:
            timeout: Optional timeout in seconds

        Returns:
            CoordinatorReport with results
        """
        self._status = CoordinatorStatus.RUNNING
        start = datetime.now()
        all_results: List[WorkResult] = []
        stuck_work: List[str] = []

        try:
            # Phase 1: Run attacker and defender in parallel
            attacker_task = asyncio.create_task(
                self._engines[AgentRole.ATTACKER].run(timeout=timeout)
            )
            defender_task = asyncio.create_task(
                self._engines[AgentRole.DEFENDER].run(timeout=timeout)
            )

            # Wait for both to complete
            attacker_results, defender_results = await asyncio.gather(
                attacker_task, defender_task
            )
            all_results.extend(attacker_results)
            all_results.extend(defender_results)

            # Phase 2: Verifier processes beads (already assigned via callbacks)
            # Create verifier propulsion engine
            if self._inboxes[AgentRole.VERIFIER].pending_count > 0:
                verifier_engine = PropulsionEngine(
                    self.runtime,
                    {AgentRole.VERIFIER: self._inboxes[AgentRole.VERIFIER]},
                    PropulsionConfig(
                        max_concurrent_per_role=self.config.agents_per_role.get("verifier", 1),
                        poll_interval_seconds=self.config.poll_interval,
                    ),
                )
                verifier_results = await verifier_engine.run(timeout=timeout)
                all_results.extend(verifier_results)

            # Phase 3: Supervisor check for stuck work
            if self.config.enable_supervisor:
                for inbox in self._inboxes.values():
                    stuck_work.extend([
                        bid for bid, claim in inbox._in_progress.items()
                    ])

            self._status = CoordinatorStatus.COMPLETE

        except Exception as e:
            logger.error(f"Coordinator failed: {e}")
            self._status = CoordinatorStatus.FAILED
            raise

        duration = (datetime.now() - start).total_seconds()

        # Compile report
        completed = sum(1 for r in all_results if r.success)
        failed = sum(1 for r in all_results if not r.success)
        by_role = {}
        for result in all_results:
            role = result.agent_role.value
            by_role[role] = by_role.get(role, 0) + 1

        return CoordinatorReport(
            status=self._status,
            total_beads=len(self._bead_registry),
            completed_beads=completed,
            failed_beads=failed,
            results_by_role=by_role,
            duration_seconds=duration,
            stuck_work=stuck_work,
        )

    def stop(self) -> None:
        """Stop all engines"""
        self._status = CoordinatorStatus.PAUSED
        for engine in self._engines.values():
            engine.stop()

    @property
    def status(self) -> CoordinatorStatus:
        return self._status
