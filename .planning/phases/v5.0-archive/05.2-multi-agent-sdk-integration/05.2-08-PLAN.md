"""CLI command for multi-agent orchestration.

Implements SDK-08: CLI + SDK drive same artifact contract.
Both CLI and SDK produce identical pool/bead artifacts.

Usage:
    alphaswarm orchestrate path/to/contracts/ --output .vrs/pools/
    alphaswarm orchestrate --pool existing-pool-id --resume
"""

import asyncio
from dataclasses import dataclass
from pathlib import Path
from typing import Optional
import json
import logging
import typer

from true_vkg.agents.runtime import AgentRole, RuntimeConfig
from true_vkg.agents.runtime.anthropic import AnthropicRuntime
from true_vkg.agents.runtime.openai_agents import OpenAIAgentsRuntime
from true_vkg.agents.propulsion import AgentCoordinator, CoordinatorConfig, CoordinatorReport
from true_vkg.orchestration.pool import PoolManager
from true_vkg.orchestration.schemas import Pool, PoolStatus, Scope
from true_vkg.beads.storage import BeadStorage
from true_vkg.kg.builder import VKGBuilder

logger = logging.getLogger(__name__)

app = typer.Typer()

@dataclass
class OrchestrateOptions:
    """Options for orchestration command"""
    sdk: str = "anthropic"  # "anthropic" or "openai"
    output_dir: Path = Path(".vrs/pools")
    timeout: int = 3600  # 1 hour default
    agents_attacker: int = 2
    agents_defender: int = 2
    agents_verifier: int = 1
    verbose: bool = False
    resume: bool = False
    headless: bool = False

def get_runtime(sdk: str, config: Optional[RuntimeConfig] = None):
    """Get agent runtime based on SDK choice.

    Ensures CLI and SDK use same runtime implementations.
    """
    config = config or RuntimeConfig()

    if sdk == "anthropic":
        return AnthropicRuntime(config)
    elif sdk == "openai":
        return OpenAIAgentsRuntime(config)
    else:
        raise ValueError(f"Unknown SDK: {sdk}. Use 'anthropic' or 'openai'")

@app.command()
def orchestrate(
    target: Path = typer.Argument(
        ...,
        help="Path to contracts directory or existing pool",
        exists=True,
    ),
    output: Path = typer.Option(
        Path(".vrs/pools"),
        "--output", "-o",
        help="Output directory for pool artifacts",
    ),
    sdk: str = typer.Option(
        "anthropic",
        "--sdk",
        help="SDK to use: anthropic or openai",
    ),
    timeout: int = typer.Option(
        3600,
        "--timeout", "-t",
        help="Timeout in seconds (default 1 hour)",
    ),
    agents_attacker: int = typer.Option(
        2,
        "--attackers",
        help="Number of attacker agents",
    ),
    agents_defender: int = typer.Option(
        2,
        "--defenders",
        help="Number of defender agents",
    ),
    agents_verifier: int = typer.Option(
        1,
        "--verifiers",
        help="Number of verifier agents",
    ),
    resume: bool = typer.Option(
        False,
        "--resume",
        help="Resume existing pool from checkpoint",
    ),
    headless: bool = typer.Option(
        False,
        "--headless",
        help="Run without interactive prompts (for CI)",
    ),
    verbose: bool = typer.Option(
        False,
        "--verbose", "-v",
        help="Verbose output",
    ),
) -> None:
    """Run multi-agent orchestration on contracts.

    Builds knowledge graph, detects patterns, creates beads,
    and runs attacker/defender/verifier agents to produce verdicts.

    Example:
        alphaswarm orchestrate ./contracts/ -o .vrs/pools/audit-001
    """
    if verbose:
        logging.basicConfig(level=logging.DEBUG)
    else:
        logging.basicConfig(level=logging.INFO)

    options = OrchestrateOptions(
        sdk=sdk,
        output_dir=output,
        timeout=timeout,
        agents_attacker=agents_attacker,
        agents_defender=agents_defender,
        agents_verifier=agents_verifier,
        verbose=verbose,
        resume=resume,
        headless=headless,
    )

    # Run async orchestration
    result = asyncio.run(_run_orchestration(target, options))

    # Output results
    _print_report(result, options)

async def _run_orchestration(target: Path, options: OrchestrateOptions) -> CoordinatorReport:
    """Run full orchestration pipeline"""

    # 1. Setup output directories
    options.output_dir.mkdir(parents=True, exist_ok=True)
    pool_manager = PoolManager(options.output_dir)
    bead_storage = BeadStorage(options.output_dir / "beads")

    # 2. Check for resume
    if options.resume:
        # Load existing pool
        pools = list(options.output_dir.glob("*/pool.yaml"))
        if not pools:
            typer.echo("No existing pools found to resume", err=True)
            raise typer.Exit(1)

        # Load most recent
        pool = pool_manager.load_pool(pools[-1].parent.name)
        beads = [bead_storage.load(bid) for bid in pool.bead_ids]
        typer.echo(f"Resuming pool {pool.id} with {len(beads)} beads")
    else:
        # Create new pool from contracts
        typer.echo(f"Building knowledge graph from {target}...")
        pool, beads = await _create_pool_from_contracts(target, pool_manager, bead_storage)
        typer.echo(f"Created pool {pool.id} with {len(beads)} beads")

    # 3. Get runtime
    typer.echo(f"Using {options.sdk} SDK")
    runtime = get_runtime(options.sdk)

    # 4. Configure coordinator
    config = CoordinatorConfig(
        agents_per_role={
            "attacker": options.agents_attacker,
            "defender": options.agents_defender,
            "verifier": options.agents_verifier,
            "test_builder": 1,
        },
        enable_supervisor=True,
    )

    # 5. Run coordination
    typer.echo("Starting agent orchestration...")
    coordinator = AgentCoordinator(runtime, config)
    coordinator.setup_for_pool(pool, beads)

    if not options.headless:
        typer.echo("Press Ctrl+C to stop early")

    try:
        report = await coordinator.run(timeout=options.timeout)
    except KeyboardInterrupt:
        typer.echo("\nStopping orchestration...")
        coordinator.stop()
        report = CoordinatorReport(
            status=coordinator.status,
            total_beads=len(beads),
            completed_beads=0,
            failed_beads=0,
            results_by_role={},
            duration_seconds=0,
            stuck_work=[],
        )

    # 6. Save results
    _save_results(pool, report, options)

    return report

async def _create_pool_from_contracts(
    target: Path,
    pool_manager: PoolManager,
    bead_storage: BeadStorage,
):
    """Create pool from contracts directory.

    Pipeline:
    1. Build BSKG from contracts
    2. Run pattern detection
    3. Create beads from findings
    4. Create pool with bead IDs
    """
    from true_vkg.kg.builder import VKGBuilder
    from true_vkg.queries.executor import QueryExecutor
    from true_vkg.beads.factory import BeadFactory

    # Build graph
    builder = VKGBuilder()
    graph = builder.build(target)

    # Detect patterns
    executor = QueryExecutor(graph)
    findings = executor.detect_all_patterns()

    # Create beads
    factory = BeadFactory(graph)
    beads = []
    for finding in findings:
        bead = factory.create_from_finding(finding)
        bead_storage.save(bead)
        beads.append(bead)

    # Create pool
    scope = Scope(
        files=[str(f) for f in target.glob("**/*.sol")],
        contracts=list(graph.contracts.keys()),
        focus_areas=[],
    )
    pool = pool_manager.create_pool(
        scope=scope,
        bead_ids=[b.id for b in beads],
    )

    return pool, beads

def _save_results(pool: Pool, report: CoordinatorReport, options: OrchestrateOptions) -> None:
    """Save orchestration results.

    Same artifact format whether called from CLI or SDK.
    Ensures SDK-08 parity.
    """
    report_path = options.output_dir / pool.id / "report.json"
    report_path.parent.mkdir(parents=True, exist_ok=True)

    with open(report_path, "w") as f:
        json.dump(report.to_dict(), f, indent=2)

    typer.echo(f"Report saved to {report_path}")

def _print_report(report: CoordinatorReport, options: OrchestrateOptions) -> None:
    """Print human-readable report"""
    typer.echo("\n" + "=" * 60)
    typer.echo("ORCHESTRATION REPORT")
    typer.echo("=" * 60)

    typer.echo(f"Status: {report.status.value}")
    typer.echo(f"Duration: {report.duration_seconds:.1f}s")
    typer.echo(f"Total beads: {report.total_beads}")
    typer.echo(f"Completed: {report.completed_beads}")
    typer.echo(f"Failed: {report.failed_beads}")

    if report.results_by_role:
        typer.echo("\nResults by role:")
        for role, count in report.results_by_role.items():
            typer.echo(f"  {role}: {count}")

    if report.stuck_work:
        typer.echo(f"\nStuck work: {len(report.stuck_work)} items")
        if options.verbose:
            for item in report.stuck_work:
                typer.echo(f"  - {item}")

    typer.echo("=" * 60)
