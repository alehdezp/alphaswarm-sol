from dataclasses import dataclass
from enum import IntEnum
from heapq import heappush, heappop
from typing import List, Optional
from datetime import datetime

from true_vkg.beads.schema import VulnerabilityBead
from true_vkg.beads.types import Severity

class BeadPriority(IntEnum):
    """Priority levels per PHILOSOPHY.md ordering:
    critical severity > exploitability > tool agreement > recency
    """
    CRITICAL_EXPLOITABLE = 1
    CRITICAL = 2
    HIGH_EXPLOITABLE = 3
    HIGH = 4
    MEDIUM_TOOL_AGREEMENT = 5
    MEDIUM = 6
    LOW = 7

@dataclass
class PrioritizedBead:
    """Wrapper for bead with computed priority"""
    priority: BeadPriority
    timestamp: datetime  # For recency tiebreaker
    bead: VulnerabilityBead

    def __lt__(self, other: "PrioritizedBead") -> bool:
        """Comparison for heapq - lower priority value = higher priority"""
        if self.priority != other.priority:
            return self.priority < other.priority
        # Recency tiebreaker: older beads first (FIFO within priority)
        return self.timestamp < other.timestamp

class PrioritizedBeadQueue:
    """Min-heap priority queue for beads"""

    def __init__(self):
        self._heap: List[PrioritizedBead] = []
        self._bead_ids: set = set()  # Track IDs for O(1) contains check

    def push(self, bead: VulnerabilityBead) -> None:
        """Add bead with computed priority"""
        if bead.id in self._bead_ids:
            return  # Already in queue
        priority = self._compute_priority(bead)
        item = PrioritizedBead(priority, datetime.now(), bead)
        heappush(self._heap, item)
        self._bead_ids.add(bead.id)

    def pop(self) -> Optional[VulnerabilityBead]:
        """Remove and return highest priority bead"""
        while self._heap:
            item = heappop(self._heap)
            if item.bead.id in self._bead_ids:
                self._bead_ids.remove(item.bead.id)
                return item.bead
        return None

    def peek(self) -> Optional[VulnerabilityBead]:
        """Return highest priority bead without removing"""
        while self._heap:
            if self._heap[0].bead.id in self._bead_ids:
                return self._heap[0].bead
            heappop(self._heap)  # Remove stale entry
        return None

    def _compute_priority(self, bead: VulnerabilityBead) -> BeadPriority:
        """Compute priority per PHILOSOPHY.md ordering"""
        is_critical = bead.severity == Severity.CRITICAL
        is_high = bead.severity == Severity.HIGH

        # Check exploitability from pattern context
        is_exploitable = "exploitable" in bead.pattern_context.why_flagged.lower()

        # Check tool agreement from notes (populated by tool integration phase)
        has_tool_agreement = any("tool_agreement" in note for note in bead.notes)

        if is_critical and is_exploitable:
            return BeadPriority.CRITICAL_EXPLOITABLE
        if is_critical:
            return BeadPriority.CRITICAL
        if is_high and is_exploitable:
            return BeadPriority.HIGH_EXPLOITABLE
        if is_high:
            return BeadPriority.HIGH
        if has_tool_agreement:
            return BeadPriority.MEDIUM_TOOL_AGREEMENT
        if bead.severity == Severity.MEDIUM:
            return BeadPriority.MEDIUM
        return BeadPriority.LOW

    def __len__(self) -> int:
        return len(self._bead_ids)

    def __contains__(self, bead_id: str) -> bool:
        return bead_id in self._bead_ids
```

from dataclasses import dataclass, field
from typing import Dict, List, Optional
from datetime import datetime

from true_vkg.agents.runtime import AgentRole
from true_vkg.beads.schema import VulnerabilityBead
from .queue import PrioritizedBeadQueue

@dataclass
class InboxConfig:
    """Configuration for agent inbox"""
    max_queue_size: int = 100
    max_retries: int = 3  # Per 05.2-CONTEXT.md
    requeue_on_failure: bool = True
    escalate_threshold: int = 3

@dataclass
class WorkClaim:
    """Represents claimed work"""
    bead: VulnerabilityBead
    agent_role: AgentRole
    claimed_at: datetime
    attempt: int = 1

class AgentInbox:
    """Per-agent work inbox with prioritized queue"""

    def __init__(self, agent_role: AgentRole, config: Optional[InboxConfig] = None):
        self.agent_role = agent_role
        self.config = config or InboxConfig()
        self._queue = PrioritizedBeadQueue()
        self._in_progress: Dict[str, WorkClaim] = {}  # bead_id -> claim
        self._failure_counts: Dict[str, int] = {}  # bead_id -> failure count

    def assign(self, bead: VulnerabilityBead) -> bool:
        """Assign bead to this inbox"""
        if len(self._queue) >= self.config.max_queue_size:
            return False
        self._queue.push(bead)
        return True

    def claim_work(self) -> Optional[WorkClaim]:
        """Claim next available work item"""
        bead = self._queue.pop()
        if bead is None:
            return None

        claim = WorkClaim(
            bead=bead,
            agent_role=self.agent_role,
            claimed_at=datetime.now(),
            attempt=self._failure_counts.get(bead.id, 0) + 1,
        )
        self._in_progress[bead.id] = claim
        return claim

    def complete_work(self, bead_id: str) -> None:
        """Mark work as completed"""
        if bead_id in self._in_progress:
            del self._in_progress[bead_id]
        if bead_id in self._failure_counts:
            del self._failure_counts[bead_id]

    def fail_work(self, bead_id: str) -> bool:
        """Mark work as failed, returns True if should escalate"""
        claim = self._in_progress.pop(bead_id, None)
        if claim is None:
            return False

        self._failure_counts[bead_id] = self._failure_counts.get(bead_id, 0) + 1

        if self._failure_counts[bead_id] >= self.config.escalate_threshold:
            return True  # Should escalate to supervisor

        if self.config.requeue_on_failure:
            self._queue.push(claim.bead)

        return False

    @property
    def pending_count(self) -> int:
        return len(self._queue)

    @property
    def in_progress_count(self) -> int:
        return len(self._in_progress)

    def to_dict(self) -> Dict:
        """Serialize for persistence"""
        return {
            "agent_role": self.agent_role.value,
            "pending_bead_ids": list(self._queue._bead_ids),
            "in_progress": {bid: claim.bead.id for bid, claim in self._in_progress.items()},
            "failure_counts": self._failure_counts,
        }
```
