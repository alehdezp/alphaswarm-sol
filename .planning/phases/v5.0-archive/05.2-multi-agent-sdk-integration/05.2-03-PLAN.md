from dataclasses import dataclass, field
from datetime import datetime, timedelta
from typing import Dict, List, Optional
import logging

from true_vkg.agents.runtime import AgentRole, AgentConfig, AgentRuntime
from true_vkg.agents.hooks import AgentInbox, WorkClaim
from true_vkg.orchestration.pool import PoolManager
from true_vkg.orchestration.schemas import Pool, PoolStatus
from .prompts import SUPERVISOR_SYSTEM_PROMPT

logger = logging.getLogger(__name__)

@dataclass
class SupervisorConfig:
    """Configuration for supervisor agent"""
    stuck_threshold_minutes: int = 30  # Work in progress > this is stuck
    check_interval_seconds: int = 60
    escalate_after_failures: int = 3
    log_only: bool = True  # Per 05.2-CONTEXT.md: log and continue

@dataclass
class StuckWorkReport:
    """Report of stuck work items"""
    pool_id: str
    bead_id: str
    agent_role: AgentRole
    in_progress_since: datetime
    stuck_minutes: int
    failure_count: int
    recommended_action: str

    def to_dict(self) -> Dict:
        return {
            "pool_id": self.pool_id,
            "bead_id": self.bead_id,
            "agent_role": self.agent_role.value,
            "in_progress_since": self.in_progress_since.isoformat(),
            "stuck_minutes": self.stuck_minutes,
            "failure_count": self.failure_count,
            "recommended_action": self.recommended_action,
        }

@dataclass
class SupervisorReport:
    """Full supervisor status report"""
    timestamp: datetime
    pool_id: str
    pool_status: PoolStatus
    total_beads: int
    completed_beads: int
    stuck_work: List[StuckWorkReport]
    queue_depths: Dict[str, int]  # role -> pending count
    escalations: List[str]  # bead_ids that need human attention

class SupervisorAgent:
    """Monitors agent queues and detects stuck work.

    Per 05.2-CONTEXT.md:
    - Log and continue (don't auto-intervene)
    - Let pool complete with what it can
    - Visible in logs for post-audit review
    """

    def __init__(
        self,
        pool_manager: PoolManager,
        inboxes: Dict[AgentRole, AgentInbox],
        config: Optional[SupervisorConfig] = None,
        runtime: Optional[AgentRuntime] = None,
    ):
        self.pool_manager = pool_manager
        self.inboxes = inboxes
        self.config = config or SupervisorConfig()
        self.runtime = runtime
        self._escalated_beads: set = set()

    def check_pool(self, pool_id: str) -> SupervisorReport:
        """Run supervisor check on a pool"""
        pool = self.pool_manager.get_pool(pool_id)
        if pool is None:
            raise ValueError(f"Pool not found: {pool_id}")

        stuck_work = []
        escalations = []
        queue_depths = {}
        now = datetime.now()

        for role, inbox in self.inboxes.items():
            queue_depths[role.value] = inbox.pending_count

            # Check in-progress work for stuck items
            for bead_id, claim in inbox._in_progress.items():
                minutes_stuck = (now - claim.claimed_at).total_seconds() / 60

                if minutes_stuck >= self.config.stuck_threshold_minutes:
                    report = StuckWorkReport(
                        pool_id=pool_id,
                        bead_id=bead_id,
                        agent_role=role,
                        in_progress_since=claim.claimed_at,
                        stuck_minutes=int(minutes_stuck),
                        failure_count=inbox._failure_counts.get(bead_id, 0),
                        recommended_action=self._recommend_action(claim, inbox),
                    )
                    stuck_work.append(report)
                    logger.warning(f"Stuck work detected: {report.to_dict()}")

            # Check for beads that exceeded failure threshold
            for bead_id, count in inbox._failure_counts.items():
                if count >= self.config.escalate_after_failures:
                    if bead_id not in self._escalated_beads:
                        escalations.append(bead_id)
                        self._escalated_beads.add(bead_id)
                        self._flag_for_human(pool_id, bead_id)

        # Count completed beads
        completed = len([bid for bid in pool.bead_ids if self._is_bead_complete(bid)])

        return SupervisorReport(
            timestamp=now,
            pool_id=pool_id,
            pool_status=pool.status,
            total_beads=len(pool.bead_ids),
            completed_beads=completed,
            stuck_work=stuck_work,
            queue_depths=queue_depths,
            escalations=escalations,
        )

    def _recommend_action(self, claim: WorkClaim, inbox: AgentInbox) -> str:
        """Generate recommended action for stuck work"""
        failures = inbox._failure_counts.get(claim.bead.id, 0)
        if failures >= self.config.escalate_after_failures:
            return "ESCALATE: Multiple failures, needs human review"
        elif claim.attempt > 1:
            return "MONITOR: Retry in progress"
        else:
            return "WAIT: First attempt, may complete"

    def _flag_for_human(self, pool_id: str, bead_id: str) -> None:
        """Flag bead for human review"""
        logger.info(f"Flagging bead {bead_id} for human review in pool {pool_id}")
        # Update bead's human_flag via bead storage
        from true_vkg.beads.storage import BeadStorage
        storage = BeadStorage(self.pool_manager.base_path.parent / "beads")
        bead = storage.load(bead_id)
        if bead:
            bead.human_flag = True
            bead.add_note(f"Supervisor escalation: exceeded failure threshold")
            storage.save(bead)

    def _is_bead_complete(self, bead_id: str) -> bool:
        """Check if bead has verdict"""
        from true_vkg.beads.storage import BeadStorage
        storage = BeadStorage(self.pool_manager.base_path.parent / "beads")
        bead = storage.load(bead_id)
        return bead is not None and bead.is_resolved

    def test_check_pool_no_issues(self, mock_pool_manager):
        """Healthy pool should report no stuck work"""
        inboxes = {r: AgentInbox(r) for r in AgentRole}
        supervisor = SupervisorAgent(mock_pool_manager, inboxes)

        report = supervisor.check_pool("test-pool")

        assert report.pool_id == "test-pool"
        assert len(report.stuck_work) == 0
        assert len(report.escalations) == 0

    def test_detect_stuck_work(self, mock_pool_manager, inboxes_with_stuck_work):
        """Should detect work exceeding stuck threshold"""
        config = SupervisorConfig(stuck_threshold_minutes=1)
        supervisor = SupervisorAgent(mock_pool_manager, inboxes_with_stuck_work, config)

        report = supervisor.check_pool("test-pool")

        assert len(report.stuck_work) > 0
        assert report.stuck_work[0].stuck_minutes >= 1

    def test_escalation_after_failures(self, mock_pool_manager):
        """Should escalate beads that exceed failure threshold"""
        config = SupervisorConfig(escalate_after_failures=2)
        inboxes = {AgentRole.ATTACKER: AgentInbox(AgentRole.ATTACKER)}

        # Simulate multiple failures

        with patch.object(supervisor, '_flag_for_human') as mock_flag:
            report = supervisor.check_pool("test-pool")
            mock_flag.assert_called_once_with("test-pool", "bead-1")

        assert "bead-1" in report.escalations

        """Supervisor should only log, not auto-intervene"""
        config = SupervisorConfig(log_only=True)
        supervisor = SupervisorAgent(mock_pool_manager, inboxes_with_stuck_work, config)

        # Check doesn't modify inbox queues
        initial_counts = {r: i.pending_count for r, i in inboxes_with_stuck_work.items()}
        supervisor.check_pool("test-pool")
        final_counts = {r: i.pending_count for r, i in inboxes_with_stuck_work.items()}

        assert initial_counts == final_counts
