class AgentRole(str, Enum):
    ATTACKER = "attacker"
    DEFENDER = "defender"
    VERIFIER = "verifier"
    TEST_BUILDER = "test_builder"
    SUPERVISOR = "supervisor"
    INTEGRATOR = "integrator"

@dataclass
class AgentConfig:
    role: AgentRole
    system_prompt: str
    tools: List[Dict[str, Any]]  # Per-SDK tool definitions
    max_tokens: int = 8192
    temperature: float = 0.1
    timeout_seconds: int = 300

@dataclass
class AgentResponse:
    content: str
    tool_calls: List[Dict[str, Any]]
    input_tokens: int
    output_tokens: int
    cache_read_tokens: int = 0
    cache_write_tokens: int = 0
    model: str
    latency_ms: int

class AgentRuntime(ABC):
    @abstractmethod
    async def execute(self, config: AgentConfig, messages: List[Dict]) -> AgentResponse: ...
    @abstractmethod
    async def spawn_agent(self, config: AgentConfig, task: str) -> AgentResponse: ...
    @abstractmethod
    def get_model_for_role(self, role: AgentRole) -> str: ...
```

class AnthropicRuntime(AgentRuntime):
    def __init__(self, config: RuntimeConfig, api_key: Optional[str] = None):
        self.config = config
        self.client = anthropic.AsyncAnthropic(api_key=api_key or os.getenv("ANTHROPIC_API_KEY"))
        self._usage_tracker = UsageTracker()

    async def execute(self, config: AgentConfig, messages: List[Dict]) -> AgentResponse:
        """Execute with prompt caching per TOKEN-EFFICIENCY-RESEARCH.md"""
        model = self.get_model_for_role(config.role)

        # Apply cache_control to stable prefix (system prompt)
        # Per research: cache VulnDocs, tools, contract source
        cached_system = {
            "type": "text",
            "text": config.system_prompt,
            "cache_control": {"type": "ephemeral"}
        }

        response = await self.client.messages.create(
            model=model,
            max_tokens=config.max_tokens,
            system=[cached_system],
            messages=messages,
            tools=config.tools,
            temperature=config.temperature,
        )

        return self._parse_response(response, model)

    async def spawn_agent(self, config: AgentConfig, task: str) -> AgentResponse:
        """Spawn context-fresh agent for single task"""
        messages = [{"role": "user", "content": task}]
        return await self.execute(config, messages)

    def get_model_for_role(self, role: AgentRole) -> str:
        return ROLE_MODEL_MAP[role]["anthropic"]
```
