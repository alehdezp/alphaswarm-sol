    """Tests for alphaswarm init command."""

    import pytest
    from pathlib import Path
    from typer.testing import CliRunner

    from true_vkg.cli.main import app
    from true_vkg.skills import get_shipped_skills_path, list_shipped_skills

    runner = CliRunner()

    class TestInitCommand:
        """Tests for alphaswarm init."""

        def test_init_creates_vrs_directory(self, tmp_path: Path):
            """Init creates .claude/vrs/ directory."""
            result = runner.invoke(app, ["init", str(tmp_path)])
            assert result.exit_code == 0
            assert (tmp_path / ".claude" / "vrs").exists()

        def test_init_creates_beads_directory(self, tmp_path: Path):
            """Init creates .beads/ directory."""
            result = runner.invoke(app, ["init", str(tmp_path)])
            assert result.exit_code == 0
            assert (tmp_path / ".beads").exists()
            assert (tmp_path / ".beads" / "index.jsonl").exists()

        def test_init_copies_skills(self, tmp_path: Path):
            """Init copies all shipped skills."""
            result = runner.invoke(app, ["init", str(tmp_path)])
            assert result.exit_code == 0

            vrs_dir = tmp_path / ".claude" / "vrs"
            skills = list(vrs_dir.glob("*.md"))
            assert len(skills) > 0
            assert any("audit" in s.name for s in skills)

        def test_init_copies_agents(self, tmp_path: Path):
            """Init copies agent definitions."""
            result = runner.invoke(app, ["init", str(tmp_path)])
            assert result.exit_code == 0

            agents_dir = tmp_path / ".claude" / "vrs" / "agents"
            if agents_dir.exists():
                agents = list(agents_dir.glob("*.md"))
                assert len(agents) > 0

        def test_init_refuses_without_force(self, tmp_path: Path):
            """Init refuses to overwrite without --force."""
            # First init
            runner.invoke(app, ["init", str(tmp_path)])

            # Second init should fail
            result = runner.invoke(app, ["init", str(tmp_path)])
            assert result.exit_code == 1
            assert "already installed" in result.output.lower()

        def test_init_force_reinstalls(self, tmp_path: Path):
            """Init with --force reinstalls."""
            runner.invoke(app, ["init", str(tmp_path)])
            result = runner.invoke(app, ["init", str(tmp_path), "--force"])
            assert result.exit_code == 0

        def test_init_does_not_modify_existing_claude_files(self, tmp_path: Path):
            """Init does not touch existing .claude/ root files."""
            claude_dir = tmp_path / ".claude"
            claude_dir.mkdir()
            existing_file = claude_dir / "settings.yaml"
            existing_file.write_text("existing: content")

            runner.invoke(app, ["init", str(tmp_path)])

            assert existing_file.read_text() == "existing: content"

        def test_init_creates_cache_gitignore(self, tmp_path: Path):
            """Init creates .gitignore in cache directory."""
            runner.invoke(app, ["init", str(tmp_path)])
            cache_dir = tmp_path / ".claude" / "vrs" / "cache"
            assert cache_dir.exists()
            assert (cache_dir / ".gitignore").exists()

    """Tests for alphaswarm bead commands."""

    import json
    import pytest
    from pathlib import Path
    from typer.testing import CliRunner

    from true_vkg.cli.main import app
    from true_vkg.beads.jsonl_storage import BeadJSONLStorage, BeadEntry

    runner = CliRunner()

    class TestBeadStorage:
        """Tests for JSONL bead storage."""

        def test_create_bead(self, tmp_path: Path):
            """Create bead returns ID."""
            storage = BeadJSONLStorage(tmp_path / ".beads")
            bead_id = storage.create(title="Test bead", severity="high")
            assert bead_id.startswith("bd-")
            assert len(bead_id) == 11  # bd- + 8 chars

        def test_bead_persists(self, tmp_path: Path):
            """Bead persists to JSONL file."""
            storage = BeadJSONLStorage(tmp_path / ".beads")
            bead_id = storage.create(title="Test bead")

            # Reload storage
            storage2 = BeadJSONLStorage(tmp_path / ".beads")
            bead = storage2.get(bead_id)
            assert bead is not None
            assert bead.title == "Test bead"

        def test_update_bead(self, tmp_path: Path):
            """Update bead changes status."""
            storage = BeadJSONLStorage(tmp_path / ".beads")
            bead_id = storage.create(title="Test bead")

            storage.update(bead_id, status="in_progress")
            bead = storage.get(bead_id)
            assert bead.status == "in_progress"

        def test_list_beads_with_filter(self, tmp_path: Path):
            """List beads with status filter."""
            storage = BeadJSONLStorage(tmp_path / ".beads")
            storage.create(title="Open bead")
            id2 = storage.create(title="Complete bead")
            storage.update(id2, status="complete")

            open_beads = storage.list(status="open")
            assert len(open_beads) == 1
            assert open_beads[0].title == "Open bead"

        def test_get_ready_excludes_blocked(self, tmp_path: Path):
            """get_ready excludes blocked beads."""
            storage = BeadJSONLStorage(tmp_path / ".beads")
            id1 = storage.create(title="Blocker")
            id2 = storage.create(title="Blocked")
            storage.update(id2, blocked_by=id1)

            ready = storage.get_ready()
            assert len(ready) == 1
            assert ready[0].id == id1

    class TestBeadCLI:
        """Tests for bead CLI commands."""

        def test_bead_create(self, tmp_path: Path, monkeypatch):
            """bead create command creates bead."""
            monkeypatch.chdir(tmp_path)
            result = runner.invoke(app, ["bead", "create", "Test bead", "--json"])
            assert result.exit_code == 0
            data = json.loads(result.output)
            assert "id" in data

        def test_bead_list_empty(self, tmp_path: Path, monkeypatch):

    """Tests for skills package structure."""

    import pytest
    from pathlib import Path

    from true_vkg.skills import get_shipped_skills_path, list_shipped_skills

    class TestSkillsPackage:
        """Tests for skills package."""

        def test_get_shipped_skills_path_exists(self):
            """Shipped skills path exists."""
            path = get_shipped_skills_path()
            assert path.exists()
            assert path.is_dir()

        def test_list_shipped_skills_returns_list(self):
            """list_shipped_skills returns skill names."""
            skills = list_shipped_skills()
            assert isinstance(skills, list)
            assert len(skills) > 0

        def test_shipped_skills_have_md_extension(self):
            """All shipped skills are .md files."""
            path = get_shipped_skills_path()
            for skill_file in path.glob("*.md"):
                assert skill_file.suffix == ".md"

        def test_shipped_skills_have_frontmatter(self):
            """All shipped skills have YAML frontmatter."""
            path = get_shipped_skills_path()
            for skill_file in path.glob("*.md"):
                if skill_file.name == "README.md":
                    continue
                content = skill_file.read_text()
                assert content.startswith("---"), f"{skill_file.name} missing frontmatter"
                # Check frontmatter closes
                parts = content.split("---", 2)
                assert len(parts) >= 3, f"{skill_file.name} frontmatter not closed"

        def test_core_skills_exist(self):
            """Core skills (audit, health-check) exist."""
            path = get_shipped_skills_path()
            skill_names = [f.name for f in path.glob("*.md")]
            assert "audit.md" in skill_names
            assert "health-check.md" in skill_names

        def test_skills_use_vrs_namespace(self):
            """Skills use vrs: namespace."""
            path = get_shipped_skills_path()
            for skill_file in path.glob("*.md"):
                if skill_file.name == "README.md":
                    continue
                content = skill_file.read_text()
                # Check slash_command uses vrs:
                if "slash_command:" in content:
                    assert "vrs:" in content, f"{skill_file.name} should use vrs: namespace"

        def test_agents_directory_exists(self):
            """Agents subdirectory exists if agents are shipped."""
            path = get_shipped_skills_path()
            agents_dir = path / "agents"
            # May or may not exist depending on implementation
            if agents_dir.exists():
                agents = list(agents_dir.glob("*.md"))
                assert len(agents) > 0
