def save(
    self,
    graph: KnowledgeGraph,
    *,
    format: Literal["toon", "json"] = "toon",
    overwrite: bool = False
) -> Path:
    """Persist knowledge graph to disk.

    Args:
        graph: Knowledge graph to save
        format: Output format - "toon" (default, LLM-optimized) or "json" (legacy)
        overwrite: If True, overwrite existing file

    Returns:
        Path to saved file
    """
    self._ensure_root()
    payload = {
        "format": FORMAT_VERSION,
        "saved_at": datetime.now(timezone.utc).isoformat(),
        "graph": graph.to_dict(),
    }

    ext = ".toon" if format == "toon" else ".json"
    target = self.root / f"graph{ext}"

    if target.exists() and not overwrite:
        timestamp = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
        target = self.root / f"graph-{timestamp}{ext}"

    if format == "toon":
        target.write_text(toon_dumps(payload), encoding="utf-8")
    else:
        target.write_text(json.dumps(payload, indent=2), encoding="utf-8")

    return target

def load(self, path: Path | None = None) -> KnowledgeGraph:
    """Load knowledge graph from disk.

    Args:
        path: Explicit path to load from. If None, tries graph.toon then graph.json

    Returns:
        Loaded KnowledgeGraph
    """
    if path is None:
        # Try TOON first (new default), fall back to JSON (legacy)
        toon_path = self.root / "graph.toon"
        json_path = self.root / "graph.json"
        path = toon_path if toon_path.exists() else json_path

    content = path.read_text(encoding="utf-8")

    if path.suffix == ".toon":
        data = toon_loads(content)
    else:
        data = json.loads(content)

    graph_data = data.get("graph") or {}
    return KnowledgeGraph.from_dict(graph_data)

"""Tests for GraphStore with TOON format support."""

import json
import pytest
from pathlib import Path

from alphaswarm_sol.kg.store import GraphStore, FORMAT_VERSION
from alphaswarm_sol.kg.schema import KnowledgeGraph, Node, Edge


@pytest.fixture
def empty_graph():
    """Create minimal empty graph."""
    return KnowledgeGraph(nodes={}, edges=[], metadata={"test": True})


@pytest.fixture
def simple_graph():
    """Create simple graph with nodes and edges."""
    nodes = {
        "func:transfer": Node(
            id="func:transfer",
            type="function",
            label="transfer",
            properties={"visibility": "public", "writes_state": True},
            evidence=[],
        ),
        "func:approve": Node(
            id="func:approve",
            type="function",
            label="approve",
            properties={"visibility": "public", "writes_state": True},
            evidence=[],
        ),
    }
    edges = [
        Edge(
            id="edge:1",
            type="calls",
            source="func:transfer",
            target="func:approve",
            properties={},
            evidence=[],
        )
    ]
    return KnowledgeGraph(nodes=nodes, edges=edges, metadata={"test": True})


class TestGraphStoreToonFormat:
    """Test TOON format output."""

    def test_save_defaults_to_toon(self, tmp_path, empty_graph):
        """Default format is TOON."""
        store = GraphStore(tmp_path)
        saved = store.save(empty_graph)

        assert saved.suffix == ".toon"
        assert saved.name == "graph.toon"

    def test_save_toon_explicit(self, tmp_path, empty_graph):
        """Explicit TOON format."""
        store = GraphStore(tmp_path)
        saved = store.save(empty_graph, format="toon")

        assert saved.suffix == ".toon"

    def test_save_json_format(self, tmp_path, empty_graph):
        """JSON format for legacy compatibility."""
        store = GraphStore(tmp_path)
        saved = store.save(empty_graph, format="json")

        assert saved.suffix == ".json"
        assert saved.name == "graph.json"

    def test_toon_roundtrip(self, tmp_path, simple_graph):
        """Graph survives TOON round-trip."""
        store = GraphStore(tmp_path)
        saved = store.save(simple_graph, format="toon")

        loaded = store.load(saved)

        assert len(loaded.nodes) == len(simple_graph.nodes)
        assert len(loaded.edges) == len(simple_graph.edges)
        assert loaded.metadata.get("test") is True

    def test_json_roundtrip(self, tmp_path, simple_graph):
        """Graph survives JSON round-trip."""
        store = GraphStore(tmp_path)
        saved = store.save(simple_graph, format="json")

        loaded = store.load(saved)

        assert len(loaded.nodes) == len(simple_graph.nodes)
        assert len(loaded.edges) == len(simple_graph.edges)

class TestGraphStoreOverwrite:
    """Test overwrite behavior with different formats."""

    def test_no_overwrite_creates_timestamped_toon(self, tmp_path, empty_graph):
        """Without overwrite, creates timestamped file."""

        saved = store.save(empty_graph, format="json")

        content = saved.read_text()
        data = json.loads(content)  # Should not raise
