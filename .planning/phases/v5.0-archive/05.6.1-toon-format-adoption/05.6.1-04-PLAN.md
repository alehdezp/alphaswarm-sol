```python
"""Token savings validation for TOON format.

This test measures the actual token reduction achieved by using TOON
instead of JSON for knowledge graph serialization.

Expected savings: 35-45% based on research (05.6.1-RESEARCH.md)
- Node arrays (variable schema): 20-30%
- Edge arrays (uniform): 40-50%
- Evidence arrays (uniform): 50-60%
"""

import json
import pytest

from alphaswarm_sol.kg.toon import toon_dumps


class TestTokenSavings:
    """Measure token reduction for graph-like structures."""

    @staticmethod
    def count_tokens(text: str) -> int:
        """Count tokens using tiktoken (Claude/GPT tokenizer).

        Falls back to character-based estimate if tiktoken unavailable.
        """
        try:
            import tiktoken
            enc = tiktoken.get_encoding("cl100k_base")
            return len(enc.encode(text))
        except ImportError:
            # Fallback: ~4 chars per token (rough estimate)
            return len(text) // 4

    def test_uniform_array_savings(self):
        """Uniform arrays (like edges/evidence) should see 40-60% savings."""
        # Simulate evidence array - highly uniform structure
        evidence = [
            {"file": "Token.sol", "line_start": i, "line_end": i + 5, "detail": None}
            for i in range(0, 100, 10)
        ]

        json_str = json.dumps(evidence, separators=(",", ":"))
        toon_str = toon_dumps(evidence)

        json_tokens = self.count_tokens(json_str)
        toon_tokens = self.count_tokens(toon_str)

        savings = (json_tokens - toon_tokens) / json_tokens * 100

        print(f"\nUniform array (evidence-like):")
        print(f"  JSON: {json_tokens} tokens ({len(json_str)} chars)")
        print(f"  TOON: {toon_tokens} tokens ({len(toon_str)} chars)")
        print(f"  Savings: {savings:.1f}%")

        # Uniform arrays should see significant savings
        assert savings > 30, f"Expected >30% savings for uniform arrays, got {savings:.1f}%"

    def test_mixed_schema_array_savings(self):
        """Mixed-schema arrays (like nodes) should see 20-30% savings."""
        # Simulate nodes array - varying properties per node type
        nodes = [
            {"id": f"func:{i}", "type": "function", "label": f"fn{i}",
             "properties": {"visibility": "public", "writes_state": True, "has_modifier": False}}
            for i in range(5)
        ] + [
            {"id": f"contract:{i}", "type": "contract", "label": f"C{i}",
             "properties": {"is_upgradeable": False, "has_proxy": True}}
            for i in range(5)
        ]

        json_str = json.dumps(nodes, separators=(",", ":"))
        toon_str = toon_dumps(nodes)

        json_tokens = self.count_tokens(json_str)
        toon_tokens = self.count_tokens(toon_str)

        savings = (json_tokens - toon_tokens) / json_tokens * 100

        print(f"\nMixed-schema array (node-like):")
        print(f"  JSON: {json_tokens} tokens ({len(json_str)} chars)")
        print(f"  TOON: {toon_tokens} tokens ({len(toon_str)} chars)")
        print(f"  Savings: {savings:.1f}%")

        # Mixed arrays still see some savings
        assert savings > 10, f"Expected >10% savings for mixed arrays, got {savings:.1f}%"

    def test_full_graph_structure_savings(self):
        """Full graph structure should see 35-45% overall savings."""
        # Simulate realistic graph payload
        graph_data = {
            "format": "alphaswarm-kg-v1",
            "saved_at": "2025-01-23T12:00:00Z",
            "graph": {
                "nodes": {
                    f"func:{i}": {
                        "id": f"func:{i}",
                        "type": "function",
                        "label": f"function_{i}",
                        "properties": {
                            "visibility": "public" if i % 2 == 0 else "internal",
                            "writes_state": i % 3 == 0,
                            "has_access_control": i % 4 == 0,
                            "is_payable": i % 5 == 0,
                            "modifiers": [],
                        },
                        "evidence": [
                            {"file": "Contract.sol", "line_start": i * 10, "line_end": i * 10 + 5, "detail": None}
                        ]
                    }
                    for i in range(20)
                },
                "edges": [
                    {
                        "id": f"edge:{i}",
                        "type": "calls",
                        "source": f"func:{i}",
                        "target": f"func:{(i+1) % 20}",
                        "properties": {"confidence": "high"},
                        "evidence": []
                    }
                    for i in range(15)
                ],
                "metadata": {
                    "solc_version": "0.8.20",
                    "build_time_ms": 1234,
                }
            }
        }

        json_str = json.dumps(graph_data, separators=(",", ":"))
        toon_str = toon_dumps(graph_data)

        json_tokens = self.count_tokens(json_str)
        toon_tokens = self.count_tokens(toon_str)

        savings = (json_tokens - toon_tokens) / json_tokens * 100

        print(f"\nFull graph structure:")
        print(f"  JSON: {json_tokens} tokens ({len(json_str)} chars)")
        print(f"  TOON: {toon_tokens} tokens ({len(toon_str)} chars)")
        print(f"  Savings: {savings:.1f}%")
        print(f"  Expected: 35-45%")

        # Full graph should see meaningful savings
        # Note: Actual savings depend on data structure; 20% is minimum acceptable
        assert savings > 20, f"Expected >20% savings for full graph, got {savings:.1f}%"

    def test_real_graph_savings(self):
        """Test with actual built graph if available."""
        from tests.graph_cache import load_graph

        try:
            graph = load_graph("NoAccessGate.sol")
        except Exception:
            pytest.skip("Test contract not available")

        graph_data = {
            "format": "alphaswarm-kg-v1",
            "saved_at": "2025-01-23T12:00:00Z",
            "graph": graph.to_dict(),
        }

        json_str = json.dumps(graph_data, separators=(",", ":"))
        toon_str = toon_dumps(graph_data)

        json_tokens = self.count_tokens(json_str)
        toon_tokens = self.count_tokens(toon_str)

        savings = (json_tokens - toon_tokens) / json_tokens * 100

        print(f"\nReal graph (NoAccessGate.sol):")
        print(f"  Nodes: {len(graph.nodes)}, Edges: {len(graph.edges)}")
        print(f"  JSON: {json_tokens} tokens ({len(json_str)} chars)")
        print(f"  TOON: {toon_tokens} tokens ({len(toon_str)} chars)")
        print(f"  Savings: {savings:.1f}%")

        # Real graphs should show meaningful improvement
        assert savings > 15, f"Expected >15% savings for real graph, got {savings:.1f}%"
```
