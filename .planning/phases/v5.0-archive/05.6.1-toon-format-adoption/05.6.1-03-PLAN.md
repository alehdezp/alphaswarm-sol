```python
@lru_cache(maxsize=None)
def load_graph(contract_path: str):
    """Load VKG from contract file with LRU caching."""
    if contract_path.startswith("projects/"):
        full_path = ROOT / "tests" / contract_path
    else:
        full_path = CONTRACTS / contract_path
    return VKGBuilder(ROOT).build(full_path)
```

This function:
1. Takes a contract path (e.g., "NoAccessGate.sol")
2. Calls `VKGBuilder.build()` to BUILD the graph fresh each time (first call)
3. Uses LRU cache to return cached graph on subsequent calls

**Key insight:** The graph_cache doesn't read/write graph files - it builds graphs directly from Solidity source. The TOON format change only affects `GraphStore.save()`/`GraphStore.load()`, which is used by:
- CLI `build-kg` command (saves to disk)
- Tests that explicitly load from saved files

**NO CHANGES NEEDED to graph_cache.py** because:
- It builds graphs in-memory from source
- It doesn't save/load graph files

```python
"""CLI integration tests for build-kg command with TOON format."""

import pytest
from pathlib import Path
from typer.testing import CliRunner

from alphaswarm_sol.cli import app


runner = CliRunner()


class TestBuildKgFormat:
    """Test build-kg --format flag."""

    def test_build_kg_default_toon(self, tmp_path):
        """Default format produces .toon file."""
        contract = Path(__file__).parent / "contracts" / "NoAccessGate.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        result = runner.invoke(app, [
            "build-kg",
            str(contract),
            "--out", str(tmp_path),
        ])

        assert result.exit_code == 0
        toon_file = tmp_path / "graph.toon"
        assert toon_file.exists(), f"Expected {toon_file} to exist"
        assert "format: toon" in result.stdout or ".toon" in result.stdout

    def test_build_kg_explicit_toon(self, tmp_path):
        """Explicit --format toon produces .toon file."""
        contract = Path(__file__).parent / "contracts" / "NoAccessGate.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        result = runner.invoke(app, [
            "build-kg",
            str(contract),
            "--out", str(tmp_path),
            "--format", "toon",
        ])

        assert result.exit_code == 0
        assert (tmp_path / "graph.toon").exists()

    def test_build_kg_json_format(self, tmp_path):
        """--format json produces .json file."""
        contract = Path(__file__).parent / "contracts" / "NoAccessGate.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        result = runner.invoke(app, [
            "build-kg",
            str(contract),
            "--out", str(tmp_path),
            "--format", "json",
        ])

        assert result.exit_code == 0
        json_file = tmp_path / "graph.json"
        assert json_file.exists(), f"Expected {json_file} to exist"

    def test_build_kg_invalid_format(self, tmp_path):
        """Invalid format value exits with error."""
        contract = Path(__file__).parent / "contracts" / "NoAccessGate.sol"
        if not contract.exists():
            pytest.skip("Test contract not found")

        result = runner.invoke(app, [
            "build-kg",
            str(contract),
            "--out", str(tmp_path),
            "--format", "xml",  # Invalid
        ])

        assert result.exit_code != 0
        assert "toon" in result.stdout.lower() or "json" in result.stdout.lower()
```
