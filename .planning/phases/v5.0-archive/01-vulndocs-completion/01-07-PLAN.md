       ```yaml
       # .github/workflows/exa-scan.yaml
       name: Weekly Exa Vulnerability Scan

       on:
         schedule:
           - cron: '0 9 * * 1'  # Every Monday at 9 AM UTC
         workflow_dispatch:  # Manual trigger

       jobs:
         scan:
           runs-on: ubuntu-latest
           steps:
             - uses: actions/checkout@v4

             - name: Set up Python
               uses: actions/setup-python@v5
               with:
                 python-version: '3.11'

             - name: Install UV
               run: pip install uv

             - name: Install dependencies
               run: uv sync

             - name: Run Exa scan
               env:
                 EXA_API_KEY: ${{ secrets.EXA_API_KEY }}
               run: uv run alphaswarm vulndocs scan-exa --days 7

             - name: Check for new discoveries
               id: check
               run: |
                 if [ -f ".true_vkg/discovery/exa_queue.yaml" ]; then
                   count=$(grep -c "pending_review:" .true_vkg/discovery/exa_queue.yaml || echo 0)
                   echo "discoveries=$count" >> $GITHUB_OUTPUT
                 fi

             - name: Create issue if discoveries found
               if: steps.check.outputs.discoveries > 0
               uses: actions/github-script@v7
               with:
                 script: |
                   github.rest.issues.create({
                     owner: context.repo.owner,
                     repo: context.repo.repo,
                     title: 'Weekly Exa Scan: New vulnerability discoveries',
                     body: 'New vulnerability discoveries found. Review queue at `.true_vkg/discovery/exa_queue.yaml`',
                     labels: ['vulndocs', 'automated']
                   })

             - name: Commit queue updates
               run: |
                 git config user.name "github-actions[bot]"
                 git config user.email "github-actions[bot]@users.noreply.github.com"
                 git add .true_vkg/discovery/
                 git diff --staged --quiet || git commit -m "chore: weekly Exa scan results"
                 git push
       ```
