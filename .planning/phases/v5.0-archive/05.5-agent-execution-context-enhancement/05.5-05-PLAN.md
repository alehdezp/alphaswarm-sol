"""ContextMergeAgent - prepares context bundle for a vulnerability class.

Per 05.5-CONTEXT.md:
- One vuln class per agent spawn (fresh context, focused reasoning)
- Uses Sonnet 4.5 (good reasoning, cost-effective)
- Quality loop: merge -> verify -> retry if fail (max 3)
- Invokes /vrs-create-bead-context-merge after verification passes
"""

from __future__ import annotations

import logging
from dataclasses import dataclass
from typing import List, Optional

from true_vkg.agents.context import (
    ContextMerger,
    ContextVerifier,
    ContextBeadFactory,
    MergeResult,
    VerificationResult,
)
from true_vkg.agents.runtime.base import AgentRole
from true_vkg.beads.context_merge import ContextMergeBead
from true_vkg.context.schema import ProtocolContextPack

logger = logging.getLogger(__name__)


@dataclass
class ContextMergeConfig:
    """Configuration for context-merge agent."""
    max_retries: int = 3
    abort_on_schema_error: bool = True
    retry_on_quality_error: bool = True


@dataclass
class ContextMergeResult:
    """Result of context-merge agent execution."""
    success: bool
    bead: Optional[ContextMergeBead]
    vuln_class: str
    attempts: int
    errors: List[str]
    final_quality_score: float


class ContextMergeAgent:
    """Agent that merges vulndoc + protocol context for one vulnerability class.

    Workflow:
    1. Receive vuln_class, protocol_pack, target_scope
    2. Run ContextMerger.merge()
    3. Run ContextVerifier.verify()
    4. If fail: retry with feedback (max 3)
    5. If pass: invoke /vrs-create-bead-context-merge
    6. Return bead or errors
    """

    # Model assignment per 05.5-CONTEXT.md
    MODEL = "claude-sonnet-4-5"
    ROLE = AgentRole.COORDINATOR  # Uses sonnet via role mapping

    def __init__(
        self,
        merger: ContextMerger,
        verifier: ContextVerifier,
        bead_factory: ContextBeadFactory,
        config: Optional[ContextMergeConfig] = None,
    ):
        self.merger = merger
        self.verifier = verifier
        self.bead_factory = bead_factory
        self.config = config or ContextMergeConfig()

    def execute(
        self,
        vuln_class: str,
        protocol_pack: ProtocolContextPack,
        target_scope: List[str],
        pool_id: Optional[str] = None,
    ) -> ContextMergeResult:
        """Execute context-merge for a vulnerability class.

        Args:
            vuln_class: Category/subcategory (e.g., "reentrancy/classic")
            protocol_pack: Protocol context pack
            target_scope: Contracts in scope
            pool_id: Optional pool to associate beads with

        Returns:
            ContextMergeResult with bead or errors
        """
        errors: List[str] = []
        attempts = 0
        additional_context = None  # Accumulates retry feedback

        while attempts < self.config.max_retries:
            attempts += 1
            logger.info(f"Context-merge attempt {attempts}/{self.config.max_retries} for {vuln_class}")

            # Step 1: Merge
            merge_result = self.merger.merge(
                vuln_class=vuln_class,
                protocol_pack=protocol_pack,
                target_scope=target_scope,
                additional_context=additional_context,
            )

            if not merge_result.success:
                errors.extend(merge_result.errors)
                logger.error(f"Merge failed: {merge_result.errors}")
                # Schema errors abort immediately
                if self.config.abort_on_schema_error:
                    break
                continue

            # Step 2: Verify
            verification_result = self.verifier.verify(merge_result.bundle)

            if verification_result.valid:
                # Success! Create bead
                logger.info(f"Verification passed with score {verification_result.quality_score}")
                try:
                    bead = self.bead_factory.create_from_verified_merge(
                        merge_result=merge_result,
                        verification_result=verification_result,
                        pool_id=pool_id,
                        created_by="context-merge-agent",
                    )
                    self.bead_factory.save_bead(bead)

                    return ContextMergeResult(
                        success=True,
                        bead=bead,
                        vuln_class=vuln_class,
                        attempts=attempts,
                        errors=[],
                        final_quality_score=verification_result.quality_score,
                    )
                except Exception as e:
                    errors.append(f"Bead creation failed: {str(e)}")
                    logger.error(f"Bead creation failed: {e}")
                    break

            # Verification failed - prepare retry
            if not self.config.retry_on_quality_error:
                errors.extend([e.message for e in verification_result.errors])
                break

            # Use feedback for next attempt
            if verification_result.feedback_for_retry:
                additional_context = {
                    "reasoning_notes": verification_result.feedback_for_retry,
                }
            errors.extend([e.message for e in verification_result.errors])
            logger.warning(f"Verification failed, retrying: {verification_result.errors}")

        # All retries exhausted
        return ContextMergeResult(
            success=False,
            bead=None,
            vuln_class=vuln_class,
            attempts=attempts,
            errors=errors,
            final_quality_score=0.0,
        )

```python
"""SubCoordinator - orchestrates parallel context-merge agents.

Per 05.5-CONTEXT.md:
- Uses Opus 4.5 (complex coordination decisions)
- Has minimal base protocol context
- Spawns context-merge agents in PARALLEL for each vuln class
- Ends job after context-merge beads created (does NOT spawn vuln-discovery)
"""

from __future__ import annotations

import asyncio
import logging
from dataclasses import dataclass, field
from typing import Dict, List, Optional

from true_vkg.agents.orchestration.context_merge_agent import (
    ContextMergeAgent,
    ContextMergeConfig,
    ContextMergeResult,
)
from true_vkg.agents.context import (
    VulndocContextExtractor,
    ContextMerger,
    ContextVerifier,
    ContextBeadFactory,
)
from true_vkg.agents.runtime.base import AgentRole
from true_vkg.beads.context_merge import ContextMergeBead
from true_vkg.context.schema import ProtocolContextPack

logger = logging.getLogger(__name__)


@dataclass
class SubCoordinatorConfig:
    """Configuration for sub-coordinator."""
    max_parallel: int = 10  # Max concurrent context-merge agents
    timeout_per_class_seconds: int = 300
    context_merge_config: ContextMergeConfig = field(default_factory=ContextMergeConfig)


@dataclass
class SubCoordinatorResult:
    """Result of sub-coordinator execution."""
    success: bool  # All vuln classes processed (may have individual failures)
    beads_created: List[ContextMergeBead]
    failed_classes: List[str]
    errors: Dict[str, List[str]]  # vuln_class -> errors
    total_classes: int
    successful_classes: int


class SubCoordinator:
    """Orchestrates parallel context-merge for multiple vulnerability classes.

    Per 05.5-CONTEXT.md:
    - Spawns context-merge agents in PARALLEL
    - Each agent handles one vulnerability class
    - Sub-coordinator ends after beads created
    - Does NOT spawn vuln-discovery agents (main orchestrator does that)

    Usage:
        coordinator = SubCoordinator(
            protocol_pack=protocol_pack,
            target_scope=["contracts/"],
            vuln_classes=["reentrancy/classic", "access-control/missing-auth"],
            pool_id="audit-2026-01",
        )
        result = await coordinator.run()
        # result.beads_created contains context-merge beads
    """

    # Model assignment per 05.5-CONTEXT.md
    MODEL = "claude-opus-4-5"
    ROLE = AgentRole.SUPERVISOR  # Uses opus via role mapping

    def __init__(
        self,
        protocol_pack: ProtocolContextPack,
        target_scope: List[str],
        vuln_classes: List[str],
        pool_id: Optional[str] = None,
        config: Optional[SubCoordinatorConfig] = None,
        vulndocs_root: Optional[str] = None,
    ):
        self.protocol_pack = protocol_pack
        self.target_scope = target_scope
        self.vuln_classes = vuln_classes
        self.pool_id = pool_id
        self.config = config or SubCoordinatorConfig()

        # Initialize shared components
        from pathlib import Path
        vulndocs_path = Path(vulndocs_root) if vulndocs_root else Path("vulndocs")
        self.extractor = VulndocContextExtractor(vulndocs_root=vulndocs_path)
        self.merger = ContextMerger(extractor=self.extractor)
        self.verifier = ContextVerifier()
        self.bead_factory = ContextBeadFactory()

    async def run(self) -> SubCoordinatorResult:
        """Run parallel context-merge for all vulnerability classes.

        Returns:
            SubCoordinatorResult with beads and any errors
        """
        logger.info(f"Starting sub-coordinator for {len(self.vuln_classes)} vuln classes")

        # Create semaphore for concurrency control
        semaphore = asyncio.Semaphore(self.config.max_parallel)

        # Run all context-merge tasks in parallel
        tasks = [
            self._run_context_merge(vuln_class, semaphore)
            for vuln_class in self.vuln_classes
        ]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # Collect results
        beads_created: List[ContextMergeBead] = []
        failed_classes: List[str] = []
        errors: Dict[str, List[str]] = {}

        for vuln_class, result in zip(self.vuln_classes, results):
            if isinstance(result, Exception):
                failed_classes.append(vuln_class)
                errors[vuln_class] = [str(result)]
            elif isinstance(result, ContextMergeResult):
                if result.success and result.bead:
                    beads_created.append(result.bead)
                else:
                    failed_classes.append(vuln_class)
                    errors[vuln_class] = result.errors

        return SubCoordinatorResult(
            success=len(failed_classes) == 0,
            beads_created=beads_created,
            failed_classes=failed_classes,
            errors=errors,
            total_classes=len(self.vuln_classes),
            successful_classes=len(beads_created),
        )

    async def _run_context_merge(
        self,
        vuln_class: str,
        semaphore: asyncio.Semaphore,
    ) -> ContextMergeResult:
        """Run context-merge for a single vulnerability class."""
        async with semaphore:
            logger.info(f"Starting context-merge for {vuln_class}")

            agent = ContextMergeAgent(
                merger=self.merger,
                verifier=self.verifier,
                bead_factory=self.bead_factory,
                config=self.config.context_merge_config,
            )

            # Run in thread pool to avoid blocking
            loop = asyncio.get_event_loop()
            result = await asyncio.wait_for(
                loop.run_in_executor(
                    None,
                    agent.execute,
                    vuln_class,
                    self.protocol_pack,
                    self.target_scope,
                    self.pool_id,
                ),
                timeout=self.config.timeout_per_class_seconds,
            )

            if result.success:
                logger.info(f"Context-merge completed for {vuln_class}: {result.bead.id}")
            else:
                logger.error(f"Context-merge failed for {vuln_class}: {result.errors}")

            return result

    def get_pending_beads(self) -> List[ContextMergeBead]:
        """Get all pending context-merge beads for this pool."""
        return self.bead_factory.list_pending_beads(pool_id=self.pool_id)
```
