```python
"""FindingBeadFactory - creates VulnerabilityBeads from discovery output.

Per 05.5-CONTEXT.md:
- Full evidence chain inline in bead
- Links to context bead that spawned this discovery
- Created via /vrs-create-bead-finding skill
"""

from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime
from pathlib import Path
from typing import Any, Dict, List, Optional

from true_vkg.beads.schema import (
    VulnerabilityBead,
    CodeSnippet,
    PatternContext,
    InvestigationGuide,
    InvestigationStep,
    TestContext,
    ExploitReference,
)
from true_vkg.beads.types import Severity, BeadStatus
from true_vkg.beads.context_merge import ContextMergeBead


@dataclass
class EvidenceChain:
    """Complete evidence chain for a finding."""
    code_locations: List[CodeSnippet]
    vulndoc_reference: str  # Path to vulndoc (e.g., "reentrancy/classic")
    reasoning_steps: List[str]
    vql_queries: List[str]
    protocol_context_applied: List[str]
    confidence: str  # confirmed, likely, uncertain, rejected
    confidence_reason: str


@dataclass
class FindingInput:
    """Input for creating a finding bead."""
    vulnerability_class: str
    severity: Severity
    summary: str
    evidence_chain: EvidenceChain
    context_bead_id: str
    pool_id: Optional[str] = None


class FindingBeadFactory:
    """Factory for creating finding beads from vuln-discovery output.

    Usage:
        factory = FindingBeadFactory(beads_dir=Path(".vrs/beads"))
        bead = factory.create_finding(
            finding=FindingInput(...),
            context_bead=context_bead,
        )
    """

    def __init__(self, beads_dir: Path = Path(".vrs/beads")):
        self.beads_dir = beads_dir
        self.beads_dir.mkdir(parents=True, exist_ok=True)

    def create_finding(
        self,
        finding: FindingInput,
        context_bead: ContextMergeBead,
    ) -> VulnerabilityBead:
        """Create VulnerabilityBead from finding input.

        Args:
            finding: FindingInput with evidence chain
            context_bead: ContextMergeBead that spawned this discovery

        Returns:
            VulnerabilityBead with full evidence
        """
        # Map confidence to float
        confidence_map = {
            "confirmed": 0.99,
            "likely": 0.85,
            "uncertain": 0.50,
            "rejected": 0.10,
        }
        confidence = confidence_map.get(finding.evidence_chain.confidence, 0.50)

        # Get primary code location
        primary_code = finding.evidence_chain.code_locations[0] if finding.evidence_chain.code_locations else CodeSnippet(
            source="",
            file_path="unknown",
            start_line=0,
            end_line=0,
        )

        # Build pattern context
        pattern_context = PatternContext(
            pattern_name=finding.vulnerability_class,
            pattern_description=f"Detection via {finding.evidence_chain.vulndoc_reference}",
            why_flagged=finding.evidence_chain.confidence_reason,
            matched_properties=[],  # Populated from VQL results
            evidence_lines=[loc.start_line for loc in finding.evidence_chain.code_locations],
        )

        # Build investigation guide from reasoning steps
        steps = [
            InvestigationStep(
                step_number=i + 1,
                action=step,
                look_for="",
                evidence_needed="",
            )
            for i, step in enumerate(finding.evidence_chain.reasoning_steps)
        ]
        investigation_guide = InvestigationGuide(
            steps=steps,
            questions_to_answer=[],
            common_false_positives=[],
            key_indicators=[],
            safe_patterns=[],
        )

        # Create bead
        bead = VulnerabilityBead(
            id=self._generate_id(finding, context_bead),
            vulnerability_class=finding.vulnerability_class,
            pattern_id=finding.evidence_chain.vulndoc_reference,
            severity=finding.severity,
            confidence=confidence,
            vulnerable_code=primary_code,
            related_code=finding.evidence_chain.code_locations[1:] if len(finding.evidence_chain.code_locations) > 1 else [],
            full_contract=None,
            inheritance_chain=[],
            pattern_context=pattern_context,
            investigation_guide=investigation_guide,
            test_context=TestContext(
                scaffold_code="",
                attack_scenario="",
                setup_requirements=[],
                expected_outcome="",
            ),
            similar_exploits=[],
            fix_recommendations=[],
            pool_id=finding.pool_id,
            metadata={
                "context_bead_id": finding.context_bead_id,
                "vulndoc_reference": finding.evidence_chain.vulndoc_reference,
                "vql_queries": finding.evidence_chain.vql_queries,
                "protocol_context_applied": finding.evidence_chain.protocol_context_applied,
                "reasoning_steps": finding.evidence_chain.reasoning_steps,
            },
        )

        return bead

    def _generate_id(self, finding: FindingInput, context_bead: ContextMergeBead) -> str:
        """Generate finding bead ID."""
        import hashlib
        content = f"{context_bead.id}:{finding.vulnerability_class}:{datetime.now().isoformat()}"
        return f"VKG-{hashlib.sha256(content.encode()).hexdigest()[:12]}"

    def save_finding(self, bead: VulnerabilityBead) -> Path:
        """Save finding bead to storage."""
        if bead.pool_id:
            bead_dir = self.beads_dir / bead.pool_id / "findings"
        else:
            bead_dir = self.beads_dir / "unassigned" / "findings"

        bead_dir.mkdir(parents=True, exist_ok=True)
        bead_path = bead_dir / f"{bead.id}.yaml"
        bead_path.write_text(bead.to_minimal_yaml())
        return bead_path

    def link_to_context_bead(
        self,
        finding_bead: VulnerabilityBead,
        context_bead: ContextMergeBead,
    ) -> None:
        """Add finding bead ID to context bead's tracking."""
        context_bead.add_finding_bead(finding_bead.id)
```

```python
"""VulnDiscoveryAgent - investigates vulnerabilities using context beads.

Per 05.5-CONTEXT.md:
- Consumes ContextMergeBead with system prompt + user context
- Uses VQL queries to search graph for evidence
- Produces findings via /vrs-create-bead-finding skill
- Model selected by task router (critical -> Opus, standard -> Sonnet)
"""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Dict, List, Optional

from true_vkg.agents.orchestration.finding_factory import (
    FindingBeadFactory,
    FindingInput,
    EvidenceChain,
)
from true_vkg.agents.runtime.base import AgentRole
from true_vkg.beads.context_merge import ContextMergeBead, ContextBeadStatus
from true_vkg.beads.schema import VulnerabilityBead, CodeSnippet
from true_vkg.beads.types import Severity

logger = logging.getLogger(__name__)


@dataclass
class VulnDiscoveryConfig:
    """Configuration for vuln-discovery agent."""
    max_findings_per_context: int = 10
    min_confidence_to_report: str = "uncertain"  # uncertain, likely, confirmed
    graph_path: Optional[Path] = None  # Path to knowledge graph


@dataclass
class VulnDiscoveryResult:
    """Result of vuln-discovery agent execution."""
    success: bool
    context_bead_id: str
    findings: List[VulnerabilityBead]
    findings_count: int
    errors: List[str]
    vql_queries_executed: int


class VulnDiscoveryAgent:
    """Agent that discovers vulnerabilities using context beads.

    Workflow:
    1. Receive ContextMergeBead
    2. Extract system prompt (reasoning template) and user context (protocol data)
    3. Load knowledge graph for target scope
    4. Execute VQL queries from context bundle
    5. Analyze results using reasoning template
    6. Create finding beads for discovered vulnerabilities
    7. Mark context bead as complete

    Model selection per 05.5-CONTEXT.md:
    - Critical tasks (high severity, novel patterns) -> Opus
    - Standard tasks (known patterns) -> Sonnet
    - Decision made by task router based on vulnerability class
    """

    def __init__(
        self,
        finding_factory: FindingBeadFactory,
        config: Optional[VulnDiscoveryConfig] = None,
    ):
        self.finding_factory = finding_factory
        self.config = config or VulnDiscoveryConfig()

    def execute(
        self,
        context_bead: ContextMergeBead,
        graph: Optional[Dict[str, Any]] = None,
    ) -> VulnDiscoveryResult:
        """Execute vuln-discovery for a context bead.

        Args:
            context_bead: ContextMergeBead with merged context
            graph: Pre-loaded knowledge graph (optional)

        Returns:
            VulnDiscoveryResult with findings or errors
        """
        errors: List[str] = []
        findings: List[VulnerabilityBead] = []
        vql_queries_executed = 0

        try:
            # Mark context bead as in progress
            context_bead.status = ContextBeadStatus.IN_PROGRESS

            # Step 1: Extract prompts from context bead
            system_prompt = context_bead.get_system_prompt()
            user_context = context_bead.get_user_context()

            # Step 2: Load or use provided graph
            if graph is None:
                graph = self._load_graph(context_bead.target_scope)

            # Step 3: Execute VQL queries from context bundle
            vql_results = []
            for query in context_bead.context_bundle.vql_queries:
                try:
                    result = self._execute_vql(query, graph)
                    vql_results.append({"query": query, "result": result})
                    vql_queries_executed += 1
                except Exception as e:
                    errors.append(f"VQL query failed: {query} - {str(e)}")

            # Step 4: Analyze results (this would invoke LLM in real implementation)
            # For now, create findings based on VQL matches
            discovered = self._analyze_results(
                context_bead=context_bead,
                vql_results=vql_results,
                system_prompt=system_prompt,
                user_context=user_context,
            )

            # Step 5: Create finding beads
            for finding_input in discovered:
                if self._meets_confidence_threshold(finding_input.evidence_chain.confidence):
                    bead = self.finding_factory.create_finding(
                        finding=finding_input,
                        context_bead=context_bead,
                    )
                    self.finding_factory.save_finding(bead)
                    self.finding_factory.link_to_context_bead(bead, context_bead)
                    findings.append(bead)

                    if len(findings) >= self.config.max_findings_per_context:
                        break

            # Mark context bead as complete
            context_bead.mark_complete()

            return VulnDiscoveryResult(
                success=True,
                context_bead_id=context_bead.id,
                findings=findings,
                findings_count=len(findings),
                errors=errors,
                vql_queries_executed=vql_queries_executed,
            )

        except Exception as e:
            context_bead.mark_failed(str(e))
            errors.append(str(e))
            return VulnDiscoveryResult(
                success=False,
                context_bead_id=context_bead.id,
                findings=findings,
                findings_count=len(findings),
                errors=errors,
                vql_queries_executed=vql_queries_executed,
            )

    def _load_graph(self, target_scope: List[str]) -> Dict[str, Any]:
        """Load knowledge graph for target scope."""
        # In real implementation, load from file or build
        # For now, return empty structure
        if self.config.graph_path and self.config.graph_path.exists():
            import json
            return json.loads(self.config.graph_path.read_text())
        return {"functions": [], "contracts": [], "edges": []}

    def _execute_vql(self, query: str, graph: Dict[str, Any]) -> List[Dict[str, Any]]:
        """Execute VQL query against graph."""
        # In real implementation, use QueryExecutor
        # For now, return empty results
        try:
            from true_vkg.queries.executor import QueryExecutor
            executor = QueryExecutor(graph)
            return executor.execute(query)
        except ImportError:
            return []

    def _analyze_results(
        self,
        context_bead: ContextMergeBead,
        vql_results: List[Dict[str, Any]],
        system_prompt: str,
        user_context: str,
    ) -> List[FindingInput]:
        """Analyze VQL results to create findings.

        In real implementation, this would:
        1. Send results + prompts to LLM
        2. LLM applies reasoning template
        3. LLM produces structured findings

        For now, create findings from VQL matches.
        """
        findings: List[FindingInput] = []

        for vql_result in vql_results:
            if not vql_result["result"]:
                continue

            for match in vql_result["result"]:
                # Create finding from match
                code_locations = []
                if "location" in match:
                    code_locations.append(CodeSnippet(
                        source=match.get("source", ""),
                        file_path=match.get("file", "unknown"),
                        start_line=match.get("start_line", 0),
                        end_line=match.get("end_line", 0),
                        function_name=match.get("function_name", ""),
                        contract_name=match.get("contract_name", ""),
                    ))

                evidence_chain = EvidenceChain(
                    code_locations=code_locations,
                    vulndoc_reference=context_bead.vulnerability_class,
                    reasoning_steps=[
                        f"VQL query matched: {vql_result['query']}",
                        "Applied reasoning template from vulndoc",
                    ],
                    vql_queries=[vql_result["query"]],
                    protocol_context_applied=[
                        f"Protocol: {context_bead.protocol_name}",
                    ],
                    confidence="uncertain",  # Default, LLM would determine
                    confidence_reason="VQL pattern matched, requires verification",
                )

                findings.append(FindingInput(
                    vulnerability_class=context_bead.vulnerability_class,
                    severity=self._infer_severity(context_bead.vulnerability_class),
                    summary=f"Potential {context_bead.vulnerability_class} in {match.get('function_name', 'unknown')}",
                    evidence_chain=evidence_chain,
                    context_bead_id=context_bead.id,
                    pool_id=context_bead.pool_id,
                ))

        return findings

    def _infer_severity(self, vuln_class: str) -> Severity:
        """Infer severity from vulnerability class."""
        severity_map = {
            "reentrancy": Severity.CRITICAL,
            "access-control": Severity.HIGH,
            "oracle": Severity.HIGH,
            "flash-loan": Severity.HIGH,
            "upgrade": Severity.MEDIUM,
        }
        category = vuln_class.split("/")[0] if "/" in vuln_class else vuln_class
        return severity_map.get(category, Severity.MEDIUM)

    def _meets_confidence_threshold(self, confidence: str) -> bool:
        """Check if confidence meets reporting threshold."""
        levels = ["rejected", "uncertain", "likely", "confirmed"]
        min_level = self.config.min_confidence_to_report
        return levels.index(confidence) >= levels.index(min_level)
```
