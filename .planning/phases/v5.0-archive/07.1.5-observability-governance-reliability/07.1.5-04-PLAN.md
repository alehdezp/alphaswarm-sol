```python
from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import List, Optional, Dict, Any, Callable
import statistics
from enum import Enum

class SLOStatus(Enum):
    """SLO health status."""
    HEALTHY = "healthy"
    WARNING = "warning"
    VIOLATED = "violated"

@dataclass
class SLO:
    """Service Level Objective definition."""
    name: str
    description: str
    target_percentile: float  # 1.0 for rate, 0.95 for P95
    target_value: float
    measurement_window: timedelta
    alert_threshold: float
    severity: str = "high"  # "critical", "high", "medium"

@dataclass
class SLOViolation:
    """SLO violation event."""
    slo_name: str
    measured_value: float
    target_value: float
    violation_percentage: float
    measurement_window: timedelta
    timestamp: datetime
    affected_pools: List[str]
    severity: str

@dataclass
class SLOMeasurement:
    """Single SLO measurement."""
    slo_name: str
    value: float
    status: SLOStatus
    timestamp: datetime
    sample_count: int

class SLOTracker:
    """Track and enforce SLOs for pool execution."""

    # Default SLO definitions
    DEFAULT_SLOS = {
        "pool_success_rate": SLO(
            name="pool_success_rate",
            description="Percentage of pools completing successfully",
            target_percentile=1.0,
            target_value=0.95,
            measurement_window=timedelta(days=7),
            alert_threshold=0.90,
            severity="critical",
        ),
        "verdict_accuracy": SLO(
            name="verdict_accuracy",
            description="Percentage of verdicts confirmed accurate",
            target_percentile=1.0,
            target_value=0.90,
            measurement_window=timedelta(days=7),
            alert_threshold=0.85,
            severity="high",
        ),
        "pool_completion_latency_p95": SLO(
            name="pool_completion_latency_p95",
            description="95th percentile pool completion time (seconds)",
            target_percentile=0.95,
            target_value=300.0,
            measurement_window=timedelta(hours=24),
            alert_threshold=360.0,
            severity="high",
        ),
        "cost_per_finding": SLO(
            name="cost_per_finding",
            description="Average cost per confirmed finding (USD)",
            target_percentile=0.95,
            target_value=2.0,
            measurement_window=timedelta(days=7),
            alert_threshold=3.0,
            severity="medium",
        ),
        "bead_mttr": SLO(
            name="bead_mttr",
            description="Mean time to recovery for failed beads (seconds)",
            target_percentile=0.95,
            target_value=60.0,
            measurement_window=timedelta(hours=24),
            alert_threshold=120.0,
            severity="high",
        ),
    }

    def __init__(
        self,
        event_store=None,
        cost_ledger=None,
        slos: Dict[str, SLO] = None,
        alert_callback: Callable[[SLOViolation], None] = None,
    ):
        self.event_store = event_store
        self.cost_ledger = cost_ledger
        self.slos = slos or self.DEFAULT_SLOS
        self.alert_callback = alert_callback
        self._measurements: List[SLOMeasurement] = []

    def measure_slo(self, slo_name: str) -> SLOMeasurement:
        """Measure current value for an SLO."""

    def check_slo(self, slo_name: str) -> Optional[SLOViolation]:
        """Check if SLO is violated."""

    def _check_pool_success_rate(self, slo: SLO) -> Optional[SLOViolation]:
        """Check pool success rate SLO."""

    def _check_pool_latency(self, slo: SLO) -> Optional[SLOViolation]:
        """Check P95 pool completion latency."""

    def _check_cost_per_finding(self, slo: SLO) -> Optional[SLOViolation]:
        """Check cost per finding SLO."""

    def _check_bead_mttr(self, slo: SLO) -> Optional[SLOViolation]:
        """Check mean time to recovery SLO."""

    def monitor_all_slos(self) -> List[SLOViolation]:
        """Check all SLOs and return violations."""

    def get_slo_status(self) -> Dict[str, SLOStatus]:
        """Get current status of all SLOs."""

def load_slos(path: Path) -> Dict[str, SLO]:
    """Load SLO definitions from YAML."""
```

from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Optional, Dict, Any
from enum import Enum

class IncidentSeverity(Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"

class IncidentStatus(Enum):
    DETECTED = "detected"
    INVESTIGATING = "investigating"
    MITIGATING = "mitigating"
    RESOLVED = "resolved"

@dataclass
class Incident:
    """Incident record."""
    incident_id: str
    title: str
    description: str
    severity: IncidentSeverity
    status: IncidentStatus
    triggered_by: str  # SLO name or metric
    affected_pools: List[str]
    detected_at: datetime
    resolved_at: Optional[datetime] = None
    playbook_executed: Optional[str] = None
    timeline: List[Dict[str, Any]] = field(default_factory=list)

class IncidentDetector:
    """Detect incidents from SLO violations and anomalies."""

    def __init__(self, slo_tracker: "SLOTracker"):
        self.slo_tracker = slo_tracker
        self.active_incidents: Dict[str, Incident] = {}

    def detect_from_slo_violations(
        self,
        violations: List["SLOViolation"],
    ) -> List[Incident]:
        """Create incidents from SLO violations."""

    def update_incident_status(
        self,
        incident_id: str,
        status: IncidentStatus,
        note: str = "",
    ) -> None:
        """Update incident status with timeline entry."""

    def resolve_incident(
        self,
        incident_id: str,
        resolution: str,
    ) -> None:
        """Mark incident as resolved."""

    def get_active_incidents(self) -> List[Incident]:
        """Get all active (non-resolved) incidents."""
```

from dataclasses import dataclass
from typing import List, Dict, Any, Callable, Optional
from enum import Enum

class StepAction(Enum):
    """Playbook step action types."""
    QUERY_EVENT_STORE = "query_event_store"
    QUERY_COST_LEDGER = "query_cost_ledger"
    AGGREGATE_ERRORS = "aggregate_errors"
    SEND_ALERT = "send_alert"
    SET_POOL_INTAKE = "set_pool_intake_status"
    UPDATE_POLICY = "update_policy"
    LOG_MESSAGE = "log_message"

@dataclass
class PlaybookStep:
    """Single step in incident playbook."""
    name: str
    action: StepAction
    params: Dict[str, Any]
    condition: Optional[str] = None  # Optional condition for execution

@dataclass
class PlaybookResult:
    """Result of playbook execution."""
    playbook_id: str
    incident_id: str
    steps_executed: int
    steps_total: int
    success: bool
    outputs: Dict[str, Any]
    errors: List[str]

@dataclass
class Playbook:
    """Incident response playbook."""
    id: str
    name: str
    description: str
    trigger_slo: Optional[str]
    trigger_threshold: Optional[float]
    steps: List[PlaybookStep]

class PlaybookExecutor:
    """Execute incident response playbooks."""

    def __init__(
        self,
        event_store=None,
        cost_ledger=None,
        alert_sender: Callable[[str, str, str], None] = None,
    ):
        self.event_store = event_store
        self.cost_ledger = cost_ledger
        self.alert_sender = alert_sender
        self._action_handlers: Dict[StepAction, Callable] = {}
        self._register_handlers()

    def _register_handlers(self) -> None:
        """Register action handlers."""
        self._action_handlers = {
            StepAction.QUERY_EVENT_STORE: self._handle_query_event_store,
            StepAction.QUERY_COST_LEDGER: self._handle_query_cost_ledger,
            StepAction.SEND_ALERT: self._handle_send_alert,
            StepAction.LOG_MESSAGE: self._handle_log_message,
            # ... other handlers
        }

    def execute(
        self,
        playbook: Playbook,
        incident: "Incident",
        context: Dict[str, Any] = None,
    ) -> PlaybookResult:
        """Execute playbook for incident."""

    def load_playbook(self, path: Path) -> Playbook:
        """Load playbook from YAML."""

def load_playbooks(path: Path) -> Dict[str, Playbook]:
    """Load all playbooks from YAML."""
```

version: "1.0"

playbooks:
  pool_success_rate_degradation:
    name: "Pool Success Rate Degradation"
    description: "Respond to pool success rate dropping below threshold"
    trigger_slo: pool_success_rate
    trigger_threshold: 0.90
    steps:
      - name: "Check recent pool failures"
        action: query_event_store
        params:
          status: FAILED
          window_hours: 1
      - name: "Aggregate error patterns"
        action: aggregate_errors
        params:
          group_by: error_type
      - name: "Alert on-call"
        action: send_alert
        params:
          severity: high
          channel: slack
          message: "Pool success rate below {threshold}%"
      - name: "Log investigation started"
        action: log_message
        params:
          message: "Investigating pool success rate degradation"

  cost_spike:
    name: "Cost Spike Response"
    description: "Respond to unexpected cost increases"
    trigger_slo: cost_per_finding
    trigger_threshold: 3.0
    steps:
      - name: "Check cost by agent"
        action: query_cost_ledger
        params:
          group_by: agent_type
          window_hours: 1
      - name: "Alert cost owner"
        action: send_alert
        params:
          severity: high
          channel: cost-alerts
