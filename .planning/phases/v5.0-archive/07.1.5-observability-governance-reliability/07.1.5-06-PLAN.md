from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import List, Dict, Any, Optional
from .latency import OutputFormat

@dataclass
class AccuracyStats:
    """Accuracy statistics."""
    total_verdicts: int
    confirmed_accurate: int
    confirmed_inaccurate: int
    unverified: int
    precision: float  # TP / (TP + FP)
    recall: float  # TP / (TP + FN) - if ground truth available

@dataclass
class PatternAccuracyBreakdown:
    """Per-pattern accuracy metrics."""
    pattern_id: str
    precision: float
    recall: float
    f1_score: float
    verdict_count: int

@dataclass
class ConfidenceCalibration:
    """Confidence calibration metrics."""
    confidence_level: str  # POSSIBLE, LIKELY, CONFIRMED
    predicted_count: int
    actually_positive: int
    calibration_score: float  # How well confidence predicts accuracy

class AccuracyDashboard:
    """Verdict accuracy dashboard."""

    def __init__(self, event_store=None, ground_truth_store=None):
        self.event_store = event_store
        self.ground_truth_store = ground_truth_store

    def get_accuracy_stats(
        self,
        window: timedelta = timedelta(days=7),
    ) -> AccuracyStats:
        """Get overall verdict accuracy statistics."""

    def get_pattern_breakdown(
        self,
        window: timedelta = timedelta(days=7),
    ) -> List[PatternAccuracyBreakdown]:
        """Get per-pattern accuracy breakdown."""

    def get_confidence_calibration(
        self,
        window: timedelta = timedelta(days=7),
    ) -> List[ConfidenceCalibration]:
        """Get confidence level calibration metrics."""

    def render(
        self,
        format: OutputFormat = OutputFormat.MARKDOWN,
        window: timedelta = timedelta(days=7),
    ) -> str:
        """Render accuracy dashboard in specified format."""

    def _render_markdown(self, accuracy: AccuracyStats, patterns: List[PatternAccuracyBreakdown]) -> str:
        """Render as Markdown."""
        md = f"""# Accuracy Dashboard

## Overall Verdict Accuracy (Last 7d)

| Metric | Value |
|--------|-------|
| Total Verdicts | {accuracy.total_verdicts} |
| Confirmed Accurate | {accuracy.confirmed_accurate} |
| Confirmed Inaccurate | {accuracy.confirmed_inaccurate} |
| Unverified | {accuracy.unverified} |
| Precision | {accuracy.precision:.1%} |

## Per-Pattern Accuracy

| Pattern | Precision | Recall | F1 | Count |
|---------|-----------|--------|-----|-------|
"""
        for pattern in patterns[:10]:  # Top 10
            md += f"| {pattern.pattern_id} | {pattern.precision:.1%} | {pattern.recall:.1%} | {pattern.f1_score:.2f} | {pattern.verdict_count} |\n"
        return md

def render_accuracy_dashboard(
    event_store=None,
    ground_truth_store=None,
    format: str = "markdown",
    window_days: int = 7,
) -> str:
    """Convenience function to render accuracy dashboard."""
    dashboard = AccuracyDashboard(event_store, ground_truth_store)
    return dashboard.render(
        format=OutputFormat(format),
        window=timedelta(days=window_days),
    )

```python
from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import List, Dict, Any, Optional
from .latency import OutputFormat, LatencyDashboard, LatencyStats
from ..reliability.slo import SLOTracker, SLOStatus, SLOMeasurement
from ..reliability.incidents import Incident, IncidentDetector

@dataclass
class PoolHealthSummary:
    """Pool health summary."""
    total_pools_24h: int
    successful_pools: int
    failed_pools: int
    in_progress_pools: int
    success_rate: float

@dataclass
class SLOStatusSummary:
    """SLO status summary."""
    slo_name: str
    status: SLOStatus  # HEALTHY, WARNING, VIOLATED
    current_value: float
    target_value: float
    trend: str  # "improving", "stable", "degrading"

class OpsDashboard:
    """Operations overview dashboard."""

    def __init__(
        self,
        event_store=None,
        cost_ledger=None,
        slo_tracker: SLOTracker = None,
        incident_detector: IncidentDetector = None,
    ):
        self.event_store = event_store
        self.cost_ledger = cost_ledger
        self.slo_tracker = slo_tracker
        self.incident_detector = incident_detector
        self.latency_dashboard = LatencyDashboard(event_store)

    def get_pool_health(
        self,
        window: timedelta = timedelta(hours=24),
    ) -> PoolHealthSummary:
        """Get pool health summary."""

    def get_slo_status_summary(self) -> List[SLOStatusSummary]:
        """Get SLO status summary with trend."""

    def get_active_incidents(self) -> List[Incident]:
        """Get active incidents."""

    def get_cost_summary(
        self,
        window: timedelta = timedelta(hours=24),
    ) -> Dict[str, float]:
        """Get cost summary."""

    def render(
        self,
        format: OutputFormat = OutputFormat.MARKDOWN,
    ) -> str:
        """Render ops dashboard in specified format."""
        if format == OutputFormat.MARKDOWN:
            return self._render_markdown()
        elif format == OutputFormat.JSON:
            return self._render_json()
        elif format == OutputFormat.TOON:
            return self._render_toon()

    def _render_markdown(self) -> str:
        """Render as Markdown."""
        health = self.get_pool_health()
        slos = self.get_slo_status_summary()
        incidents = self.get_active_incidents()
        costs = self.get_cost_summary()
        latency = self.latency_dashboard.get_pool_latency_stats()

        md = f"""# AlphaSwarm Operations Dashboard

_Generated: {datetime.utcnow().strftime('%Y-%m-%d %H:%M:%S')} UTC_

## Pool Health (Last 24h)

| Metric | Value |
|--------|-------|
| Total Pools | {health.total_pools_24h} |
| Successful | {health.successful_pools} |
| Failed | {health.failed_pools} |
| In Progress | {health.in_progress_pools} |
| **Success Rate** | **{health.success_rate:.1%}** |

## SLO Status

| SLO | Status | Current | Target | Trend |
|-----|--------|---------|--------|-------|
"""
        for slo in slos:
            status_emoji = {"HEALTHY": "[OK]", "WARNING": "[WARN]", "VIOLATED": "[FAIL]"}
            md += f"| {slo.slo_name} | {status_emoji.get(slo.status.value.upper(), '?')} | {slo.current_value:.2f} | {slo.target_value:.2f} | {slo.trend} |\n"

        md += f"""
## Latency

| Metric | Value |
|--------|-------|
| P50 | {latency.p50:.1f}s |
| P95 | {latency.p95:.1f}s |
| P99 | {latency.p99:.1f}s |

## Cost (Last 24h)

| Category | Amount |
|----------|--------|
| Total | ${costs.get('total', 0):.2f} |
| LLM Calls | ${costs.get('llm', 0):.2f} |
| Tool Calls | ${costs.get('tools', 0):.2f} |

## Active Incidents

"""
        if incidents:
            for incident in incidents[:5]:
                md += f"- **[{incident.severity.value.upper()}]** {incident.title} (detected {incident.detected_at.strftime('%H:%M')})\n"
        else:
            md += "_No active incidents_\n"

        return md

    def _render_json(self) -> str:
        """Render as JSON."""
        import json
        return json.dumps({
            "generated_at": datetime.utcnow().isoformat(),
            "pool_health": self.get_pool_health().__dict__,
            "slo_status": [s.__dict__ for s in self.get_slo_status_summary()],
            "active_incidents": len(self.get_active_incidents()),
            "cost_summary": self.get_cost_summary(),
        }, indent=2, default=str)

    def _render_toon(self) -> str:
        """Render as TOON format for LLM consumption."""
        # TOON is optimized for token efficiency
        health = self.get_pool_health()
        slos = self.get_slo_status_summary()

        lines = [
            "# ops_dashboard",
            f"generated={datetime.utcnow().isoformat()}",
            "",
            "## pool_health",
            f"total={health.total_pools_24h}",
            f"success={health.successful_pools}",
            f"failed={health.failed_pools}",
            f"rate={health.success_rate:.3f}",
            "",
            "## slo_status",
        ]
        for slo in slos:
            lines.append(f"{slo.slo_name}={slo.status.value}|{slo.current_value:.2f}|{slo.target_value:.2f}")

        return "\n".join(lines)

def render_ops_dashboard(
    event_store=None,
    cost_ledger=None,
    slo_tracker=None,
    format: str = "markdown",
) -> str:
    """Convenience function to render ops dashboard."""
    dashboard = OpsDashboard(event_store, cost_ledger, slo_tracker)
    return dashboard.render(format=OutputFormat(format))

"""Dashboard CLI commands.

Usage:
    alphaswarm ops dashboard [--format markdown|json|toon] [--type ops|latency|accuracy|cost]
    alphaswarm ops dashboard --all  # Generate all dashboards
"""

import click
from pathlib import Path
from typing import Optional

@click.group("ops")
def ops_group():
    """Operations commands."""
    pass

@ops_group.command("dashboard")
@click.option(
    "--format", "-f",
    type=click.Choice(["markdown", "json", "toon"]),
    default="markdown",
    help="Output format",
)
@click.option(
    "--type", "-t",
    type=click.Choice(["ops", "latency", "accuracy", "cost", "all"]),
    default="ops",
    help="Dashboard type",
)
@click.option(
    "--output", "-o",
    type=click.Path(),
    help="Output file path (default: stdout)",
)
@click.option(
    "--window-hours",
    type=int,
    default=24,
    help="Time window in hours for metrics",
)
def dashboard_cmd(
    format: str,
    type: str,
    output: Optional[str],
    window_hours: int,
):
    """Generate operations dashboard."""
    from alphaswarm_sol.dashboards import (
        render_ops_dashboard,
        render_latency_dashboard,
        render_accuracy_dashboard,
        OutputFormat,
    )
    from alphaswarm_sol.report.cost_dashboard import render_cost_dashboard

    if type == "ops":
        content = render_ops_dashboard(format=format)
    elif type == "latency":
        content = render_latency_dashboard(format=format, window_hours=window_hours)
    elif type == "accuracy":
        content = render_accuracy_dashboard(format=format)
    elif type == "cost":
        content = render_cost_dashboard(format=format)
    elif type == "all":
        sections = [
            render_ops_dashboard(format=format),
            render_latency_dashboard(format=format, window_hours=window_hours),
            render_accuracy_dashboard(format=format),
        ]
        if format == "markdown":
            content = "\n\n---\n\n".join(sections)
        else:
            content = "\n".join(sections)

    if output:
        Path(output).write_text(content)
        click.echo(f"Dashboard written to {output}")
    else:
        click.echo(content)

# Register with main CLI
def register_cli(cli):
    """Register ops commands with main CLI."""
    cli.add_command(ops_group)
```
