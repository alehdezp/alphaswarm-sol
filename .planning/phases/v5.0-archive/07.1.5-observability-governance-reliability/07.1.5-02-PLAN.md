import structlog
from enum import Enum
from typing import List, Optional, Dict, Any
from datetime import datetime

class AuditCategory(Enum):
    """Audit log categories."""
    VERDICT_ASSIGNMENT = "verdict_assignment"
    CONFIDENCE_UPGRADE = "confidence_upgrade"
    EVIDENCE_USAGE = "evidence_usage"
    POLICY_VIOLATION = "policy_violation"
    POOL_LIFECYCLE = "pool_lifecycle"
    COST_TRACKING = "cost_tracking"
    TOOL_EXECUTION = "tool_execution"
    HANDOFF = "handoff"

class AuditLogger:
    """Structured audit logger with evidence lineage."""

    def __init__(self, log_path: Optional[Path] = None):
        # Configure structlog with JSON processor
        # If log_path provided, also write to file
        self._configure_structlog()
        self.logger = structlog.get_logger("alphaswarm.audit")

    def log_verdict(
        self,
        pool_id: str,
        bead_id: str,
        verdict: str,
        confidence: str,
        evidence_refs: List[str],
        agent_type: str,
        trace_id: Optional[str] = None,
    ) -> None:
        """Log verdict assignment with evidence lineage."""

    def log_confidence_upgrade(
        self,
        pool_id: str,
        bead_id: str,
        from_confidence: str,
        to_confidence: str,
        evidence_refs: List[str],
        agent_type: str,
        trace_id: Optional[str] = None,
    ) -> None:
        """Log confidence level upgrade with justification."""

    def log_evidence_usage(
        self,
        evidence_id: str,
        source_type: str,  # "bskg", "tool", "manual"
        source_id: str,
        used_by_agent: str,
        used_in_verdict: str,
        pool_id: str,
        trace_id: Optional[str] = None,
    ) -> None:
        """Log evidence usage for lineage tracking."""

    def log_policy_violation(
        self,
        pool_id: str,
        policy_id: str,
        violation_type: str,
        actor: str,
        severity: str,  # "critical", "high", "medium", "low"
        suggested_action: str,  # "block", "alert", "sanitize"
        details: Dict[str, Any],
        trace_id: Optional[str] = None,
    ) -> None:
        """Log policy violation for governance."""

    def log_pool_event(
        self,
        pool_id: str,
        event_type: str,  # "created", "started", "completed", "failed"
        details: Dict[str, Any],
        trace_id: Optional[str] = None,
    ) -> None:
        """Log pool lifecycle event."""
```

from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Optional, Dict, Any
from enum import Enum

class SourceType(Enum):
    """Evidence source types."""
    BSKG = "bskg"
    TOOL = "tool"
    MANUAL = "manual"
    DERIVED = "derived"

@dataclass
class LineageStep:
    """Single step in evidence lineage chain."""
    step_type: str  # "origin", "extraction", "transformation", "verification"
    source_type: SourceType
    source_id: str
    agent: Optional[str]
    timestamp: str
    metadata: Dict[str, Any] = field(default_factory=dict)

@dataclass
class EvidenceLineage:
    """Complete evidence lineage from origin to current."""
    evidence_id: str
    current_source_type: SourceType
    current_source_id: str
    chain: List[LineageStep]
    created_at: str
    last_accessed: str

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to dict for logging."""

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "EvidenceLineage":
        """Deserialize from dict."""

class LineageTracker:
    """Track and query evidence lineage."""

    def __init__(self, storage_path: Optional[Path] = None):
        self._lineages: Dict[str, EvidenceLineage] = {}
        self._storage_path = storage_path

    def create_lineage(
        self,
        evidence_id: str,
        source_type: SourceType,
        source_id: str,
        extracting_agent: str,
    ) -> EvidenceLineage:
        """Create new lineage for evidence extracted from source."""

    def add_transformation(
        self,
        evidence_id: str,
        transform_type: str,
        transforming_agent: str,
        metadata: Dict[str, Any] = None,
    ) -> None:
        """Add transformation step to existing lineage."""

    def add_verification(
        self,
        evidence_id: str,
        verifying_agent: str,
        verification_result: str,
    ) -> None:
        """Add verification step to lineage."""

    def get_lineage(self, evidence_id: str) -> Optional[EvidenceLineage]:
        """Get lineage for evidence ID."""

    def query_by_source(
        self,
        source_type: SourceType,
        source_id: str,
    ) -> List[EvidenceLineage]:
        """Find all evidence derived from a source."""

    def build_lineage_chain(self, evidence_id: str) -> List[Dict[str, Any]]:
        """Build full lineage chain for audit logging."""
```
