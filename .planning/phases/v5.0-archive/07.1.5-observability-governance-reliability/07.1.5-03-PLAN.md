from dataclasses import dataclass
from enum import Enum
from typing import Dict, List, Any, Optional

class PolicyAction(Enum):
    """Action to take on policy violation."""
    ALLOW = "allow"
    BLOCK = "block"
    SANITIZE = "sanitize"
    ALERT = "alert"

@dataclass
class PolicyViolation:
    """Policy violation details."""
    policy_id: str
    violation_type: str
    severity: str  # "critical", "high", "medium", "low"
    message: str
    suggested_action: PolicyAction
    details: Dict[str, Any] = None

class PolicyViolationError(Exception):
    """Raised when blocking policy violated."""
    def __init__(self, violation: PolicyViolation):
        super().__init__(violation.message)
        self.violation = violation

@dataclass
class CostPolicy:
    """Cost budget policy."""
    hard_limit_usd: float
    soft_limit_usd: float
    per_agent_limit_usd: Optional[float] = None
    action_on_hard_limit: PolicyAction = PolicyAction.BLOCK
    action_on_soft_limit: PolicyAction = PolicyAction.ALERT

@dataclass
class ToolAccessPolicy:
    """Tool access policy per agent type."""
    agent_type: str
    allowed_tools: List[str]
    forbidden_tools: List[str]
    max_calls_per_pool: Optional[int] = None

@dataclass
class EvidencePolicy:
    """Evidence integrity policy."""
    confidence_upgrade_requires_evidence: bool = True
    min_evidence_refs_for_likely: int = 1
    min_evidence_refs_for_confirmed: int = 2
    require_provenance_chain: bool = True

@dataclass
class ModelUsagePolicy:
    """Model usage policy."""
    approved_models: List[str]
    tier_routing_required: bool = True
    fallback_model: Optional[str] = None

def load_policies(path: Path) -> Dict[str, Any]:
    """Load policies from YAML file."""
```

from functools import wraps
from .policies import PolicyAction, PolicyViolation, PolicyViolationError, load_policies
from .validators import (
    validate_cost_budget,

    validate_evidence_integrity,
    validate_model_usage,

from ..observability.audit import AuditLogger

class PolicyEnforcer:
    """Runtime policy enforcement for agents and tools."""

    def __init__(
        self,
        policies_path: Optional[Path] = None,
        cost_ledger = None,
        audit_logger: Optional[AuditLogger] = None,
    ):

        self.cost_ledger = cost_ledger
        self.audit_logger = audit_logger or AuditLogger()

    def check_input_policy(
        self,
        agent_type: str,
        input_data: Dict[str, Any],
        pool_id: str,
        trace_id: Optional[str] = None,
    ) -> Optional[PolicyViolation]:
        """Check all input policies before agent execution."""
        # 1. Check cost budget
        # 2. Check tool access
        # 3. Check model usage
        # Return first violation or None

    def check_output_policy(
        self,
        agent_type: str,
        output_data: Dict[str, Any],
        pool_id: str,
        trace_id: Optional[str] = None,
    ) -> Optional[PolicyViolation]:
        """Check output policies after agent execution."""
        # 1. Check evidence integrity
        # Return violation or None

    def handle_violation(
        self,
        violation: PolicyViolation,
        pool_id: str,
        agent_type: str,
        trace_id: Optional[str] = None,
    ) -> None:
        """Handle policy violation based on suggested action."""
        # Log to audit
        self.audit_logger.log_policy_violation(
            pool_id=pool_id,
            policy_id=violation.policy_id,
            violation_type=violation.violation_type,
            actor=agent_type,
            severity=violation.severity,
            suggested_action=violation.suggested_action.value,
            details=violation.details or {},
            trace_id=trace_id,
        )

        if violation.suggested_action == PolicyAction.BLOCK:
            raise PolicyViolationError(violation)

def enforce_policy(enforcer: PolicyEnforcer):
    """Decorator for policy enforcement."""
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Extract context from kwargs
            pool_id = kwargs.get("pool_id")
            agent_type = kwargs.get("agent_type")
            trace_id = kwargs.get("trace_id")
            input_data = kwargs.get("input_data", {})

            # Check input policy
            violation = enforcer.check_input_policy(
                agent_type, input_data, pool_id, trace_id
            )
            if violation:
                enforcer.handle_violation(violation, pool_id, agent_type, trace_id)

            # Execute function
            result = func(*args, **kwargs)

            # Check output policy
            violation = enforcer.check_output_policy(
                agent_type, result, pool_id, trace_id
            )
            if violation:
                enforcer.handle_violation(violation, pool_id, agent_type, trace_id)

            return result
        return wrapper
    return decorator
```
