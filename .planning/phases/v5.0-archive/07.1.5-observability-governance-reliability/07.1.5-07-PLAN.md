class TestFullAuditWorkflow:
    """Complete audit workflow with all observability components."""

    @pytest.fixture
    def full_observability_stack(self, tmp_path):
        """Setup complete observability stack."""
        # Tracing
        exporter = InMemorySpanExporter()
        provider = TracerProvider()
        provider.add_span_processor(SimpleSpanProcessor(exporter))
        trace.set_tracer_provider(provider)

        # Audit & Lineage
        audit_logger = AuditLogger(log_path=tmp_path / "audit.jsonl")
        lineage_tracker = LineageTracker(storage_path=tmp_path / "lineage.json")

        # Governance
        cost_ledger = CostLedger()
        policy_enforcer = PolicyEnforcer(
            cost_ledger=cost_ledger,
            audit_logger=audit_logger,
        )

        # Reliability
        event_store = EventStore(storage_path=tmp_path / "events")
        slo_tracker = SLOTracker(event_store=event_store, cost_ledger=cost_ledger)

        # Dashboards
        ops_dashboard = OpsDashboard(
            event_store=event_store,
            cost_ledger=cost_ledger,
            slo_tracker=slo_tracker,
        )

        return {
            "exporter": exporter,
            "audit_logger": audit_logger,
            "lineage_tracker": lineage_tracker,
            "policy_enforcer": policy_enforcer,
            "cost_ledger": cost_ledger,
            "event_store": event_store,
            "slo_tracker": slo_tracker,
            "ops_dashboard": ops_dashboard,
            "tmp_path": tmp_path,
        }

    def test_audit_produces_traceable_evidence(self, full_observability_stack, mock_llm):
        """Complete audit run produces traceable evidence chain."""
        stack = full_observability_stack

        # Create pool manager with full stack
        pool_manager = PoolManager(
            audit_logger=stack["audit_logger"],
            lineage_tracker=stack["lineage_tracker"],
            policy_enforcer=stack["policy_enforcer"],
            cost_ledger=stack["cost_ledger"],
            event_store=stack["event_store"],
        )

        # Execute audit
        pool = pool_manager.create_pool(
            scope="tests/contracts/ReentrancyVulnerable.sol",
            pool_type="standard",
        )
        result = pool_manager.execute_pool(pool.id, mock_llm=mock_llm)

        # Verify trace chain
        spans = stack["exporter"].get_finished_spans()
        assert len(spans) > 0, "No spans recorded"

        # Verify audit trail
        audit_path = stack["tmp_path"] / "audit.jsonl"
        if audit_path.exists():
            entries = [json.loads(l) for l in audit_path.read_text().strip().split("\n") if l]
            assert len(entries) > 0, "No audit entries"

            # Verdicts should have trace_id
            verdicts = [e for e in entries if "verdict" in e.get("event", "")]
            for v in verdicts:
                assert "trace_id" in v or "pool_id" in v, "Verdict missing trace correlation"

        # Verify cost tracking
        pool_cost = stack["cost_ledger"].get_pool_cost(pool.id)
        assert pool_cost >= 0, "Cost not tracked"

    def test_dashboard_reflects_audit_results(self, full_observability_stack, mock_llm):
        """Dashboard accurately reflects audit execution results."""
        stack = full_observability_stack

        pool_manager = PoolManager(
            event_store=stack["event_store"],
            cost_ledger=stack["cost_ledger"],
        )

        # Run multiple pools
        for i in range(3):
            pool = pool_manager.create_pool(scope="tests/contracts/SimpleVault.sol")
            pool_manager.execute_pool(pool.id, mock_llm=mock_llm)

        # Generate dashboard
        dashboard_output = stack["ops_dashboard"].render(format=OutputFormat.MARKDOWN)

        # Verify dashboard structure
        assert "Pool Health" in dashboard_output
        assert "SLO Status" in dashboard_output
        assert "Cost" in dashboard_output or "Latency" in dashboard_output

    def test_policy_violation_appears_in_audit_and_dashboard(self, full_observability_stack):
        """Policy violations tracked in audit and reflected in dashboard."""
        stack = full_observability_stack

        # Trigger policy violation
        stack["cost_ledger"].record(
            pool_id="expensive-pool",
            tokens=200000,
            cost_usd=25.0,  # Way over $10 limit
            agent="test",
        )

        try:
            stack["policy_enforcer"].check_input_policy(
                agent_type="vrs-attacker",
                input_data={},
                pool_id="expensive-pool",
            )
        except Exception:
            pass  # Expected violation

        # Verify audit log
        audit_path = stack["tmp_path"] / "audit.jsonl"
        if audit_path.exists():
            content = audit_path.read_text()
            assert "policy" in content.lower() or "violation" in content.lower()

    def test_chaos_results_feed_into_slo_tracking(self, full_observability_stack, mock_llm):
        """Chaos test results affect SLO measurements."""
        stack = full_observability_stack

        # Setup chaos harness
        harness = ChaosTestHarness(enabled=True, seed=42)

        pool_manager = PoolManager(
            event_store=stack["event_store"],
            chaos_harness=harness,
        )

        # Run pools under chaos (some will fail)
        for i in range(10):
            try:
                pool = pool_manager.create_pool(scope="tests/contracts/SimpleVault.sol")
                pool_manager.execute_pool(pool.id, mock_llm=mock_llm)
            except Exception:
                # Record failure
                stack["event_store"].record_pool_complete(
                    pool_id=f"chaos-pool-{i}",
                    status="FAILED",
                    completed_at=datetime.utcnow(),
                )

        # SLO should reflect chaos impact
        measurement = stack["slo_tracker"].measure_slo("pool_success_rate")
        # Under chaos, success rate may be lower
        assert measurement.sample_count >= 0
