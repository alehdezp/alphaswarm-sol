from typing import List, Optional, Tuple
import hashlib

class SemanticDeduplicator:
    """Two-stage deduplication: location + semantic similarity."""

    def __init__(
        self,
        use_embeddings: bool = True,
        similarity_threshold: float = 0.85,
        line_tolerance: int = 5
    ):
        self.use_embeddings = use_embeddings
        self.similarity_threshold = similarity_threshold
        self.line_tolerance = line_tolerance
        self._embedding_model = None

    @property
    def embedding_model(self):
        """Lazy load embedding model."""
        if self._embedding_model is None and self.use_embeddings:
            try:
                from sentence_transformers import SentenceTransformer
                self._embedding_model = SentenceTransformer('all-MiniLM-L6-v2')
            except ImportError:
                # Fallback to no embeddings
                self.use_embeddings = False
        return self._embedding_model

    def deduplicate(
        self,
        findings: List[VKGFinding]
    ) -> Tuple[List[DeduplicatedFinding], DeduplicationStats]:
        """Two-stage deduplication."""
        # Stage 1: Location-based clustering (existing logic)
        location_clusters = self._cluster_by_location(findings)

        # Stage 2: Semantic similarity within location clusters
        if self.use_embeddings:
            final_clusters = []
            for cluster in location_clusters:
                if len(cluster) > 1:
                    # Check semantic similarity
                    semantic_subclusters = self._cluster_by_semantics(cluster)
                    final_clusters.extend(semantic_subclusters)
                else:
                    final_clusters.append(cluster)
        else:
            final_clusters = location_clusters

        # Convert to DeduplicatedFinding
        results = [self._cluster_to_finding(c) for c in final_clusters]

        # Calculate stats
        stats = DeduplicationStats(
            input_count=len(findings),
            output_count=len(results),
            reduction_percent=round(100 * (1 - len(results) / max(len(findings), 1)), 1),
            location_matches=sum(1 for c in location_clusters if len(c) > 1),
            semantic_matches=sum(1 for c in final_clusters if len(c) > 1) if self.use_embeddings else 0
        )

        return results, stats

    def _cluster_by_semantics(self, findings: List[VKGFinding]) -> List[List[VKGFinding]]:
        """Group by semantic similarity of descriptions."""
        if not self.embedding_model or len(findings) < 2:
            return [findings]

        # Get descriptions
        descriptions = [f"{f.title} {f.description}" for f in findings]
        embeddings = self.embedding_model.encode(descriptions)

        # Compute similarity matrix
        from sklearn.metrics.pairwise import cosine_similarity
        similarities = cosine_similarity(embeddings)

        # Cluster based on threshold
        clusters = []
        used = set()
        for i, finding in enumerate(findings):
            if i in used:
                continue
            cluster = [finding]
            used.add(i)
            for j in range(i + 1, len(findings)):
                if j not in used and similarities[i][j] >= self.similarity_threshold:
                    cluster.append(findings[j])
                    used.add(j)
            clusters.append(cluster)

        return clusters
```

from .mapping import get_patterns_covered_by_tools, TOOL_DETECTOR_MAP

class ToolCoordinator:
    # Coverage threshold: if tool precision >= this, skip BSKG pattern
    SKIP_THRESHOLD = 0.80

    # Patterns that should NEVER be skipped (edge cases, complex logic)
    NEVER_SKIP_PATTERNS = {
        "business-logic-violation",  # Tools can't detect
        "economic-manipulation",      # Context-dependent
        "governance-attack",          # Multi-step logic
        "cross-function-reentrancy",  # Complex patterns
    }

    def create_strategy(
        self,
        analysis: ProjectAnalysis,
        available_tools: List[str]
    ) -> ToolStrategy:
        # ... existing tool selection logic ...

        # Calculate pattern skip list
        patterns_to_skip, skip_rationale = self._calculate_pattern_skips(
            tools_to_run, analysis
        )

        return ToolStrategy(
            tools_to_run=tools_to_run,
            parallel_groups=parallel_groups,
            tool_configs=tool_configs,
            estimated_time_seconds=estimated_time,
            rationale=rationale,
            skip_reasons=skip_reasons,
            patterns_to_skip=patterns_to_skip,
            pattern_skip_rationale=skip_rationale
        )

    def _calculate_pattern_skips(
        self,
        tools: List[str],
        analysis: ProjectAnalysis
    ) -> Tuple[List[str], Dict[str, str]]:
        """Determine which BSKG patterns to skip based on tool coverage.

        Returns:
            patterns_to_skip: List of pattern IDs BSKG should not run
            skip_rationale: Mapping of pattern -> explanation
        """
        coverage = get_patterns_covered_by_tools(tools)

        patterns_to_skip = []
        skip_rationale = {}

        for pattern, score in coverage.items():
            # Never skip edge-case patterns
            if pattern in self.NEVER_SKIP_PATTERNS:
                continue

            # Skip if tool coverage is high enough
            if score >= self.SKIP_THRESHOLD:
                patterns_to_skip.append(pattern)
                tools_covering = self._get_tools_for_pattern(pattern, tools)
                skip_rationale[pattern] = (
                    f"Covered by {', '.join(tools_covering)} "
                    f"(precision={score:.0%})"
                )

        return patterns_to_skip, skip_rationale

    def _get_tools_for_pattern(
        self,
        pattern: str,
        tools: List[str]
    ) -> List[str]:
        """Get which tools cover a pattern."""
        covering = []
        for tool in tools:
            for detector, mapping in TOOL_DETECTOR_MAP.get(tool, {}).items():
                if mapping.vkg_pattern == pattern:
                    covering.append(tool)
                    break
        return covering

    def get_edge_case_patterns(self) -> List[str]:
        """Return patterns that require BSKG regardless of tools.

        These patterns detect things tools can't:
        - Business logic violations
        - Economic attacks
        - Multi-step exploits
        """
        return list(self.NEVER_SKIP_PATTERNS)
```
