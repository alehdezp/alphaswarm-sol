**Parallel execution:**
Use ThreadPoolExecutor for I/O-bound tools (network, disk).
Each parallel group runs concurrently, groups run sequentially.

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def execute_strategy(self, strategy: ToolStrategy, project_path: Path):
    all_results = []

    for group in strategy.parallel_groups:
        # Filter to tools we're actually running
        tools_in_group = [t for t in group if t in strategy.tools_to_run]
        if not tools_in_group:
            continue

        # Run group in parallel
        with ThreadPoolExecutor(max_workers=len(tools_in_group)) as executor:
            futures = {
                executor.submit(
                    self.execute_tool,
                    tool,
                    strategy.tool_configs.get(tool),
                    project_path
                ): tool
                for tool in tools_in_group
            }

            for future in as_completed(futures):
                tool = futures[future]
                try:
                    result = future.result()
                    all_results.append(result)
                except Exception as e:
                    # Tool failure is not fatal
                    all_results.append(ExecutionResult(
                        tool=tool,
                        success=False,
                        findings=[],
                        execution_time=0,
                        error=str(e),
                        from_cache=False,
                        partial=False
                    ))

    return all_results
```

**Timeout handling:**
- Per-tool timeouts from config
- Capture partial output on timeout
- Mark result as partial=True

**Error recovery:**
- Retry once with modified config on certain errors
- Log all failures with recovery hints

**File hashing:**
```python
import hashlib
from pathlib import Path

def get_file_hash(self, file_path: Path) -> str:
    sha256 = hashlib.sha256()
    with open(file_path, 'rb') as f:
        for chunk in iter(lambda: f.read(8192), b''):
            sha256.update(chunk)
    return sha256.hexdigest()

def get_project_hash(self, project_path: Path) -> str:
    """Hash all .sol files to detect any changes."""
    sha256 = hashlib.sha256()
    sol_files = sorted(project_path.rglob("*.sol"))

    for sol_file in sol_files:
        # Skip common library paths
        if any(skip in str(sol_file) for skip in ["node_modules", "lib/", ".cache"]):
            continue
        sha256.update(str(sol_file.relative_to(project_path)).encode())
        sha256.update(self.get_file_hash(sol_file).encode())

    return sha256.hexdigest()
```
