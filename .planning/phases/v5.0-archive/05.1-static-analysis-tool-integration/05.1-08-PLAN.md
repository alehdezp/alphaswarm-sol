```python
class BatchDismissal:
    """Batch dismissal with cross-agent verification."""

    def __init__(self, storage_path: Path):
        self.storage_path = storage_path
        self.log_file = storage_path / "dismissal_log.yaml"
        self._logs: List[DismissalLog] = []
        self._load_logs()

    def propose_dismissal(
        self,
        bead_ids: List[str],
        reason: DismissalReason,
        proposing_agent: str
    ) -> str:
        """Propose dismissal - needs verification."""
        log_id = self._generate_id()
        log = DismissalLog(
            id=log_id,
            bead_ids=bead_ids,
            reason=reason,
            dismissed_by=proposing_agent,
            dismissed_at=datetime.now(),
            verified_by=None,
            verified_at=None,
            consensus=False  # Not yet verified
        )
        self._logs.append(log)
        self._save_logs()
        return log_id

    def verify_dismissal(
        self,
        log_id: str,
        verifying_agent: str,
        agrees: bool,
        counter_reason: Optional[str] = None
    ) -> bool:
        """Second agent verifies dismissal.

        Returns True if dismissal is finalized (both agree).
        Returns False if disagreement (beads NOT dismissed).
        """
        log = self._get_log(log_id)
        if not log:
            raise ValueError(f"Dismissal log not found: {log_id}")

        if log.dismissed_by == verifying_agent:
            raise ValueError("Cannot self-verify dismissal")

        log.verified_by = verifying_agent
        log.verified_at = datetime.now()
        log.consensus = agrees

        if not agrees:
            # Disagreement -> NOT dismissed, log counter reason
            log.challenge_result = counter_reason

        self._save_logs()
        return agrees

    def challenge_dismissal(
        self,
        log_id: str,
        challenger: str,
        reason: str
    ) -> bool:
        """User challenges a dismissal.

        Marks dismissal for re-verification.
        """
        log = self._get_log(log_id)
        if not log:
            raise ValueError(f"Dismissal log not found: {log_id}")

        if log.challenged:
            raise ValueError("Dismissal already challenged (one challenge limit)")

        log.challenged = True
        log.challenge_result = reason
        log.consensus = False  # Reopen for verification

        self._save_logs()
        return True

    def get_dismissed_beads(self) -> List[str]:
        """Get all successfully dismissed bead IDs."""
        dismissed = []
        for log in self._logs:
            if log.consensus and not log.challenged:
                dismissed.extend(log.bead_ids)
        return dismissed

    def get_pending_verifications(self) -> List[DismissalLog]:
        """Get dismissals awaiting verification."""
        return [log for log in self._logs if not log.consensus and log.verified_by is None]

    def get_challenges(self) -> List[DismissalLog]:
        """Get challenged dismissals."""
        return [log for log in self._logs if log.challenged]
```

```python
from ..llm.providers import get_provider, LLMProvider

class CrossModelVerifier:
    """Verify decisions using multiple LLM models.

    Per CONTEXT.md: Different models required for cross-agent verification.
    This reduces correlated hallucination risk.
    """

    # Model pairs for cross-verification
    MODEL_PAIRS = [
        ("claude-sonnet-4", "gpt-4o"),        # Claude + OpenAI
        ("claude-sonnet-4", "claude-haiku"),   # Different Claude tiers
        ("gpt-4o", "claude-opus-4"),          # OpenAI + Claude Opus
    ]

    def __init__(
        self,
        primary_provider: Optional[LLMProvider] = None,
        secondary_provider: Optional[LLMProvider] = None
    ):
        self.primary = primary_provider or get_provider("anthropic")
        self.secondary = secondary_provider

    def verify_dismissal(
        self,
        bead_ids: List[str],
        dismissal_reason: DismissalReason,
        beads: List[VulnerabilityBead]
    ) -> VerificationResult:
        """Verify dismissal decision with two models.

        Both models must agree for dismissal to proceed.
        Conflict = conservative (NOT dismissed).
        """
        task_id = f"verify-dismiss-{datetime.now().strftime('%Y%m%d%H%M%S')}"

        # Build verification prompt
        prompt = self._build_verification_prompt(beads, dismissal_reason)

        # Get primary opinion
        primary_response = self.primary.generate(
            system="You are a security auditor verifying dismissal decisions. "
                   "Respond with 'AGREE' or 'DISAGREE' followed by reasoning.",
            user=prompt
        )
        primary_agrees = "AGREE" in primary_response.content.upper()

        # Get secondary opinion (different model)
        secondary_response = None
        secondary_agrees = False

        if self.secondary:
            secondary_response = self.secondary.generate(
                system="You are a security auditor verifying dismissal decisions. "
                       "Respond with 'AGREE' or 'DISAGREE' followed by reasoning.",
                user=prompt
            )
            secondary_agrees = "AGREE" in secondary_response.content.upper()
        else:
            # Fallback: use same provider with lower temperature
            secondary_response = self.primary.generate(
                system="You are a skeptical security auditor. "
                       "Challenge dismissal decisions. "
                       "Respond with 'AGREE' or 'DISAGREE' followed by reasoning.",
                user=prompt,
                temperature=0.1  # More deterministic
            )
            secondary_agrees = "AGREE" in secondary_response.content.upper()

        consensus = primary_agrees and secondary_agrees

        return VerificationResult(
            task_id=task_id,
            primary_model=self.primary.model_name,
            secondary_model=self.secondary.model_name if self.secondary else f"{self.primary.model_name}-skeptical",
            primary_agrees=primary_agrees,
            secondary_agrees=secondary_agrees,
            consensus=consensus,
            primary_reasoning=primary_response.content,
            secondary_reasoning=secondary_response.content if secondary_response else "",
            conflict_details=self._explain_conflict(primary_agrees, secondary_agrees) if not consensus else None
        )

    def _build_verification_prompt(
        self,
        beads: List[VulnerabilityBead],
        reason: DismissalReason
    ) -> str:
        """Build prompt for verification."""
        bead_summaries = "\n".join([
            f"- {b.id}: {b.vulnerability_class} in {b.vulnerable_code.function_name}"
            for b in beads
        ])

        return f"""
Verify this batch dismissal decision:

## Beads to Dismiss
{bead_summaries}

## Dismissal Reason
Category: {reason.category.value}
Explanation: {reason.explanation}

Evidence:
{chr(10).join('- ' + e for e in reason.evidence)}

## Your Task
1. Review the evidence supporting dismissal
2. Consider if any bead might still be a real vulnerability
3. Decide: AGREE (safe to dismiss) or DISAGREE (keep for verification)

Respond with AGREE or DISAGREE on the first line, then explain your reasoning.
"""

    def _explain_conflict(self, primary: bool, secondary: bool) -> str:
        """Explain why models disagreed."""
        if primary and not secondary:
            return "Primary model agreed but secondary model raised concerns"
        elif not primary and secondary:
            return "Secondary model agreed but primary model raised concerns"
        else:
            return "Both models raised different concerns"


def get_diverse_verifier() -> CrossModelVerifier:
    """Get verifier with diverse model providers.

    Tries to use different providers (Anthropic + OpenAI).
    Falls back to same provider with different prompting if only one available.
    """
    try:
        anthropic = get_provider("anthropic")
        openai = get_provider("openai")
        return CrossModelVerifier(anthropic, openai)
    except Exception:
        # Fallback to single provider
        return CrossModelVerifier(get_provider("anthropic"), None)
```
