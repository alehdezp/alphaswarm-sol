"""CLI commands for static analysis tool integration."""

from pathlib import Path
from typing import Optional, List
import typer
from rich.console import Console
from rich.table import Table
from rich.progress import Progress, SpinnerColumn, TextColumn

tools_app = typer.Typer(
    name="tools",
    help="Static analysis tool integration commands"
)
console = Console()


@tools_app.command("list")
def list_tools(
    check_health: bool = typer.Option(False, "--health", "-h", help="Check tool health"),
):
    """List available static analysis tools."""
    from ..tools.registry import ToolRegistry

    registry = ToolRegistry()
    tools = registry.TOOL_DEFINITIONS

    table = Table(title="VKG Tool Registry")
    table.add_column("Tool", style="cyan")
    table.add_column("Tier", style="magenta")
    table.add_column("Status", style="green")
    table.add_column("Version", style="yellow")
    table.add_column("Install", style="dim")

    for name, info in tools.items():
        if check_health:
            health = registry.check_tool(name)
            status = "[green]OK" if health.healthy else "[red]Missing"
            version = health.version or "-"
        else:
            installed = registry.check_tool_installed(name)
            status = "[green]Installed" if installed else "[yellow]Not installed"
            version = "-"

        table.add_row(
            name,
            str(info.tier),
            status,
            version,
            info.install_hint[:50] + "..." if len(info.install_hint) > 50 else info.install_hint
        )

    console.print(table)


@tools_app.command("run")
def run_tool(
    tool: str = typer.Argument(..., help="Tool name to run"),
    project_path: Path = typer.Argument(..., help="Path to Solidity project"),
    output: Optional[Path] = typer.Option(None, "--output", "-o", help="Output file"),
    format: str = typer.Option("json", "--format", "-f", help="Output format (json, sarif, yaml)"),
    timeout: int = typer.Option(120, "--timeout", "-t", help="Timeout in seconds"),
):
    """Run a single analysis tool."""
    from ..tools.executor import ToolExecutor
    from ..tools.config import get_optimal_config
    from ..tools.registry import ToolRegistry
    import json

    # Check tool is available
    registry = ToolRegistry()
    if not registry.check_tool_installed(tool):
        health = registry.check_tool(tool)
        console.print(f"[red]Tool not installed: {tool}")
        console.print(f"[yellow]Install with: {health.error or registry.get_install_hint(tool)}")
        raise typer.Exit(1)

    # Run tool
    config = get_optimal_config(tool)
    config.timeout = timeout

    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        console=console
    ) as progress:
        progress.add_task(f"Running {tool}...", total=None)
        executor = ToolExecutor()
        result = executor.execute_tool(tool, config, project_path)

    if result.success:
        console.print(f"[green]Tool completed: {len(result.findings)} findings in {result.execution_time:.1f}s")

        if output:
            if format == "sarif":
                from ..tools.adapters.sarif import SARIFAdapter
                adapter = SARIFAdapter()
                sarif = adapter.to_sarif(result.findings, tool, "latest")
                output.write_text(json.dumps(sarif, indent=2))
            else:
                output.write_text(json.dumps([f.to_dict() for f in result.findings], indent=2))
            console.print(f"[green]Output written to {output}")
        else:
            # Print summary
            for finding in result.findings[:5]:
                console.print(f"  - {finding.severity.upper()}: {finding.title[:60]}")
            if len(result.findings) > 5:
                console.print(f"  ... and {len(result.findings) - 5} more")
    else:
        console.print(f"[red]Tool failed: {result.error}")
        raise typer.Exit(1)


@tools_app.command("analyze")
def analyze_project(
    project_path: Path = typer.Argument(..., help="Path to Solidity project"),
    tools: Optional[str] = typer.Option(None, "--tools", "-t", help="Comma-separated tools to run"),
    output_dir: Optional[Path] = typer.Option(None, "--output", "-o", help="Output directory"),
    no_cache: bool = typer.Option(False, "--no-cache", help="Skip cache"),
    create_beads: bool = typer.Option(True, "--beads/--no-beads", help="Create beads from findings"),
):
    """Run full tool analysis pipeline."""
    from ..tools.coordinator import ToolCoordinator
    from ..tools.executor import ToolExecutor
    from ..orchestration.dedup import SemanticDeduplicator
    from ..beads.from_tools import create_beads_from_tools
    import json

    # Analyze project and create strategy
    console.print("[bold]Analyzing project...")
    coordinator = ToolCoordinator()
    analysis = coordinator.analyze_project(project_path)
    console.print(f"  Contracts: {analysis.contract_count}")
    console.print(f"  Lines: {analysis.total_lines}")

    # Get strategy
    if tools:
        tool_list = [t.strip() for t in tools.split(",")]
        strategy = coordinator.create_strategy(analysis, tool_list)
    else:
        strategy = coordinator.create_strategy(analysis, None)

    console.print(f"\n[bold]Strategy: {len(strategy.tools_to_run)} tools")
    console.print(f"  {coordinator.explain_strategy(strategy)}")

    # Execute tools
    console.print("\n[bold]Running tools...")
    executor = ToolExecutor()
    results = executor.execute_strategy(strategy, project_path)

    # Report results
    total_findings = sum(len(r.findings) for r in results)
    console.print(f"\n[bold]Results: {total_findings} raw findings")
    for result in results:
        status = "[green]OK" if result.success else "[red]FAIL"
        cache = "[dim](cached)" if result.from_cache else ""
        console.print(f"  {result.tool}: {len(result.findings)} findings {status} {cache}")

    # Deduplicate
    console.print("\n[bold]Deduplicating...")
    all_findings = []
    for result in results:
        all_findings.extend(result.findings)

    deduplicator = SemanticDeduplicator()
    deduped, stats = deduplicator.deduplicate(all_findings)
    console.print(f"  {stats.input_count} -> {stats.output_count} ({stats.reduction_percent}% reduction)")

    # Create beads if requested
    if create_beads and deduped:
        console.print("\n[bold]Creating beads...")
        beads = create_beads_from_tools(
            [f.findings[0] for f in deduped],  # Use representative finding
            project_path
        )
        console.print(f"  {len(beads)} beads created (confidence=uncertain)")

        if output_dir:
            output_dir.mkdir(parents=True, exist_ok=True)
            beads_file = output_dir / "tool_beads.yaml"
            # Save beads
            import yaml
            with open(beads_file, "w") as f:
                yaml.dump([b.to_dict() for b in beads], f)
            console.print(f"  [green]Beads saved to {beads_file}")

    # Save findings if output specified
    if output_dir:
        output_dir.mkdir(parents=True, exist_ok=True)
        findings_file = output_dir / "findings.json"
        with open(findings_file, "w") as f:
            json.dump([d.to_dict() for d in deduped], f, indent=2)
        console.print(f"[green]Findings saved to {findings_file}")


@tools_app.command("install")
def install_tools(
    tools: Optional[str] = typer.Option(None, "--tools", "-t", help="Comma-separated tools to install"),
    tier: Optional[int] = typer.Option(None, "--tier", help="Install all tools of this tier"),
    all_tools: bool = typer.Option(False, "--all", "-a", help="Install all tools"),
):
    """Install analysis tools."""
    console.print("[bold]Tool Installation")
    console.print("Run the install script:")
    console.print("  [cyan]bash .vrs/install-tools.sh --help")

    if all_tools:
        console.print("  [cyan]bash .vrs/install-tools.sh --all")
    elif tier is not None:
        console.print(f"  [cyan]bash .vrs/install-tools.sh --tier-{tier}")
    elif tools:
        for tool in tools.split(","):
            console.print(f"  [cyan]bash .vrs/install-tools.sh --tool={tool.strip()}")


@tools_app.command("config")
def show_config(
    tool: str = typer.Argument(..., help="Tool name"),
    project_path: Optional[Path] = typer.Option(None, "--project", "-p", help="Project path for overrides"),
):
    """Show tool configuration."""
    from ..tools.config import get_optimal_config, load_tool_config
    import yaml

    if project_path:
        config = load_tool_config(tool, project_path)
    else:
        config = get_optimal_config(tool)

    console.print(f"[bold]Configuration for {tool}:")
    console.print(yaml.dump(config.__dict__, default_flow_style=False))
