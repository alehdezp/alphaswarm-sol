```python
from typing import List, Optional, Dict, Any
from ..tools.adapters.sarif import VKGFinding
from .schema import VulnerabilityBead, PatternContext, TestContext, InvestigationGuide

class ToolFindingToBead:
    """Convert tool findings to VulnerabilityBeads.

    Model tier: haiku-4.5 (parsing, conversion)
    """

    MODEL_TIER = "haiku-4.5"

    def __init__(self, project_path: Path, graph: Optional[KnowledgeGraph] = None):
        self.project_path = project_path
        self.graph = graph
        self._bead_counter = 0

    def create_bead(
        self,
        finding: VKGFinding,
        dedup_sources: List[str] = None
    ) -> VulnerabilityBead:
        """Create a bead from a tool finding.

        Per PHILOSOPHY.md:
        - All tool findings start as "uncertain"
        - Must go through multi-agent verification
        """
        bead_id = self._generate_id(finding)

        # Extract code context
        code_snippet = self._extract_code_snippet(finding)

        # Build pattern context from tool info
        pattern_context = self._build_pattern_context(finding, dedup_sources)

        # Create investigation guide for tool finding
        investigation_guide = self._tool_investigation_guide(finding)

        # Build test context scaffold
        test_context = self._build_test_context(finding)

        # Create bead with UNCERTAIN confidence
        return VulnerabilityBead(
            id=bead_id,
            vulnerability_class=finding.category,
            pattern_id=finding.rule_id,
            function_id=finding.function or "",
            severity=self._map_severity(finding.severity),
            confidence=0.3,  # UNCERTAIN - must verify
            vulnerable_code=code_snippet,
            related_code=[],  # Will be populated by verification
            full_contract=None,
            inheritance_chain=[],
            pattern_context=pattern_context,
            investigation_guide=investigation_guide,
            test_context=test_context,
            similar_exploits=[],
            fix_recommendations=self._get_fix_recommendations(finding.category),
            status=BeadStatus.PENDING,
            graph_context=None,
            graph_context_category="",
            full_graph_available=self.graph is not None,
            # Tool-specific metadata
            metadata={
                "tool_sources": dedup_sources or [finding.source],
                "detector_ids": [finding.rule_id],
                "tool_confidence": finding.confidence,
                "from_tool": True,
            }
        )

    def create_beads_from_findings(
        self,
        findings: List[VKGFinding]
    ) -> List[VulnerabilityBead]:
        """Create beads from multiple findings."""
        return [self.create_bead(f) for f in findings]
```

```python
class BeadTriager:
    """Prioritize beads for efficient verification.

    Model tier: sonnet-4.5 (triage decisions require reasoning)
    """

    MODEL_TIER = "sonnet-4.5"

    # Known false positive patterns (per tool experience)
    FALSE_POSITIVE_PATTERNS = {
        "slither": ["naming-convention", "solc-version", "pragma"],
        "semgrep": ["informational-*"],
    }

    def triage_beads(
        self,
        beads: List[VulnerabilityBead]
    ) -> List[TriageResult]:
        """Prioritize and relate beads."""
        results = []

        # Build relationship graph
        relationships = self._find_relationships(beads)

        for bead in beads:
            priority = self._calculate_priority(bead)
            related = relationships.get(bead.id, [])
            cascade = self._calculate_cascade_potential(bead, related, beads)
            agent = self._recommend_agent(bead)

            results.append(TriageResult(
                bead_id=bead.id,
                priority=priority,
                reasoning=self._explain_priority(bead, priority),
                related_beads=related,
                cascade_potential=cascade,
                recommended_agent=agent
            ))

        # Sort by priority
        priority_order = {
            TriagePriority.CRITICAL: 0,
            TriagePriority.HIGH: 1,
            TriagePriority.MEDIUM: 2,
            TriagePriority.LOW: 3,
            TriagePriority.SKIP: 4,
        }
        results.sort(key=lambda r: priority_order[r.priority])

        return results

    def _calculate_priority(self, bead: VulnerabilityBead) -> TriagePriority:
        """Calculate priority based on multiple factors."""
        # Check for skip patterns
        if self._is_likely_false_positive(bead):
            return TriagePriority.SKIP

        # Severity-based initial priority
        severity_map = {
            Severity.CRITICAL: TriagePriority.CRITICAL,
            Severity.HIGH: TriagePriority.HIGH,
            Severity.MEDIUM: TriagePriority.MEDIUM,
            Severity.LOW: TriagePriority.LOW,
            Severity.INFO: TriagePriority.LOW,
        }
        base_priority = severity_map.get(bead.severity, TriagePriority.MEDIUM)

        # Boost priority if multiple tools agree
        tool_sources = bead.metadata.get("tool_sources", [])
        if len(tool_sources) >= 3:
            base_priority = self._boost_priority(base_priority)
        elif len(tool_sources) >= 2:
            if base_priority == TriagePriority.MEDIUM:
                base_priority = TriagePriority.HIGH

        return base_priority

    def _find_relationships(
        self,
        beads: List[VulnerabilityBead]
    ) -> Dict[str, List[str]]:
        """Find related beads by file, function, or pattern."""
        relationships = {}

        for bead in beads:
            related = []
            for other in beads:
                if other.id == bead.id:
                    continue
                # Same file
                if (bead.vulnerable_code.file_path ==
                    other.vulnerable_code.file_path):
                    related.append(other.id)
                # Same function
                elif (bead.function_id and
                      bead.function_id == other.function_id):
                    related.append(other.id)
                # Same vulnerability class (may cascade)
                elif bead.vulnerability_class == other.vulnerability_class:
                    related.append(other.id)

            relationships[bead.id] = related

        return relationships

    def _recommend_agent(self, bead: VulnerabilityBead) -> str:
        """Recommend which agent should verify first."""
        # High severity -> attacker first (prove exploitability)
        if bead.severity in [Severity.CRITICAL, Severity.HIGH]:
            return "attacker"
        # Access control -> defender first (prove guards exist)
        if bead.vulnerability_class == "access_control":
            return "defender"
        # Default -> verifier for balanced analysis
        return "verifier"
```

```python
def create_bead_from_tool_finding(
    self,
    finding: VKGFinding,
    tool_sources: List[str] = None
) -> VulnerabilityBead:
    """Create bead from tool finding, optionally enriched with BSKG context.

    If graph is available, enriches bead with:
    - Related code from call graph
    - Graph context for the vulnerability category
    - Similar exploits from VulnDocs
    """
    # Use ToolFindingToBead for base conversion
    from .from_tools import ToolFindingToBead
    converter = ToolFindingToBead(self.project_path, self.graph)
    bead = converter.create_bead(finding, tool_sources)

    # Enrich with BSKG context if graph available
    if self.graph and self.config.include_graph_context:
        bead = self._enrich_with_graph_context(bead, finding)

    return bead

def _enrich_with_graph_context(
    self,
    bead: VulnerabilityBead,
    finding: VKGFinding
) -> VulnerabilityBead:
    """Enrich tool-generated bead with BSKG context."""
    # Find matching node in graph
    node = self._find_node_for_finding(finding)
    if node:
        # Add related code
        bead.related_code = self._extract_related_code(node)

        # Add graph context
        if self.config.include_graph_context:
            graph_context, category = self._extract_graph_context(
                {"node_id": node.id},
                bead.vulnerability_class
            )
            bead.graph_context = graph_context
            bead.graph_context_category = category

        # Add similar exploits
        bead.similar_exploits = self._find_similar_exploits(
            {"pattern_id": finding.rule_id},
            bead.vulnerability_class
        )

    return bead

def _find_node_for_finding(self, finding: VKGFinding) -> Optional[Node]:
    """Find graph node matching the tool finding location."""
    if not self.graph:
        return None

    # Search by function name and file
    for node in self.graph.nodes.values():
        if node.type != "Function":
            continue

        props = node.properties
        # Match by function name
        if finding.function and props.get("name") == finding.function:
            return node
        # Match by file and line
        if node.evidence:
            for ev in node.evidence:
                if (ev.file and finding.file in ev.file and
                    ev.line_start and abs(ev.line_start - finding.line) < 5):
                    return node

    return None
```
