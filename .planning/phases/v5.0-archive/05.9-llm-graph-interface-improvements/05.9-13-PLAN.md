#!/usr/bin/env python3
"""
Validate patterns against taxonomy registry.

Usage:
  python scripts/validate_patterns.py --strict
  python scripts/validate_patterns.py --check-taxonomy --patterns vulndocs/
  python scripts/validate_patterns.py --check-vql --vql-files tests/
"""

import argparse
import sys
from pathlib import Path
import yaml
import os

def load_taxonomy_registry():
    """Load canonical taxonomy from registry."""
    from alphaswarm_sol.kg.taxonomy import TaxonomyRegistry
    return TaxonomyRegistry()

def validate_pattern_operations(pattern_path: Path, registry) -> list[str]:
    """Check pattern uses registered operations."""
    errors = []
    warnings = []

    with open(pattern_path) as f:
        pattern = yaml.safe_load(f)

    # Extract operations used in pattern
    operations = extract_operations_from_pattern(pattern)

    for op in operations:
        if not registry.is_registered(op):
            errors.append(f"{pattern_path}: Unknown operation '{op}'")
        elif registry.is_deprecated(op):
            canonical = registry.resolve(op)
            warnings.append(
                f"{pattern_path}: Deprecated operation '{op}', use '{canonical}'"
            )

    return errors, warnings

def extract_operations_from_pattern(pattern: dict) -> set[str]:
    """Extract all operation references from pattern."""
    operations = set()

    # Check match.tier_a.all
    for clause in pattern.get("match", {}).get("tier_a", {}).get("all", []):
        if "has_operation" in clause:
            operations.add(clause["has_operation"])
        if "has_all_operations" in clause:
            operations.update(clause["has_all_operations"])
        if "sequence_order" in clause:
            # Extract operation names from sequence
            pass

    # Check match.tier_c
    for clause in pattern.get("match", {}).get("tier_c", []):
        if "has_label" in clause:
            # Labels may reference operations
            pass

    return operations

def validate_vql_files(vql_dir: Path, registry) -> list[str]:
    """Check VQL files use registered operations."""
    errors = []
    warnings = []

    for vql_file in vql_dir.rglob("*.vql"):
        with open(vql_file) as f:
            content = f.read()

        # Parse VQL for operation references
        # This is simplified - real implementation would use VQL parser
        import re
        ops = re.findall(r'\b([A-Z_]+_[A-Z_]+)\b', content)

        for op in ops:
            if op.startswith(("TRANSFERS_", "READS_", "WRITES_", "CHECKS_")):
                if not registry.is_registered(op):
                    errors.append(f"{vql_file}: Unknown operation '{op}'")

    return errors, warnings

def main():
    parser = argparse.ArgumentParser(description="Validate patterns against taxonomy")
    parser.add_argument("--strict", action="store_true",
                       help="Fail on any warning")
    parser.add_argument("--check-taxonomy", action="store_true",
                       help="Check pattern operations against registry")
    parser.add_argument("--check-vql", action="store_true",
                       help="Check VQL files against registry")
    parser.add_argument("--patterns", type=Path, default=Path("vulndocs/"),
                       help="Pattern directory")
    parser.add_argument("--vql-files", type=Path, default=Path("tests/"),
                       help="VQL files directory")
    args = parser.parse_args()

    fail_on_warning = args.strict or os.environ.get("FAIL_ON_WARNING") == "true"
    fail_on_deprecated = os.environ.get("FAIL_ON_DEPRECATED") == "true"

    registry = load_taxonomy_registry()
    all_errors = []
    all_warnings = []

    if args.check_taxonomy:
        for pattern_file in args.patterns.rglob("*.yaml"):
            if "patterns" in str(pattern_file):
                errors, warnings = validate_pattern_operations(pattern_file, registry)
                all_errors.extend(errors)
                all_warnings.extend(warnings)

    if args.check_vql:
        errors, warnings = validate_vql_files(args.vql_files, registry)
        all_errors.extend(errors)
        all_warnings.extend(warnings)

    # Report
    for warning in all_warnings:
        print(f"WARNING: {warning}")
    for error in all_errors:
        print(f"ERROR: {error}")

    # Exit codes
    if all_errors:
        print(f"\n{len(all_errors)} error(s) found")
        sys.exit(1)
    if all_warnings and (fail_on_warning or fail_on_deprecated):
        print(f"\n{len(all_warnings)} warning(s) treated as errors")
        sys.exit(1)

    print("Pattern validation passed")
    sys.exit(0)

if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
Validate outputs against Graph Interface Contract v2.

Usage:
  python scripts/validate_contract.py --schema schemas/graph_interface_v2.json --samples tests/fixtures/
  python scripts/validate_contract.py --check-coverage --min-coverage 0.5
"""

import argparse
import json
import sys
from pathlib import Path
import os

def load_schema(schema_path: Path) -> dict:
    """Load JSON schema."""
    with open(schema_path) as f:
        return json.load(f)

def validate_sample(sample_path: Path, schema: dict) -> list[str]:
    """Validate sample output against schema."""
    import jsonschema

    errors = []

    with open(sample_path) as f:
        try:
            sample = json.load(f)
        except json.JSONDecodeError as e:
            return [f"{sample_path}: Invalid JSON - {e}"]

    # Schema validation
    try:
        jsonschema.validate(sample, schema)
    except jsonschema.ValidationError as e:
        errors.append(f"{sample_path}: Schema violation - {e.message}")

    # Contract-specific checks (beyond schema)
    contract_errors = validate_contract_invariants(sample, sample_path)
    errors.extend(contract_errors)

    return errors

def validate_contract_invariants(output: dict, path: Path) -> list[str]:
    """Check contract invariants beyond JSON schema."""
    errors = []

    # Check interface version
    if output.get("interface_version") != 2:
        errors.append(f"{path}: interface_version must be 2")

    # Check findings have required evidence
    for i, finding in enumerate(output.get("findings", [])):
        # matched_clauses implies evidence_refs or evidence_missing
        if finding.get("matched_clauses"):
            if not finding.get("evidence_refs") and not finding.get("evidence_missing"):
                errors.append(
                    f"{path}: finding[{i}] has matched_clauses but no evidence"
                )

        # unknown_clauses implies omission or availability=false
        if finding.get("unknown_clauses"):
            has_omission = finding.get("omissions") or \
                any(c.get("omissions_refs") for c in finding.get("clause_matrix", []))
            if not has_omission:
                errors.append(
                    f"{path}: finding[{i}] has unknown_clauses but no omissions"
                )

        # clause_matrix must align with lists
        if finding.get("clause_matrix"):
            matrix_matched = {c["clause"] for c in finding["clause_matrix"]
                            if c.get("status") == "matched"}
            list_matched = set(finding.get("matched_clauses", []))
            if matrix_matched != list_matched:
                errors.append(
                    f"{path}: finding[{i}] clause_matrix doesn't match matched_clauses"
                )

    # Check summary has coverage_score
    if "summary" in output:
        if "coverage_score" not in output["summary"]:
            errors.append(f"{path}: summary missing coverage_score")

    return errors

def validate_coverage_threshold(output: dict, min_coverage: float) -> list[str]:
    """Check coverage score meets threshold."""
    errors = []

    coverage = output.get("summary", {}).get("coverage_score", 0)
    if coverage < min_coverage:
        errors.append(
            f"Coverage score {coverage:.2f} below threshold {min_coverage}"
        )

    return errors

def main():
    parser = argparse.ArgumentParser(description="Validate contract compliance")
    parser.add_argument("--schema", type=Path, required=True,
                       help="Path to JSON schema")
    parser.add_argument("--samples", type=Path,
                       help="Directory of sample outputs to validate")
    parser.add_argument("--check-coverage", action="store_true",
                       help="Check coverage score thresholds")
    parser.add_argument("--min-coverage", type=float, default=0.5,
                       help="Minimum coverage score (default: 0.5)")
    args = parser.parse_args()

    fail_on_warning = os.environ.get("FAIL_ON_WARNING") == "true"

    schema = load_schema(args.schema)
    all_errors = []

    if args.samples:
        for sample_file in args.samples.rglob("*.json"):
            errors = validate_sample(sample_file, schema)
            all_errors.extend(errors)

            if args.check_coverage:
                with open(sample_file) as f:
                    sample = json.load(f)
                coverage_errors = validate_coverage_threshold(
                    sample, args.min_coverage
                )
                all_errors.extend(coverage_errors)

    # Report
    for error in all_errors:
        print(f"ERROR: {error}")

    if all_errors:
        print(f"\n{len(all_errors)} contract violation(s) found")
        sys.exit(1)

    print("Contract validation passed")
    sys.exit(0)

if __name__ == "__main__":
    main()
```
