"""Structured debate protocol (iMAD-inspired).

Protocol from 04-CONTEXT.md:
1. CLAIM ROUND - Attacker/Defender present evidence-anchored arguments
2. REBUTTAL ROUND - Challenge each other's evidence
3. SYNTHESIS - Verifier weighs evidence, produces verdict
4. HUMAN CHECKPOINT - Always flagged for human review

Key principle: Evidence anchoring - all claims must reference code locations.
"""

from dataclasses import dataclass, field
from typing import Optional, List, Dict, Any
from enum import Enum
from datetime import datetime
import logging

from .schemas import (
    Verdict, VerdictConfidence, DebateRecord, DebateClaim,
    EvidencePacket, EvidenceItem,
)
from .confidence import ConfidenceEnforcer

logger = logging.getLogger(__name__)


class DebatePhase(Enum):
    """Phases of the debate protocol."""
    CLAIM = "claim"
    REBUTTAL = "rebuttal"
    SYNTHESIS = "synthesis"
    COMPLETE = "complete"


@dataclass
class DebateRound:
    """A single round of debate."""
    phase: DebatePhase
    attacker_argument: Optional[DebateClaim] = None
    defender_argument: Optional[DebateClaim] = None
    timestamp: datetime = field(default_factory=datetime.now)


@dataclass
class DebateConfig:
    """Configuration for debate."""
    max_rebuttal_rounds: int = 2  # Max back-and-forth
    require_evidence: bool = True  # Claims must have evidence
    auto_flag_human: bool = True  # Always flag for human review


class DebateOrchestrator:
    """Orchestrates structured debate between attacker and defender.

    The debate follows iMAD protocol:
    - Claims must be evidence-anchored (code locations)
    - Rebuttals challenge specific evidence
    - Verifier synthesizes, doesn't add new analysis
    - Human always reviews debate outcomes
    """

    def __init__(
        self,
        attacker_agent: Any = None,  # AttackerAgent
        defender_agent: Any = None,  # DefenderAgent
        verifier_agent: Any = None,  # VerifierAgent
        config: Optional[DebateConfig] = None,
    ):
        self.attacker = attacker_agent
        self.defender = defender_agent
        self.verifier = verifier_agent
        self.config = config or DebateConfig()
        self.enforcer = ConfidenceEnforcer()

    def run_debate(
        self,
        bead_id: str,
        evidence: EvidencePacket,
        attacker_context: Dict[str, Any],
        defender_context: Dict[str, Any],
    ) -> Verdict:
        """Run full debate protocol.

        Args:
            bead_id: The bead being debated
            evidence: Evidence packet with behavioral signature, code
            attacker_context: Context for attacker (exploit paths, etc.)
            defender_context: Context for defender (guards, specs)

        Returns:
            Verdict with debate record, always human-flagged
        """
        rounds: List[DebateRound] = []

        # Phase 1: CLAIM ROUND
        claim_round = self._run_claim_round(evidence, attacker_context, defender_context)
        rounds.append(claim_round)

        # Phase 2: REBUTTAL ROUND(S)
        for i in range(self.config.max_rebuttal_rounds):
            rebuttal_round = self._run_rebuttal_round(
                rounds[-1], evidence, attacker_context, defender_context
            )
            rounds.append(rebuttal_round)

            # Early exit if no new arguments
            if not rebuttal_round.attacker_argument and not rebuttal_round.defender_argument:
                break

        # Phase 3: SYNTHESIS
        debate_record = self._build_debate_record(rounds)
        verdict = self._run_synthesis(bead_id, debate_record, evidence)

        # Phase 4: HUMAN FLAG (always)
        verdict.human_flag = True

        return verdict

    def _run_claim_round(
        self,
        evidence: EvidencePacket,
        attacker_ctx: Dict[str, Any],
        defender_ctx: Dict[str, Any],
    ) -> DebateRound:
        """Run claim round - each side presents their case."""
        round = DebateRound(phase=DebatePhase.CLAIM)

        # Attacker claim
        if self.attacker:
            attacker_result = self._get_attacker_claim(evidence, attacker_ctx)
            round.attacker_argument = attacker_result

        # Defender claim
        if self.defender:
            defender_result = self._get_defender_claim(evidence, defender_ctx)
            round.defender_argument = defender_result

        return round

    def _run_rebuttal_round(
        self,
        previous_round: DebateRound,
        evidence: EvidencePacket,
        attacker_ctx: Dict[str, Any],
        defender_ctx: Dict[str, Any],
    ) -> DebateRound:
        """Run rebuttal round - challenge each other's claims."""
        round = DebateRound(phase=DebatePhase.REBUTTAL)

        # Attacker rebuts defender's claim
        if self.attacker and previous_round.defender_argument:
            round.attacker_argument = self._get_attacker_rebuttal(
                previous_round.defender_argument, evidence, attacker_ctx
            )

        # Defender rebuts attacker's claim
        if self.defender and previous_round.attacker_argument:
            round.defender_argument = self._get_defender_rebuttal(
                previous_round.attacker_argument, evidence, defender_ctx
            )

        return round

    def _get_attacker_claim(
        self, evidence: EvidencePacket, context: Dict[str, Any]
    ) -> DebateClaim:
        """Get attacker's initial claim with evidence anchoring."""
        if not self.attacker:
            return DebateClaim(
                agent_role="attacker",
                claim="No attacker agent configured",
            )

        # Use existing AttackerAgent
        # The agent's analyze() returns AttackerResult with attacks
        try:
            result = self.attacker.analyze(context.get("agent_context"))
            if result.matched and result.attack:
                evidence_items = [
                    EvidenceItem(
                        type="attack_vector",
                        value=result.attack.postconditions[0] if result.attack.postconditions else "exploit",
                        location=result.attack.target_function,
                        confidence=result.attack.strength,
                        source="attacker_agent",
                    )
                ]
                return DebateClaim(
                    agent_role="attacker",
                    claim=result.summary or "Exploit path identified",
                    evidence=evidence_items,
                )
        except Exception as e:
            logger.warning(f"Attacker analysis failed: {e}")

        return DebateClaim(
            agent_role="attacker",
            claim="Unable to construct exploit",
        )

    def _get_defender_claim(
        self, evidence: EvidencePacket, context: Dict[str, Any]
    ) -> DebateClaim:
        """Get defender's initial claim with evidence anchoring."""
        if not self.defender:
            return DebateClaim(
                agent_role="defender",
                claim="No defender agent configured",
            )

        try:
            result = self.defender.analyze(context.get("agent_context"))
            if result.matched and result.defenses:
                evidence_items = [
                    EvidenceItem(
                        type="guard",
                        value=d.claim,
                        confidence=d.strength,
                        source="defender_agent",
                    )
                    for d in result.defenses[:3]  # Top 3 defenses
                ]
                return DebateClaim(
                    agent_role="defender",
                    claim=result.summary or "Protected by guards",
                    evidence=evidence_items,
                )
        except Exception as e:
            logger.warning(f"Defender analysis failed: {e}")

        return DebateClaim(
            agent_role="defender",
            claim="Unable to identify protections",
        )

    def _get_attacker_rebuttal(
        self, defender_claim: DebateClaim, evidence: EvidencePacket, context: Dict[str, Any]
    ) -> Optional[DebateClaim]:
        """Attacker rebuts defender's claim."""
        # Simplified: In full implementation, would analyze defender's guards
        # and find bypass paths
        if not self.attacker:
            return None

        # Check if defender's guards actually block the attack
        guards_mentioned = [e.value for e in defender_claim.evidence]

        # For now, return simple rebuttal
        return DebateClaim(
            agent_role="attacker",
            claim=f"Guards ({', '.join(guards_mentioned[:2])}) can be bypassed",
            evidence=[
                EvidenceItem(
                    type="bypass",
                    value="Guard bypass analysis",
                    source="attacker_agent",
                )
            ],
        )

    def _get_defender_rebuttal(
        self, attacker_claim: DebateClaim, evidence: EvidencePacket, context: Dict[str, Any]
    ) -> Optional[DebateClaim]:
        """Defender rebuts attacker's claim."""
        if not self.defender:
            return None

        return DebateClaim(
            agent_role="defender",
            claim="Attack preconditions cannot be satisfied",
            evidence=[
                EvidenceItem(
                    type="precondition_block",
                    value="Precondition analysis",
                    source="defender_agent",
                )
            ],
        )

    def _build_debate_record(self, rounds: List[DebateRound]) -> DebateRecord:
        """Build debate record from rounds."""
        record = DebateRecord()

        # Extract first claims
        if rounds and rounds[0].attacker_argument:
            record.attacker_claim = rounds[0].attacker_argument
        if rounds and rounds[0].defender_argument:
            record.defender_claim = rounds[0].defender_argument

        # Extract rebuttals
        for round in rounds[1:]:
            if round.attacker_argument:
                record.rebuttals.append(round.attacker_argument)
            if round.defender_argument:
                record.rebuttals.append(round.defender_argument)

        return record

    def _run_synthesis(
        self, bead_id: str, debate: DebateRecord, evidence: EvidencePacket
    ) -> Verdict:
        """Verifier synthesizes debate into verdict."""
        # Determine confidence based on debate
        confidence = self._assess_debate_outcome(debate)

        # Build rationale from debate
        rationale = self._synthesize_rationale(debate)

        # Check for dissent
        dissent = self._check_for_dissent(debate)

        verdict = Verdict(
            bead_id=bead_id,
            confidence=confidence,
            rationale=rationale,
            dissenting_opinion=dissent,
            debate=debate,
            evidence_summary=[
                e.value for e in (debate.attacker_claim.evidence if debate.attacker_claim else [])
            ] + [
                e.value for e in (debate.defender_claim.evidence if debate.defender_claim else [])
            ],
        )

        # Verifier adds synthesis
        if debate:
            debate.verifier_synthesis = rationale

        # Enforce confidence rules
        return self.enforcer.enforce(verdict)

    def _assess_debate_outcome(self, debate: DebateRecord) -> VerdictConfidence:
        """Assess confidence from debate."""
        attacker_strength = 0.0
        defender_strength = 0.0

        if debate.attacker_claim:
            attacker_strength = sum(
                e.confidence for e in debate.attacker_claim.evidence
            ) / max(len(debate.attacker_claim.evidence), 1)

        if debate.defender_claim:
            defender_strength = sum(
                e.confidence for e in debate.defender_claim.evidence
            ) / max(len(debate.defender_claim.evidence), 1)

        # If close, uncertain
        if abs(attacker_strength - defender_strength) < 0.2:
            return VerdictConfidence.UNCERTAIN

        # Attacker wins
        if attacker_strength > defender_strength:
            return VerdictConfidence.LIKELY

        # Defender wins
        return VerdictConfidence.REJECTED

    def _synthesize_rationale(self, debate: DebateRecord) -> str:
        """Synthesize rationale from debate."""
        parts = []

        if debate.attacker_claim:
            parts.append(f"Attacker: {debate.attacker_claim.claim}")
        if debate.defender_claim:
            parts.append(f"Defender: {debate.defender_claim.claim}")

        if debate.rebuttals:
            parts.append(f"Rebuttals: {len(debate.rebuttals)} rounds")

        return " | ".join(parts) if parts else "No debate conducted"

    def _check_for_dissent(self, debate: DebateRecord) -> Optional[str]:
        """Check if there's significant dissent."""
        if not debate.defender_claim:
            return None

        # If defender had strong evidence, note the dissent
        if debate.defender_claim.evidence:
            avg_strength = sum(e.confidence for e in debate.defender_claim.evidence) / len(
                debate.defender_claim.evidence
            )
            if avg_strength > 0.7:
                return f"Defender notes: {debate.defender_claim.claim}"

        return None


def run_debate(
    bead_id: str,
    evidence: EvidencePacket,
    attacker_context: Dict[str, Any],
    defender_context: Dict[str, Any],
    attacker_agent: Any = None,
    defender_agent: Any = None,
) -> Verdict:
    """Convenience function for running debate."""
    orchestrator = DebateOrchestrator(
        attacker_agent=attacker_agent,
        defender_agent=defender_agent,
    )
    return orchestrator.run_debate(bead_id, evidence, attacker_context, defender_context)

"""Phase handlers for execution loop.

Handlers implement the actual work for each RouteAction.
They are injected into ExecutionLoop to keep it thin.
"""

from dataclasses import dataclass
from typing import Optional, List, Dict, Any, Callable
from pathlib import Path
import logging

from .schemas import Pool, PoolStatus, Verdict, Scope, EvidencePacket
from .loop import PhaseResult, LoopPhase
from .router import RouteAction
from .pool import PoolManager
from .debate import DebateOrchestrator

logger = logging.getLogger(__name__)


@dataclass
class HandlerConfig:
    """Configuration for handlers."""
    project_root: Path = Path(".")
    pattern_dir: Optional[Path] = None
    use_llm: bool = False
    verbose: bool = False


class BaseHandler:
    """Base class for phase handlers."""

    def __init__(self, manager: PoolManager, config: Optional[HandlerConfig] = None):
        self.manager = manager
        self.config = config or HandlerConfig()

    def __call__(self, pool: Pool, target_beads: Optional[List[str]] = None) -> PhaseResult:
        """Execute handler. Subclasses implement this."""
        raise NotImplementedError


class BuildGraphHandler(BaseHandler):
    """Handler for BUILD_GRAPH action."""

    def __call__(self, pool: Pool, target_beads: Optional[List[str]] = None) -> PhaseResult:
        """Build knowledge graph from scope."""
        try:
            from true_vkg.kg.builder import VKGBuilder
            from true_vkg.kg.store import GraphStore

            if not pool.scope:
                return PhaseResult(
                    success=False,
                    phase=LoopPhase.INTAKE,
                    message="No scope defined for pool",
                )

            # Build graph
            project_root = self.config.project_root
            builder = VKGBuilder(project_root)

            # Build from scope files
            for file_path in pool.scope.files:
                target = project_root / file_path
                if target.exists():
                    graph = builder.build(target)
                    break
            else:
                return PhaseResult(
                    success=False,
                    phase=LoopPhase.INTAKE,
                    message="No valid files in scope",
                )

            # Save graph
            output_dir = project_root / ".true_vkg" / "graphs"
            store = GraphStore(output_dir)
            store.save(graph, overwrite=True)

            # Update pool metadata
            pool.metadata["graph_built"] = True
            pool.metadata["graph_nodes"] = len(graph.nodes)
            self.manager.storage.save(pool)

            return PhaseResult(
                success=True,
                phase=LoopPhase.INTAKE,
                message=f"Built graph with {len(graph.nodes)} nodes",
                artifacts={"graph_path": str(output_dir / "graph.json")},
            )

        except Exception as e:
            logger.error(f"Build graph failed: {e}")
            return PhaseResult(
                success=False,
                phase=LoopPhase.INTAKE,
                message=f"Build failed: {e}",
            )


class DetectPatternsHandler(BaseHandler):
    """Handler for DETECT_PATTERNS action."""

    def __call__(self, pool: Pool, target_beads: Optional[List[str]] = None) -> PhaseResult:
        """Run pattern detection on graph."""
        try:
            from true_vkg.kg.store import GraphStore
            from true_vkg.queries.patterns import PatternEngine

            # Load graph
            graph_path = self.config.project_root / ".true_vkg" / "graphs"
            store = GraphStore(graph_path)
            graph = store.load()

            # Run patterns
            engine = PatternEngine(pattern_dir=self.config.pattern_dir)
            matches = engine.run_all_patterns(graph)

            # Update pool
            pool.metadata["patterns_detected"] = True
            pool.metadata["match_count"] = len(matches)
            pool.metadata["matches"] = matches  # Store for bead creation
            self.manager.storage.save(pool)

            return PhaseResult(
                success=True,
                phase=LoopPhase.CONTEXT,
                message=f"Detected {len(matches)} pattern matches",
                artifacts={"matches": matches},
            )

        except Exception as e:
            logger.error(f"Pattern detection failed: {e}")
            return PhaseResult(
                success=False,
                phase=LoopPhase.CONTEXT,
                message=f"Detection failed: {e}",
            )


class CreateBeadsHandler(BaseHandler):
    """Handler for CREATE_BEADS action."""

    def __call__(self, pool: Pool, target_beads: Optional[List[str]] = None) -> PhaseResult:
        """Create beads from pattern matches."""
        try:
            from true_vkg.kg.store import GraphStore
            from true_vkg.beads.creator import BeadCreator
            from true_vkg.beads.storage import BeadStorage

            # Load graph
            graph_path = self.config.project_root / ".true_vkg" / "graphs"
            store = GraphStore(graph_path)
            graph = store.load()

            # Get matches from pool metadata
            matches = pool.metadata.get("matches", [])
            if not matches:
                return PhaseResult(
                    success=True,
                    phase=LoopPhase.BEADS,
                    message="No matches to create beads from",
                )

            # Create beads
            creator = BeadCreator(graph)
            beads = creator.create_beads_from_findings(matches)

            # Save beads to pool
            bead_storage = BeadStorage(self.config.project_root / ".true_vkg" / "beads")
            for bead in beads:
                bead_storage.save_to_pool(bead, pool.id)
                pool.beads.append(bead.id)

            self.manager.storage.save(pool)

            return PhaseResult(
                success=True,
                phase=LoopPhase.BEADS,
                message=f"Created {len(beads)} beads",
                artifacts={"bead_ids": [b.id for b in beads]},
            )

        except Exception as e:
            logger.error(f"Bead creation failed: {e}")
            return PhaseResult(
                success=False,
                phase=LoopPhase.BEADS,
                message=f"Creation failed: {e}",
            )

    """Handler for SPAWN_VERIFIERS action."""

    def __call__(self, pool: Pool, target_beads: Optional[List[str]] = None) -> PhaseResult:
        """Spawn verifier agents to synthesize attacker/defender results."""
        try:
            from true_vkg.agents.verifier import VerifierAgent
            from true_vkg.beads.storage import BeadStorage

            beads_to_process = target_beads or pool.beads
            bead_storage = BeadStorage(self.config.project_root / ".true_vkg" / "beads")

            verifier = VerifierAgent()
            processed = []

            for bead_id in beads_to_process:
                bead = bead_storage.load(bead_id)
                if not bead:
                    continue

                # Get attacker/defender results from work_state
                work_state = bead.work_state
                attacker_matched = work_state.get("attacker_matched", False)
                defender_matched = work_state.get("defender_matched", False)

                # Simple synthesis
                if attacker_matched and not defender_matched:
                    confidence = "likely"
                elif defender_matched and not attacker_matched:
                    confidence = "rejected"
                else:
                    confidence = "uncertain"

                bead_storage.update_work_state(
                    bead_id,
                    {
                        **work_state,
                        "verifier_confidence": confidence,
                        "verifier_processed": True,
                    },
                    agent="verifier",
                )
                processed.append(bead_id)

            pool.metadata["verifier_processed"] = processed
            self.manager.storage.save(pool)

            return PhaseResult(
                success=True,
                phase=LoopPhase.EXECUTE,
                message=f"Verifier processed {len(processed)} beads",
            )

        except Exception as e:
            logger.error(f"Verifier spawn failed: {e}")
            return PhaseResult(
                success=False,
                phase=LoopPhase.EXECUTE,
                message=f"Spawn failed: {e}",
            )

class CollectVerdictsHandler(BaseHandler):
    """Handler for COLLECT_VERDICTS action."""

    def __call__(self, pool: Pool, target_beads: Optional[List[str]] = None) -> PhaseResult:
        """Collect and finalize all verdicts."""
        from true_vkg.beads.storage import BeadStorage
        from .confidence import ConfidenceEnforcer

        bead_storage = BeadStorage(self.config.project_root / ".true_vkg" / "beads")
        enforcer = ConfidenceEnforcer()

        for bead_id in pool.beads:
            if bead_id in pool.verdicts:
                continue

            bead = bead_storage.load(bead_id)
            if not bead:
                continue

            # Create verdict from work_state
            work_state = bead.work_state
            confidence_str = work_state.get("verifier_confidence", "uncertain")
            confidence = {
                "confirmed": VerdictConfidence.CONFIRMED,
                "likely": VerdictConfidence.LIKELY,
                "uncertain": VerdictConfidence.UNCERTAIN,
                "rejected": VerdictConfidence.REJECTED,
            }.get(confidence_str, VerdictConfidence.UNCERTAIN)

            verdict = Verdict(
                bead_id=bead_id,
                confidence=confidence,
                rationale=f"Attacker: {work_state.get('attacker_summary', 'N/A')} | "
                         f"Defender: {work_state.get('defender_summary', 'N/A')}",
            )

            # Enforce rules
            verdict = enforcer.enforce(verdict)

            self.manager.record_verdict(pool.id, verdict)

        return PhaseResult(
            success=True,
            phase=LoopPhase.VERIFY,
            message=f"Collected {len(pool.verdicts)} verdicts",
        )
