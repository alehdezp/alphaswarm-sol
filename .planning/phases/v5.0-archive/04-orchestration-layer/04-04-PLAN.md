"""Confidence enforcement rules for orchestration.

ORCH-09: "no likely/confirmed without evidence + verification"
ORCH-10: "default missing context to uncertain bucket"

This module enforces quality standards on verdicts.
"""

from dataclasses import dataclass
from typing import Optional, List, Dict, Any
from enum import Enum

from .schemas import (
    Verdict, VerdictConfidence, EvidencePacket, EvidenceItem,
    DebateRecord, DebateClaim,
)


class ConfidenceRequirement(Enum):
    """Requirements for each confidence level."""
    CONFIRMED = "confirmed"  # Requires: test OR multi-agent consensus
    LIKELY = "likely"        # Requires: evidence + attacker/defender analysis
    UNCERTAIN = "uncertain"  # Default bucket, no requirements
    REJECTED = "rejected"    # Requires: defender rebuttal with evidence


@dataclass
class ValidationResult:
    """Result of confidence validation."""
    valid: bool
    original_confidence: VerdictConfidence
    enforced_confidence: VerdictConfidence
    reason: str
    missing: List[str] = None  # What's missing


class ConfidenceEnforcer:
    """Enforces confidence rules on verdicts.

    Rules from 04-CONTEXT.md:
    - CONFIRMED: Test passes OR multi-agent consensus
    - LIKELY: Strong evidence from attacker + defender analysis
    - UNCERTAIN: Missing context (default bucket)
    - REJECTED: Defender rebuttal accepted

    Per decision: "Debate always human-flagged"
    """

    # Minimum evidence items for each level
    EVIDENCE_REQUIREMENTS = {
        VerdictConfidence.CONFIRMED: 2,  # At least 2 strong evidence items
        VerdictConfidence.LIKELY: 1,     # At least 1 evidence item
        VerdictConfidence.UNCERTAIN: 0,   # No requirement
        VerdictConfidence.REJECTED: 1,    # Need evidence for rejection
    }

    def validate(self, verdict: Verdict) -> ValidationResult:
        """Validate verdict meets confidence requirements.

        May downgrade confidence if requirements not met.
        Never upgrades confidence (that's for test results).
        """
        original = verdict.confidence
        reasons = []
        missing = []

        # Rule: CONFIRMED requires test OR consensus
        if original == VerdictConfidence.CONFIRMED:
            if not self._has_test_verification(verdict):
                if not self._has_multi_agent_consensus(verdict):
                    missing.append("test_verification or multi_agent_consensus")
                    reasons.append("CONFIRMED requires test or consensus")

        # Rule: LIKELY requires evidence + analysis
        if original == VerdictConfidence.LIKELY:
            evidence_count = len(verdict.evidence_summary)
            if evidence_count < self.EVIDENCE_REQUIREMENTS[VerdictConfidence.LIKELY]:
                missing.append("evidence_items")
                reasons.append(f"LIKELY requires evidence (found {evidence_count})")

            if not self._has_analysis(verdict):
                missing.append("attacker_or_defender_analysis")
                reasons.append("LIKELY requires attacker/defender analysis")

        # Rule: REJECTED requires rebuttal evidence
        if original == VerdictConfidence.REJECTED:
            if not self._has_defender_rebuttal(verdict):
                missing.append("defender_rebuttal")
                reasons.append("REJECTED requires defender rebuttal")

        # Determine enforced confidence
        if missing:
            # Downgrade to uncertain if requirements not met
            enforced = VerdictConfidence.UNCERTAIN
            return ValidationResult(
                valid=False,
                original_confidence=original,
                enforced_confidence=enforced,
                reason="; ".join(reasons),
                missing=missing,
            )

        return ValidationResult(
            valid=True,
            original_confidence=original,
            enforced_confidence=original,
            reason="Requirements met",
        )

    def enforce(self, verdict: Verdict) -> Verdict:
        """Enforce confidence rules, returning corrected verdict.

        Modifies verdict in place AND returns it.
        """
        result = self.validate(verdict)

        if not result.valid:
            verdict.confidence = result.enforced_confidence
            # Append enforcement note to rationale
            verdict.rationale = (
                f"{verdict.rationale} [ENFORCED: {result.reason}]"
            )

        # Rule: Debate outcomes always human-flagged
        if verdict.debate and verdict.debate.verifier_synthesis:
            verdict.human_flag = True

        return verdict

    def bucket_uncertain(self, verdict: Verdict) -> Verdict:
        """Apply ORCH-10: Missing context -> uncertain.

        Call this when context is incomplete.
        """
        verdict.confidence = VerdictConfidence.UNCERTAIN
        verdict.rationale = f"{verdict.rationale} [BUCKETED: missing context]"
        return verdict

    def elevate_on_test(self, verdict: Verdict, test_passed: bool) -> Verdict:
        """Elevate to CONFIRMED if test passes.

        This is the only way to reach CONFIRMED without multi-agent consensus.
        """
        if test_passed:
            verdict.confidence = VerdictConfidence.CONFIRMED
            verdict.evidence_summary.append("test_passed: true")
            verdict.rationale = f"{verdict.rationale} [ELEVATED: test passed]"
        return verdict

    def _has_test_verification(self, verdict: Verdict) -> bool:
        """Check if verdict has test verification."""
        return any("test_passed" in e for e in verdict.evidence_summary)

    def _has_multi_agent_consensus(self, verdict: Verdict) -> bool:
        """Check if attacker and defender agree (rare)."""
        if not verdict.debate:
            return False
        # Consensus = both agree on outcome
        # In practice, this is rare and requires specific synthesis
        return "consensus" in (verdict.debate.verifier_synthesis or "").lower()

    def _has_analysis(self, verdict: Verdict) -> bool:
        """Check if attacker or defender provided analysis."""
        if verdict.debate:
            return (
                verdict.debate.attacker_claim is not None or
                verdict.debate.defender_claim is not None
            )
        return False

    def _has_defender_rebuttal(self, verdict: Verdict) -> bool:
        """Check if defender provided rebuttal evidence."""
        if not verdict.debate:
            return False
        return (
            verdict.debate.defender_claim is not None and
            len(verdict.debate.defender_claim.evidence) > 0
        )


def enforce_confidence(verdict: Verdict) -> Verdict:
    """Convenience function for confidence enforcement."""
    return ConfidenceEnforcer().enforce(verdict)


def bucket_on_missing_context(verdict: Verdict) -> Verdict:
    """Convenience function for ORCH-10."""
    return ConfidenceEnforcer().bucket_uncertain(verdict)


def elevate_on_test_pass(verdict: Verdict, test_passed: bool) -> Verdict:
    """Convenience function for test elevation."""
    return ConfidenceEnforcer().elevate_on_test(verdict, test_passed)

"""Orchestration rules and constraints.

Central location for all orchestration policies.
These are enforced by the router and loop.
"""

from dataclasses import dataclass
from typing import List, Optional, Dict, Any
from enum import Enum

from .schemas import Pool, PoolStatus, Verdict, VerdictConfidence


@dataclass
class RuleViolation:
    """A rule violation."""
    rule_id: str
    message: str
    severity: str  # "error", "warning"
    fix_hint: Optional[str] = None


class OrchestrationRules:
    """Orchestration rules from PHILOSOPHY.md and 04-CONTEXT.md.

    Rules:
    1. No CONFIRMED without test OR multi-agent consensus
    2. No LIKELY without evidence
    3. Debate outcomes always human-flagged
    4. Missing context -> uncertain
    5. Phase order is fixed (no skipping)
    6. Batch by role (attackers, then defenders, then verifiers)
    """

    @staticmethod
    def check_verdict_rules(verdict: Verdict) -> List[RuleViolation]:
        """Check verdict against rules."""
        violations = []

        # Rule 1: CONFIRMED requirements
        if verdict.confidence == VerdictConfidence.CONFIRMED:
            has_test = any("test_passed" in e for e in verdict.evidence_summary)
            has_consensus = verdict.debate and "consensus" in (
                verdict.debate.verifier_synthesis or ""
            ).lower()

            if not has_test and not has_consensus:
                violations.append(RuleViolation(
                    rule_id="ORCH-09-CONFIRMED",
                    message="CONFIRMED verdict without test or consensus",
                    severity="error",
                    fix_hint="Run verification test or require multi-agent agreement",
                ))

        # Rule 2: LIKELY requirements
        if verdict.confidence == VerdictConfidence.LIKELY:
            if not verdict.evidence_summary:
                violations.append(RuleViolation(
                    rule_id="ORCH-09-LIKELY",
                    message="LIKELY verdict without evidence",
                    severity="error",
                    fix_hint="Add evidence items to verdict",
                ))

        # Rule 3: Debate -> human flag
        if verdict.debate and verdict.debate.verifier_synthesis:
            if not verdict.human_flag:
                violations.append(RuleViolation(
                    rule_id="ORCH-DEBATE-HUMAN",
                    message="Debate outcome not flagged for human",
                    severity="error",
                    fix_hint="Set human_flag=True",
                ))

        return violations

    @staticmethod
    def check_pool_rules(pool: Pool) -> List[RuleViolation]:
        """Check pool against rules."""
        violations = []

        # Rule 5: Phase order
        if pool.phases_complete:
            expected_order = ["intake", "context", "beads", "execute", "verify", "integrate"]
            for i, phase in enumerate(pool.phases_complete):
                if i < len(expected_order) and phase != expected_order[i]:
                    violations.append(RuleViolation(
                        rule_id="ORCH-07-ORDER",
                        message=f"Phase {phase} out of order (expected {expected_order[i]})",
                        severity="warning",
                    ))

        return violations

    @staticmethod
    def can_advance_phase(pool: Pool) -> tuple[bool, str]:
        """Check if pool can advance to next phase."""
        # Check current phase completion
        status_requirements = {
            PoolStatus.PENDING: lambda p: True,  # Always can start
            PoolStatus.INTAKE: lambda p: p.metadata.get("graph_built", False),
            PoolStatus.CONTEXT: lambda p: p.metadata.get("patterns_detected", False),
            PoolStatus.BEADS: lambda p: len(p.beads) > 0,
            PoolStatus.EXECUTE: lambda p: all(
                p.metadata.get(f"{role}_processed", []) for role in ["attacker", "defender", "verifier"]
            ),
            PoolStatus.VERIFY: lambda p: all(
                bead_id in p.verdicts for bead_id in p.beads
            ),
            PoolStatus.INTEGRATE: lambda p: p.metadata.get("report_generated", False),
        }

        requirement = status_requirements.get(pool.status)
        if requirement is None:
            return False, f"Unknown status: {pool.status}"

        if not requirement(pool):
            return False, f"Phase {pool.status.value} requirements not met"

        return True, "Ready to advance"


@dataclass
class BatchingPolicy:
    """Policy for agent batching."""
    role_order: List[str] = None  # Order to spawn agents
    parallel_within_role: bool = True  # Can spawn multiple of same role
    max_parallel: int = 5  # Max parallel agents per role

    def __post_init__(self):
        if self.role_order is None:
            self.role_order = ["attacker", "defender", "verifier"]


# Default batching policy from 04-CONTEXT.md decision
DEFAULT_BATCHING = BatchingPolicy(
    role_order=["attacker", "defender", "verifier"],
    parallel_within_role=True,
    max_parallel=5,
)

import pytest
from datetime import datetime

from true_vkg.orchestration.schemas import (
    Verdict, VerdictConfidence, DebateRecord, DebateClaim, EvidenceItem,

class TestConfidenceEnforcement:
    """Test ORCH-09: No likely/confirmed without evidence."""

    def test_confirmed_without_test_downgraded(self):
        """CONFIRMED without test -> uncertain."""
        verdict = Verdict(
            bead_id="VKG-001",
            confidence=VerdictConfidence.CONFIRMED,
            rationale="Pattern matched strongly",
        )

        enforcer = ConfidenceEnforcer()
        result = enforcer.validate(verdict)

        assert not result.valid
        assert result.enforced_confidence == VerdictConfidence.UNCERTAIN
        assert "test" in result.reason.lower() or "consensus" in result.reason.lower()

    def test_confirmed_with_test_passes(self):
        """CONFIRMED with test passes validation."""
        verdict = Verdict(
            bead_id="VKG-001",
            confidence=VerdictConfidence.CONFIRMED,
            rationale="Verified",
            evidence_summary=["test_passed: true", "exploit successful"],
        )

        enforcer = ConfidenceEnforcer()
        result = enforcer.validate(verdict)

        assert result.valid
        assert result.enforced_confidence == VerdictConfidence.CONFIRMED

    def test_likely_without_evidence_downgraded(self):
        """LIKELY without evidence -> uncertain."""
        verdict = Verdict(
            bead_id="VKG-001",
            confidence=VerdictConfidence.LIKELY,
            rationale="Looks suspicious",
            evidence_summary=[],  # No evidence
        )

        enforcer = ConfidenceEnforcer()
        result = enforcer.validate(verdict)
