"""Orchestration CLI commands (ORCH-06).

Provides CLI interface for:
- Starting audits
- Checking pool status
- Resuming from checkpoints
- Listing pools and beads
"""

from pathlib import Path
from typing import Optional
import json

import typer
import structlog

from true_vkg.orchestration.pool import PoolManager, PoolStorage
from true_vkg.orchestration.schemas import Scope, PoolStatus, VerdictConfidence
from true_vkg.orchestration.loop import ExecutionLoop, LoopConfig
from true_vkg.orchestration.handlers import create_default_handlers, HandlerConfig

logger = structlog.get_logger()

orchestrate_app = typer.Typer(help="Orchestration commands for multi-agent audits")


def get_manager(project: Optional[Path] = None) -> PoolManager:
    """Get pool manager for project."""
    project_root = project or Path.cwd()
    storage = PoolStorage(project_root / ".vkg" / "pools")
    return PoolManager(storage)


@orchestrate_app.command("start")
def start_audit(
    path: str = typer.Argument(..., help="Path to contracts to audit"),
    focus: Optional[str] = typer.Option(
        None,
        "--focus",
        "-f",
        help="Comma-separated focus areas (reentrancy, access_control, etc.)",
    ),
    project: Optional[str] = typer.Option(
        None,
        "--project",
        "-p",
        help="Project root directory",
    ),
    auto_advance: bool = typer.Option(
        True,
        "--auto-advance/--no-auto-advance",
        help="Automatically advance through phases",
    ),
    verbose: bool = typer.Option(
        False,
        "--verbose",
        "-v",
        help="Show detailed output",
    ),
) -> None:
    """Start a new orchestrated audit.

    Creates a pool, builds the knowledge graph, runs pattern detection,
    creates beads, and spawns multi-agent verification.

    Examples:
        # Start audit on contracts directory
        vkg orchestrate start contracts/

        # Start audit with focus areas
        vkg orchestrate start contracts/ --focus reentrancy,access_control

        # Start audit with verbose output
        vkg orchestrate start contracts/ -v
    """
    try:
        project_root = Path(project) if project else Path.cwd()
        target_path = Path(path)

        # Resolve relative paths
        if not target_path.is_absolute():
            target_path = project_root / target_path

        if not target_path.exists():
            typer.echo(f"Error: Path not found: {target_path}", err=True)
            raise typer.Exit(code=1)

        # Parse focus areas
        focus_areas = []
        if focus:
            focus_areas = [f.strip() for f in focus.split(",")]

        # Get files to audit
        if target_path.is_file():
            files = [str(target_path.relative_to(project_root))]
        else:
            files = [
                str(f.relative_to(project_root))
                for f in target_path.rglob("*.sol")
                if "node_modules" not in str(f) and "lib" not in str(f)
            ]

        if not files:
            typer.echo("No Solidity files found", err=True)
            raise typer.Exit(code=1)

        typer.echo(f"Starting audit on {len(files)} file(s)")
        if verbose:
            for f in files[:5]:
                typer.echo(f"  - {f}")
            if len(files) > 5:
                typer.echo(f"  ... and {len(files) - 5} more")

        # Create scope and pool
        manager = get_manager(project_root)
        scope = Scope(
            id=f"audit-{target_path.stem}",
            files=files,
            focus_areas=focus_areas,
        )
        pool = manager.create_pool(scope, initiated_by="cli:orchestrate:start")

        typer.echo(f"Created pool: {pool.id}")

        # Create and run loop
        config = HandlerConfig(
            project_root=project_root,
            verbose=verbose,
        )
        handlers = create_default_handlers(manager, config)

        loop = ExecutionLoop(
            manager,
            LoopConfig(
                auto_advance=auto_advance,
                pause_on_human_flag=True,
                verbose=verbose,
            ),
        )

        for action, handler in handlers.items():
            loop.register_handler(action, handler)

        typer.echo("Running execution loop...")
        result = loop.run(pool.id)

        # Display result
        if result.checkpoint:
            typer.echo("")
            typer.echo("=" * 60)
            typer.echo("CHECKPOINT: Human review required")
            typer.echo("=" * 60)
            typer.echo(f"Phase: {result.phase.value}")
            typer.echo(f"Message: {result.message}")
            if result.artifacts and "beads_for_review" in result.artifacts:
                beads = result.artifacts["beads_for_review"]
                typer.echo(f"Beads for review: {len(beads)}")
                for bead_id in beads[:5]:
                    typer.echo(f"  - {bead_id}")
            typer.echo("")
            typer.echo(f"Resume with: vkg orchestrate resume {pool.id}")
        elif result.success:
            typer.echo("")
            typer.echo("=" * 60)
            typer.echo("AUDIT COMPLETE")
            typer.echo("=" * 60)
            _print_pool_summary(manager.storage.load(pool.id))
        else:
            typer.echo(f"Audit failed: {result.message}", err=True)
            raise typer.Exit(code=1)

    except typer.Exit:
        raise
    except Exception as e:
        logger.error("start_audit_failed", error=str(e))
        typer.echo(f"Error: {e}", err=True)
        raise typer.Exit(code=1)


@orchestrate_app.command("resume")
def resume_audit(
    pool_id: str = typer.Argument(..., help="Pool ID to resume"),
    project: Optional[str] = typer.Option(None, "--project", "-p"),
    verbose: bool = typer.Option(False, "--verbose", "-v"),
) -> None:
    """Resume an audit from checkpoint.

    Continues execution from where the pool paused (usually for human review).

    Example:
        vkg orchestrate resume audit-vault-2026-01-20
    """
    try:
        project_root = Path(project) if project else Path.cwd()
        manager = get_manager(project_root)

        pool = manager.storage.load(pool_id)
        if pool is None:
            typer.echo(f"Pool not found: {pool_id}", err=True)
            raise typer.Exit(code=1)

        typer.echo(f"Resuming pool: {pool_id}")
        typer.echo(f"Current phase: {pool.status.value}")

        # Create and run loop
        config = HandlerConfig(project_root=project_root, verbose=verbose)
        handlers = create_default_handlers(manager, config)

        loop = ExecutionLoop(
            manager,
            LoopConfig(auto_advance=True, pause_on_human_flag=True, verbose=verbose),
        )

        for action, handler in handlers.items():
            loop.register_handler(action, handler)

        result = loop.resume(pool_id)

        if result.checkpoint:
            typer.echo(f"Paused at checkpoint: {result.message}")
            typer.echo(f"Resume with: vkg orchestrate resume {pool_id}")
        elif result.success:
            typer.echo("Audit complete")
            _print_pool_summary(manager.storage.load(pool_id))
        else:
            typer.echo(f"Resume failed: {result.message}", err=True)
            raise typer.Exit(code=1)

    except typer.Exit:
        raise
    except Exception as e:
        logger.error("resume_audit_failed", error=str(e))
        typer.echo(f"Error: {e}", err=True)
        raise typer.Exit(code=1)


@orchestrate_app.command("status")
def pool_status(
    pool_id: str = typer.Argument(..., help="Pool ID to check"),
    project: Optional[str] = typer.Option(None, "--project", "-p"),
    json_output: bool = typer.Option(False, "--json", help="Output as JSON"),
) -> None:
    """Show status of an audit pool.

    Example:
        vkg orchestrate status audit-vault-2026-01-20
    """
    try:
        project_root = Path(project) if project else Path.cwd()
        manager = get_manager(project_root)

        pool = manager.storage.load(pool_id)
        if pool is None:
            typer.echo(f"Pool not found: {pool_id}", err=True)
            raise typer.Exit(code=1)

        if json_output:
            status = {
                "id": pool.id,
                "status": pool.status.value,
                "beads": len(pool.beads),
                "verdicts": len(pool.verdicts),
                "phases_complete": pool.phases_complete,
                "human_flagged": len([
                    v for v in pool.verdicts.values() if v.human_flag
                ]),
            }
            typer.echo(json.dumps(status, indent=2))
        else:
            _print_pool_summary(pool)

    except typer.Exit:
        raise
    except Exception as e:
        typer.echo(f"Error: {e}", err=True)
        raise typer.Exit(code=1)


@orchestrate_app.command("list")
def list_pools(
    project: Optional[str] = typer.Option(None, "--project", "-p"),
    status: Optional[str] = typer.Option(None, "--status", "-s", help="Filter by status"),
    json_output: bool = typer.Option(False, "--json"),
) -> None:
    """List all audit pools.

    Example:
        vkg orchestrate list
        vkg orchestrate list --status execute
    """
    try:
        project_root = Path(project) if project else Path.cwd()
        manager = get_manager(project_root)

        status_filter = None
        if status:
            try:
                status_filter = PoolStatus(status)
            except ValueError:
                typer.echo(f"Invalid status: {status}", err=True)
                typer.echo(f"Valid: {[s.value for s in PoolStatus]}")
                raise typer.Exit(code=1)

        pools = manager.storage.list_pools(status=status_filter)

        if json_output:
            data = [
                {
                    "id": p.id,
                    "status": p.status.value,
                    "beads": len(p.beads),
                    "created": str(p.created),
                }
                for p in pools
            ]
            typer.echo(json.dumps(data, indent=2))
        else:
            if not pools:
                typer.echo("No pools found")
                return

            typer.echo(f"Found {len(pools)} pool(s):")
            typer.echo("")
            for p in pools:
                human_flagged = len([v for v in p.verdicts.values() if v.human_flag])
                typer.echo(f"  {p.id}")
                typer.echo(f"    Status: {p.status.value}")
                typer.echo(f"    Beads: {len(p.beads)}")
                typer.echo(f"    Human flagged: {human_flagged}")
                typer.echo("")

    except typer.Exit:
        raise
    except Exception as e:
        typer.echo(f"Error: {e}", err=True)
        raise typer.Exit(code=1)


@orchestrate_app.command("beads")
def list_beads(
    pool_id: str = typer.Argument(..., help="Pool ID"),
    project: Optional[str] = typer.Option(None, "--project", "-p"),
    pending: bool = typer.Option(False, "--pending", help="Show only pending beads"),
    human_flagged: bool = typer.Option(False, "--human-flagged", help="Show human-flagged only"),
) -> None:
    """List beads in a pool.

    Example:
        vkg orchestrate beads audit-vault-2026-01-20
        vkg orchestrate beads audit-vault-2026-01-20 --human-flagged
    """
    try:
        project_root = Path(project) if project else Path.cwd()
        manager = get_manager(project_root)

        pool = manager.storage.load(pool_id)
        if pool is None:
            typer.echo(f"Pool not found: {pool_id}", err=True)
            raise typer.Exit(code=1)

        for bead_id in pool.beads:
            verdict = pool.verdicts.get(bead_id)

            if pending and verdict is not None:
                continue
            if human_flagged and (verdict is None or not verdict.human_flag):
                continue

            if verdict:
                typer.echo(f"  {bead_id}: {verdict.confidence.value}")
                if verdict.human_flag:
                    typer.echo(f"    [HUMAN REVIEW REQUIRED]")
                typer.echo(f"    {verdict.rationale[:60]}...")
            else:
                typer.echo(f"  {bead_id}: pending")

    except typer.Exit:
        raise
    except Exception as e:
        typer.echo(f"Error: {e}", err=True)
        raise typer.Exit(code=1)


def _print_pool_summary(pool) -> None:
    """Print pool summary."""
    typer.echo(f"Pool: {pool.id}")
    typer.echo(f"Status: {pool.status.value}")
    typer.echo(f"Beads: {len(pool.beads)}")
    typer.echo(f"Phases complete: {', '.join(pool.phases_complete) if pool.phases_complete else 'none'}")
    typer.echo("")

    if pool.verdicts:
        typer.echo("Verdicts:")
        confidence_counts = {c: 0 for c in VerdictConfidence}
        human_flagged = 0
        for v in pool.verdicts.values():
            confidence_counts[v.confidence] += 1
            if v.human_flag:
                human_flagged += 1

        for conf, count in confidence_counts.items():
            if count > 0:
                typer.echo(f"  {conf.value}: {count}")

        if human_flagged > 0:
            typer.echo(f"  Human flagged: {human_flagged}")
