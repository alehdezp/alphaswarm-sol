from dataclasses import dataclass, field
from enum import Enum
from typing import List, Dict, Optional, Any
from datetime import datetime

class PoolStatus(Enum):
    PENDING = "pending"
    INTAKE = "intake"
    CONTEXT = "context"
    BEADS = "beads"
    EXECUTE = "execute"
    VERIFY = "verify"
    INTEGRATE = "integrate"
    COMPLETE = "complete"
    FAILED = "failed"

class VerdictConfidence(Enum):
    CONFIRMED = "confirmed"      # Test OR multi-agent consensus
    LIKELY = "likely"            # Strong evidence, needs verification
    UNCERTAIN = "uncertain"      # Missing context (default bucket)
    REJECTED = "rejected"        # Disproven by defender

@dataclass
class Scope:
    """Defines audit scope - what to analyze."""
    id: str
    files: List[str]
    contracts: List[str] = field(default_factory=list)
    focus_areas: List[str] = field(default_factory=list)  # e.g., ["reentrancy", "access_control"]
    exclude_patterns: List[str] = field(default_factory=list)
    metadata: Dict[str, Any] = field(default_factory=dict)

@dataclass
class EvidenceItem:
    """Single piece of evidence."""
    type: str  # "behavioral_signature", "code_location", "test_result", "agent_analysis"
    value: str
    location: Optional[str] = None  # file:line
    confidence: float = 0.0
    source: Optional[str] = None  # Which agent/tool produced this

@dataclass
class EvidencePacket:
    """Collection of evidence for a finding."""
    bead_id: str
    items: List[EvidenceItem] = field(default_factory=list)
    behavioral_signature: Optional[str] = None
    protocol_context: Optional[str] = None  # From Phase 3
    graph_context: Optional[Dict[str, Any]] = None

@dataclass
class DebateClaim:
    """A claim in the debate protocol."""
    agent_role: str  # "attacker" or "defender"
    claim: str
    evidence: List[EvidenceItem] = field(default_factory=list)
    timestamp: datetime = field(default_factory=datetime.now)

@dataclass
class DebateRecord:
    """Record of attacker/defender debate."""
    attacker_claim: Optional[DebateClaim] = None
    defender_claim: Optional[DebateClaim] = None
    rebuttals: List[DebateClaim] = field(default_factory=list)
    verifier_synthesis: Optional[str] = None

@dataclass
class Verdict:
    """Final verdict on a bead."""
    bead_id: str
    confidence: VerdictConfidence
    rationale: str
    dissenting_opinion: Optional[str] = None
    human_flag: bool = False  # Always True for debate outcomes
    evidence_summary: List[str] = field(default_factory=list)
    debate: Optional[DebateRecord] = None
    created_at: datetime = field(default_factory=datetime.now)

@dataclass
class Pool:
    """Batch container for audit wave (renamed from convoy)."""
    id: str
    status: PoolStatus = PoolStatus.PENDING
    created: datetime = field(default_factory=datetime.now)
    beads: List[str] = field(default_factory=list)  # Bead IDs
    verdicts: Dict[str, Verdict] = field(default_factory=dict)  # bead_id -> Verdict
    scope: Optional[Scope] = None
    metadata: Dict[str, Any] = field(default_factory=dict)
    phases_complete: List[str] = field(default_factory=list)
    current_phase: str = "pending"
    initiated_by: Optional[str] = None  # e.g., "/vkg:audit"

"""Tests for orchestration schemas."""
import pytest
from datetime import datetime
from pathlib import Path
import tempfile

from true_vkg.orchestration.schemas import (
    Pool, PoolStatus, Verdict, VerdictConfidence,
    Scope, EvidencePacket, EvidenceItem, DebateRecord, DebateClaim,
    to_yaml, pool_from_yaml,
)
from true_vkg.orchestration.pool import PoolManager, PoolStorage


class TestSchemas:
    """Test canonical schema definitions."""

    def test_pool_creation(self):
        """Pool can be created with required fields."""
        pool = Pool(id="test-pool-001")
        assert pool.id == "test-pool-001"
        assert pool.status == PoolStatus.PENDING
        assert pool.beads == []
        assert pool.current_phase == "pending"

    def test_pool_status_transitions(self):
        """Pool status enum has all required phases."""
        statuses = [s.value for s in PoolStatus]
        assert "intake" in statuses
        assert "context" in statuses
        assert "beads" in statuses
        assert "execute" in statuses
        assert "verify" in statuses
        assert "integrate" in statuses
        assert "complete" in statuses

    def test_verdict_requires_rationale(self):
        """Verdict must have rationale for confidence level."""
        verdict = Verdict(
            bead_id="VKG-001",
            confidence=VerdictConfidence.LIKELY,
            rationale="Strong behavioral signature match",
        )
        assert verdict.rationale
        assert verdict.human_flag == False  # Default

    def test_verdict_confidence_levels(self):
        """Verdict confidence matches bucket system."""
        levels = [v.value for v in VerdictConfidence]
        assert "confirmed" in levels
        assert "likely" in levels
        assert "uncertain" in levels
        assert "rejected" in levels

    def test_scope_definition(self):
        """Scope captures audit target."""
        scope = Scope(
            id="scope-vault",
            files=["src/Vault.sol"],
            contracts=["Vault"],
            focus_areas=["reentrancy", "access_control"],
        )
        assert len(scope.files) == 1
        assert "reentrancy" in scope.focus_areas

    def test_evidence_packet_structure(self):
        """Evidence packet links evidence to bead."""
        evidence = EvidenceItem(
            type="behavioral_signature",
            value="R:bal→X:out→W:bal",
            location="Vault.sol:142",
            confidence=0.9,
            source="pattern_engine",
        )
        packet = EvidencePacket(
            bead_id="VKG-001",
            items=[evidence],
            behavioral_signature="R:bal→X:out→W:bal",
        )
        assert packet.bead_id == "VKG-001"
        assert len(packet.items) == 1
        assert packet.items[0].confidence == 0.9


class TestYAMLSerialization:
    """Test YAML serialization for human readability."""

    def test_pool_to_yaml(self):
        """Pool serializes to readable YAML."""
        pool = Pool(
            id="audit-vault-2026-01-20",
            status=PoolStatus.EXECUTE,
            beads=["VKG-001", "VKG-002"],
        )
        yaml_str = to_yaml(pool)
        assert "audit-vault-2026-01-20" in yaml_str
        assert "execute" in yaml_str
        assert "VKG-001" in yaml_str

    def test_pool_roundtrip(self):
        """Pool survives YAML roundtrip."""
        original = Pool(
            id="test-pool",
            status=PoolStatus.VERIFY,
            beads=["VKG-001"],
            phases_complete=["intake", "context", "beads", "execute"],
        )
        yaml_str = to_yaml(original)
        restored = pool_from_yaml(yaml_str)
        assert restored.id == original.id
        assert restored.status == original.status
        assert restored.beads == original.beads


class TestPoolStorage:
    """Test pool storage operations."""

    def test_save_and_load(self):
        """Pool saves to and loads from disk."""
        with tempfile.TemporaryDirectory() as tmpdir:
            storage = PoolStorage(Path(tmpdir))
            pool = Pool(id="test-save", beads=["VKG-001"])

            saved_path = storage.save(pool)
            assert saved_path.exists()

            loaded = storage.load("test-save")
            assert loaded is not None
            assert loaded.id == "test-save"
            assert loaded.beads == ["VKG-001"]

    def test_list_pools_by_status(self):
        """Can filter pools by status."""
        with tempfile.TemporaryDirectory() as tmpdir:
            storage = PoolStorage(Path(tmpdir))

            pool1 = Pool(id="pool-1", status=PoolStatus.EXECUTE)
            pool2 = Pool(id="pool-2", status=PoolStatus.COMPLETE)
            storage.save(pool1)
            storage.save(pool2)

            executing = storage.list_pools(status=PoolStatus.EXECUTE)
            assert len(executing) == 1
            assert executing[0].id == "pool-1"


class TestPoolManager:
    """Test high-level pool operations."""

            updated = storage.load(pool.id)
            assert "VKG-001" in updated.beads
            assert "VKG-002" in updated.beads

        """Can record verdict for bead."""
        with tempfile.TemporaryDirectory() as tmpdir:
            storage = PoolStorage(Path(tmpdir))
            manager = PoolManager(storage)

            scope = Scope(id="test", files=["Test.sol"])
            pool = manager.create_pool(scope)
            manager.add_bead(pool.id, "VKG-001")

            verdict = Verdict(
                bead_id="VKG-001",
                confidence=VerdictConfidence.LIKELY,
                rationale="Behavioral signature matches reentrancy",
                human_flag=True,
            )
            manager.record_verdict(pool.id, verdict)

            updated = storage.load(pool.id)
            assert "VKG-001" in updated.verdicts
            assert updated.verdicts["VKG-001"].confidence == VerdictConfidence.LIKELY
