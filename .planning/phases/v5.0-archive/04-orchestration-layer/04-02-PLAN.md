def save_to_pool(self, bead: VulnerabilityBead, pool_id: str) -> Path:
    """Save bead to pool directory structure.

    Storage path: .vrs/pools/{pool_id}/beads/{bead_id}.yaml

    Args:
        bead: The bead to save
        pool_id: Pool to associate with

    Returns:
        Path to saved bead file
    """
    # Set pool_id on bead
    bead.pool_id = pool_id

    # Determine base path
    pool_path = self.base_path.parent / "pools" / pool_id / "beads"
    pool_path.mkdir(parents=True, exist_ok=True)

    bead_file = pool_path / f"{bead.id}.yaml"
    bead_file.write_text(bead.to_minimal_yaml())

    # Also update main storage
    self.save(bead)

    return bead_file

def load_from_pool(self, bead_id: str, pool_id: str) -> Optional[VulnerabilityBead]:
    """Load bead from pool directory."""
    bead_file = self.base_path.parent / "pools" / pool_id / "beads" / f"{bead_id}.yaml"
    if not bead_file.exists():
        return None

def list_pool_beads(self, pool_id: str) -> List[str]:
    """List all bead IDs in a pool."""
    pool_path = self.base_path.parent / "pools" / pool_id / "beads"
    if not pool_path.exists():
        return []
    return [f.stem for f in pool_path.glob("*.yaml")]

def update_work_state(self, bead_id: str, state: Dict[str, Any], agent: str) -> None:
    """Update bead's work state for resumption (ORCH-08).

    Args:
        bead_id: Bead to update
        state: Arbitrary state dict
        agent: Which agent is updating
    """
    bead = self.load(bead_id)
    if bead:
        bead.work_state = state
        bead.last_agent = agent
        bead.last_updated = datetime.now()
        self.save(bead)

def get_resumable_beads(self, pool_id: str) -> List[VulnerabilityBead]:
    """Get beads that can be resumed (have work_state)."""
    beads = []
    for bead_id in self.list_pool_beads(pool_id):
        bead = self.load(bead_id)
        if bead and bead.work_state:
            beads.append(bead)
    return beads

from pathlib import Path

class TestBeadPoolIntegration:
    """Test bead integration with pools."""

    @pytest.fixture
    def storage(self):
        """Create temp storage."""
        with tempfile.TemporaryDirectory() as tmpdir:
            yield BeadStorage(Path(tmpdir) / "beads")

    @pytest.fixture
    def sample_bead(self):
        """Create sample bead for testing."""
        return VulnerabilityBead(
            id="VKG-001",
            vulnerability_class="reentrancy",
            pattern_id="vm-001",
            function_id="Vault.withdraw",
            severity=Severity.HIGH,
            confidence=0.85,
        )

    def test_bead_has_pool_fields(self, sample_bead):
        """Bead schema includes pool integration fields."""
        assert hasattr(sample_bead, "pool_id")
        assert hasattr(sample_bead, "human_flag")
        assert hasattr(sample_bead, "work_state")
        assert hasattr(sample_bead, "debate_summary")

    def test_save_to_pool(self, storage, sample_bead):
        """Bead can be saved to pool directory."""
        path = storage.save_to_pool(sample_bead, "audit-vault-2026")

        assert path.exists()
        assert "pools" in str(path)
        assert "audit-vault-2026" in str(path)
        assert sample_bead.pool_id == "audit-vault-2026"

    def test_load_from_pool(self, storage, sample_bead):
        """Bead can be loaded from pool."""
        storage.save_to_pool(sample_bead, "test-pool")
