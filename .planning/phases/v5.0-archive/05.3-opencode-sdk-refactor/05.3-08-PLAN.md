3. Update execute_agent method with full flow:
   ```python
   async def execute_agent(self, config, messages, task_type: TaskType = TaskType.ANALYZE):
       # 1. Use ModelSelector to pick best model
       profile = TaskProfile(task_type=task_type, ...)
       model = self.selector.select(profile)

       # 2. Route to appropriate runtime
       runtime = self.router.route(config.role, task_type)

       # 3. Execute with selected model
       start_time = time.monotonic()
       response = await runtime.execute(config, messages, model=model)
       latency_ms = int((time.monotonic() - start_time) * 1000)

       # 4. Record feedback
       feedback = TaskFeedback(
           task_id=str(uuid.uuid4()),
           model_id=model,
           task_type=task_type.value,
           success=response.success,
           latency_ms=latency_ms,
           tokens_used=response.usage.total_tokens,
           quality_score=self._estimate_quality(response),
           cost_usd=response.usage.cost_usd,
           timestamp=datetime.now()
       )
       self.feedback_collector.record(feedback)

       return response
   ```
