   async def execute(self, config, messages, task_type="analyze") -> AgentResponse:
       model = self._select_model(task_type, config.role)
       prompt = self._build_prompt(config, messages)
       cmd = self._build_command(model, prompt)  # Returns: ["opencode", "-p", prompt, "--model", model, "-f", "json", "-q"]

       start_time = time.monotonic()
       result = await self._run_subprocess(cmd, self.config.timeout_seconds)
       latency_ms = int((time.monotonic() - start_time) * 1000)

       response = self._parse_response(result, model, latency_ms)
       await self._record_feedback(task_type, config.role, model, response)
       return response

5. Loop prevention (CRITICAL - all three mechanisms):
   ```python
   @dataclass
   class LoopState:
       iteration_count: int = 0          # Incremented per LLM call
       output_hashes: List[str] = field(default_factory=list)  # SHA256 of outputs
       total_tokens_used: int = 0        # Cumulative tokens

   # Constants
   MAX_ITERATIONS = 10                   # Hard limit per task
   MAX_REPEATED_OUTPUTS = 3              # Consecutive identical outputs
   TOKEN_CEILING = 100_000               # Max tokens per task

   def _check_loop_prevention(self, state: LoopState, output: str) -> bool:
       """Check if we should abort due to loop detection.

       Returns True if loop detected (should abort).
       """
       # Check iteration count
       if state.iteration_count >= MAX_ITERATIONS:
           logger.warning(f"Max iterations ({MAX_ITERATIONS}) reached")
           return True

       # Check repeated outputs
       output_hash = hashlib.sha256(output.encode()).hexdigest()[:16]
       if len(state.output_hashes) >= MAX_REPEATED_OUTPUTS:
           if all(h == output_hash for h in state.output_hashes[-MAX_REPEATED_OUTPUTS:]):
               logger.warning(f"Repeated output detected {MAX_REPEATED_OUTPUTS} times")
               return True
       state.output_hashes.append(output_hash)

       # Check token ceiling
       if state.total_tokens_used >= TOKEN_CEILING:
           logger.warning(f"Token ceiling ({TOKEN_CEILING}) exceeded")
           return True

       return False
   ```
