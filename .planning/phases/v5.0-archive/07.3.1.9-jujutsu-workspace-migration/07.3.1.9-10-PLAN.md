### T3: Verify Parallel Workspace Execution

Test that multiple workspaces can run in parallel:
```python
import asyncio
from alphaswarm_sol.orchestration.workspace import WorkspaceManager

async def test_parallel_workspaces():
    """Test parallel workspace creation and isolation."""
    manager = WorkspaceManager()

    # Create 3 workspaces in parallel
    paths = await asyncio.gather(
        asyncio.to_thread(manager.allocate, "pool-1", "agent-1", {}),
        asyncio.to_thread(manager.allocate, "pool-1", "agent-2", {}),
        asyncio.to_thread(manager.allocate, "pool-1", "agent-3", {}),
    )

    # Verify all exist and are isolated
    for i, path in enumerate(paths):
        assert path.exists()
        # Create unique file in each
        (path / f"unique-{i}.txt").write_text(f"workspace-{i}")

    # Verify isolation - files don't leak
    for i, path in enumerate(paths):
        assert (path / f"unique-{i}.txt").exists()
        for j in range(3):
            if i != j:
                assert not (path / f"unique-{j}.txt").exists()

    # Cleanup
    manager.cleanup_pool("pool-1")
```
