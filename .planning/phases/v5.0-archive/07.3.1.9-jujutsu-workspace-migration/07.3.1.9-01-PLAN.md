### T7: Update `_remove_worktree()` Method

**Current (lines 398-431):**
```python
def _remove_worktree(self, key: str) -> bool:
    """Remove a single worktree."""
    if key not in self._metadata:
        return False

    metadata = self._metadata[key]
    worktree_path = Path(metadata.worktree_path)

    # Remove the worktree
    if worktree_path.exists():
        subprocess.run(
            ["git", "worktree", "remove", "--force", str(worktree_path)],
            cwd=self.repo_root,
            capture_output=True,
            text=True,
            check=False,
        )

    # Delete the branch
    branch_name = metadata.branch_name
    if branch_name:
        subprocess.run(
            ["git", "branch", "-D", branch_name],
            cwd=self.repo_root,
            capture_output=True,
            text=True,
            check=False,
        )
```

**New:**
```python
def _remove_workspace(self, key: str) -> bool:
    """Remove a single workspace."""
    if key not in self._metadata:
        return False

    metadata = self._metadata[key]
    workspace_path = Path(metadata.workspace_path)
    workspace_name = metadata.workspace_name

    # Forget the workspace from jj tracking
    if workspace_name:
        subprocess.run(
            ["jj", "workspace", "forget", workspace_name],
            cwd=self.repo_root,
            capture_output=True,
            text=True,
            check=False,
        )

    # Remove the workspace directory
    if workspace_path.exists():
        shutil.rmtree(workspace_path, ignore_errors=True)
```
