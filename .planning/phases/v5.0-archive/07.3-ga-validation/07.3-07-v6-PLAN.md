## Flexible Workflow Integration

This E2E test validates the complete pipeline AND the flexible workflow commands:

### Core Commands Tested

| Command | Test Scenario |
|---------|---------------|
| `/vrs-audit contracts/` | Default full pipeline |
| `/vrs-audit --mode swarm` | Full pipeline with debate |
| `/vrs-audit --mode solo` | Fast mode without debate |
| `/vrs-audit --skip-stages 3,6` | Skip context + debate |
| `/vrs-audit --only-stages 2,4` | Graph + patterns only |
| `/vrs-audit --stop-after 4` | Stop after patterns |
| `/vrs-audit --tier A` | Only Tier A patterns |
| `/vrs-audit --category reentrancy` | Category filtering |

### Flexible Workflow Test Scenarios

**Scenario 1: Full Swarm Mode (Default)**
```bash
/vrs-audit contracts/ --mode swarm
# Expected: All 7 stages complete, verdicts.json exists
```

**Scenario 2: Solo Mode (No Debate)**
```bash
/vrs-audit contracts/ --mode solo
# Expected: Stages 1-5,7 complete, no verdicts.json
# Faster execution, still produces report
```

**Scenario 3: Skip Setup (Pre-initialized)**
```bash
/vrs-audit contracts/ --skip-setup
# Expected: Skips Stage 1, uses existing slither install
# Useful for repeated runs
```

**Scenario 4: Pattern Filtering**
```bash
/vrs-audit contracts/ --only-stages 2,4 --tier A
# Expected: Only graph + Tier A pattern matching
# No agents, no debate, no report
```

**Scenario 5: Targeted Investigation**
```bash
/vrs-audit contracts/ --only-stages 2,4,5 --category reentrancy
# Expected: Graph + reentrancy patterns + reentrancy agents only
```

**Scenario 6: Incremental Run**
```bash
/vrs-audit contracts/ --stop-after 4
# Later: /vrs-audit contracts/ --only-stages 5,6,7
# Expected: Can resume from intermediate state
```

### Task 2: Create Comprehensive Findings Comparison Script

**File:** `scripts/compare_findings.py` (Enhanced Version)

This script was created in 07.3-00-v6, but here's an enhanced version for the full pipeline:

```python
#!/usr/bin/env python3
"""
Compare audit findings to external ground truth with comprehensive metrics.

Usage:
    uv run python scripts/compare_findings.py --worktree /path/to/worktree
"""

import argparse
import json
import yaml
from pathlib import Path
from dataclasses import dataclass, field
from typing import List, Optional, Set
import sys


@dataclass
class Vulnerability:
    id: str
    type: str
    severity: str
    function: str
    line: Optional[int] = None
    description: Optional[str] = None


@dataclass
class Finding:
    id: str
    type: str
    function: str
    severity: str = ""
    confidence: float = 0.0
    line: Optional[int] = None


@dataclass
class Metrics:
    true_positives: int
    false_positives: int
    false_negatives: int
    precision: float
    recall: float
    f1_score: float
    tp_list: List[str] = field(default_factory=list)
    fp_list: List[str] = field(default_factory=list)
    fn_list: List[str] = field(default_factory=list)


def load_ground_truth(worktree: Path) -> List[Vulnerability]:
    """Load ground truth from YAML file."""
    gt_file = worktree / "ground-truth.yaml"
    if not gt_file.exists():
        raise FileNotFoundError(f"Ground truth file not found: {gt_file}")

    data = yaml.safe_load(gt_file.read_text())
    vulns = []

    for v in data.get("vulnerabilities", []):
        vulns.append(Vulnerability(
            id=v["id"],
            type=v["type"],
            severity=v["severity"],
            function=v["function"],
            line=v.get("line"),
            description=v.get("description"),
        ))

    return vulns


def load_safe_functions(worktree: Path) -> List[dict]:
    """Load safe functions that should NOT be flagged."""
    gt_file = worktree / "ground-truth.yaml"
    if not gt_file.exists():
        return []

    data = yaml.safe_load(gt_file.read_text())
    return data.get("safe_functions", [])


def load_findings(worktree: Path) -> List[Finding]:
    """Load findings from all available audit outputs."""
    findings = []
    seen_ids: Set[str] = set()

    # Priority order: verdicts > agent-investigations > pattern-matches
    sources = [
        worktree / ".vrs" / "findings" / "verdicts.json",
        worktree / ".vrs" / "findings" / "agent-investigations.json",
        worktree / ".vrs" / "findings" / "pattern-matches.json",
    ]

    for source in sources:
        if source.exists():
            data = json.loads(source.read_text())

            # Handle different formats
            if isinstance(data, list):
                items = data
            elif isinstance(data, dict):
                items = data.get("verdicts", data.get("investigations", data.get("matches", data.get("findings", []))))
            else:
                continue

            for item in items:
                finding_id = item.get("id") or item.get("finding_id") or f"{item.get('pattern_id', 'unknown')}@{item.get('function', 'unknown')}"

                if finding_id in seen_ids:
                    continue
                seen_ids.add(finding_id)

                findings.append(Finding(
                    id=finding_id,
                    type=item.get("pattern_id") or item.get("type") or item.get("pattern", ""),
                    function=item.get("function") or item.get("function_name", ""),
                    severity=item.get("severity", ""),
                    confidence=item.get("confidence", 0.0),
                    line=item.get("line") or (item.get("evidence", {}) or {}).get("line"),
                ))

    return findings


def match_finding_to_vuln(finding: Finding, vuln: Vulnerability) -> bool:
    """Determine if a finding matches a ground truth vulnerability."""
    # Match by function name (case-insensitive, partial match)
    finding_func = finding.function.lower()
    vuln_func = vuln.function.lower()

    if finding_func == vuln_func or vuln_func in finding_func or finding_func in vuln_func:
        # Also check type similarity
        finding_type = finding.type.lower()
        vuln_type = vuln.type.lower()

        # Type keywords mapping
        type_keywords = {
            "reentrancy": ["reentrancy", "reentrant", "re-entry"],
            "access-control": ["access", "auth", "owner", "permission", "control"],
            "oracle": ["oracle", "price", "feed"],
            "logic": ["logic", "invariant", "overflow", "underflow", "math"],
        }

        for category, keywords in type_keywords.items():
            if any(kw in vuln_type for kw in keywords):
                if any(kw in finding_type for kw in keywords):
                    return True

        # Direct function match is enough
        if finding_func == vuln_func:
            return True

    # Match by line number proximity
    if vuln.line and finding.line:
        if abs(finding.line - vuln.line) <= 10:
            return True

    return False


def check_false_positive(finding: Finding, safe_functions: List[dict]) -> bool:
    """Check if a finding is a false positive (flagging a safe function)."""
    for safe in safe_functions:
        if finding.function.lower() == safe.get("function", "").lower():
            return True
    return False


def calculate_metrics(
    ground_truth: List[Vulnerability],
    findings: List[Finding],
    safe_functions: List[dict]
) -> Metrics:
    """Calculate precision, recall, and F1 score."""
    matched_vulns: Set[str] = set()
    matched_findings: Set[str] = set()

    # Find true positives
    for finding in findings:
        for vuln in ground_truth:
            if vuln.id not in matched_vulns and match_finding_to_vuln(finding, vuln):
                matched_vulns.add(vuln.id)
                matched_findings.add(finding.id)
                break

    # Check for false positives (findings that hit safe functions)
    fp_safe = 0
    for finding in findings:
        if check_false_positive(finding, safe_functions):
            fp_safe += 1

    tp = len(matched_vulns)
    fn = len(ground_truth) - tp
    # FP = findings not matched to any vuln + findings hitting safe functions
    fp = (len(findings) - len(matched_findings)) + fp_safe

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0.0

    return Metrics(
        true_positives=tp,
        false_positives=fp,
        false_negatives=fn,
        precision=precision,
        recall=recall,
        f1_score=f1,
        tp_list=[v.id for v in ground_truth if v.id in matched_vulns],
        fp_list=[f.id for f in findings if f.id not in matched_findings],
        fn_list=[v.id for v in ground_truth if v.id not in matched_vulns],
    )


def main():
    parser = argparse.ArgumentParser(description="Compare findings to ground truth")
    parser.add_argument("--worktree", required=True, help="Path to worktree")
    parser.add_argument("--output", help="Output file for metrics JSON")
    parser.add_argument("--verbose", action="store_true", help="Verbose output")
    args = parser.parse_args()

    worktree = Path(args.worktree)

    try:
        ground_truth = load_ground_truth(worktree)
        safe_functions = load_safe_functions(worktree)
        findings = load_findings(worktree)
        metrics = calculate_metrics(ground_truth, findings, safe_functions)

        result = {
            "ground_truth_count": len(ground_truth),
            "findings_count": len(findings),
            "safe_functions_count": len(safe_functions),
            "true_positives": metrics.true_positives,
            "false_positives": metrics.false_positives,
            "false_negatives": metrics.false_negatives,
            "precision": round(metrics.precision, 4),
            "recall": round(metrics.recall, 4),
            "f1_score": round(metrics.f1_score, 4),
            "tp_list": metrics.tp_list,
            "fp_list": metrics.fp_list,
            "fn_list": metrics.fn_list,
        }

        if args.output:
            Path(args.output).write_text(json.dumps(result, indent=2))

        if args.verbose:
            print(json.dumps(result, indent=2))
        else:
            # Summary output
            print(f"Precision: {metrics.precision:.2%}")
            print(f"Recall: {metrics.recall:.2%}")
            print(f"F1 Score: {metrics.f1_score:.2%}")

        # Exit with success if F1 >= 0.7

    except FileNotFoundError as e:

if __name__ == "__main__":
```
