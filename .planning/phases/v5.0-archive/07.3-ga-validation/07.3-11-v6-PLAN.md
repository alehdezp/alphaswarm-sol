```bash
#!/bin/bash
# Run SmartBugs access control suite test

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

echo " SMARTBUGS ACCESS CONTROL SUITE TEST"

"$SCRIPT_DIR/manage_worktrees.sh" create-from-fixture "$WORKTREE_NAME" "smartbugs/access_control"

CONTRACT_COUNT=$(ls "$WORKTREE_PATH/contracts/"*.sol 2>/dev/null | wc -l)

# Step 3: Launch Claude Code

    echo "ERROR: Failed to launch session"
    exit 1
fi

sleep 15

# Step 4: Send audit command
echo "[3/8] Sending /vrs-audit command..."
claude-code-controller send "/vrs-audit contracts/ --mode swarm" --pane="$PANE_ID"

# Step 5: Wait for completion
echo "[4/8] Waiting for completion (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)
claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=90 --timeout="$TIMEOUT" || true

# Step 6: Capture and validate
echo "[5/8] Capturing output..."
claude-code-controller capture --pane="$PANE_ID" --lines=10000 > "$WORKTREE_PATH/audit-transcript.txt"

echo "[6/8] Validating stage outputs..."
"$SCRIPT_DIR/manage_worktrees.sh" validate "$WORKTREE_NAME" all

# Step 7: Compare to ground truth
echo "[7/8] Comparing to ground truth..."
uv run python "$SCRIPT_DIR/compare_findings.py" \
    --worktree "$WORKTREE_PATH" \
    --output "$WORKTREE_PATH/metrics.json"

# Step 8: Detailed analysis
echo "[8/8] Running detailed analysis..."
uv run python "$SCRIPT_DIR/analyze_smartbugs_results.py" \
    --worktree "$WORKTREE_PATH" \
    --output "$WORKTREE_PATH/detailed-analysis.json"

# Display results
echo ""
echo "========================================"
echo " AGGREGATE RESULTS"
echo "========================================"
echo "Duration: ${DURATION}s"
echo "Contracts: $CONTRACT_COUNT"
echo ""
cat "$WORKTREE_PATH/metrics.json"

# Cleanup
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

# Check success
RECALL=$(jq -r '.recall' "$WORKTREE_PATH/metrics.json")
PRECISION=$(jq -r '.precision' "$WORKTREE_PATH/metrics.json")
F1=$(jq -r '.f1_score' "$WORKTREE_PATH/metrics.json")

echo ""
echo "========================================"
echo " SUMMARY"
echo "========================================"
echo "Precision: $PRECISION (target: >= 0.60)"
echo "Recall:    $RECALL (target: >= 0.80)"
echo "F1 Score:  $F1 (target: >= 0.65)"

# Validate against targets
PASS=true
if (( $(echo "$RECALL < 0.80" | bc -l) )); then
    echo "FAIL: Recall below 0.80"
    PASS=false
fi
if (( $(echo "$PRECISION < 0.60" | bc -l) )); then
    echo "FAIL: Precision below 0.60"
    PASS=false
fi

if $PASS; then
    echo ""
    echo "SUCCESS: All metrics meet targets"
    exit 0
else
    echo ""
    echo "FAILURE: Some metrics below targets"
    exit 1
fi
```

## Subtype Analysis Script

**File:** `scripts/analyze_by_subtype.py`

```python
#!/usr/bin/env python3
"""
Analyze detection by vulnerability subtype.

Usage:
    uv run python scripts/analyze_by_subtype.py --worktree /path/to/worktree
"""

import argparse
import json
import yaml
from pathlib import Path
from collections import defaultdict
from typing import Dict, List, Any


def load_ground_truth(worktree: Path) -> Dict[str, List[Dict]]:
    """Load ground truth, organized by subtype."""
    gt_file = worktree / "ground-truth.yaml"
    data = yaml.safe_load(gt_file.read_text())

    by_subtype = defaultdict(list)
    for contract in data.get("contracts", []):
        for vuln in contract.get("vulnerabilities", []):
            vuln["contract_file"] = contract["file"]
            by_subtype[vuln["type"]].append(vuln)

    return dict(by_subtype)


def load_findings(worktree: Path) -> List[Dict]:
    """Load findings from audit output."""
    verdicts_file = worktree / ".vrs" / "findings" / "verdicts.json"
    if verdicts_file.exists():
        return json.loads(verdicts_file.read_text())
    return []


def analyze_by_subtype(
    ground_truth: Dict[str, List[Dict]],
    findings: List[Dict]
) -> Dict[str, Any]:
    """Analyze detection by vulnerability subtype."""
    results = {"by_subtype": {}, "summary": {}}

    total_gt = 0
    total_detected = 0

    for subtype, vulns in ground_truth.items():
        detected = []
        missed = []

        for vuln in vulns:
            found = any(
                f.get("type", "").lower() == subtype.lower() or
                vuln["contract_file"] in f.get("file", "") and
                f.get("function", "").lower() == vuln.get("function", "").lower()
                for f in findings
            )
            if found:
                detected.append(vuln["id"])
            else:
                missed.append(vuln["id"])

        results["by_subtype"][subtype] = {
            "total": len(vulns),
            "detected": len(detected),
            "missed": len(missed),
            "detection_rate": len(detected) / len(vulns) if vulns else 0,
            "detected_ids": detected,
            "missed_ids": missed,
        }

        total_gt += len(vulns)
        total_detected += len(detected)

    results["summary"] = {
        "total_vulnerabilities": total_gt,
        "total_detected": total_detected,
        "overall_recall": total_detected / total_gt if total_gt else 0,
        "subtypes_with_perfect_detection": sum(
            1 for s in results["by_subtype"].values()
            if s["detection_rate"] == 1.0
        ),
    }

    return results


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--worktree", required=True)
    parser.add_argument("--output", required=True)
    args = parser.parse_args()

    worktree = Path(args.worktree)
    ground_truth = load_ground_truth(worktree)
    findings = load_findings(worktree)

    results = analyze_by_subtype(ground_truth, findings)

    Path(args.output).write_text(json.dumps(results, indent=2))
    print(json.dumps(results, indent=2))


if __name__ == "__main__":
    main()
```

## Notes

1. **Ground Truth Integrity:** SmartBugs is an academic benchmark with labeled vulnerabilities confirmed by security researchers. This provides objective ground truth.

2. **Access Control Variants:** The SmartBugs dataset includes multiple access control variants:
   - Missing access control (no modifier on sensitive function)
   - Unprotected selfdestruct
   - tx.origin authentication (phishing vulnerability)
   - Ownership manipulation
   - Unauthorized transfers

3. **Expected Detection:** AlphaSwarm's access control specialist should detect most patterns using:
   - `has_access_gate = false` property
   - `MODIFIES_OWNER` operation without guard
   - `TRANSFERS_VALUE_OUT` without authorization
   - tx.origin usage detection

4. **Parity Wallet Cases:** The Parity wallet bugs (SB-AC-002, SB-AC-003) are historically significant vulnerabilities that caused $30M+ in losses. Detecting these validates real-world relevance.

5. **Safe Contract Validation:** Including contracts with proper OpenZeppelin Ownable or custom modifiers validates that AlphaSwarm doesn't produce false positives on correctly-implemented access control.

6. **Benchmark Comparison:** SmartBugs paper benchmarks include Slither, Mythril, Securify, and others. AlphaSwarm's results can be compared against these published baselines.

7. **Flexible Workflow Reference:** See `FLEXIBLE-WORKFLOW-DESIGN.md` for all available pattern control commands including `--tier` and `--category` filtering.
