## Goal

Validate that Stage 3 of the `/vrs-audit` pipeline correctly performs protocol context research:
1. Exa MCP search is invoked for similar protocol exploits
2. Protocol type is correctly detected (vault, lending, DEX, etc.)
3. Roles, oracles, and economic assumptions are extracted
4. `protocol-pack.yaml` is generated with correct structure

**Test Method:** Real `/vrs-audit` execution via claude-code-controller with Exa MCP validation.

### Task 1: Create Stage-Specific Test Script

**File:** `scripts/test_stage_context.sh`

```bash
#!/bin/bash
# Test Stage 3: Protocol Context Research
#
# Usage:
#   ./scripts/test_stage_context.sh [--timeout 300]

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
MAIN_REPO="$(dirname "$SCRIPT_DIR")"
WORKTREES_BASE="/tmp/vrs-worktrees"
FIXTURE="foundry-vault"
WORKTREE_NAME="wt-context-test-$(date +%s)"
TIMEOUT=${TIMEOUT:-300}

echo "========================================"
echo " STAGE 3: PROTOCOL CONTEXT VALIDATION"
echo "========================================"
echo "Fixture:   $FIXTURE"
echo "Worktree:  $WORKTREE_NAME"
echo "Timeout:   ${TIMEOUT}s"
echo "========================================"

# Step 1: Create worktree from fixture
echo ""
echo "[Step 1/6] Creating worktree from fixture..."
"$SCRIPT_DIR/manage_worktrees.sh" create-from-fixture "$WORKTREE_NAME" "$FIXTURE"
WORKTREE_PATH="$WORKTREES_BASE/$WORKTREE_NAME"

# First, build the graph (required for context generation)
echo ""
echo "[Step 1b/6] Building graph (prerequisite)..."
uv run alphaswarm build-kg "$WORKTREE_PATH/contracts/" --output "$WORKTREE_PATH/.vrs/graphs/" 2>/dev/null || true

# Step 2: Launch Claude Code session via claude-code-controller
echo ""
echo "[Step 2/6] Launching Claude Code session..."
PANE_ID=$(claude-code-controller launch "cd $WORKTREE_PATH && claude --dangerously-skip-permissions" 2>&1 | grep -oP 'pane_id:\s*\K\S+' || echo "")

if [ -z "$PANE_ID" ]; then
    echo "ERROR: Failed to launch Claude Code session"
    "$SCRIPT_DIR/manage_worktrees.sh" cleanup "$WORKTREE_NAME"
    exit 1
fi

echo "Session started: $PANE_ID"

# Wait for Claude Code to initialize
sleep 15

# Step 3: Send context generation command
echo ""
echo "[Step 3/6] Sending context generation command..."
# Ask Claude to generate protocol context pack
claude-code-controller send "Generate a protocol context pack for the contracts in contracts/ directory. Detect the protocol type, extract roles (owner, user), identify any oracles or external dependencies, and save to .vrs/context/protocol-pack.yaml. Use Exa search to find similar protocol vulnerabilities." --pane="$PANE_ID"

# Step 4: Wait for completion
echo ""
echo "[Step 4/6] Waiting for context generation (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)

if ! claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=45 --timeout="$TIMEOUT"; then
    echo "WARNING: Context generation may have timed out"
fi

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))
echo "Duration: ${DURATION}s"

# Step 5: Capture output
echo ""
echo "[Step 5/6] Capturing output..."
TRANSCRIPT="$WORKTREE_PATH/context-transcript.txt"
claude-code-controller capture --pane="$PANE_ID" --lines=3000 > "$TRANSCRIPT"
echo "Transcript saved: $TRANSCRIPT"

# Step 6: Validate context output
echo ""
echo "[Step 6/6] Validating context output..."

VALIDATION_PASSED=true

# Check protocol-pack.yaml exists
if [ -f "$WORKTREE_PATH/.vrs/context/protocol-pack.yaml" ]; then
    echo "  [PASS] protocol-pack.yaml exists"
    CONTEXT_FILE="$WORKTREE_PATH/.vrs/context/protocol-pack.yaml"
else
    echo "  [FAIL] protocol-pack.yaml not found"
    # Check if it was created with different name
    if ls "$WORKTREE_PATH/.vrs/context/"*.yaml 1>/dev/null 2>&1; then
        echo "  [INFO] Found alternate context file(s):"
        ls "$WORKTREE_PATH/.vrs/context/"*.yaml
        CONTEXT_FILE=$(ls "$WORKTREE_PATH/.vrs/context/"*.yaml | head -1)
    else
        VALIDATION_PASSED=false
        CONTEXT_FILE=""
    fi
fi

# Validate context content using Python script
if [ -n "$CONTEXT_FILE" ] && [ -f "$CONTEXT_FILE" ]; then
    echo ""
    echo "  Validating context content..."
    if uv run python "$SCRIPT_DIR/validate_context.py" \
        --context "$CONTEXT_FILE" \
        --expected "$WORKTREE_PATH/expected/context-checks.json" \
        --output "$WORKTREE_PATH/context-validation.json"; then
        echo "  [PASS] Context content validated"
    else
        echo "  [WARN] Context validation had issues (may be partial)"
    fi
fi

# Check transcript for Exa search activity
echo ""
echo "  Checking for Exa search invocation..."
if grep -qi "exa" "$TRANSCRIPT" || grep -qi "web_search" "$TRANSCRIPT" || grep -qi "similar.*protocol" "$TRANSCRIPT"; then
    echo "  [PASS] Exa search appears to have been invoked"
else
    echo "  [WARN] Could not confirm Exa search invocation"
fi

# Cleanup
echo ""
echo "========================================"
echo " STAGE 3 TEST COMPLETE"
echo "========================================"
echo "Duration:    ${DURATION}s"
echo "Worktree:    $WORKTREE_PATH"
echo "Transcript:  $TRANSCRIPT"
echo "Validation:  $($VALIDATION_PASSED && echo 'PASSED' || echo 'PARTIAL')"

# Kill Claude session
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

# Return exit code based on validation
if $VALIDATION_PASSED; then
    exit 0
else
    exit 1
fi
```

### Task 2: Create Context Validation Script

**File:** `scripts/validate_context.py`

```python
#!/usr/bin/env python3
"""
Validate protocol context pack output.

Usage:
    uv run python scripts/validate_context.py --context .vrs/context/protocol-pack.yaml
"""

import argparse
import json
import yaml
from pathlib import Path
from dataclasses import dataclass
from typing import Any, Optional
import sys


@dataclass
class ValidationResult:
    passed: bool
    check_name: str
    expected: Any
    actual: Any
    message: str


def load_context(context_path: Path) -> dict:
    """Load context pack from YAML."""
    content = context_path.read_text()
    return yaml.safe_load(content) or {}


def validate_protocol_type(context: dict) -> ValidationResult:
    """Check if protocol type is detected."""
    protocol_type = context.get("protocol_type") or context.get("type")

    if not protocol_type:
        return ValidationResult(
            passed=False,
            check_name="protocol_type",
            expected="vault|lending|dex|...",
            actual=None,
            message="Protocol type not detected",
        )

    # For foundry-vault fixture, expect "vault" type
    expected_types = ["vault", "defi", "custody", "storage"]
    if protocol_type.lower() in expected_types:
        return ValidationResult(
            passed=True,
            check_name="protocol_type",
            expected=expected_types,
            actual=protocol_type,
            message=f"Protocol type detected: {protocol_type}",
        )

    return ValidationResult(
        passed=True,  # Pass if any type detected
        check_name="protocol_type",
        expected=expected_types,
        actual=protocol_type,
        message=f"Protocol type detected (not vault): {protocol_type}",
    )


def validate_roles(context: dict) -> ValidationResult:
    """Check if roles are extracted."""
    roles = context.get("roles", [])

    if not roles:
        return ValidationResult(
            passed=False,
            check_name="roles",
            expected=["owner", "user"],
            actual=[],
            message="No roles extracted",
        )

    # Normalize role names
    role_names = []
    for role in roles:
        if isinstance(role, dict):
            role_names.append(role.get("name", "").lower())
        else:
            role_names.append(str(role).lower())

    # For foundry-vault, expect owner and user roles
    expected_roles = {"owner", "user", "admin", "depositor"}
    found_expected = set(role_names) & expected_roles

    if found_expected:
        return ValidationResult(
            passed=True,
            check_name="roles",
            expected=list(expected_roles),
            actual=role_names,
            message=f"Found roles: {role_names}",
        )

    return ValidationResult(
        passed=True,  # Pass if any roles found
        check_name="roles",
        expected=list(expected_roles),
        actual=role_names,
        message=f"Found roles (not expected): {role_names}",
    )


def validate_structure(context: dict) -> ValidationResult:
    """Check if context pack has basic required structure."""
    required_fields = ["protocol_type", "roles"]
    optional_fields = ["oracles", "assumptions", "invariants", "economic_model", "similar_exploits"]

    missing_required = [f for f in required_fields if f not in context and f.replace("_", "") not in context]
    has_optional = [f for f in optional_fields if f in context or f.replace("_", "") in context]

    if missing_required:
        # Check for alternate field names
        has_type = any(k in context for k in ["type", "protocol_type", "protocolType"])
        has_roles = any(k in context for k in ["roles", "actors", "participants"])

        if has_type and has_roles:
            return ValidationResult(
                passed=True,
                check_name="structure",
                expected=required_fields,
                actual=list(context.keys()),
                message="Structure valid (alternate field names)",
            )

        return ValidationResult(
            passed=False,
            check_name="structure",
            expected=required_fields,
            actual=list(context.keys()),
            message=f"Missing required fields: {missing_required}",
        )

    return ValidationResult(
        passed=True,
        check_name="structure",
        expected=required_fields + optional_fields,
        actual=list(context.keys()),
        message=f"Structure valid. Optional fields present: {has_optional}",
    )


def validate_economic_context(context: dict) -> ValidationResult:
    """Check if economic context is present."""
    economic_keys = ["economic_model", "economics", "incentives", "fees", "rewards"]
    found = [k for k in economic_keys if k in context]

    if found:
        return ValidationResult(
            passed=True,
            check_name="economic_context",
            expected="economic model present",
            actual=found,
            message=f"Economic context found: {found}",
        )

    # Not required, just informational
    return ValidationResult(
        passed=True,
        check_name="economic_context",
        expected="economic model (optional)",
        actual=None,
        message="No economic context (optional)",
    )


def run_validation(context: dict, expected_checks: Optional[dict] = None) -> list[ValidationResult]:
    """Run all validation rules against context."""
    results = []

    results.append(validate_structure(context))
    results.append(validate_protocol_type(context))
    results.append(validate_roles(context))
    results.append(validate_economic_context(context))

    return results


def main():
    parser = argparse.ArgumentParser(description="Validate protocol context pack")
    parser.add_argument("--context", required=True, help="Path to context pack YAML")
    parser.add_argument("--expected", help="Path to expected checks JSON (optional)")
    parser.add_argument("--output", help="Output file for validation results")
    args = parser.parse_args()

    context_path = Path(args.context)

    # Load context
    try:
        context = load_context(context_path)
    except Exception as e:
        print(f"ERROR: Failed to load context: {e}")
        sys.exit(1)

    if not context:
        print("ERROR: Context pack is empty")
        sys.exit(1)

    # Load expected checks if provided
    expected_checks = None
    if args.expected and Path(args.expected).exists():
        expected_checks = json.loads(Path(args.expected).read_text())

    # Run validation
    results = run_validation(context, expected_checks)

    # Output results
    all_passed = all(r.passed for r in results)

    print("")
    print("Context Validation Results:")
    print("-" * 60)

    for result in results:
        status = "[PASS]" if result.passed else "[FAIL]"
        print(f"  {status} {result.check_name}")
        print(f"         {result.message}")

    print("-" * 60)
    print(f"Total: {sum(1 for r in results if r.passed)}/{len(results)} passed")

    # Write output file
    if args.output:
        output_data = {
            "passed": all_passed,
            "total_checks": len(results),
            "passed_checks": sum(1 for r in results if r.passed),
            "context_summary": {
                "protocol_type": context.get("protocol_type") or context.get("type"),
                "roles_count": len(context.get("roles", [])),
                "has_economic": any(k in context for k in ["economic_model", "economics"]),
            },
            "results": [
                {
                    "check": r.check_name,
                    "passed": r.passed,
                    "message": r.message,
                }
                for r in results
            ],
        }
        Path(args.output).write_text(json.dumps(output_data, indent=2))

    sys.exit(0 if all_passed else 1)


if __name__ == "__main__":
    main()
```

1. **Direct Exa invocation:** `mcp__exa_search__web_search_exa`

3. **Similar protocol references:** References to known vault exploits

**Expected Exa Query Patterns:**
- `"vault protocol exploit vulnerability 2024 2025"`
- `"ethereum vault reentrancy attack"`
