- Pattern control: `--tier`, `--category`, `/vrs-run-pattern`
- Mode control: `--mode solo`, `--mode swarm`

The `validate_stage` function validates individual stage outputs for isolated testing.

FIXTURES_DIR="$MAIN_REPO/tests/fixtures"

function create_from_fixture() {
    local name=$1
    local fixture=$2
    local path="$WORKTREES_BASE/$name"
    local fixture_path="$FIXTURES_DIR/$fixture"

    echo "Creating worktree from fixture: $fixture"

    if [ ! -d "$fixture_path" ]; then
        echo "ERROR: Fixture not found: $fixture_path"
        exit 1
    fi

    # Create base worktree
    create_worktree "$name"
    setup_worktree "$name"

    # Copy fixture files
    echo "Copying fixture files..."
    if [ -d "$fixture_path/src" ]; then
        mkdir -p "$path/contracts"
        cp -r "$fixture_path/src/"* "$path/contracts/"
    fi

    # Copy expected outputs
    if [ -d "$fixture_path/expected" ]; then
        mkdir -p "$path/expected"
        cp -r "$fixture_path/expected/"* "$path/expected/"
    fi

    # Copy ground truth
    if [ -f "$fixture_path/ground-truth.yaml" ]; then
        cp "$fixture_path/ground-truth.yaml" "$path/"
    fi

    # Copy foundry.toml if exists
    if [ -f "$fixture_path/foundry.toml" ]; then
        cp "$fixture_path/foundry.toml" "$path/"
    fi

    echo "✓ Worktree created from fixture: $path"
    echo ""
    echo "Contents:"
    ls -la "$path/contracts/" 2>/dev/null || echo "No contracts"
    ls -la "$path/expected/" 2>/dev/null || echo "No expected files"
}

function validate_stage() {
    local name=$1
    local stage=$2
    local path="$WORKTREES_BASE/$name"

    echo "Validating stage: $stage"

    case "$stage" in
        graph)
            if ls "$path/.vrs/graphs/"*.toon 1>/dev/null 2>&1 || ls "$path/.vrs/graphs/"*.json 1>/dev/null 2>&1; then
                echo "✓ Graph file exists"
                return 0
            else
                echo "✗ No graph file found"
                return 1
            fi
            ;;

        context)
            if [ -f "$path/.vrs/context/protocol-pack.yaml" ]; then
                echo "✓ Context pack exists"
                return 0
            else
                echo "✗ No context pack found"
                return 1
            fi
            ;;

        patterns)
            if [ -f "$path/.vrs/findings/pattern-matches.json" ]; then
                echo "✓ Pattern matches file exists"
                return 0
            else
                echo "✗ No pattern matches found"
                return 1
            fi
            ;;

        agents)
            if [ -f "$path/.vrs/findings/agent-investigations.json" ]; then
                echo "✓ Agent investigations file exists"
                return 0
            else
                echo "✗ No agent investigations found"
                return 1
            fi
            ;;

        debate)
            if [ -f "$path/.vrs/findings/verdicts.json" ]; then
                echo "✓ Verdicts file exists"
                return 0
            else
                echo "✗ No verdicts found"
                return 1
            fi
            ;;

        report)
            if [ -f "$path/.vrs/reports/audit-report.md" ]; then
                echo "✓ Audit report exists"
                return 0
            else
                echo "✗ No audit report found"
                return 1
            fi
            ;;

        all)
            local all_passed=true
            for s in graph context patterns agents debate report; do
                if ! validate_stage "$name" "$s"; then
                    all_passed=false
                fi
            done
            if $all_passed; then
                echo "✓ All stages validated"
                return 0
            else
                echo "✗ Some stages failed"
                return 1
            fi
            ;;

        *)
            echo "Unknown stage: $stage"
            echo "Valid stages: graph, context, patterns, agents, debate, report, all"
            return 1
            ;;
    esac
}

### Task 2: Create run_audit_test.sh Script

**File:** `scripts/run_audit_test.sh`

```bash
#!/bin/bash
# Run a full audit test in an isolated worktree via claude-code-controller
#
# Usage:
#   ./scripts/run_audit_test.sh <fixture> [--timeout 600] [--mode swarm]

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
MAIN_REPO="$(dirname "$SCRIPT_DIR")"
WORKTREES_BASE="/tmp/vrs-worktrees"

# Parse arguments
FIXTURE=$1
TIMEOUT=${TIMEOUT:-600}
MODE=${MODE:-swarm}
WORKTREE_NAME="wt-test-$(date +%s)"

if [ -z "$FIXTURE" ]; then
    echo "Usage: $0 <fixture> [--timeout N] [--mode swarm|solo]"
    echo ""
    echo "Example:"
    echo "  $0 foundry-vault --timeout 600 --mode swarm"
    echo ""
    echo "Available fixtures:"
    ls "$MAIN_REPO/tests/fixtures/" 2>/dev/null || echo "No fixtures found"
    exit 1
fi

# Parse optional args
shift
while [[ $# -gt 0 ]]; do
    case $1 in
        --timeout)
            TIMEOUT="$2"
            shift 2
            ;;
        --mode)
            MODE="$2"
            shift 2
            ;;
        *)
            echo "Unknown option: $1"
            exit 1
            ;;
    esac
done

echo "========================================"
echo " ALPHASWARM AUDIT TEST"
echo "========================================"
echo "Fixture:   $FIXTURE"
echo "Worktree:  $WORKTREE_NAME"
echo "Timeout:   ${TIMEOUT}s"
echo "Mode:      $MODE"
echo "========================================"

# Step 1: Create worktree from fixture
echo ""
echo "[Step 1/7] Creating worktree from fixture..."
"$SCRIPT_DIR/manage_worktrees.sh" create-from-fixture "$WORKTREE_NAME" "$FIXTURE"

WORKTREE_PATH="$WORKTREES_BASE/$WORKTREE_NAME"

# Step 2: Launch Claude Code session via claude-code-controller
echo ""
echo "[Step 2/7] Launching Claude Code session..."
PANE_ID=$(claude-code-controller launch "cd $WORKTREE_PATH && claude --dangerously-skip-permissions" 2>&1 | grep -oP 'pane_id:\s*\K\S+' || echo "")

if [ -z "$PANE_ID" ]; then
    echo "ERROR: Failed to launch Claude Code session"
    "$SCRIPT_DIR/manage_worktrees.sh" cleanup "$WORKTREE_NAME"
    exit 1
fi

echo "Session started: $PANE_ID"

# Wait for Claude Code to initialize
sleep 10

# Step 3: Send /vrs-audit command
echo ""
echo "[Step 3/7] Sending /vrs-audit command..."
claude-code-controller send "/vrs-audit contracts/ --mode $MODE" --pane="$PANE_ID"

# Step 4: Wait for completion
echo ""
echo "[Step 4/7] Waiting for audit completion (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)

if ! claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=60 --timeout="$TIMEOUT"; then
    echo "WARNING: Audit may have timed out"
fi

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))
echo "Duration: ${DURATION}s"

# Step 5: Capture output
echo ""
echo "[Step 5/7] Capturing output..."
TRANSCRIPT="$WORKTREE_PATH/audit-transcript.txt"
claude-code-controller capture --pane="$PANE_ID" --lines=5000 > "$TRANSCRIPT"
echo "Transcript saved: $TRANSCRIPT"

# Step 6: Validate stage outputs
echo ""
echo "[Step 6/7] Validating stage outputs..."
VALIDATION_PASSED=true

for stage in graph patterns agents debate report; do
    if "$SCRIPT_DIR/manage_worktrees.sh" validate "$WORKTREE_NAME" "$stage"; then
        echo "  ✓ $stage"
    else
        echo "  ✗ $stage"
        VALIDATION_PASSED=false
    fi
done

# Step 7: Compare to ground truth (if exists)
echo ""
echo "[Step 7/7] Comparing to ground truth..."
if [ -f "$WORKTREE_PATH/ground-truth.yaml" ]; then
    # Run Python comparison script
    uv run python "$SCRIPT_DIR/compare_findings.py" \
        --worktree "$WORKTREE_PATH" \
        --output "$WORKTREE_PATH/metrics.json"

    if [ -f "$WORKTREE_PATH/metrics.json" ]; then
        echo "Metrics:"
        cat "$WORKTREE_PATH/metrics.json"
    fi
else
    echo "No ground truth file found, skipping comparison"
fi

# Cleanup
echo ""
echo "========================================"
echo " TEST COMPLETE"
echo "========================================"
echo "Duration:    ${DURATION}s"
echo "Worktree:    $WORKTREE_PATH"
echo "Transcript:  $TRANSCRIPT"
echo "Validation:  $($VALIDATION_PASSED && echo 'PASSED' || echo 'FAILED')"
echo ""
echo "To cleanup: $SCRIPT_DIR/manage_worktrees.sh cleanup $WORKTREE_NAME"

# Kill Claude session
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

# Return exit code based on validation
if $VALIDATION_PASSED; then
    exit 0
else
    exit 1
fi
```

### Task 3: Create compare_findings.py

**File:** `scripts/compare_findings.py`

```python
#!/usr/bin/env python3
"""
Compare audit findings to external ground truth.

Usage:
    uv run python scripts/compare_findings.py --worktree /path/to/worktree
"""

import argparse
import json
import yaml
from pathlib import Path
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class Vulnerability:
    id: str
    type: str
    severity: str
    function: str
    line: Optional[int] = None
    description: Optional[str] = None

@dataclass
class Metrics:
    true_positives: int
    false_positives: int
    false_negatives: int
    precision: float
    recall: float
    f1_score: float
    tp_list: List[str]
    fp_list: List[str]
    fn_list: List[str]


def load_ground_truth(worktree: Path) -> List[Vulnerability]:
    """Load ground truth from YAML file."""
    gt_file = worktree / "ground-truth.yaml"
    if not gt_file.exists():
        raise FileNotFoundError(f"Ground truth file not found: {gt_file}")

    data = yaml.safe_load(gt_file.read_text())
    vulns = []

    for v in data.get("vulnerabilities", []):
        vulns.append(Vulnerability(
            id=v["id"],
            type=v["type"],
            severity=v["severity"],
            function=v["function"],
            line=v.get("line"),
            description=v.get("description"),
        ))

    return vulns


def load_findings(worktree: Path) -> List[dict]:
    """Load findings from audit output."""
    # Try verdicts first (most complete)
    verdicts_file = worktree / ".vrs" / "findings" / "verdicts.json"
    if verdicts_file.exists():
        return json.loads(verdicts_file.read_text())

    # Fall back to agent investigations
    agent_file = worktree / ".vrs" / "findings" / "agent-investigations.json"
    if agent_file.exists():
        return json.loads(agent_file.read_text())

    # Fall back to pattern matches
    pattern_file = worktree / ".vrs" / "findings" / "pattern-matches.json"
    if pattern_file.exists():
        return json.loads(pattern_file.read_text())

    return []


def match_finding_to_vuln(finding: dict, vuln: Vulnerability) -> bool:
    """Determine if a finding matches a ground truth vulnerability."""
    # Match by function name
    finding_func = finding.get("function", "").lower()
    vuln_func = vuln.function.lower()

    if finding_func == vuln_func:
        return True

    # Match by location (if line numbers provided)
    if vuln.line:
        finding_line = finding.get("line")
        if finding_line and abs(finding_line - vuln.line) <= 5:
            return True

    # Match by type
    finding_type = finding.get("type", "").lower()
    vuln_type = vuln.type.lower()

    if finding_type == vuln_type and finding_func in vuln_func:
        return True

    return False


def calculate_metrics(ground_truth: List[Vulnerability], findings: List[dict]) -> Metrics:
    """Calculate precision, recall, and F1 score."""
    matched_vulns = set()
    matched_findings = set()

    # Find true positives
    for i, finding in enumerate(findings):
        for vuln in ground_truth:
            if vuln.id not in matched_vulns and match_finding_to_vuln(finding, vuln):
                matched_vulns.add(vuln.id)
                matched_findings.add(i)
                break

    tp = len(matched_vulns)
    fn = len(ground_truth) - tp
    fp = len(findings) - len(matched_findings)

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0.0

    return Metrics(
        true_positives=tp,
        false_positives=fp,
        false_negatives=fn,
        precision=precision,
        recall=recall,
        f1_score=f1,
        tp_list=[v.id for v in ground_truth if v.id in matched_vulns],
        fp_list=[f.get("id", f"finding-{i}") for i, f in enumerate(findings) if i not in matched_findings],
        fn_list=[v.id for v in ground_truth if v.id not in matched_vulns],
    )


def main():
    parser = argparse.ArgumentParser(description="Compare findings to ground truth")
    parser.add_argument("--worktree", required=True, help="Path to worktree")
    parser.add_argument("--output", help="Output file for metrics JSON")
    args = parser.parse_args()

    worktree = Path(args.worktree)

    try:
        ground_truth = load_ground_truth(worktree)
        findings = load_findings(worktree)
        metrics = calculate_metrics(ground_truth, findings)

        result = {
            "ground_truth_count": len(ground_truth),
            "findings_count": len(findings),
            "true_positives": metrics.true_positives,
            "false_positives": metrics.false_positives,
            "false_negatives": metrics.false_negatives,
            "precision": round(metrics.precision, 4),
            "recall": round(metrics.recall, 4),
            "f1_score": round(metrics.f1_score, 4),
            "tp_list": metrics.tp_list,
            "fp_list": metrics.fp_list,
            "fn_list": metrics.fn_list,
        }

        if args.output:
            Path(args.output).write_text(json.dumps(result, indent=2))

        print(json.dumps(result, indent=2))

    except FileNotFoundError as e:
        print(f"ERROR: {e}")
        exit(1)


if __name__ == "__main__":
    main()
```
