```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

/**
 * @title Vault
 * @notice A vulnerable vault contract for testing AlphaSwarm detection
 * @dev Contains intentional vulnerabilities for testing:
 *      - Reentrancy in withdraw()
 *      - Missing access control on setOwner()
 */
contract Vault {
    mapping(address => uint256) public balances;
    address public owner;

    event Deposit(address indexed user, uint256 amount);
    event Withdrawal(address indexed user, uint256 amount);
    event OwnerChanged(address indexed oldOwner, address indexed newOwner);

    constructor() {
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Not owner");
        _;
    }

    /**
     * @notice Deposit ETH into the vault
     */
    function deposit() external payable {
        balances[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    /**
     * @notice Withdraw ETH from the vault
     * @dev VULNERABLE: External call before state update (reentrancy)
     * @param amount Amount to withdraw
     */
    function withdraw(uint256 amount) external {
        require(balances[msg.sender] >= amount, "Insufficient balance");

        // VULNERABLE: Send ETH before updating balance
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");

        // State update AFTER external call = REENTRANCY
        balances[msg.sender] -= amount;

        emit Withdrawal(msg.sender, amount);
    }

    /**
     * @notice Withdraw all ETH (safe version with CEI pattern)
     */
    function withdrawSafe() external {
        uint256 amount = balances[msg.sender];
        require(amount > 0, "No balance");

        // CEI Pattern: Check-Effects-Interactions
        balances[msg.sender] = 0;  // Effect first

        (bool success, ) = msg.sender.call{value: amount}("");  // Interaction last
        require(success, "Transfer failed");

        emit Withdrawal(msg.sender, amount);
    }

    /**
     * @notice Emergency withdrawal by owner
     */
    function emergencyWithdraw() external onlyOwner {
        uint256 balance = address(this).balance;
        (bool success, ) = owner.call{value: balance}("");
        require(success, "Transfer failed");
    }

    /**
     * @notice Change the vault owner
     * @dev VULNERABLE: No access control, anyone can call
     * @param newOwner New owner address
     */
    function setOwner(address newOwner) external {
        // VULNERABLE: Missing onlyOwner modifier
        emit OwnerChanged(owner, newOwner);
        owner = newOwner;
    }

    /**
     * @notice Get vault balance
     */
    function getBalance() external view returns (uint256) {
        return address(this).balance;
    }
}
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/interfaces/IERC3156FlashLender.sol";
import "@openzeppelin/contracts/interfaces/IERC3156FlashBorrower.sol";

/**
 * @title NaiveReceiverPool
 * @dev A pool offering flash loans, but with a design flaw:
 *      Anyone can call flashLoan on behalf of any receiver,
 *      draining their ETH through fees.
 */
contract NaiveReceiverPool is ReentrancyGuard, IERC3156FlashLender {

    address public constant ETH = 0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE;
    uint256 private constant FIXED_FEE = 1 ether;

    error RepayFailed();
    error UnsupportedCurrency();
    error CallbackFailed();

    constructor() payable {}

    function maxFlashLoan(address token) external view returns (uint256) {
        if (token != ETH) return 0;
        return address(this).balance;
    }

    function flashFee(address token, uint256) external pure returns (uint256) {
        if (token != ETH) revert UnsupportedCurrency();
        return FIXED_FEE;
    }

    /**
     * @notice Execute a flash loan
     * @dev VULNERABLE: Anyone can specify any receiver, causing them to pay fees
     * @param receiver The contract that will receive the flash loan
     * @param token Must be ETH address
     * @param amount Amount to borrow
     * @param data Callback data
     */
    function flashLoan(
        IERC3156FlashBorrower receiver,
        address token,
        uint256 amount,
        bytes calldata data
    ) external nonReentrant returns (bool) {
        if (token != ETH) revert UnsupportedCurrency();

        uint256 balanceBefore = address(this).balance;

        // Send ETH to receiver
        (bool sent, ) = address(receiver).call{value: amount}("");
        require(sent, "Transfer failed");

        // Callback - receiver must pay back loan + fee
        if (receiver.onFlashLoan(msg.sender, token, amount, FIXED_FEE, data) != keccak256("ERC3156FlashBorrower.onFlashLoan")) {
            revert CallbackFailed();
        }

        // Check repayment
        if (address(this).balance < balanceBefore + FIXED_FEE) {
            revert RepayFailed();
        }

        return true;
    }

    receive() external payable {}
}
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

interface IFlashLoanEtherReceiver {
    function execute() external payable;
}

/**
 * @title SideEntranceLenderPool
 * @dev A pool offering flash loans with a critical accounting bug:
 *      deposit() can be called during a flash loan, crediting the
 *      borrowed funds to the borrower's balance.
 */
contract SideEntranceLenderPool {

    mapping(address => uint256) public balances;

    error RepayFailed();

    /**
     * @notice Deposit ETH into the pool
     * @dev VULNERABLE: Can be called during flash loan to credit borrowed funds
     */
    function deposit() external payable {
        unchecked {
            balances[msg.sender] += msg.value;
        }
    }

    /**
     * @notice Withdraw deposited ETH
     */
    function withdraw() external {
        uint256 amount = balances[msg.sender];
        balances[msg.sender] = 0;

        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Withdraw failed");
    }

    /**
     * @notice Execute a flash loan
     * @dev The accounting bug: deposit() can be called during execute(),
     *      allowing the borrower to credit the loan amount to their balance
     */
    function flashLoan(uint256 amount) external {
        uint256 balanceBefore = address(this).balance;

        IFlashLoanEtherReceiver(msg.sender).execute{value: amount}();

        if (address(this).balance < balanceBefore) {
            revert RepayFailed();
        }
    }

    receive() external payable {}
}
```

**File:** `scripts/import_smartbugs.sh`
```bash
#!/bin/bash
# Import SmartBugs curated dataset as test fixtures

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
FIXTURES_DIR="$SCRIPT_DIR/../tests/fixtures/smartbugs"

mkdir -p "$FIXTURES_DIR"

# Clone SmartBugs curated
if [ ! -d "$FIXTURES_DIR/repo" ]; then
    git clone https://github.com/smartbugs/smartbugs-curated "$FIXTURES_DIR/repo"
fi

# Create ground truth from labels
echo "Creating ground truth files..."

for vuln_type in reentrancy access_control arithmetic; do
    mkdir -p "$FIXTURES_DIR/$vuln_type/src"

    # Copy contracts
    if [ -d "$FIXTURES_DIR/repo/dataset/$vuln_type" ]; then
        for contract in "$FIXTURES_DIR/repo/dataset/$vuln_type"/*.sol; do
            if [ -f "$contract" ]; then
                cp "$contract" "$FIXTURES_DIR/$vuln_type/src/"
            fi
        done
    fi
done

echo "SmartBugs fixtures imported to $FIXTURES_DIR"
```
