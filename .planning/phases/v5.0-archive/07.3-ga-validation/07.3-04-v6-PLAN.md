## Goal

Validate that Stage 4 of the `/vrs-audit` pipeline correctly matches vulnerability patterns:
1. 556+ patterns are matched against the BSKG graph
2. Tier A (deterministic), Tier B (exploratory), and Tier C (label-dependent) patterns are classified
3. Expected patterns for foundry-vault are matched (reentrancy-classic, weak-access-control)
4. `pattern-matches.json` is generated with correct findings

**Test Method:** Real pattern matching execution via claude-code-controller with ground truth comparison.

### Task 1: Create Stage-Specific Test Script

**File:** `scripts/test_stage_patterns.sh`

```bash
#!/bin/bash
# Test Stage 4: Pattern Matching
#
# Usage:
#   ./scripts/test_stage_patterns.sh [--timeout 300]

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
MAIN_REPO="$(dirname "$SCRIPT_DIR")"
WORKTREES_BASE="/tmp/vrs-worktrees"
FIXTURE="foundry-vault"
WORKTREE_NAME="wt-patterns-test-$(date +%s)"
TIMEOUT=${TIMEOUT:-300}

echo "========================================"
echo " STAGE 4: PATTERN MATCHING VALIDATION"
echo "========================================"
echo "Fixture:   $FIXTURE"
echo "Worktree:  $WORKTREE_NAME"
echo "Timeout:   ${TIMEOUT}s"
echo "========================================"

# Step 1: Create worktree from fixture
echo ""
echo "[Step 1/7] Creating worktree from fixture..."
"$SCRIPT_DIR/manage_worktrees.sh" create-from-fixture "$WORKTREE_NAME" "$FIXTURE"
WORKTREE_PATH="$WORKTREES_BASE/$WORKTREE_NAME"

# Step 2: Build the graph (prerequisite for pattern matching)
echo ""
echo "[Step 2/7] Building BSKG graph..."
mkdir -p "$WORKTREE_PATH/.vrs/graphs"
if uv run alphaswarm build-kg "$WORKTREE_PATH/contracts/" --with-labels --output "$WORKTREE_PATH/.vrs/graphs/"; then
    echo "  [PASS] Graph built successfully"
    GRAPH_FILE=$(ls "$WORKTREE_PATH/.vrs/graphs/"*.toon 2>/dev/null || ls "$WORKTREE_PATH/.vrs/graphs/"*.json 2>/dev/null | head -1)
    echo "  Graph: $GRAPH_FILE"
else
    echo "  [FAIL] Graph build failed"
    "$SCRIPT_DIR/manage_worktrees.sh" cleanup "$WORKTREE_NAME"
    exit 1
fi

# Step 3: Launch Claude Code session via claude-code-controller
echo ""
echo "[Step 3/7] Launching Claude Code session..."
PANE_ID=$(claude-code-controller launch "cd $WORKTREE_PATH && claude --dangerously-skip-permissions" 2>&1 | grep -oP 'pane_id:\s*\K\S+' || echo "")

if [ -z "$PANE_ID" ]; then
    echo "ERROR: Failed to launch Claude Code session"
    "$SCRIPT_DIR/manage_worktrees.sh" cleanup "$WORKTREE_NAME"
    exit 1
fi

echo "Session started: $PANE_ID"

# Wait for Claude Code to initialize
sleep 15

# Step 4: Send pattern matching command
echo ""
echo "[Step 4/7] Sending pattern matching command..."
mkdir -p "$WORKTREE_PATH/.vrs/findings"

# Run pattern matching via alphaswarm query
claude-code-controller send "Run pattern matching on the BSKG graph. Execute: alphaswarm query 'pattern:*' --graph $GRAPH_FILE --output .vrs/findings/pattern-matches.json" --pane="$PANE_ID"

# Step 5: Wait for completion
echo ""
echo "[Step 5/7] Waiting for pattern matching (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)

if ! claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=30 --timeout="$TIMEOUT"; then
    echo "WARNING: Pattern matching may have timed out"
fi

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))
echo "Duration: ${DURATION}s"

# Step 6: Capture output
echo ""
echo "[Step 6/7] Capturing output..."
TRANSCRIPT="$WORKTREE_PATH/patterns-transcript.txt"
claude-code-controller capture --pane="$PANE_ID" --lines=3000 > "$TRANSCRIPT"
echo "Transcript saved: $TRANSCRIPT"

# Step 7: Validate pattern matches
echo ""
echo "[Step 7/7] Validating pattern matches..."

VALIDATION_PASSED=true

# Check pattern-matches.json exists
if [ -f "$WORKTREE_PATH/.vrs/findings/pattern-matches.json" ]; then
    echo "  [PASS] pattern-matches.json exists"
    MATCHES_FILE="$WORKTREE_PATH/.vrs/findings/pattern-matches.json"
else
    echo "  [FAIL] pattern-matches.json not found"
    # Try alternate locations
    if ls "$WORKTREE_PATH/.vrs/findings/"*.json 1>/dev/null 2>&1; then
        MATCHES_FILE=$(ls "$WORKTREE_PATH/.vrs/findings/"*.json | head -1)
        echo "  [INFO] Found alternate file: $MATCHES_FILE"
    else
        VALIDATION_PASSED=false
        MATCHES_FILE=""
    fi
fi

# Validate pattern matches using Python script
if [ -n "$MATCHES_FILE" ] && [ -f "$MATCHES_FILE" ]; then
    echo ""
    echo "  Validating pattern matches against ground truth..."
    if uv run python "$SCRIPT_DIR/validate_patterns.py" \
        --matches "$MATCHES_FILE" \
        --expected "$WORKTREE_PATH/expected/pattern-matches.json" \
        --output "$WORKTREE_PATH/patterns-validation.json"; then
        echo "  [PASS] Pattern validation passed"
    else
        echo "  [FAIL] Pattern validation failed"
        VALIDATION_PASSED=false
    fi
fi

# Cleanup
echo ""
echo "========================================"
echo " STAGE 4 TEST COMPLETE"
echo "========================================"
echo "Duration:    ${DURATION}s"
echo "Worktree:    $WORKTREE_PATH"
echo "Transcript:  $TRANSCRIPT"
echo "Validation:  $($VALIDATION_PASSED && echo 'PASSED' || echo 'FAILED')"

# Kill Claude session
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

# Return exit code based on validation
if $VALIDATION_PASSED; then
    exit 0
else
    exit 1
fi
```

### Task 2: Create Pattern Validation Script

**File:** `scripts/validate_patterns.py`

```python
#!/usr/bin/env python3
"""
Validate pattern matching output against expected ground truth.

Usage:
    uv run python scripts/validate_patterns.py --matches .vrs/findings/pattern-matches.json --expected expected/pattern-matches.json
"""

import argparse
import json
from pathlib import Path
from dataclasses import dataclass
from typing import Any, Optional
import sys


@dataclass
class ValidationResult:
    passed: bool
    check_name: str
    expected: Any
    actual: Any
    message: str


def load_matches(matches_path: Path) -> list[dict]:
    """Load pattern matches from JSON."""
    content = matches_path.read_text()
    data = json.loads(content)

    # Handle different output formats
    if isinstance(data, list):
        return data
    elif isinstance(data, dict):
        return data.get("matches", data.get("findings", data.get("results", [])))

    return []


def find_match(matches: list[dict], pattern_id: str, function_name: str) -> Optional[dict]:
    """Find a specific pattern match."""
    for match in matches:
        match_pattern = match.get("pattern_id") or match.get("pattern") or match.get("id")
        match_func = match.get("function") or match.get("function_name") or match.get("location", {}).get("function")

        if match_pattern and function_name:
            if pattern_id.lower() in match_pattern.lower() and function_name.lower() in str(match_func).lower():
                return match

    return None


def validate_expected_match(
    matches: list[dict], expected: dict
) -> ValidationResult:
    """Validate that an expected pattern was matched."""
    pattern_id = expected["pattern_id"]
    function_name = expected["function"]

    match = find_match(matches, pattern_id, function_name)

    if match is None:
        return ValidationResult(
            passed=False,
            check_name=f"expected_match:{pattern_id}@{function_name}",
            expected=expected,
            actual=None,
            message=f"Expected pattern {pattern_id} not found for {function_name}",
        )

    # Validate tier if specified
    expected_tier = expected.get("tier")
    if expected_tier:
        actual_tier = match.get("tier")
        if actual_tier and actual_tier.upper() != expected_tier.upper():
            return ValidationResult(
                passed=False,
                check_name=f"expected_match:{pattern_id}@{function_name}",
                expected=expected,
                actual=match,
                message=f"Tier mismatch: expected {expected_tier}, got {actual_tier}",
            )

    # Validate confidence if specified
    expected_conf = expected.get("confidence")
    if expected_conf:
        actual_conf = match.get("confidence", 0)
        if actual_conf < expected_conf * 0.8:  # Allow 20% tolerance
            return ValidationResult(
                passed=False,
                check_name=f"expected_match:{pattern_id}@{function_name}",
                expected=expected,
                actual=match,
                message=f"Confidence too low: expected >= {expected_conf * 0.8}, got {actual_conf}",
            )

    return ValidationResult(
        passed=True,
        check_name=f"expected_match:{pattern_id}@{function_name}",
        expected=expected,
        actual=match,
        message=f"Pattern {pattern_id} correctly matched for {function_name}",
    )


def validate_expected_no_match(
    matches: list[dict], expected: dict
) -> ValidationResult:
    """Validate that a pattern was NOT matched (false positive check)."""
    pattern_id = expected["pattern_id"]
    function_name = expected["function"]

    match = find_match(matches, pattern_id, function_name)

    if match is not None:
        return ValidationResult(
            passed=False,
            check_name=f"expected_no_match:{pattern_id}@{function_name}",
            expected=f"No match for {pattern_id}@{function_name}",
            actual=match,
            message=f"FALSE POSITIVE: Pattern {pattern_id} incorrectly matched for {function_name}. Reason: {expected.get('reason', 'unknown')}",
        )

    return ValidationResult(
        passed=True,
        check_name=f"expected_no_match:{pattern_id}@{function_name}",
        expected=f"No match for {pattern_id}@{function_name}",
        actual=None,
        message=f"Correctly did NOT match {pattern_id} for {function_name}",
    )


def validate_match_count(matches: list[dict], min_count: int = 1) -> ValidationResult:
    """Validate minimum number of matches."""
    count = len(matches)

    if count < min_count:
        return ValidationResult(
            passed=False,
            check_name="match_count",
            expected=f">= {min_count}",
            actual=count,
            message=f"Too few matches: expected >= {min_count}, got {count}",
        )

    return ValidationResult(
        passed=True,
        check_name="match_count",
        expected=f">= {min_count}",
        actual=count,
        message=f"Match count OK: {count} matches found",
    )


def run_validation(matches: list[dict], expected_data: dict) -> list[ValidationResult]:
    """Run all validation rules."""
    results = []

    # Basic count check
    results.append(validate_match_count(matches, min_count=1))

    # Check expected matches (true positives)
    for expected in expected_data.get("expected_matches", []):
        results.append(validate_expected_match(matches, expected))

    # Check expected NO matches (false positive prevention)
    for expected in expected_data.get("expected_no_matches", []):
        results.append(validate_expected_no_match(matches, expected))

    return results


def calculate_metrics(results: list[ValidationResult], expected_data: dict) -> dict:
    """Calculate precision/recall metrics."""
    expected_matches = expected_data.get("expected_matches", [])
    expected_no_matches = expected_data.get("expected_no_matches", [])

    # True positives: expected matches that were found
    tp = sum(1 for r in results if r.check_name.startswith("expected_match:") and r.passed)
    # False negatives: expected matches that were NOT found
    fn = sum(1 for r in results if r.check_name.startswith("expected_match:") and not r.passed)
    # False positives: expected NO matches that WERE found
    fp = sum(1 for r in results if r.check_name.startswith("expected_no_match:") and not r.passed)
    # True negatives: expected NO matches that were NOT found
    tn = sum(1 for r in results if r.check_name.startswith("expected_no_match:") and r.passed)

    precision = tp / (tp + fp) if (tp + fp) > 0 else 1.0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 1.0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0.0

    return {
        "true_positives": tp,
        "false_negatives": fn,
        "false_positives": fp,
        "true_negatives": tn,
        "precision": round(precision, 4),
        "recall": round(recall, 4),
        "f1_score": round(f1, 4),
    }


def main():
    parser = argparse.ArgumentParser(description="Validate pattern matches")
    parser.add_argument("--matches", required=True, help="Path to pattern-matches.json")
    parser.add_argument("--expected", required=True, help="Path to expected matches JSON")
    parser.add_argument("--output", help="Output file for validation results")
    args = parser.parse_args()

    matches_path = Path(args.matches)
    expected_path = Path(args.expected)

    # Load matches
    try:
        matches = load_matches(matches_path)
    except Exception as e:
        print(f"ERROR: Failed to load matches: {e}")
        sys.exit(1)

    # Load expected
    if not expected_path.exists():
        print(f"WARNING: No expected file found: {expected_path}")
        print(f"Found {len(matches)} pattern matches")
        sys.exit(0)

    expected_data = json.loads(expected_path.read_text())

    # Run validation
    results = run_validation(matches, expected_data)
    metrics = calculate_metrics(results, expected_data)

    # Output results
    all_passed = all(r.passed for r in results)

    print("")
    print("Pattern Validation Results:")
    print("-" * 60)

    for result in results:
        status = "[PASS]" if result.passed else "[FAIL]"
        print(f"  {status} {result.check_name}")
        if not result.passed:
            print(f"         {result.message}")

    print("-" * 60)
    print(f"Total: {sum(1 for r in results if r.passed)}/{len(results)} checks passed")
    print("")
    print("Metrics:")
    print(f"  Precision: {metrics['precision']}")
    print(f"  Recall:    {metrics['recall']}")
    print(f"  F1 Score:  {metrics['f1_score']}")
    print(f"  TP: {metrics['true_positives']}, FN: {metrics['false_negatives']}, FP: {metrics['false_positives']}, TN: {metrics['true_negatives']}")

    # Write output file
    if args.output:
        output_data = {
            "passed": all_passed,
            "total_matches": len(matches),
            "metrics": metrics,
            "results": [
                {
                    "check": r.check_name,
                    "passed": r.passed,
                    "message": r.message,
                }
                for r in results
            ],
        }
        Path(args.output).write_text(json.dumps(output_data, indent=2))

    sys.exit(0 if all_passed else 1)


if __name__ == "__main__":
    main()
```

| Tier | Confidence | Method | Example |
|------|------------|--------|---------|
