## Goal

This plan validates two related areas:

### Part A: A/B Mode Comparison
Quantify the incremental value of multi-agent debate (swarm mode) compared to single-agent analysis (solo mode) by running identical contracts through both modes and comparing:

1. **Precision/Recall** - Does multi-agent debate improve accuracy?
2. **Token Usage** - What is the cost overhead of multi-agent debate?
3. **Execution Time** - How much longer does swarm mode take?
4. **False Positive Reduction** - Does debate filter out noise?

### Part B: Flexible Workflow Control (Merged from 07.3-16)
Validate that users can exercise granular control over the `/vrs-audit` pipeline through:

1. **Solo mode stage validation** - Verify Stage 6 (debate) is correctly skipped
2. **Single pattern execution** - Query specific patterns against the graph
3. **Targeted pattern query** - Wildcard pattern queries
4. **Pause/resume simulation** - Interrupted audits can be resumed
5. **Status during execution** - Real-time status monitoring
6. **Skip-setup flag** - Faster execution when dependencies pre-installed

**CRITICAL:** Use the same external ground truth (DVD, SmartBugs) for both modes to ensure fair comparison.

```bash
#!/bin/bash
# Run a single A/B test (solo or swarm mode)
#
# Usage: ./scripts/run_ab_test.sh <worktree-name> <mode> <timeout>

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
WORKTREE_NAME=$1
MODE=$2
TIMEOUT=${3:-600}
WORKTREE_PATH="/tmp/vrs-worktrees/$WORKTREE_NAME"

if [ -z "$WORKTREE_NAME" ] || [ -z "$MODE" ]; then
    echo "Usage: $0 <worktree-name> <mode> [timeout]"
    echo "  mode: solo or swarm"
    exit 1
fi

echo "========================================"
echo " A/B TEST: $WORKTREE_NAME ($MODE mode)"
echo "========================================"

# Launch Claude Code
echo "[1/5] Launching Claude Code session..."
PANE_ID=$(claude-code-controller launch "cd $WORKTREE_PATH && claude --dangerously-skip-permissions" 2>&1 | grep -oP 'pane_id:\s*\K\S+' || echo "")

if [ -z "$PANE_ID" ]; then
    echo "ERROR: Failed to launch session"
    exit 1
fi

sleep 15

# Send audit command with specified mode
echo "[2/5] Sending /vrs-audit --mode $MODE..."
claude-code-controller send "/vrs-audit contracts/ --mode $MODE" --pane="$PANE_ID"

# Wait for completion
echo "[3/5] Waiting for completion (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)
claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=60 --timeout="$TIMEOUT" || true
END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))

# Capture output
echo "[4/5] Capturing output..."
claude-code-controller capture --pane="$PANE_ID" --lines=5000 > "$WORKTREE_PATH/audit-transcript.txt"

# Record duration
echo "{\"mode\": \"$MODE\", \"duration_seconds\": $DURATION}" > "$WORKTREE_PATH/run-info.json"

# Compare to ground truth
echo "[5/5] Computing metrics..."
uv run python "$SCRIPT_DIR/compare_findings.py" \
    --worktree "$WORKTREE_PATH" \
    --output "$WORKTREE_PATH/metrics.json"

# Cleanup session
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

echo ""
echo "========================================"
echo " COMPLETE: $WORKTREE_NAME ($MODE mode)"
echo "========================================"
echo "Duration: ${DURATION}s"
cat "$WORKTREE_PATH/metrics.json"
```

## Metrics Collection Script

**File:** `scripts/collect_ab_metrics.py`

```python
#!/usr/bin/env python3
"""
Collect and compare metrics from A/B test runs.

Usage:
    uv run python scripts/collect_ab_metrics.py \
        --solo-worktrees "wt1,wt2,wt3" \
        --swarm-worktrees "wt1,wt2,wt3" \
        --output results.json
"""

import argparse
import json
from pathlib import Path
from typing import Dict, List, Any


def load_metrics(worktree_path: str) -> Dict[str, Any]:
    """Load metrics from a worktree."""
    path = Path(f"/tmp/vrs-worktrees/{worktree_path}")

    metrics = {}

    # Load comparison metrics
    metrics_file = path / "metrics.json"
    if metrics_file.exists():
        metrics["findings"] = json.loads(metrics_file.read_text())

    # Load run info
    run_info_file = path / "run-info.json"
    if run_info_file.exists():
        metrics["run_info"] = json.loads(run_info_file.read_text())

    return metrics


def aggregate_metrics(worktrees: List[str], mode: str) -> Dict[str, Any]:
    """Aggregate metrics across multiple worktrees."""
    all_metrics = []
    total_tp = 0
    total_fp = 0
    total_fn = 0
    total_duration = 0

    for wt in worktrees:
        metrics = load_metrics(wt)
        all_metrics.append({"worktree": wt, **metrics})

        if "findings" in metrics:
            total_tp += metrics["findings"].get("true_positives", 0)
            total_fp += metrics["findings"].get("false_positives", 0)
            total_fn += metrics["findings"].get("false_negatives", 0)

        if "run_info" in metrics:
            total_duration += metrics["run_info"].get("duration_seconds", 0)

    precision = total_tp / (total_tp + total_fp) if (total_tp + total_fp) > 0 else 0
    recall = total_tp / (total_tp + total_fn) if (total_tp + total_fn) > 0 else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0

    return {
        "mode": mode,
        "worktrees": worktrees,
        "aggregate": {
            "true_positives": total_tp,
            "false_positives": total_fp,
            "false_negatives": total_fn,
            "precision": round(precision, 4),
            "recall": round(recall, 4),
            "f1_score": round(f1, 4),
            "total_duration_seconds": total_duration,
            "avg_duration_seconds": round(total_duration / len(worktrees), 2),
        },
        "per_worktree": all_metrics,
    }


def compare_modes(solo: Dict, swarm: Dict) -> Dict[str, Any]:
    """Compare solo vs swarm mode results."""
    return {
        "precision_delta": round(swarm["aggregate"]["precision"] - solo["aggregate"]["precision"], 4),
        "recall_delta": round(swarm["aggregate"]["recall"] - solo["aggregate"]["recall"], 4),
        "f1_delta": round(swarm["aggregate"]["f1_score"] - solo["aggregate"]["f1_score"], 4),
        "fp_reduction": solo["aggregate"]["false_positives"] - swarm["aggregate"]["false_positives"],
        "fn_reduction": solo["aggregate"]["false_negatives"] - swarm["aggregate"]["false_negatives"],
        "time_overhead_ratio": round(
            swarm["aggregate"]["total_duration_seconds"] /
            max(solo["aggregate"]["total_duration_seconds"], 1), 2
        ),
        "swarm_improves_precision": swarm["aggregate"]["precision"] > solo["aggregate"]["precision"],
        "swarm_improves_recall": swarm["aggregate"]["recall"] > solo["aggregate"]["recall"],
        "swarm_improves_f1": swarm["aggregate"]["f1_score"] > solo["aggregate"]["f1_score"],
    }


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--solo-worktrees", required=True, help="Comma-separated list")
    parser.add_argument("--swarm-worktrees", required=True, help="Comma-separated list")
    parser.add_argument("--output", required=True)
    args = parser.parse_args()

    solo_wts = args.solo_worktrees.split(",")
    swarm_wts = args.swarm_worktrees.split(",")

    solo_metrics = aggregate_metrics(solo_wts, "solo")
    swarm_metrics = aggregate_metrics(swarm_wts, "swarm")
    comparison = compare_modes(solo_metrics, swarm_metrics)

    results = {
        "solo": solo_metrics,
        "swarm": swarm_metrics,
        "comparison": comparison,
    }

    Path(args.output).write_text(json.dumps(results, indent=2))
    print(json.dumps(results, indent=2))


if __name__ == "__main__":
    main()
```

## Notes

2. **Expected Swarm Benefits:**

   - 1.5-2.5x execution time for sequential agent calls
   - Higher complexity in debugging failures
