## Goal

Validate that Stage 2 of the `/vrs-audit` pipeline correctly builds the BSKG (Behavioral Smart-contract Knowledge Graph) with:
1. 50+ security properties per function
2. Semantic operations (TRANSFERS_VALUE_OUT, WRITES_USER_BALANCE, etc.)
3. Behavioral signatures (operation ordering)
4. Correct graph structure and node relationships

**Test Method:** Real `/vrs-audit` execution via claude-code-controller with automated validation.

### Task 1: Create Stage-Specific Test Script

**File:** `scripts/test_stage_graph.sh`

```bash
#!/bin/bash
# Test Stage 2: BSKG Graph Building
#
# Usage:
#   ./scripts/test_stage_graph.sh [--timeout 300]

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
MAIN_REPO="$(dirname "$SCRIPT_DIR")"
WORKTREES_BASE="/tmp/vrs-worktrees"
FIXTURE="foundry-vault"
WORKTREE_NAME="wt-graph-test-$(date +%s)"
TIMEOUT=${TIMEOUT:-300}

echo "========================================"
echo " STAGE 2: BSKG GRAPH VALIDATION"
echo "========================================"
echo "Fixture:   $FIXTURE"
echo "Worktree:  $WORKTREE_NAME"
echo "Timeout:   ${TIMEOUT}s"
echo "========================================"

# Step 1: Create worktree from fixture
echo ""
echo "[Step 1/6] Creating worktree from fixture..."
"$SCRIPT_DIR/manage_worktrees.sh" create-from-fixture "$WORKTREE_NAME" "$FIXTURE"
WORKTREE_PATH="$WORKTREES_BASE/$WORKTREE_NAME"

# Step 2: Launch Claude Code session via claude-code-controller
echo ""
echo "[Step 2/6] Launching Claude Code session..."
PANE_ID=$(claude-code-controller launch "cd $WORKTREE_PATH && claude --dangerously-skip-permissions" 2>&1 | grep -oP 'pane_id:\s*\K\S+' || echo "")

if [ -z "$PANE_ID" ]; then
    echo "ERROR: Failed to launch Claude Code session"
    "$SCRIPT_DIR/manage_worktrees.sh" cleanup "$WORKTREE_NAME"
    exit 1
fi

echo "Session started: $PANE_ID"

# Wait for Claude Code to initialize
sleep 15

# Step 3: Send command to build graph only
echo ""
echo "[Step 3/6] Sending graph build command..."
# Use direct alphaswarm command for isolated stage testing
claude-code-controller send "alphaswarm build-kg contracts/ --with-labels --output .vrs/graphs/" --pane="$PANE_ID"

# Step 4: Wait for completion
echo ""
echo "[Step 4/6] Waiting for graph build completion (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)

if ! claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=30 --timeout="$TIMEOUT"; then
    echo "WARNING: Build may have timed out"
fi

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))
echo "Duration: ${DURATION}s"

# Step 5: Capture output
echo ""
echo "[Step 5/6] Capturing output..."
TRANSCRIPT="$WORKTREE_PATH/graph-build-transcript.txt"
claude-code-controller capture --pane="$PANE_ID" --lines=3000 > "$TRANSCRIPT"
echo "Transcript saved: $TRANSCRIPT"

# Step 6: Validate graph output
echo ""
echo "[Step 6/6] Validating graph output..."

VALIDATION_PASSED=true

# Check graph file exists
if ls "$WORKTREE_PATH/.vrs/graphs/"*.toon 1>/dev/null 2>&1 || ls "$WORKTREE_PATH/.vrs/graphs/"*.json 1>/dev/null 2>&1; then
    echo "  [PASS] Graph file exists"
    GRAPH_FILE=$(ls "$WORKTREE_PATH/.vrs/graphs/"*.toon 2>/dev/null || ls "$WORKTREE_PATH/.vrs/graphs/"*.json 2>/dev/null | head -1)
    echo "  Graph file: $GRAPH_FILE"
else
    echo "  [FAIL] No graph file found"
    VALIDATION_PASSED=false
fi

# Validate graph properties using Python script
if [ -f "$GRAPH_FILE" ]; then
    echo ""
    echo "  Validating graph properties..."
    if uv run python "$SCRIPT_DIR/validate_graph.py" \
        --graph "$GRAPH_FILE" \
        --expected "$WORKTREE_PATH/expected/graph-checks.json" \
        --output "$WORKTREE_PATH/graph-validation.json"; then
        echo "  [PASS] Graph properties validated"
    else
        echo "  [FAIL] Graph property validation failed"
        VALIDATION_PASSED=false
    fi
fi

# Cleanup
echo ""
echo "========================================"
echo " STAGE 2 TEST COMPLETE"
echo "========================================"
echo "Duration:    ${DURATION}s"
echo "Worktree:    $WORKTREE_PATH"
echo "Transcript:  $TRANSCRIPT"
echo "Validation:  $($VALIDATION_PASSED && echo 'PASSED' || echo 'FAILED')"

# Kill Claude session
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

# Return exit code based on validation
if $VALIDATION_PASSED; then
    exit 0
else
    exit 1
fi
```

### Task 2: Create Graph Validation Script

**File:** `scripts/validate_graph.py`

```python
#!/usr/bin/env python3
"""
Validate BSKG graph output against expected properties.

Usage:
    uv run python scripts/validate_graph.py --graph .vrs/graphs/project.toon --expected expected/graph-checks.json
"""

import argparse
import json
from pathlib import Path
from dataclasses import dataclass
from typing import Any, Optional
import sys

# Add src to path for TOON parser
sys.path.insert(0, str(Path(__file__).parent.parent / "src"))

try:
    from alphaswarm_sol.formats.toon import parse_toon
except ImportError:
    parse_toon = None


@dataclass
class ValidationResult:
    passed: bool
    check_name: str
    expected: Any
    actual: Any
    message: str


def load_graph(graph_path: Path) -> dict:
    """Load graph from TOON or JSON format."""
    content = graph_path.read_text()

    if graph_path.suffix == ".toon":
        if parse_toon is None:
            raise ImportError("TOON parser not available")
        return parse_toon(content)
    else:
        return json.loads(content)


def find_function(graph: dict, func_name: str) -> Optional[dict]:
    """Find a function node by name."""
    for node in graph.get("nodes", []):
        if node.get("type") == "function" and node.get("name") == func_name:
            return node
    return None


def validate_operations(graph: dict, required_ops: list[str]) -> ValidationResult:
    """Check if graph contains required semantic operations."""
    found_ops = set()

    for node in graph.get("nodes", []):
        if "operations" in node:
            found_ops.update(node["operations"])

    missing = set(required_ops) - found_ops
    if missing:
        return ValidationResult(
            passed=False,
            check_name="has_operations",
            expected=required_ops,
            actual=list(found_ops),
            message=f"Missing operations: {missing}",
        )

    return ValidationResult(
        passed=True,
        check_name="has_operations",
        expected=required_ops,
        actual=list(found_ops),
        message="All required operations present",
    )


def validate_function_properties(
    graph: dict, func_name: str, expected_props: dict
) -> ValidationResult:
    """Validate properties of a specific function."""
    func = find_function(graph, func_name)

    if func is None:
        return ValidationResult(
            passed=False,
            check_name=f"function_properties:{func_name}",
            expected=expected_props,
            actual=None,
            message=f"Function not found: {func_name}",
        )

    mismatches = []
    for prop, expected_value in expected_props.items():
        actual_value = func.get("properties", {}).get(prop)

        # Also check direct node properties
        if actual_value is None:
            actual_value = func.get(prop)

        if actual_value != expected_value:
            mismatches.append(f"{prop}: expected {expected_value}, got {actual_value}")

    if mismatches:
        return ValidationResult(
            passed=False,
            check_name=f"function_properties:{func_name}",
            expected=expected_props,
            actual=func.get("properties", {}),
            message=f"Property mismatches: {', '.join(mismatches)}",
        )

    return ValidationResult(
        passed=True,
        check_name=f"function_properties:{func_name}",
        expected=expected_props,
        actual=func.get("properties", {}),
        message="All properties match",
    )


def validate_behavioral_signature(
    graph: dict, func_name: str, expected_sig: str
) -> ValidationResult:
    """Validate behavioral signature of a function."""
    func = find_function(graph, func_name)

    if func is None:
        return ValidationResult(
            passed=False,
            check_name=f"behavioral_signature:{func_name}",
            expected=expected_sig,
            actual=None,
            message=f"Function not found: {func_name}",
        )

    actual_sig = func.get("behavioral_signature") or func.get("properties", {}).get(
        "behavioral_signature"
    )

    if actual_sig != expected_sig:
        return ValidationResult(
            passed=False,
            check_name=f"behavioral_signature:{func_name}",
            expected=expected_sig,
            actual=actual_sig,
            message=f"Signature mismatch",
        )

    return ValidationResult(
        passed=True,
        check_name=f"behavioral_signature:{func_name}",
        expected=expected_sig,
        actual=actual_sig,
        message="Signature matches",
    )


def run_validation(graph: dict, rules: list[dict]) -> list[ValidationResult]:
    """Run all validation rules against graph."""
    results = []

    for rule in rules:
        check_type = rule.get("check")

        if check_type == "has_operations":
            results.append(validate_operations(graph, rule["required"]))

        elif check_type == "function_properties":
            results.append(
                validate_function_properties(
                    graph, rule["function"], rule["expected"]
                )
            )

        elif check_type == "behavioral_signature":
            results.append(
                validate_behavioral_signature(
                    graph, rule["function"], rule["expected"]
                )
            )

    return results


def main():
    parser = argparse.ArgumentParser(description="Validate BSKG graph properties")
    parser.add_argument("--graph", required=True, help="Path to graph file (.toon or .json)")
    parser.add_argument("--expected", required=True, help="Path to expected checks JSON")
    parser.add_argument("--output", help="Output file for validation results")
    args = parser.parse_args()

    graph_path = Path(args.graph)
    expected_path = Path(args.expected)

    # Load graph
    try:
        graph = load_graph(graph_path)
    except Exception as e:
        print(f"ERROR: Failed to load graph: {e}")
        sys.exit(1)

    # Load expected checks
    if not expected_path.exists():
        print(f"WARNING: No expected checks file found: {expected_path}")
        print("Performing basic graph structure validation only")
        # Basic validation
        if not graph.get("nodes"):
            print("ERROR: Graph has no nodes")
            sys.exit(1)
        print(f"Graph has {len(graph.get('nodes', []))} nodes")
        sys.exit(0)

    expected = json.loads(expected_path.read_text())
    rules = expected.get("validation_rules", [])

    # Run validation
    results = run_validation(graph, rules)

    # Output results
    all_passed = all(r.passed for r in results)

    print("")
    print("Validation Results:")
    print("-" * 60)

    for result in results:
        status = "[PASS]" if result.passed else "[FAIL]"
        print(f"  {status} {result.check_name}")
        if not result.passed:
            print(f"         {result.message}")

    print("-" * 60)
    print(f"Total: {sum(1 for r in results if r.passed)}/{len(results)} passed")

    # Write output file
    if args.output:
        output_data = {
            "passed": all_passed,
            "total_checks": len(results),
            "passed_checks": sum(1 for r in results if r.passed),
            "failed_checks": sum(1 for r in results if not r.passed),
            "results": [
                {
                    "check": r.check_name,
                    "passed": r.passed,
                    "message": r.message,
                    "expected": r.expected,
                    "actual": r.actual,
                }
                for r in results
            ],
        }
        Path(args.output).write_text(json.dumps(output_data, indent=2))

    sys.exit(0 if all_passed else 1)


if __name__ == "__main__":
    main()
```
