```bash
#!/bin/bash
# Run SmartBugs reentrancy suite test

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
MAIN_REPO="$(dirname "$SCRIPT_DIR")"
WORKTREE_NAME="smartbugs-reentrancy"
WORKTREE_PATH="/tmp/vrs-worktrees/$WORKTREE_NAME"
TIMEOUT=900

echo "========================================"
echo " SMARTBUGS REENTRANCY SUITE TEST"
echo "========================================"
echo "Target: 10+ reentrancy contracts"
echo "Source: SmartBugs Curated Dataset"
echo ""

# Step 1: Import SmartBugs (if not already done)
if [ ! -d "$MAIN_REPO/tests/fixtures/smartbugs/reentrancy/src" ]; then
    echo "[0/7] Importing SmartBugs dataset..."
    "$SCRIPT_DIR/import_smartbugs.sh"
fi

# Step 2: Create worktree
echo "[1/7] Creating worktree from fixture..."
"$SCRIPT_DIR/manage_worktrees.sh" create-from-fixture "$WORKTREE_NAME" "smartbugs/reentrancy"

CONTRACT_COUNT=$(ls "$WORKTREE_PATH/contracts/"*.sol 2>/dev/null | wc -l)
echo "Contracts to audit: $CONTRACT_COUNT"

# Step 3: Launch Claude Code
echo "[2/7] Launching Claude Code session..."

if [ -z "$PANE_ID" ]; then
    echo "ERROR: Failed to launch session"
    exit 1
fi

sleep 15

# Step 4: Send audit command
echo "[3/7] Sending /vrs-audit command..."
claude-code-controller send "/vrs-audit contracts/ --mode swarm" --pane="$PANE_ID"

# Step 5: Wait for completion
echo "[4/7] Waiting for completion (timeout: ${TIMEOUT}s)..."
START_TIME=$(date +%s)
claude-code-controller wait_idle --pane="$PANE_ID" --idle-time=90 --timeout="$TIMEOUT" || true
END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))

# Step 6: Capture and validate
echo "[5/7] Capturing output..."
claude-code-controller capture --pane="$PANE_ID" --lines=10000 > "$WORKTREE_PATH/audit-transcript.txt"

echo "[6/7] Validating stage outputs..."
"$SCRIPT_DIR/manage_worktrees.sh" validate "$WORKTREE_NAME" all

# Step 7: Compare to ground truth
echo "[7/7] Comparing to ground truth..."
uv run python "$SCRIPT_DIR/compare_findings.py" \
    --worktree "$WORKTREE_PATH" \
    --output "$WORKTREE_PATH/metrics.json"

# Display results
echo ""
echo "========================================"
echo " AGGREGATE RESULTS"
echo "========================================"
echo "Duration: ${DURATION}s"
echo "Contracts: $CONTRACT_COUNT"
echo ""
cat "$WORKTREE_PATH/metrics.json"

# Cleanup
claude-code-controller kill --pane="$PANE_ID" 2>/dev/null || true

# Check success
RECALL=$(jq -r '.recall' "$WORKTREE_PATH/metrics.json")
PRECISION=$(jq -r '.precision' "$WORKTREE_PATH/metrics.json")
F1=$(jq -r '.f1_score' "$WORKTREE_PATH/metrics.json")

echo ""
echo "========================================"
echo " SUMMARY"
echo "========================================"
echo "Precision: $PRECISION (target: >= 0.60)"
echo "Recall:    $RECALL (target: >= 0.80)"
echo "F1 Score:  $F1 (target: >= 0.65)"

# Validate against targets
PASS=true
if (( $(echo "$RECALL < 0.80" | bc -l) )); then
    echo "FAIL: Recall below 0.80"
    PASS=false
fi
if (( $(echo "$PRECISION < 0.60" | bc -l) )); then
    echo "FAIL: Precision below 0.60"
    PASS=false
fi

if $PASS; then
    echo ""
    echo "SUCCESS: All metrics meet targets"
    exit 0
else
    echo ""
    echo "FAILURE: Some metrics below targets"
    exit 1
fi
```

## Per-Contract Analysis Script

**File:** `scripts/analyze_smartbugs_results.py`

```python
#!/usr/bin/env python3
"""
Detailed per-contract analysis for SmartBugs suite.

Usage:
    uv run python scripts/analyze_smartbugs_results.py --worktree /path/to/worktree
"""

import argparse
import json
import yaml
from pathlib import Path
from typing import Dict, List, Any


def load_ground_truth(worktree: Path) -> Dict[str, List[Dict]]:
    """Load ground truth, organized by contract file."""
    gt_file = worktree / "ground-truth.yaml"
    data = yaml.safe_load(gt_file.read_text())

    by_contract = {}
    for contract in data.get("contracts", []):
        filename = contract["file"]
        by_contract[filename] = contract.get("vulnerabilities", [])

    return by_contract, data.get("safe_contracts", [])


def load_findings(worktree: Path) -> List[Dict]:
    """Load findings from audit output."""
    verdicts_file = worktree / ".vrs" / "findings" / "verdicts.json"
    if verdicts_file.exists():
        return json.loads(verdicts_file.read_text())
    return []


def analyze_per_contract(
    ground_truth: Dict[str, List[Dict]],
    safe_contracts: List[Dict],
    findings: List[Dict]
) -> Dict[str, Any]:
    """Analyze detection per contract."""
    results = {
        "per_contract": {},
        "summary": {
            "total_contracts": len(ground_truth),
            "detected_contracts": 0,
            "missed_contracts": 0,
            "safe_flagged": 0,
        }
    }

    # Analyze vulnerable contracts
    for filename, vulns in ground_truth.items():
        contract_findings = [f for f in findings if filename in f.get("file", "")]

        detected = []
        missed = []
        for vuln in vulns:
            found = any(
                f.get("function", "").lower() == vuln.get("function", "").lower()
                or f.get("line") and abs(f.get("line", 0) - vuln.get("line", 0)) <= 5
                for f in contract_findings
            )
            if found:
                detected.append(vuln["id"])
            else:
                missed.append(vuln["id"])

        results["per_contract"][filename] = {
            "ground_truth_count": len(vulns),
            "detected": detected,
            "missed": missed,
            "findings_count": len(contract_findings),
        }

        if len(detected) > 0:
            results["summary"]["detected_contracts"] += 1
        if len(missed) > 0:
            results["summary"]["missed_contracts"] += 1

    # Check safe contracts for false positives
    for safe in safe_contracts:
        filename = safe["file"]
        contract_findings = [f for f in findings if filename in f.get("file", "")]

        if len(contract_findings) > 0:
            results["summary"]["safe_flagged"] += 1
            results["per_contract"][filename] = {
                "is_safe": True,
                "false_positives": len(contract_findings),
                "reason": safe.get("reason"),
            }

    return results


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--worktree", required=True)
    parser.add_argument("--output", required=True)
    args = parser.parse_args()

    worktree = Path(args.worktree)
    ground_truth, safe_contracts = load_ground_truth(worktree)
    findings = load_findings(worktree)

    results = analyze_per_contract(ground_truth, safe_contracts, findings)

    Path(args.output).write_text(json.dumps(results, indent=2))
    print(json.dumps(results, indent=2))


if __name__ == "__main__":
    main()
```

## Notes

1. **Ground Truth Integrity:** SmartBugs is an academic benchmark with labeled vulnerabilities confirmed by security researchers. This provides objective ground truth.

2. **Reentrancy Variants:** The SmartBugs dataset includes multiple reentrancy variants:
   - Classic reentrancy (external call before state update)
   - Cross-function reentrancy
   - Modifier-based reentrancy
   - The DAO pattern

3. **Expected Detection:** AlphaSwarm's reentrancy specialist should detect most patterns using:
   - Behavioral signatures (R:bal->X:out->W:bal)
   - Semantic operations (TRANSFERS_VALUE_OUT, WRITES_USER_BALANCE)
   - Missing reentrancy guard property

4. **Safe Contract Validation:** Including safe contracts (with CEI pattern or ReentrancyGuard) validates that AlphaSwarm doesn't produce false positives on correctly-implemented contracts.

5. **Benchmark Comparison:** SmartBugs paper benchmarks include Slither, Mythril, Securify, and others. AlphaSwarm's results can be compared against these published baselines.

6. **Flexible Workflow Reference:** See `FLEXIBLE-WORKFLOW-DESIGN.md` for all available pattern control commands including `--tier` and `--category` filtering.
