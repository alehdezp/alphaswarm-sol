"""Integration tests for context-merge to bead flow.

These tests validate the complete flow from context-merge
request through verification to bead creation.
"""

import pytest
from unittest.mock import AsyncMock, MagicMock, patch
from pathlib import Path

from true_vkg.agents.orchestration.sub_coordinator import SubCoordinator


class TestContextMergeFlow:
    """Integration tests for context-merge flow."""

    @pytest.fixture
    def mock_runtime_factory(self):
        """Create factory that produces mock runtimes."""
        def factory(model_name="test-model"):
            runtime = MagicMock()
            runtime.model = model_name
            runtime.execute = AsyncMock(return_value=MagicMock(
                content='{"status": "complete", "result": "test"}',
                tool_calls=[],
                input_tokens=100,
                output_tokens=50,
            ))
            return runtime
        return factory

    @pytest.fixture
    def mock_bead_storage(self, tmp_path):
        """Create mock bead storage."""
        storage = MagicMock()
        storage.pool_path = tmp_path / "beads"
        storage.pool_path.mkdir(parents=True, exist_ok=True)
        storage.create_bead = MagicMock(return_value="CTX-001")
        storage.get_bead = MagicMock(return_value=MagicMock(
            id="CTX-001",
            status="verified",
        ))
        return storage

    @pytest.mark.asyncio
    async def test_full_context_merge_flow(self, mock_runtime_factory, mock_bead_storage):
        """Test complete context-merge flow."""
        # This test validates the flow:
        # 1. Request context merge for vuln class
        # 2. Context-merge agent produces merged context
        # 3. Verifier validates quality
        # 4. Bead is created with context

        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
        )

        with patch.object(coordinator, '_spawn_context_merge_agent') as mock_spawn:
            mock_spawn.return_value = MagicMock(
                merged_context="Merged reentrancy context",
                quality_score=0.90,
            )

            result = await coordinator.spawn_context_merge(
                vuln_class="reentrancy",
                scope_path=Path("contracts/"),
            )

            assert result is not None
            mock_spawn.assert_called_once()

    @pytest.mark.asyncio
    async def test_context_merge_to_verifier_flow(self, mock_runtime_factory):
        """Test context-merge output flows to verifier."""
        # Verifier should receive context-merge output for quality check

        merge_runtime = mock_runtime_factory("merge-model")
        merge_runtime.execute = AsyncMock(return_value=MagicMock(
            content='{"merged_context": "test", "quality_score": 0.85}',
            tool_calls=[],
            input_tokens=100,
            output_tokens=50,
        ))

        verify_runtime = mock_runtime_factory("verify-model")
        verify_runtime.execute = AsyncMock(return_value=MagicMock(
            content='{"quality_pass": true, "issues": []}',
            tool_calls=[],
            input_tokens=80,
            output_tokens=30,
        ))

        # Flow: merge -> verify
        # Both should be called in sequence
        assert merge_runtime is not verify_runtime

    @pytest.mark.asyncio
    async def test_failed_verification_triggers_retry(self, mock_runtime_factory, mock_bead_storage):
        """Test that failed verification triggers quality loop retry."""
        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
            max_quality_retries=3,
        )

        # First attempt fails, second succeeds
        attempt_count = 0

        async def mock_spawn(*args, **kwargs):
            nonlocal attempt_count
            attempt_count += 1
            if attempt_count == 1:
                return MagicMock(quality_score=0.50)  # Below threshold
            return MagicMock(quality_score=0.90)  # Above threshold

        with patch.object(coordinator, '_spawn_context_merge_agent', side_effect=mock_spawn):
            with patch.object(coordinator, '_verify_quality') as mock_verify:
                mock_verify.side_effect = [False, True]  # Fail then pass

                result = await coordinator.spawn_context_merge(
                    vuln_class="reentrancy",
                    scope_path=Path("contracts/"),
                )

                # Should have retried
                assert attempt_count >= 1

    @pytest.mark.asyncio
    async def test_bead_created_on_success(self, mock_runtime_factory, mock_bead_storage):
        """Test that bead is created on successful flow."""
        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
        )

        with patch.object(coordinator, '_spawn_context_merge_agent') as mock_spawn:
            mock_spawn.return_value = MagicMock(
                merged_context="Valid merged context",
                quality_score=0.90,
            )

            with patch.object(coordinator, '_verify_quality', return_value=True):
                await coordinator.spawn_context_merge(
                    vuln_class="oracle",
                    scope_path=Path("contracts/"),
                )

                # Bead storage should be called
                mock_bead_storage.create_bead.assert_called()

    @pytest.mark.asyncio
    async def test_parallel_context_merge(self, mock_runtime_factory, mock_bead_storage):
        """Test parallel context-merge for multiple vuln classes."""
        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
            max_parallel=3,
        )

        with patch.object(coordinator, '_spawn_context_merge_agent') as mock_spawn:
            mock_spawn.return_value = MagicMock(quality_score=0.90)

            with patch.object(coordinator, '_verify_quality', return_value=True):
                # Spawn multiple context merges
                vuln_classes = ["reentrancy", "oracle", "access-control"]

                results = await coordinator.spawn_parallel_context_merges(
                    vuln_classes=vuln_classes,
                    scope_path=Path("contracts/"),
                )

                assert len(results) == len(vuln_classes)

    @pytest.mark.asyncio
    async def test_orchestrator_delegates_not_executes(self, mock_runtime_factory, mock_bead_storage):
        """Test that orchestrator delegates work to agents."""
        # Per context doc: orchestrators don't do work directly
        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
        )

        # Coordinator should not have direct LLM execution
        assert not hasattr(coordinator, 'execute_llm')
        assert not hasattr(coordinator, 'call_model')

    @pytest.mark.asyncio
    async def test_concurrency_limits_enforced(self, mock_runtime_factory, mock_bead_storage):
        """Test that concurrency limits are enforced."""
        # Per context doc: max 5 subagents, max 2 sub-orchestrators
        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
            max_parallel=5,
        )

        assert coordinator.max_parallel <= 5

    @pytest.mark.asyncio
    async def test_context_merge_to_vuln_discovery_flow(self, mock_runtime_factory, mock_bead_storage):
        """Test that context-merge bead triggers vuln-discovery."""
        coordinator = SubCoordinator(
            runtime_factory=mock_runtime_factory,
            bead_storage=mock_bead_storage,
        )

        # After context-merge completes, vuln-discovery should be spawned
        with patch.object(coordinator, '_spawn_context_merge_agent') as mock_merge:
            mock_merge.return_value = MagicMock(
                merged_context="Context for discovery",
                quality_score=0.90,
            )

            with patch.object(coordinator, '_verify_quality', return_value=True):
                with patch.object(coordinator, '_spawn_vuln_discovery') as mock_discovery:
                    mock_discovery.return_value = MagicMock(findings=[])

                    await coordinator.run_full_analysis(
                        vuln_class="reentrancy",
                        scope_path=Path("contracts/"),
                    )

                    # Both should be called
                    mock_merge.assert_called()
                    # Discovery may or may not be called depending on flow


class TestBeadStateFlow:
    """Tests for bead state transitions in flows."""

    @pytest.fixture
    def mock_bead(self):
        """Create mock bead with state tracking."""
        bead = MagicMock()
        bead.id = "CTX-001"
        bead.status = "pending"
        bead.work_state = {}
        return bead

    def test_bead_status_transitions(self, mock_bead):
        """Test valid bead status transitions."""
        # pending -> in_progress -> verified -> complete
        valid_transitions = {
            "pending": ["in_progress"],
            "in_progress": ["verified", "failed"],
            "verified": ["complete"],
            "failed": ["pending"],  # Can retry
        }

        assert mock_bead.status in valid_transitions

    def test_bead_preserves_evidence_chain(self, mock_bead):
        """Test that bead preserves evidence chain through flow."""
        mock_bead.evidence_chain = []

        # Add evidence at each stage
        mock_bead.evidence_chain.append({"stage": "context_merge", "data": "merged"})
        mock_bead.evidence_chain.append({"stage": "verification", "data": "verified"})

        assert len(mock_bead.evidence_chain) == 2
        assert mock_bead.evidence_chain[0]["stage"] == "context_merge"
