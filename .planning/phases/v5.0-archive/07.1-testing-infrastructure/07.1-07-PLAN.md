"""Component tests for ContextMergeAgent.

These tests validate the context-merge agent in isolation
with mocked LLM responses for deterministic behavior.
"""

import pytest
from unittest.mock import AsyncMock, MagicMock, patch
from pathlib import Path

from true_vkg.agents.context.types import ContextBundle, RiskProfile
from true_vkg.agents.orchestration.context_merge_agent import ContextMergeAgent


class TestContextMergeAgentComponent:
    """Component tests for ContextMergeAgent."""

    @pytest.fixture
    def mock_runtime(self):
        """Create mock runtime for testing."""
        runtime = MagicMock()
        runtime.execute = AsyncMock(return_value=MagicMock(
            content='{"merged_context": "test context", "quality_score": 0.85}',
            tool_calls=[],
            input_tokens=100,
            output_tokens=50,
        ))
        return runtime

    @pytest.fixture
    def sample_bundles(self):
        """Create sample context bundles for testing."""
        return [
            ContextBundle(
                vuln_class="reentrancy",
                reasoning_template="Check CEI pattern",
                risk_profile=RiskProfile(
                    oracle_dependency=False,
                    liquidity_risks=[],
                    access_control_issues=[],
                    upgrade_risks=[],
                    integration_risks=[],
                    timing_risks=["CEI violation possible"],
                    economic_risks=[],
                    governance_risks=[],
                ),
                semantic_triggers=["TRANSFERS_VALUE_OUT", "WRITES_USER_BALANCE"],
                vql_queries=["FIND functions WHERE transfers_eth"],
                token_count=500,
            ),
        ]

    @pytest.mark.asyncio
    async def test_merge_single_bundle(self, mock_runtime, sample_bundles):
        """Test merging a single context bundle."""
        agent = ContextMergeAgent(runtime=mock_runtime)

        result = await agent.merge_contexts(
            vuln_class="reentrancy",
            bundles=sample_bundles,
        )

        assert result is not None
        mock_runtime.execute.assert_called_once()

    @pytest.mark.asyncio
    async def test_merge_respects_token_budget(self, mock_runtime, sample_bundles):
        """Test that merging respects token budget."""
        agent = ContextMergeAgent(runtime=mock_runtime, max_tokens=6000)

        # Create bundles that exceed budget
        large_bundles = sample_bundles * 20  # Each ~500 tokens = 10k total

        result = await agent.merge_contexts(
            vuln_class="reentrancy",
            bundles=large_bundles,
        )

        # Should not fail, should trim to budget
        assert result is not None

    @pytest.mark.asyncio
    async def test_merge_output_format(self, mock_runtime, sample_bundles):
        """Test that merge output has expected format."""
        mock_runtime.execute = AsyncMock(return_value=MagicMock(
            content='''{
                "merged_context": "Combined reentrancy analysis context",
                "quality_score": 0.90,
                "reasoning_template": "Check CEI pattern thoroughly",
                "semantic_triggers": ["TRANSFERS_VALUE_OUT"],
                "preserved_risks": ["CEI violation"]
            }''',
            tool_calls=[],
            input_tokens=100,
            output_tokens=80,
        ))

        agent = ContextMergeAgent(runtime=mock_runtime)
        result = await agent.merge_contexts(
            vuln_class="reentrancy",
            bundles=sample_bundles,
        )

        assert result is not None
        # Result should be parseable as merge output

    @pytest.mark.asyncio
    async def test_merge_handles_empty_bundles(self, mock_runtime):
        """Test handling of empty bundle list."""
        agent = ContextMergeAgent(runtime=mock_runtime)

        result = await agent.merge_contexts(
            vuln_class="reentrancy",
            bundles=[],
        )

        # Should handle gracefully
        assert result is not None or mock_runtime.execute.call_count == 0

    def test_agent_does_not_execute_directly(self, mock_runtime):
        """Test that agent delegates to runtime, doesn't execute directly."""
        agent = ContextMergeAgent(runtime=mock_runtime)

        # Agent should have no direct LLM client
        assert not hasattr(agent, '_llm_client')
        assert not hasattr(agent, 'anthropic_client')

    @pytest.mark.asyncio
    async def test_merge_includes_vuln_class(self, mock_runtime, sample_bundles):
        """Test that vuln_class is included in merge request."""
        agent = ContextMergeAgent(runtime=mock_runtime)

        await agent.merge_contexts(
            vuln_class="oracle-manipulation",
            bundles=sample_bundles,
        )

        # Check that vuln_class was passed to runtime
        call_args = mock_runtime.execute.call_args
        assert call_args is not None
