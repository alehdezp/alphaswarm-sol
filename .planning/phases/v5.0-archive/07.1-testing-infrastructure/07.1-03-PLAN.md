from dataclasses import dataclass
from pathlib import Path
from typing import List, Dict, Optional
import yaml

@dataclass
class BenchmarkConfig:
    """Configuration for a benchmark run."""
    name: str
    description: str
    time_limit_minutes: int
    corpus_segment: str  # recent-audits, mutations, adversarial, or all
    contracts: List[Dict]  # contract_id, expected_finding, severity
    patterns_to_test: List[str]
    models: List[str]
    accuracy_gates: Dict[str, float]  # precision_min, recall_critical_min, etc.

    @classmethod
    def from_yaml(cls, yaml_path: Path) -> "BenchmarkConfig":
        """Load benchmark config from YAML file."""
        with open(yaml_path) as f:
            data = yaml.safe_load(f)
        return cls(**data)

    def validate(self) -> List[str]:
        """Validate config, return list of errors."""
        errors = []
        if self.time_limit_minutes <= 0:
            errors.append("time_limit_minutes must be positive")
        if not self.patterns_to_test:
            errors.append("patterns_to_test cannot be empty")
        return errors


def load_benchmark(name: str, base_path: Optional[Path] = None) -> BenchmarkConfig:
    """Load a named benchmark configuration.

    Args:
        name: Benchmark name (quick, standard, thorough)
        base_path: Base path for config files (default: .vrs/testing/benchmarks/)

    Returns:
        BenchmarkConfig loaded from YAML
    """
    if base_path is None:
        base_path = Path(".vrs/testing/benchmarks")

    yaml_path = base_path / f"{name}.yaml"
    if not yaml_path.exists():
        raise FileNotFoundError(f"Benchmark config not found: {yaml_path}")

    return BenchmarkConfig.from_yaml(yaml_path)

"""Tests for testing infrastructure."""

import pytest
import tempfile
from pathlib import Path

    CorpusDB,
    initialize_corpus_db,
    TestMetrics,
    CategoryMetrics,
    SegmentMetrics,
    BenchmarkConfig,

class TestCorpusDB:
    """Tests for corpus database operations."""

    def test_initialize_creates_tables(self, tmp_path):
        """Test database initialization creates required tables."""
        db_path = tmp_path / "corpus.db"
        db = initialize_corpus_db(db_path)

        # Verify tables exist
        cursor = db._conn.execute(
            "SELECT name FROM sqlite_master WHERE type='table'"
        )
        tables = {row[0] for row in cursor.fetchall()}
        assert "contracts" in tables
        assert "ground_truth" in tables
        assert "test_runs" in tables

    def test_add_contract(self, tmp_path):
        """Test adding a contract to corpus."""
        db = initialize_corpus_db(tmp_path / "corpus.db")
        db.add_contract(
            contract_id="test-001",
            source_path="contracts/Test.sol",
            category="recent-audits",
            vulnerability_class="reentrancy",
            corpus_version="v1"
        )

        contracts = db.get_contracts()
        assert len(contracts) == 1
        assert contracts[0]["contract_id"] == "test-001"

    def test_add_ground_truth(self, tmp_path):
        """Test adding ground truth label."""
        db = initialize_corpus_db(tmp_path / "corpus.db")
        db.add_contract("test-001", "Test.sol", "recent-audits", corpus_version="v1")
        db.add_ground_truth(
            gt_id="gt-001",
            contract_id="test-001",
            vulnerability_pattern="reentrancy-classic",
            severity="critical",
            code_location="Test.sol:L42"
        )

        gt = db.get_ground_truth("test-001")
        assert len(gt) == 1
        assert gt[0]["severity"] == "critical"

    def test_record_test_run(self, tmp_path):
        """Test recording test run metrics."""
        db = initialize_corpus_db(tmp_path / "corpus.db")
        metrics = TestMetrics(
            precision=0.90,
            recall=0.85,
            f1_score=0.87,
            recall_critical=0.98,
            recall_high=0.88,
            recall_medium=0.75,
            recall_low=0.72,
            execution_time_ms=5000,
            contracts_tested=10,
            patterns_tested=5,
            model_used="claude-sonnet-4.5"
        )
        db.record_test_run("run-001", metrics, corpus_version="v1")

        run = db.get_latest_run("v1")
        assert run is not None
        assert run["precision"] == 0.90

    def test_filter_by_category(self, tmp_path):
        """Test filtering contracts by category."""
        db = initialize_corpus_db(tmp_path / "corpus.db")
        db.add_contract("audit-001", "A.sol", "recent-audits", corpus_version="v1")
        db.add_contract("mut-001", "B.sol", "mutations", corpus_version="v1")

        audits = db.get_contracts(category="recent-audits")
        assert len(audits) == 1
        assert audits[0]["contract_id"] == "audit-001"
