"""Test orchestration runner for AlphaSwarm Test Forge.

This module provides the main test runner that orchestrates:
- Corpus loading and validation
- Benchmark execution (quick/standard/thorough)
- Metrics collection and GA gate validation
- Gap tracking and reporting
- Ranking updates

Test modes per context doc:
- QUICK: < 2 min, smoke tests, critical path only
- STANDARD: < 10 min, full regression, pre-commit
- THOROUGH: < 30 min, GA validation, model comparison

## Execution Wiring

The TestRunner connects testing infrastructure to actual BSKG detection:

1. **Graph Building:** Invokes `uv run alphaswarm build-kg <contract_path>` for each
   contract in the corpus. This produces the BSKG knowledge graph.

2. **Pattern Execution:** Calls pattern engine via `true_vkg.patterns.engine.match_patterns()`
   against the built graph. Returns raw findings for accuracy measurement.

3. **Agent Verification:** For THOROUGH mode, spawns verification agents:
   - Uses SubCoordinator from Phase 5.5 agent infrastructure
   - Spawns vrs-attacker, vrs-defender, vrs-verifier for debate protocol
   - Collects verdicts for accuracy measurement

4. **Metrics Collection:** Compares findings against corpus ground_truth to compute
   precision, recall, F1 per severity level.

Note: This is INFRASTRUCTURE wiring. Full corpus population and execution at scale
happens in Phase 7.2.
"""

@dataclass

    include_model_comparison: bool = False

@dataclass
class TestRunResult:
    """Results from a test run.

    Attributes:
        run_id: Unique identifier for this run
        mode: Test mode used
        metrics: Overall test metrics
        ga_gate_passed: Whether GA gates passed
        gaps_found: Number of gaps found
        duration_seconds: Total execution time
        timestamp: When run completed
        model_results: Per-model results (if model comparison)
        error: Error message if run failed
    """
    run_id: str
    mode: TestMode
    metrics: TestMetrics
    ga_gate_passed: bool
    gaps_found: int = 0
    duration_seconds: float = 0.0
    timestamp: str = field(default_factory=lambda: datetime.now().isoformat())
    model_results: Dict[str, TestMetrics] = field(default_factory=dict)
    error: Optional[str] = None

    def summary(self) -> str:
        """Generate human-readable summary."""
        status = "PASS" if self.ga_gate_passed else "FAIL"
        return f"""
Test Run: {self.run_id}
Mode: {self.mode.value}
Status: {status}
Duration: {self.duration_seconds:.1f}s

Metrics:
  Precision: {self.metrics.precision:.2%}
  Recall (Critical): {self.metrics.recall_critical:.2%}
  Recall (High): {self.metrics.recall_high:.2%}
  F1 Score: {self.metrics.f1_score:.2%}

GA Gates:
  Precision >= 85%: {"PASS" if self.metrics.precision >= 0.85 else "FAIL"}
  Recall Critical >= 95%: {"PASS" if self.metrics.recall_critical >= 0.95 else "FAIL"}
  Recall High >= 85%: {"PASS" if self.metrics.recall_high >= 0.85 else "FAIL"}

Gaps Found: {self.gaps_found}
""".strip()


class TestRunner:
    """Main test orchestration runner.

    Coordinates all testing components for full test execution.
    Wires infrastructure to actual BSKG detection pipeline.
    """

    def __init__(
        self,
        config: TestRunConfig,
        corpus_db: Optional[CorpusDB] = None,
        gap_tracker: Optional[GapTracker] = None,
        ranking_updater: Optional[RankingUpdater] = None,
    ):
        """Initialize test runner.

        Args:
            config: Test run configuration
            corpus_db: Corpus database (created if None)
            gap_tracker: Gap tracker (created if None)
            ranking_updater: Ranking updater (created if None)
        """
        self.config = config

        # Initialize components
        corpus_path = config.corpus_path or Path(".vrs/testing/corpus.db")
        self.corpus_db = corpus_db or initialize_corpus_db(corpus_path)
        self.gap_tracker = gap_tracker or GapTracker()
        self.ranking_updater = ranking_updater or RankingUpdater()

        # Load benchmark config
        self.benchmark = load_benchmark(config.mode.value)

        # Execution state
        self._run_id = f"run-{datetime.now().strftime('%Y%m%d-%H%M%S')}"
        self._start_time: Optional[float] = None
        self._metrics_by_model: Dict[str, TestMetrics] = {}

    def run(self) -> TestRunResult:
        """Execute the test run.

        Returns:
            TestRunResult with metrics and status
        """
        self._start_time = time.time()
        logger.info(f"Starting test run {self._run_id} in {self.config.mode.value} mode")

        try:
            # Phase 1: Health check
            self._run_health_check()

            # Phase 2: Load corpus
            contracts = self._load_corpus()

            # Phase 3: Run tests by mode
            if self.config.mode == TestMode.QUICK:
                metrics = self._run_quick_tests(contracts)
            elif self.config.mode == TestMode.STANDARD:
                metrics = self._run_standard_tests(contracts)
            else:  # THOROUGH
                metrics = self._run_thorough_tests(contracts)

            # Phase 4: Model comparison (if enabled)
            if self.config.include_model_comparison:
                self._run_model_comparison(contracts)

            # Phase 5: Gap analysis
            gaps_found = self._run_gap_analysis(metrics)

            # Phase 6: Record results
            self.corpus_db.record_test_run(
                self._run_id,
                metrics,
                corpus_version=self.corpus_db.get_corpus_version(),
            )

            duration = time.time() - self._start_time

            return TestRunResult(
                run_id=self._run_id,
                mode=self.config.mode,
                metrics=metrics,
                ga_gate_passed=metrics.meets_ga_gates(),
                gaps_found=gaps_found,
                duration_seconds=duration,
                model_results=self._metrics_by_model,
            )

        except Exception as e:
            logger.error(f"Test run failed: {e}")
            duration = time.time() - self._start_time if self._start_time else 0

            return TestRunResult(
                run_id=self._run_id,
                mode=self.config.mode,
                metrics=TestMetrics(
                    precision=0, recall=0, f1_score=0,
                    recall_critical=0, recall_high=0,
                    recall_medium=0, recall_low=0,
                    execution_time_ms=int(duration * 1000),
                    contracts_tested=0, patterns_tested=0,
                ),
                ga_gate_passed=False,
                duration_seconds=duration,
                error=str(e),
            )

    def _run_health_check(self) -> None:
        """Run health check to verify infrastructure."""
        logger.info("Running health check...")

        # Verify corpus accessible
        contracts = self.corpus_db.get_contracts()

        logger.info(f"Rankings has {len(rankings)} models for 'detect' task")

    def _load_corpus(self) -> List[Dict[str, Any]]:

        segment = self.benchmark.corpus_segment

        logger.info(f"Loaded {len(contracts)} contracts for {self.config.mode.value} tests")
        return contracts

    def _build_graph(self, contract_path: Path) -> Optional[Path]:
        """Build BSKG knowledge graph for a contract.

        Invokes: `uv run alphaswarm build-kg <contract_path>`

        Args:
            contract_path: Path to Solidity contract

        Returns:
            Path to generated graph JSON, or None on failure
        """
        output_path = contract_path.with_suffix(".vkg.json")
        try:
            result = subprocess.run(
                ["uv", "run", "alphaswarm", "build-kg", str(contract_path), "-o", str(output_path)],
                capture_output=True,
                text=True,
                timeout=60,
            )
            if result.returncode == 0:
                return output_path
            else:
                logger.warning(f"Graph build failed for {contract_path}: {result.stderr}")
                return None
        except subprocess.TimeoutExpired:
            logger.warning(f"Graph build timed out for {contract_path}")
            return None
        except Exception as e:
            logger.warning(f"Graph build error for {contract_path}: {e}")
            return None

                    root_cause="Missing critical vulnerability patterns",
                    affected_component="pattern_engine",

                    root_cause="Missing high-severity vulnerability patterns",
                    affected_component="pattern_engine",
                    discovered_by="test_runner",

        return gaps_found


def run_tests(
    mode: str = "standard",
    verbose: bool = False,
) -> TestRunResult:
    """Convenience function to run tests.

    Args:
        mode: Test mode (quick/standard/thorough)
        verbose: Enable verbose output

    Returns:
        TestRunResult with metrics and status
    """
    test_mode = TestMode(mode)

    if test_mode == TestMode.QUICK:
        config = TestRunConfig.for_quick()
    elif test_mode == TestMode.THOROUGH:
        config = TestRunConfig.for_thorough()
    else:
        config = TestRunConfig.for_standard()

    config.verbose = verbose

    runner = TestRunner(config)
    return runner.run()
