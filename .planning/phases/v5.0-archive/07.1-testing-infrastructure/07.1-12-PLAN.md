"""Message Recording for Inter-Agent Communication.

Captures and analyzes messages between agents in orchestration flows.
Used to verify:
- Correct message routing
- Expected exchange patterns
- State transitions
- Conversation completeness
"""

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any
import json
import logging

logger = logging.getLogger(__name__)


class MessageType(Enum):
    """Types of inter-agent messages."""

    # Debate protocol
    CLAIM = "claim"
    COUNTERCLAIM = "counterclaim"
    EVIDENCE = "evidence"
    VERDICT = "verdict"

    # Verification loop
    FINDING = "finding"
    VERIFICATION_REQUEST = "verification_request"
    VERIFICATION_RESULT = "verification_result"

    # Routing
    ROUTE_REQUEST = "route_request"
    ROUTE_DECISION = "route_decision"
    HANDOFF = "handoff"

    # General
    QUERY = "query"
    RESPONSE = "response"
    ERROR = "error"


@dataclass
class AgentMessage:
    """A message in an orchestration flow."""

    # Identification
    message_id: str
    timestamp: datetime

    # Routing
    from_agent: str
    to_agent: str
    message_type: MessageType

    # Content
    content: dict
    metadata: dict = field(default_factory=dict)

    # For chained messages
    in_reply_to: str | None = None

    def to_dict(self) -> dict:
        """Convert to dictionary."""
        return {
            "message_id": self.message_id,
            "timestamp": self.timestamp.isoformat(),
            "from_agent": self.from_agent,
            "to_agent": self.to_agent,
            "message_type": self.message_type.value,
            "content": self.content,
            "metadata": self.metadata,
            "in_reply_to": self.in_reply_to,
        }


@dataclass
class ConversationTrace:
    """A trace of a complete conversation in a flow."""

    flow_type: str  # "debate", "verification", "routing"
    started_at: datetime
    ended_at: datetime | None = None
    messages: list[AgentMessage] = field(default_factory=list)
    outcome: str | None = None
    metadata: dict = field(default_factory=dict)

    @property
    def duration_ms(self) -> int | None:
        """Duration of conversation in milliseconds."""
        if self.ended_at and self.started_at:
            return int((self.ended_at - self.started_at).total_seconds() * 1000)
        return None

    @property
    def message_count(self) -> int:
        """Total number of messages."""
        return len(self.messages)

    @property
    def agents_involved(self) -> set[str]:
        """Set of all agents that participated."""
        agents = set()
        for msg in self.messages:
            agents.add(msg.from_agent)
            agents.add(msg.to_agent)
        return agents

    def get_messages_from(self, agent: str) -> list[AgentMessage]:
        """Get all messages from a specific agent."""
        return [m for m in self.messages if m.from_agent == agent]

    def get_messages_to(self, agent: str) -> list[AgentMessage]:
        """Get all messages to a specific agent."""
        return [m for m in self.messages if m.to_agent == agent]

    def get_messages_of_type(self, msg_type: MessageType) -> list[AgentMessage]:
        """Get all messages of a specific type."""
        return [m for m in self.messages if m.message_type == msg_type]


class MessageRecorder:
    """Record and analyze inter-agent messages.

    Example:
        recorder = MessageRecorder()

        # Start recording a flow
        recorder.start_flow("debate", bead_id="bead-123")

        # Record messages
        recorder.record(AgentMessage(
            message_id="msg-1",
            timestamp=datetime.now(),
            from_agent="vkg-attacker",
            to_agent="vkg-defender",
            message_type=MessageType.CLAIM,
            content={"claim": "Vulnerable to reentrancy"}
        ))

        # End flow
        recorder.end_flow(outcome="CONFIRMED")

        # Analyze
        trace = recorder.get_trace()
        assert recorder.contains_exchange("vkg-attacker", "vkg-defender", MessageType.CLAIM)
    """

    def __init__(self):
        self._traces: list[ConversationTrace] = []
        self._current_trace: ConversationTrace | None = None
        self._message_counter = 0

    def start_flow(self, flow_type: str, **metadata) -> None:
        """Start recording a new flow."""
        if self._current_trace:
            logger.warning("Starting new flow while previous flow is active")
            self.end_flow(outcome="INTERRUPTED")

        self._current_trace = ConversationTrace(
            flow_type=flow_type,
            started_at=datetime.now(),
            metadata=metadata,
        )

    def record(self, message: AgentMessage) -> None:
        """Record a message."""
        if not self._current_trace:
            logger.warning("Recording message without active flow")
            self.start_flow("unknown")

        self._current_trace.messages.append(message)

    def record_message(
        self,
        from_agent: str,
        to_agent: str,
        message_type: MessageType,
        content: dict,
        in_reply_to: str | None = None,
        **metadata
    ) -> str:
        """Convenience method to record a message."""
        self._message_counter += 1
        message_id = f"msg-{self._message_counter}"

        message = AgentMessage(
            message_id=message_id,
            timestamp=datetime.now(),
            from_agent=from_agent,
            to_agent=to_agent,
            message_type=message_type,
            content=content,
            metadata=metadata,
            in_reply_to=in_reply_to,
        )

        self.record(message)
        return message_id

    def end_flow(self, outcome: str | None = None) -> ConversationTrace:
        """End the current flow and return its trace."""
        if not self._current_trace:
            raise RuntimeError("No active flow to end")

        self._current_trace.ended_at = datetime.now()
        self._current_trace.outcome = outcome

        trace = self._current_trace
        self._traces.append(trace)
        self._current_trace = None

        return trace

    def get_trace(self) -> ConversationTrace | None:
        """Get the current or most recent trace."""
        return self._current_trace or (self._traces[-1] if self._traces else None)

    def get_all_traces(self) -> list[ConversationTrace]:
        """Get all recorded traces."""
        return self._traces

    # Analysis methods

    def contains_exchange(
        self,
        from_agent: str,
        to_agent: str,
        message_type: MessageType
    ) -> bool:
        """Check if trace contains a specific exchange."""
        trace = self.get_trace()
        if not trace:
            return False

        return any(
            m.from_agent == from_agent and
            m.to_agent == to_agent and
            m.message_type == message_type
            for m in trace.messages
        )

    def count_exchanges(self, from_agent: str, to_agent: str) -> int:
        """Count exchanges between two agents."""
        trace = self.get_trace()
        if not trace:
            return 0

        return sum(
            1 for m in trace.messages
            if m.from_agent == from_agent and m.to_agent == to_agent
        )

    def get_conversation_flow(self) -> list[str]:
        """Get simplified flow: ["attacker->defender", "defender->attacker", ...]"""
        trace = self.get_trace()
        if not trace:
            return []

        return [f"{m.from_agent}->{m.to_agent}" for m in trace.messages]

    def verify_pattern(self, expected_pattern: list[tuple[str, str, MessageType]]) -> tuple[bool, list[str]]:
        """Verify conversation follows expected pattern.

        Args:
            expected_pattern: List of (from_agent, to_agent, message_type) tuples

        Returns:
            (matches, list_of_mismatches)
        """
        trace = self.get_trace()
        if not trace:
            return False, ["No trace available"]

        mismatches = []

        for i, (from_a, to_a, msg_type) in enumerate(expected_pattern):
            if i >= len(trace.messages):
                mismatches.append(f"Step {i}: Expected message but conversation ended")
                continue

            msg = trace.messages[i]
            if msg.from_agent != from_a:
                mismatches.append(f"Step {i}: Expected from {from_a}, got {msg.from_agent}")
            if msg.to_agent != to_a:
                mismatches.append(f"Step {i}: Expected to {to_a}, got {msg.to_agent}")
            if msg.message_type != msg_type:
                mismatches.append(f"Step {i}: Expected type {msg_type.value}, got {msg.message_type.value}")

        return len(mismatches) == 0, mismatches

    def export_trace(self, format: str = "json") -> str:
        """Export trace for debugging."""
        trace = self.get_trace()
        if not trace:
            return ""

        if format == "json":
            return json.dumps({
                "flow_type": trace.flow_type,
                "started_at": trace.started_at.isoformat(),
                "ended_at": trace.ended_at.isoformat() if trace.ended_at else None,
                "outcome": trace.outcome,
                "messages": [m.to_dict() for m in trace.messages],
            }, indent=2)
        elif format == "mermaid":
            # Generate Mermaid sequence diagram
            lines = ["sequenceDiagram"]
            for msg in trace.messages:
                lines.append(f"    {msg.from_agent}->>{msg.to_agent}: {msg.message_type.value}")
            return "\n".join(lines)

        return ""

"""Flow Simulator for Controlled Orchestration Testing.

Simulates orchestration flows with controlled inputs and outputs.
Allows testing without actual agent execution.

Flow Types:
- Debate: attacker claim -> defender counterclaim -> verifier verdict
- Verification: finding -> verify -> confirm/reject
- Routing: bead -> route decision -> agent assignment
"""

from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Callable
import logging

from true_vkg.testing.integration.message_recorder import (
    MessageRecorder,
    MessageType,
    ConversationTrace,
)

logger = logging.getLogger(__name__)


class FlowOutcome(Enum):
    """Possible outcomes of an orchestration flow."""

    # Debate outcomes
    CONFIRMED = "confirmed"
    REJECTED = "rejected"
    UNCERTAIN = "uncertain"

    # Verification outcomes
    VERIFIED = "verified"
    REFUTED = "refuted"
    NEEDS_MORE_EVIDENCE = "needs_more_evidence"

    # Routing outcomes
    ROUTED = "routed"
    ESCALATED = "escalated"
    DROPPED = "dropped"

    # Errors
    TIMEOUT = "timeout"
    ERROR = "error"
    MAX_ROUNDS_EXCEEDED = "max_rounds_exceeded"


@dataclass
class SimulatedStep:
    """A step in a simulated flow."""

    step_number: int
    agent: str
    action: str
    input_data: dict
    output_data: dict | None = None
    error: str | None = None
    duration_ms: int = 0

    @property
    def succeeded(self) -> bool:
        return self.error is None and self.output_data is not None


@dataclass
class SimulatedFlow:
    """A complete simulated orchestration flow."""

    flow_type: str
    steps: list[SimulatedStep] = field(default_factory=list)
    outcome: FlowOutcome | None = None
    final_output: dict = field(default_factory=dict)

    def add_step(
        self,
        agent: str,
        action: str,
        input_data: dict,
        output_data: dict | None = None,
        error: str | None = None,
    ) -> SimulatedStep:
        """Add a step to the flow."""
        step = SimulatedStep(
            step_number=len(self.steps) + 1,
            agent=agent,
            action=action,
            input_data=input_data,
            output_data=output_data,
            error=error,
        )
        self.steps.append(step)
        return step

    @property
    def total_steps(self) -> int:
        return len(self.steps)

    @property
    def all_succeeded(self) -> bool:
        return all(s.succeeded for s in self.steps)


class FlowSimulator:
    """Simulate orchestration flows with controlled behavior.

    Example:
        simulator = FlowSimulator()

        # Configure agent responses
        simulator.set_agent_response(
            "vkg-attacker",
            "analyze",
            {"exploit_path": [...], "confidence": 0.8}
        )
        simulator.set_agent_response(
            "vkg-defender",
            "defend",
            {"guards": [...], "strength": 0.3}
        )

        # Run flow
        result = simulator.run_debate_flow(bead_data)

        assert result.outcome == FlowOutcome.CONFIRMED
    """

    def __init__(self):
        self.recorder = MessageRecorder()
        self._agent_responses: dict[str, dict[str, Any]] = {}
        self._agent_errors: dict[str, dict[str, Exception]] = {}

    def set_agent_response(
        self,
        agent: str,
        action: str,
        response: dict
    ) -> "FlowSimulator":
        """Set a predetermined response for an agent action."""
        if agent not in self._agent_responses:
            self._agent_responses[agent] = {}
        self._agent_responses[agent][action] = response
        return self

    def set_agent_error(
        self,
        agent: str,
        action: str,
        error: Exception
    ) -> "FlowSimulator":
        """Set an agent to fail for a specific action."""
        if agent not in self._agent_errors:
            self._agent_errors[agent] = {}
        self._agent_errors[agent][action] = error
        return self

    def _get_agent_response(self, agent: str, action: str) -> dict | None:
        """Get configured response for agent action."""
        if agent in self._agent_errors and action in self._agent_errors[agent]:
            raise self._agent_errors[agent][action]
        if agent in self._agent_responses and action in self._agent_responses[agent]:
            return self._agent_responses[agent][action]
        return None

    def run_debate_flow(
        self,
        bead_data: dict,
        max_rounds: int = 3
    ) -> SimulatedFlow:
        """Simulate a debate flow.

        Debate pattern:
        1. Attacker makes claim
        2. Defender responds with counterclaim
        3. (Optional) Additional rounds
        4. Verifier produces verdict

        Args:
            bead_data: Input bead with finding details
            max_rounds: Maximum debate rounds before forcing verdict

        Returns:
            SimulatedFlow with all steps and outcome
        """
        flow = SimulatedFlow(flow_type="debate")
        self.recorder.start_flow("debate", bead_id=bead_data.get("id"))

        rounds = 0

        # Step 1: Attacker claim
        attacker_response = self._get_agent_response("vkg-attacker", "claim")
        if attacker_response is None:
            attacker_response = {
                "claim": "Vulnerability is exploitable",
                "exploit_path": ["step1", "step2"],
                "confidence": 0.7
            }

        flow.add_step(
            agent="vkg-attacker",
            action="claim",
            input_data=bead_data,
            output_data=attacker_response
        )

        self.recorder.record_message(
            from_agent="vkg-attacker",
            to_agent="vkg-defender",
            message_type=MessageType.CLAIM,
            content=attacker_response
        )

        # Step 2: Defender counterclaim
        defender_response = self._get_agent_response("vkg-defender", "counterclaim")
        if defender_response is None:
            defender_response = {
                "counterclaim": "Guards prevent exploitation",
                "guards": ["ReentrancyGuard"],
                "strength": 0.5
            }

        flow.add_step(
            agent="vkg-defender",
            action="counterclaim",
            input_data=attacker_response,
            output_data=defender_response
        )

        self.recorder.record_message(
            from_agent="vkg-defender",
            to_agent="vkg-attacker",
            message_type=MessageType.COUNTERCLAIM,
            content=defender_response
        )

        rounds += 1

        # Optional additional rounds
        while rounds < max_rounds:
            # Check if either side wants to continue
            attacker_continue = self._get_agent_response("vkg-attacker", "continue")
            if attacker_continue is None:
                break  # Default: don't continue

            defender_continue = self._get_agent_response("vkg-defender", "continue")
            if defender_continue is None:
                break

            # Add round
            flow.add_step(
                agent="vkg-attacker",
                action="evidence",
                input_data=defender_response,
                output_data=attacker_continue
            )
            flow.add_step(
                agent="vkg-defender",
                action="evidence",
                input_data=attacker_continue,
                output_data=defender_continue
            )

            self.recorder.record_message(
                from_agent="vkg-attacker",
                to_agent="vkg-defender",
                message_type=MessageType.EVIDENCE,
                content=attacker_continue
            )
            self.recorder.record_message(
                from_agent="vkg-defender",
                to_agent="vkg-attacker",
                message_type=MessageType.EVIDENCE,
                content=defender_continue
            )

            rounds += 1

        # Step 3: Verifier verdict
        verifier_response = self._get_agent_response("vkg-verifier", "verdict")
        if verifier_response is None:
            # Determine verdict based on confidence comparison
            attacker_conf = attacker_response.get("confidence", 0.5)
            defender_strength = defender_response.get("strength", 0.5)

            if attacker_conf > defender_strength + 0.2:
                verifier_response = {"verdict": "CONFIRMED", "confidence": 0.8}
            elif defender_strength > attacker_conf + 0.2:
                verifier_response = {"verdict": "REJECTED", "confidence": 0.8}
            else:
                verifier_response = {"verdict": "UNCERTAIN", "confidence": 0.5}

        flow.add_step(
            agent="vkg-verifier",
            action="verdict",
            input_data={
                "attacker_claim": attacker_response,
                "defender_counterclaim": defender_response,
                "rounds": rounds
            },
            output_data=verifier_response
        )

        self.recorder.record_message(
            from_agent="vkg-verifier",
            to_agent="orchestrator",
            message_type=MessageType.VERDICT,
            content=verifier_response
        )

        # Set outcome
        verdict = verifier_response.get("verdict", "UNCERTAIN")
        flow.outcome = FlowOutcome[verdict]
        flow.final_output = verifier_response

        self.recorder.end_flow(outcome=verdict)

        return flow

    def run_verification_flow(
        self,
        finding: dict,
        verification_mode: str = "standard"
    ) -> SimulatedFlow:
        """Simulate a verification flow.

        Verification pattern:
        1. Receive finding
        2. Run verification checks
        3. Produce verified/refuted result

        Args:
            finding: Finding to verify
            verification_mode: "standard", "thorough", or "quick"

        Returns:
            SimulatedFlow with verification result
        """
        flow = SimulatedFlow(flow_type="verification")
        self.recorder.start_flow("verification", finding_id=finding.get("id"))

        # Step 1: Receive finding
        flow.add_step(
            agent="orchestrator",
            action="receive",
            input_data=finding,
            output_data={"status": "received"}
        )

        self.recorder.record_message(
            from_agent="bead_source",
            to_agent="vkg-verifier",
            message_type=MessageType.VERIFICATION_REQUEST,
            content=finding
        )

        # Step 2: Run verification
        verifier_response = self._get_agent_response("vkg-verifier", "verify")
        if verifier_response is None:
            # Default: verify based on confidence
            confidence = finding.get("confidence", 0.5)
            if confidence >= 0.8:
                verifier_response = {
                    "status": "VERIFIED",
                    "confidence": confidence,
                    "evidence": ["High confidence original finding"]
                }
            elif confidence <= 0.3:
                verifier_response = {
                    "status": "REFUTED",
                    "confidence": 1 - confidence,
                    "evidence": ["Low confidence, likely false positive"]
                }
            else:
                verifier_response = {
                    "status": "NEEDS_MORE_EVIDENCE",
                    "confidence": 0.5,
                    "evidence": ["Inconclusive"]
                }

        flow.add_step(
            agent="vkg-verifier",
            action="verify",
            input_data=finding,
            output_data=verifier_response
        )

        self.recorder.record_message(
            from_agent="vkg-verifier",
            to_agent="orchestrator",
            message_type=MessageType.VERIFICATION_RESULT,
            content=verifier_response
        )

        # Set outcome
        status = verifier_response.get("status", "NEEDS_MORE_EVIDENCE")
        flow.outcome = FlowOutcome[status]
        flow.final_output = verifier_response

        self.recorder.end_flow(outcome=status)

        return flow

    def run_routing_flow(
        self,
        bead: dict
    ) -> SimulatedFlow:
        """Simulate a routing flow.

        Routing pattern:
        1. Analyze bead characteristics
        2. Decide which agent(s) to assign
        3. Hand off to selected agent(s)

        Args:
            bead: Bead to route

        Returns:
            SimulatedFlow with routing decision
        """
        flow = SimulatedFlow(flow_type="routing")
        self.recorder.start_flow("routing", bead_id=bead.get("id"))

        # Step 1: Analyze
        bead_type = bead.get("type", "finding")
        severity = bead.get("severity", "medium")

        analysis = {
            "bead_type": bead_type,
            "severity": severity,
            "requires_debate": severity in ["critical", "high"],
            "requires_verification": True
        }

        flow.add_step(
            agent="orchestrator",
            action="analyze",
            input_data=bead,
            output_data=analysis
        )

        # Step 2: Route decision
        routing_response = self._get_agent_response("orchestrator", "route")
        if routing_response is None:
            if analysis["requires_debate"]:
                routing_response = {
                    "route": "debate",
                    "agents": ["vkg-attacker", "vkg-defender", "vkg-verifier"],
                    "reason": "High severity requires adversarial verification"
                }
            else:
                routing_response = {
                    "route": "direct_verification",
                    "agents": ["vkg-verifier"],
                    "reason": "Standard verification sufficient"
                }

        flow.add_step(
            agent="orchestrator",
            action="route",
            input_data=analysis,
            output_data=routing_response
        )

        self.recorder.record_message(
            from_agent="orchestrator",
            to_agent=routing_response["agents"][0],
            message_type=MessageType.ROUTE_DECISION,
            content=routing_response
        )

        # Set outcome
        flow.outcome = FlowOutcome.ROUTED
        flow.final_output = routing_response

        self.recorder.end_flow(outcome="ROUTED")

        return flow

    def get_recorder(self) -> MessageRecorder:
        """Get the message recorder for analysis."""
        return self.recorder

from typing import Callable, Any
import logging

    FlowSimulator,

@dataclass
class StepAssertion:
    """An assertion about a step in the flow."""

    step_number: int | None = None  # None = any step
    agent: str | None = None
    action: str | None = None
    output_contains: dict | None = None
    custom_check: Callable[[Any], bool] | None = None
    description: str = ""

    def check(self, flow: SimulatedFlow) -> tuple[bool, str]:
        """Check assertion against flow.

        Returns:
            (passed, reason)
        """
        steps_to_check = flow.steps

        if self.step_number is not None:
            if self.step_number > len(flow.steps):
                return False, f"Step {self.step_number} does not exist"
            steps_to_check = [flow.steps[self.step_number - 1]]

        for step in steps_to_check:
            if self.agent and step.agent != self.agent:
                continue
            if self.action and step.action != self.action:
                continue

            # Found matching step, check output
            if self.output_contains:
                for key, expected in self.output_contains.items():
                    actual = step.output_data.get(key) if step.output_data else None
                    if actual != expected:
                        return False, f"Step {step.step_number}: {key}={actual}, expected {expected}"

            if self.custom_check:
                if not self.custom_check(step):
                    return False, f"Custom check failed for step {step.step_number}"

            return True, "Assertion passed"

        return False, f"No matching step found for assertion: {self.description}"


@dataclass
class FlowTestResult:
    """Result of testing an orchestration flow."""

    # Test identification
    test_name: str
    flow_type: str

    # Pass/fail
    passed: bool
    error: str | None = None

    # Flow details
    flow: SimulatedFlow | None = None
    trace: ConversationTrace | None = None

    # Assertion results
    assertions_passed: int = 0
    assertions_failed: int = 0
    assertion_details: list[tuple[str, bool, str]] = field(default_factory=list)

    # Metrics
    total_steps: int = 0
    total_messages: int = 0
    duration_ms: int = 0

    def add_assertion_result(self, description: str, passed: bool, reason: str) -> None:
        """Add an assertion result."""
        self.assertion_details.append((description, passed, reason))
        if passed:
            self.assertions_passed += 1
        else:
            self.assertions_failed += 1

    @property
    def all_assertions_passed(self) -> bool:
        """Check if all assertions passed."""
        return self.assertions_failed == 0


class OrchestratorTester:
    """Test sub-orchestration flows.

    Example:
        tester = OrchestratorTester()

        # Test debate flow
        result = tester.test_debate_flow(
            test_name="reentrancy_debate",
            bead_data={"id": "bead-1", "finding": "reentrancy"},
            expected_outcome=FlowOutcome.CONFIRMED,
            assertions=[
                StepAssertion(agent="vkg-attacker", action="claim"),
                StepAssertion(agent="vkg-defender", action="counterclaim"),
                StepAssertion(agent="vkg-verifier", action="verdict"),
            ]
        )

        assert result.passed
        assert result.all_assertions_passed
    """

    def __init__(self, mode: InvocationMode = InvocationMode.MOCK):
        """Initialize tester.

        Args:
            mode: Invocation mode for underlying subagent tests
        """
        self.mode = mode
        self.simulator = FlowSimulator()
        self.subagent_tester = SubagentTester(mode=mode)

    def configure_agent_response(
        self,
        agent: str,
        action: str,
        response: dict
    ) -> "OrchestratorTester":
        """Configure a predetermined agent response."""
        self.simulator.set_agent_response(agent, action, response)
        return self

    def configure_agent_error(
        self,
        agent: str,
        action: str,
        error: Exception
    ) -> "OrchestratorTester":
        """Configure an agent to fail."""
        self.simulator.set_agent_error(agent, action, error)
        return self

    def test_debate_flow(
        self,
        test_name: str,
        bead_data: dict,
        expected_outcome: FlowOutcome | None = None,
        expected_rounds: int | None = None,
        assertions: list[StepAssertion] | None = None,
        max_rounds: int = 3,
    ) -> FlowTestResult:
        """Test a debate flow.

        Args:
            test_name: Name of this test
            bead_data: Input bead data
            expected_outcome: Expected flow outcome
            expected_rounds: Expected number of debate rounds
            assertions: List of assertions to check
            max_rounds: Maximum rounds before forcing verdict

        Returns:
            FlowTestResult with detailed analysis
        """
        try:
            # Run simulation
            flow = self.simulator.run_debate_flow(bead_data, max_rounds)
            trace = self.simulator.get_recorder().get_trace()

            result = FlowTestResult(
                test_name=test_name,
                flow_type="debate",
                passed=True,
                flow=flow,
                trace=trace,
                total_steps=flow.total_steps,
                total_messages=trace.message_count if trace else 0,
            )

            # Check expected outcome
            if expected_outcome and flow.outcome != expected_outcome:
                result.passed = False
                result.error = f"Expected outcome {expected_outcome.value}, got {flow.outcome.value if flow.outcome else None}"

            # Check round count
            if expected_rounds is not None:
                actual_rounds = self._count_debate_rounds(flow)
                if actual_rounds != expected_rounds:
                    result.passed = False
                    result.error = f"Expected {expected_rounds} rounds, got {actual_rounds}"

            # Check assertions
            if assertions:
                for assertion in assertions:
                    passed, reason = assertion.check(flow)
                    result.add_assertion_result(
                        assertion.description or f"Assertion for {assertion.agent}",
                        passed,
                        reason
                    )

                if not result.all_assertions_passed:
                    result.passed = False

            return result

        except Exception as e:
            return FlowTestResult(
                test_name=test_name,
                flow_type="debate",
                passed=False,
                error=str(e),
            )

    def test_routing_flow(
        self,
        test_name: str,
        bead: dict,
        expected_route: str | None = None,
        expected_agents: list[str] | None = None,
        assertions: list[StepAssertion] | None = None,
    ) -> FlowTestResult:
        """Test a routing flow.

        Args:
            test_name: Name of this test
            bead: Bead to route
            expected_route: Expected routing decision
            expected_agents: Expected agent assignment
            assertions: List of assertions to check

        Returns:
            FlowTestResult with detailed analysis

            flow = self.simulator.run_routing_flow(bead)

                if actual_route != expected_route:
