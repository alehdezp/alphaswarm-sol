"""ClaudeCodeRunner - Programmatic Claude Code invocation.

This module wraps the Claude Code CLI for automated testing.
Uses `claude -p` (headless mode) with JSON output for parsing.

Key Features:
- Pre-approve tools via --allowedTools (no human intervention)
- Structured output via --output-format json
- Schema validation via --json-schema
- Session continuation via --resume

Limitations:
- Skills (/vrs-*) don't work in headless mode
- Use prompt descriptions instead of skill invocations
"""

import subprocess
import json
import logging
from pathlib import Path
from dataclasses import dataclass
from typing import Any

logger = logging.getLogger(__name__)

@dataclass
class ClaudeCodeResult:
    """Result from Claude Code execution."""
    raw_output: str
    parsed_json: dict | None
    result_text: str
    session_id: str | None
    structured_output: dict | None
    return_code: int
    stderr: str

class ClaudeCodeRunner:
    """Programmatic Claude Code invocation for autonomous testing.

    Example:
        runner = ClaudeCodeRunner(project_root=Path("."))

        result = runner.run_analysis(
            prompt="Analyze contracts/ for reentrancy",
            allowed_tools=["Bash(uv run true-vkg*)", "Read", "Glob"],
            json_schema={"type": "object", "properties": {...}}
        )

        print(result.structured_output)
    """

    def __init__(
        self,
        project_root: Path,
        default_timeout: int = 300,
        default_model: str | None = None
    ):
        self.project_root = Path(project_root)
        self.default_timeout = default_timeout
        self.default_model = default_model

    def run_analysis(
        self,
        prompt: str,
        allowed_tools: list[str] | None = None,
        system_prompt: str | None = None,
        append_system_prompt: str | None = None,
        json_schema: dict | None = None,
        timeout_seconds: int | None = None,
        resume_session: str | None = None,
        model: str | None = None
    ) -> ClaudeCodeResult:
        """Run Claude Code in headless mode.

        Args:
            prompt: The analysis prompt
            allowed_tools: Tools to pre-approve (no human confirmation needed)
                           Supports wildcards: "Bash(git*)", "Bash(uv run*)"
            system_prompt: Replace system prompt entirely
            append_system_prompt: Append to default system prompt
            json_schema: Schema for structured output validation
            timeout_seconds: Override default timeout
            resume_session: Session ID to continue from
            model: Model to use (default: claude-sonnet-4)

        Returns:
            ClaudeCodeResult with parsed JSON, structured output, etc.

        Raises:
            RuntimeError: If Claude Code fails with non-zero exit
            TimeoutError: If execution exceeds timeout
        """
        cmd = self._build_command(
            prompt=prompt,
            allowed_tools=allowed_tools,
            system_prompt=system_prompt,
            append_system_prompt=append_system_prompt,
            json_schema=json_schema,
            resume_session=resume_session,
            model=model or self.default_model
        )

        timeout = timeout_seconds or self.default_timeout

        logger.info(f"Running Claude Code: {' '.join(cmd[:5])}...")

        try:
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=timeout,
                cwd=self.project_root
            )
        except subprocess.TimeoutExpired as e:
            raise TimeoutError(
                f"Claude Code timed out after {timeout}s: {e.stderr}"
            )

        return self._parse_result(result)

    def _build_command(
        self,
        prompt: str,
        allowed_tools: list[str] | None,
        system_prompt: str | None,
        append_system_prompt: str | None,
        json_schema: dict | None,
        resume_session: str | None,
        model: str | None
    ) -> list[str]:
        """Build the claude CLI command."""
        cmd = ["claude", "-p", prompt, "--output-format", "json"]

        if allowed_tools:
            cmd.extend(["--allowedTools", ",".join(allowed_tools)])

        if system_prompt:
            cmd.extend(["--system-prompt", system_prompt])

        if append_system_prompt:
            cmd.extend(["--append-system-prompt", append_system_prompt])

        if json_schema:
            cmd.extend(["--json-schema", json.dumps(json_schema)])

        if resume_session:
            cmd.extend(["--resume", resume_session])

        if model:
            cmd.extend(["--model", model])

        return cmd

    def _parse_result(self, result: subprocess.CompletedProcess) -> ClaudeCodeResult:
        """Parse subprocess result into ClaudeCodeResult."""
        parsed_json = None
        result_text = ""
        session_id = None
        structured_output = None

        if result.stdout:
            try:
                parsed_json = json.loads(result.stdout)
                result_text = parsed_json.get("result", "")
                session_id = parsed_json.get("session_id")
                structured_output = parsed_json.get("structured_output")
            except json.JSONDecodeError:
                result_text = result.stdout

        return ClaudeCodeResult(
            raw_output=result.stdout,
            parsed_json=parsed_json,
            result_text=result_text,
            session_id=session_id,
            structured_output=structured_output,
            return_code=result.returncode,
            stderr=result.stderr
        )

    def run_vkg_analysis(
        self,
        contract_path: str,
        vuln_class: str | None = None,
        include_labels: bool = False
    ) -> ClaudeCodeResult:
        """Run VKG analysis on a contract.

        This is a convenience method that constructs the appropriate
        prompt and tools for VKG-based vulnerability analysis.
        """
        prompt = f"""
        Analyze {contract_path} for security vulnerabilities using the VKG.

        Steps:
        1. Build the knowledge graph: uv run true-vkg build-kg {contract_path}
        2. Query for suspicious patterns using VQL
        3. Verify findings with semantic operations

        {"Focus on: " + vuln_class if vuln_class else "Check all vulnerability classes."}

        Return structured findings with:
        - has_vulnerability: boolean
        - findings: array of {{pattern, severity, location, confidence}}
        - reasoning: explanation of analysis
        """

        build_cmd = "uv run true-vkg build-kg*"
        if include_labels:
            build_cmd = "uv run true-vkg build-kg*--with-labels*"

        return self.run_analysis(
            prompt=prompt,
            allowed_tools=[
                f"Bash({build_cmd})",
                "Bash(uv run true-vkg query*)",
                "Read",
                "Glob",
                "Grep"
            ],
            json_schema={
                "type": "object",
                "properties": {
                    "has_vulnerability": {"type": "boolean"},
                    "findings": {
                        "type": "array",
                        "items": {
                            "type": "object",
                            "properties": {
                                "pattern": {"type": "string"},
                                "severity": {"type": "string"},
                                "location": {"type": "string"},
                                "confidence": {"type": "number"}
                            }
                        }
                    },
                    "reasoning": {"type": "string"}
                },
                "required": ["has_vulnerability", "findings", "reasoning"]
            }
        )

"""Output parser for Claude Code results.

Extracts structured data from Claude Code JSON output
and converts to TestMetrics for accuracy measurement.
"""

from dataclasses import dataclass
from typing import Any

from true_vkg.testing.metrics import TestMetrics, CategoryMetrics

@dataclass
class ParsedFinding:
    """A parsed vulnerability finding."""
    pattern: str
    severity: str
    location: str
    confidence: float
    reasoning: str | None = None

class OutputParser:
    """Parse Claude Code output into structured findings."""

    @staticmethod
    def parse_findings(structured_output: dict | None) -> list[ParsedFinding]:
        """Extract findings from structured output."""
        if not structured_output:
            return []

        findings = structured_output.get("findings", [])
        return [
            ParsedFinding(
                pattern=f.get("pattern", "unknown"),
                severity=f.get("severity", "medium"),
                location=f.get("location", "unknown"),
                confidence=f.get("confidence", 0.5),
                reasoning=f.get("reasoning")
            )
            for f in findings
        ]

    @staticmethod
    def calculate_accuracy(
        findings: list[ParsedFinding],
        ground_truth: list[dict],
        severity_weights: dict[str, float] | None = None
    ) -> dict[str, float]:
        """Calculate precision, recall, F1 against ground truth.

        Args:
            findings: Parsed findings from analysis
            ground_truth: Expected findings with pattern, severity, location
            severity_weights: Weight for each severity level

        Returns:
            Dict with precision, recall, f1_score, by_severity breakdown
        """
        weights = severity_weights or {
            "critical": 1.0,
            "high": 0.8,
            "medium": 0.5,
            "low": 0.3
        }

        # Match findings to ground truth
        tp = 0  # True positives
        fp = 0  # False positives
        fn = 0  # False negatives

        matched_gt = set()

        for finding in findings:
            matched = False
            for i, gt in enumerate(ground_truth):
                if i in matched_gt:
                    continue
                if OutputParser._is_match(finding, gt):
                    tp += 1
                    matched_gt.add(i)
                    matched = True
                    break
            if not matched:
                fp += 1

        fn = len(ground_truth) - len(matched_gt)

        precision = tp / (tp + fp) if (tp + fp) > 0 else 0.0
        recall = tp / (tp + fn) if (tp + fn) > 0 else 0.0
        f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0.0

        return {
            "precision": precision,
            "recall": recall,
            "f1_score": f1,
            "true_positives": tp,
            "false_positives": fp,
            "false_negatives": fn
        }

    @staticmethod
    def _is_match(finding: ParsedFinding, ground_truth: dict) -> bool:
        """Check if finding matches ground truth entry."""
        # Pattern must match
        if finding.pattern != ground_truth.get("pattern"):
            return False

        # Location should be similar (fuzzy match on function/line)
        gt_location = ground_truth.get("location", "")
        if gt_location and finding.location:
            # Simple substring match for now
            if gt_location not in finding.location and finding.location not in gt_location:
                return False

        return True

"""AgentSpawner - Parallel agent orchestration for autonomous testing.

Spawns multiple Claude Code agents in parallel for:
- Model comparison benchmarks
- Parallel vulnerability analysis
- Stress testing with multiple scenarios

Uses subprocess pools for concurrent execution.
"""

import asyncio
import logging
from concurrent.futures import ThreadPoolExecutor, as_completed
from dataclasses import dataclass
from pathlib import Path
from typing import Callable

from true_vkg.testing.harness.runner import ClaudeCodeRunner, ClaudeCodeResult

logger = logging.getLogger(__name__)

@dataclass
class AgentTask:
    """A task for an agent to execute."""
    task_id: str
    prompt: str
    allowed_tools: list[str]
    json_schema: dict | None = None
    model: str | None = None
    timeout_seconds: int = 300

@dataclass
class AgentResult:
    """Result from an agent task."""
    task_id: str
    result: ClaudeCodeResult | None
    error: Exception | None
    duration_ms: int

class AgentSpawner:
    """Spawn multiple agents in parallel for autonomous testing.

    Example:
        spawner = AgentSpawner(runner, max_workers=4)

        tasks = [
            AgentTask(
                task_id="reentrancy-opus",
                prompt="Analyze for reentrancy",
                allowed_tools=["Bash(uv run*)", "Read"],
                model="claude-opus-4"
            ),
            AgentTask(
                task_id="reentrancy-sonnet",
                prompt="Analyze for reentrancy",
                allowed_tools=["Bash(uv run*)", "Read"],
                model="claude-sonnet-4"
            ),
        ]

        results = spawner.run_parallel(tasks)
    """

    def __init__(
        self,
        runner: ClaudeCodeRunner,
        max_workers: int = 4
    ):
        self.runner = runner
        self.max_workers = max_workers

    def run_parallel(
        self,
        tasks: list[AgentTask],
        progress_callback: Callable[[str, int, int], None] | None = None
    ) -> list[AgentResult]:
        """Run multiple agent tasks in parallel.

        Args:
            tasks: List of AgentTask to execute
            progress_callback: Called with (task_id, completed, total)

        Returns:
            List of AgentResult in same order as input tasks
        """
        results: dict[str, AgentResult] = {}
        completed = 0

        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            future_to_task = {
                executor.submit(self._run_task, task): task
                for task in tasks
            }

            for future in as_completed(future_to_task):
                task = future_to_task[future]
                result = future.result()
                results[task.task_id] = result

                completed += 1
                if progress_callback:
                    progress_callback(task.task_id, completed, len(tasks))

        # Return in original order
        return [results[task.task_id] for task in tasks]

    def _run_task(self, task: AgentTask) -> AgentResult:
        """Execute a single agent task."""
        import time
        start = time.monotonic()

        try:
            result = self.runner.run_analysis(
                prompt=task.prompt,
                allowed_tools=task.allowed_tools,
                json_schema=task.json_schema,
                model=task.model,
                timeout_seconds=task.timeout_seconds
            )
            error = None
        except Exception as e:
            result = None
            error = e
            logger.error(f"Task {task.task_id} failed: {e}")

        duration_ms = int((time.monotonic() - start) * 1000)

        return AgentResult(
            task_id=task.task_id,
            result=result,
            error=error,
            duration_ms=duration_ms
        )

    def run_model_comparison(
        self,
        prompt: str,
        allowed_tools: list[str],
        models: list[str],
        json_schema: dict | None = None
    ) -> dict[str, AgentResult]:
        """Run same prompt against multiple models for comparison.

        Args:
            prompt: Analysis prompt
            allowed_tools: Tools to pre-approve
            models: List of model IDs to compare
            json_schema: Output schema

        Returns:
            Dict mapping model -> AgentResult
        """
        tasks = [
            AgentTask(
                task_id=model,
                prompt=prompt,
                allowed_tools=allowed_tools,
                json_schema=json_schema,
                model=model
            )
            for model in models
        ]

        results = self.run_parallel(tasks)
        return {task.task_id: result for task, result in zip(tasks, results)}

"""ScenarioLoader - Load test scenarios from YAML configuration.

Scenarios define:
- Contract(s) to analyze
- Expected findings (ground truth)
- Prompt template
- Allowed tools
- Output schema
- Timeout and model settings
"""

import yaml
import logging
from pathlib import Path
from dataclasses import dataclass, field

logger = logging.getLogger(__name__)

@dataclass
class ContractCase:
    """A contract test case within a scenario."""
    path: str
    has_vulnerability: bool
    expected_pattern: str | None = None
    expected_severity: str | None = None
    ground_truth: list[dict] = field(default_factory=list)

@dataclass
class TestScenario:
    """A complete test scenario configuration."""
    name: str
    category: str
    description: str
    contracts: list[ContractCase]
    prompt_template: str
    allowed_tools: list[str]
    json_schema: dict
    timeout_seconds: int = 120
    model: str | None = None

    # Isolation settings for blind testing
    exclude_knowledge: bool = False  # Don't include vulndocs
    blind_prompt: bool = False       # Don't hint at vulnerability type

class ScenarioLoader:
    """Load test scenarios from YAML files.

    Example:
        loader = ScenarioLoader(Path("src/true_vkg/testing/scenarios"))
        scenarios = loader.load_all()

        for name, scenario in scenarios.items():
            for contract in scenario.contracts:
                # Run analysis and compare to ground truth
                ...
    """

    def __init__(self, scenarios_dir: Path):
        self.scenarios_dir = Path(scenarios_dir)

    def load_all(self) -> dict[str, TestScenario]:
        """Load all scenarios from directory."""
        scenarios = {}

        if not self.scenarios_dir.exists():
            logger.warning(f"Scenarios directory not found: {self.scenarios_dir}")
            return scenarios

        for config_file in self.scenarios_dir.rglob("config.yaml"):
            try:
                scenario = self.load_scenario(config_file)
                scenarios[scenario.name] = scenario
            except Exception as e:
                logger.error(f"Failed to load {config_file}: {e}")

        return scenarios

    def load_scenario(self, config_path: Path) -> TestScenario:
        """Load a single scenario from YAML config."""
        with open(config_path) as f:
            data = yaml.safe_load(f)

        contracts = [
            ContractCase(
                path=c["path"],
                has_vulnerability=c.get("has_vulnerability", True),
                expected_pattern=c.get("expected_pattern"),
                expected_severity=c.get("expected_severity"),
                ground_truth=c.get("ground_truth", [])
            )
            for c in data.get("contracts", [])
        ]

        return TestScenario(
            name=data["name"],
            category=data.get("category", "unknown"),
            description=data.get("description", ""),
            contracts=contracts,
            prompt_template=data["prompt_template"],
            allowed_tools=data.get("allowed_tools", []),
            json_schema=data.get("json_schema", {}),
            timeout_seconds=data.get("timeout_seconds", 120),
            model=data.get("model"),
            exclude_knowledge=data.get("isolation", {}).get("exclude_knowledge", False),
            blind_prompt=data.get("isolation", {}).get("blind_prompt", False)
        )

    def load_by_category(self, category: str) -> list[TestScenario]:
        """Load all scenarios for a specific category."""
        all_scenarios = self.load_all()
        return [s for s in all_scenarios.values() if s.category == category]
