"""SubagentTester - Unit testing infrastructure for individual subagents.

Supports both CLI invocation (subprocess) and direct Task spawning
for comprehensive subagent testing.

Level 1 Testing Goals:
- Verify subagent produces expected output schema
- Verify subagent uses VKG tools correctly
- Verify subagent handles edge cases
- Verify subagent follows its defined persona
"""

import json
import logging
import time
from dataclasses import dataclass, field
from enum import Enum
from pathlib import Path
from typing import Any, Callable

from true_vkg.testing.harness import ClaudeCodeRunner
from true_vkg.testing.unit.contracts import ExpectedOutput, OutputSchema, AgentContract

logger = logging.getLogger(__name__)


class InvocationMode(Enum):
    """How to invoke the subagent."""
    CLI = "cli"              # Via subprocess `claude -p`
    DIRECT = "direct"        # Via Task tool (within Claude Code session)
    MOCK = "mock"            # With mocked tools for fast unit tests


@dataclass
class SubagentTestResult:
    """Result from a subagent unit test."""

    # Identification
    agent_type: str
    test_name: str
    invocation_mode: InvocationMode

    # Status
    passed: bool
    error: str | None = None

    # Output analysis
    raw_output: str = ""
    parsed_output: dict | None = None
    schema_valid: bool = False
    schema_errors: list[str] = field(default_factory=list)

    # Behavior analysis
    used_expected_tools: bool = False
    tools_called: list[str] = field(default_factory=list)
    followed_persona: bool = True
    persona_violations: list[str] = field(default_factory=list)

    # Metrics
    duration_ms: int = 0
    token_count: int = 0

    def to_dict(self) -> dict:
        """Convert to dictionary for serialization."""
        return {
            "agent_type": self.agent_type,
            "test_name": self.test_name,
            "invocation_mode": self.invocation_mode.value,
            "passed": self.passed,
            "error": self.error,
            "schema_valid": self.schema_valid,
            "used_expected_tools": self.used_expected_tools,
            "tools_called": self.tools_called,
            "duration_ms": self.duration_ms,
        }


class SubagentTester:
    """Test individual subagents in isolation.

    Supports three invocation modes:
    1. CLI - Invoke via subprocess for black-box testing
    2. DIRECT - Invoke via Task tool for white-box testing
    3. MOCK - Use mocked tools for fast unit tests

    Example:
        tester = SubagentTester(
            mode=InvocationMode.CLI,
            project_root=Path("."),
            timeout=120
        )

        # Test attacker agent
        result = tester.test_agent(
            agent_type="vkg-attacker",
            test_name="exploit_path_generation",
            prompt="Analyze this reentrancy for exploitability: ...",
            expected_schema=OutputSchema(
                required_fields=["exploit_path", "preconditions", "confidence"],
                field_types={"confidence": float}
            ),
            expected_tools=["Read", "Grep"]
        )

        assert result.passed
        assert result.schema_valid
    """

    # Known VKG subagent types
    VKG_AGENTS = [
        "vkg-attacker",
        "vkg-defender",
        "vkg-verifier",
        "vkg-test-builder",
        "vkg-supervisor",
        "vkg-integrator",
    ]

    def __init__(
        self,
        mode: InvocationMode = InvocationMode.CLI,
        project_root: Path | None = None,
        timeout: int = 120,
        model: str = "haiku",  # Fast model for unit tests
    ):
        """Initialize tester.

        Args:
            mode: How to invoke subagents
            project_root: Root directory for CLI invocation
            timeout: Default timeout in seconds
            model: Default model for testing (haiku for speed)
        """
        self.mode = mode
        self.project_root = project_root or Path(".")
        self.timeout = timeout
        self.model = model

        if mode == InvocationMode.CLI:
            self._runner = ClaudeCodeRunner(
                project_root=self.project_root,
                default_timeout=timeout,
                default_model=model
            )
        else:
            self._runner = None

    def test_agent(
        self,
        agent_type: str,
        test_name: str,
        prompt: str,
        expected_schema: OutputSchema | None = None,
        expected_tools: list[str] | None = None,
        expected_outputs: list[ExpectedOutput] | None = None,
        contract: AgentContract | None = None,
        allowed_tools: list[str] | None = None,
        timeout: int | None = None,
    ) -> SubagentTestResult:
        """Run a unit test on a subagent.

        Args:
            agent_type: Type of subagent (e.g., "vkg-attacker")
            test_name: Name of this test case
            prompt: Prompt to send to agent
            expected_schema: Schema that output must conform to
            expected_tools: Tools the agent should use
            expected_outputs: Specific outputs to check for
            contract: Full agent contract specification
            allowed_tools: Tools to pre-approve (CLI mode)
            timeout: Override default timeout

        Returns:
            SubagentTestResult with pass/fail and analysis
        """
        start_time = time.monotonic()

        try:
            if self.mode == InvocationMode.CLI:
                result = self._test_via_cli(
                    agent_type=agent_type,
                    prompt=prompt,
                    allowed_tools=allowed_tools or self._default_tools_for_agent(agent_type),
                    expected_schema=expected_schema,
                    timeout=timeout or self.timeout,
                )
            elif self.mode == InvocationMode.DIRECT:
                result = self._test_via_direct(
                    agent_type=agent_type,
                    prompt=prompt,
                    expected_schema=expected_schema,
                    timeout=timeout or self.timeout,
                )
            else:
                result = self._test_via_mock(
                    agent_type=agent_type,
                    prompt=prompt,
                    expected_schema=expected_schema,
                )

            # Analyze result
            result.test_name = test_name
            result.duration_ms = int((time.monotonic() - start_time) * 1000)

            # Check expected tools
            if expected_tools:
                result.used_expected_tools = all(
                    any(exp in tool for tool in result.tools_called)
                    for exp in expected_tools
                )

            # Check expected outputs
            if expected_outputs and result.parsed_output:
                self._check_expected_outputs(result, expected_outputs)

            # Check contract compliance
            if contract:
                self._check_contract_compliance(result, contract)

            return result

        except Exception as e:
            logger.exception(f"Test {test_name} failed with exception")
            return SubagentTestResult(
                agent_type=agent_type,
                test_name=test_name,
                invocation_mode=self.mode,
                passed=False,
                error=str(e),
                duration_ms=int((time.monotonic() - start_time) * 1000),
            )

    def _test_via_cli(
        self,
        agent_type: str,
        prompt: str,
        allowed_tools: list[str],
        expected_schema: OutputSchema | None,
        timeout: int,
    ) -> SubagentTestResult:
        """Test subagent via CLI subprocess."""
        # Build prompt that spawns the subagent
        full_prompt = f"""
        Use the Task tool to spawn a {agent_type} agent with the following task:

        {prompt}

        Return the agent's complete output.
        """

        # Define JSON schema for output
        json_schema = {
            "type": "object",
            "properties": {
                "agent_output": {"type": "object"},
                "tools_used": {"type": "array", "items": {"type": "string"}},
                "agent_id": {"type": "string"},
            }
        }

        result = self._runner.run_analysis(
            prompt=full_prompt,
            allowed_tools=["Task"] + allowed_tools,
            json_schema=json_schema,
            timeout_seconds=timeout,
            model=self.model,
        )

        # Parse result
        parsed = result.structured_output or {}

        test_result = SubagentTestResult(
            agent_type=agent_type,
            test_name="",  # Set by caller
            invocation_mode=InvocationMode.CLI,
            passed=result.return_code == 0,
            error=result.stderr if result.return_code != 0 else None,
            raw_output=result.raw_output,
            parsed_output=parsed.get("agent_output"),
            tools_called=parsed.get("tools_used", []),
        )

        # Validate schema
        if expected_schema and test_result.parsed_output:
            test_result.schema_valid, test_result.schema_errors = expected_schema.validate(
                test_result.parsed_output
            )

        return test_result

    def _test_via_direct(
        self,
        agent_type: str,
        prompt: str,
        expected_schema: OutputSchema | None,
        timeout: int,
    ) -> SubagentTestResult:
        """Test subagent via direct Task tool invocation.

        Note: This method generates code/instructions for use within
        a Claude Code session. It cannot be executed standalone.
        """
        # This would be used within a Claude Code session where we have
        # access to the Task tool directly
        task_call = {
            "tool": "Task",
            "params": {
                "prompt": prompt,
                "subagent_type": agent_type,
                "model": self.model,
                "description": f"Unit test: {agent_type}",
            }
        }

        # In actual execution, this would call the Task tool
        # For now, return a placeholder indicating how to use
        return SubagentTestResult(
            agent_type=agent_type,
            test_name="",
            invocation_mode=InvocationMode.DIRECT,
            passed=True,  # Placeholder
            raw_output=json.dumps(task_call),
            parsed_output=task_call,
        )

    def _test_via_mock(
        self,
        agent_type: str,
        prompt: str,
        expected_schema: OutputSchema | None,
    ) -> SubagentTestResult:
        """Test with mocked tools for fast unit testing.

        Uses MockToolProvider to simulate tool responses without
        actual execution.
        """
        from true_vkg.testing.unit.mock_context import MockToolProvider

        mock_provider = MockToolProvider()

        # Simulate tool calls based on agent type
        tools_called = self._simulate_agent_tools(agent_type, prompt, mock_provider)

        return SubagentTestResult(
            agent_type=agent_type,
            test_name="",
            invocation_mode=InvocationMode.MOCK,
            passed=True,
            tools_called=tools_called,
            schema_valid=True,  # Mocked response always valid
        )

    def _default_tools_for_agent(self, agent_type: str) -> list[str]:
        """Get default allowed tools for an agent type."""
        tool_map = {
            "vkg-attacker": ["Read", "Glob", "Grep", "Bash(uv run python*)"],
            "vkg-defender": ["Read", "Glob", "Grep", "Bash(uv run python*)"],
            "vkg-verifier": ["Read", "Glob", "Grep", "Bash(uv run python*)"],
            "vkg-test-builder": ["Read", "Glob", "Grep", "Bash(uv run python*)", "Bash(forge*)", "Write"],
            "vkg-supervisor": ["Read", "Glob", "Grep", "Bash(uv run python*)"],
            "vkg-integrator": ["Read", "Glob", "Grep", "Bash(uv run python*)"],
        }
        return tool_map.get(agent_type, ["Read", "Glob", "Grep"])

    def _simulate_agent_tools(
        self,
        agent_type: str,
        prompt: str,
        mock_provider: "MockToolProvider"
    ) -> list[str]:
        """Simulate which tools an agent would call."""
        # Simple heuristic based on agent type and prompt
        tools = []

        if "vkg-attacker" in agent_type:
            tools = ["Read", "Grep"]  # Attacker reads and searches code
        elif "vkg-defender" in agent_type:
            tools = ["Read", "Glob", "Grep"]  # Defender searches for guards
        elif "vkg-verifier" in agent_type:
            tools = ["Read", "Grep"]  # Verifier cross-checks evidence

        return tools

    def _check_expected_outputs(
        self,
        result: SubagentTestResult,
        expected_outputs: list[ExpectedOutput]
    ) -> None:
        """Check that result contains expected outputs."""
        for expected in expected_outputs:
            if not expected.check(result.parsed_output):
                result.passed = False
                result.schema_errors.append(f"Missing expected output: {expected.description}")

    def _check_contract_compliance(
        self,
        result: SubagentTestResult,
        contract: AgentContract
    ) -> None:
        """Check that result complies with agent contract."""
        violations = contract.check_compliance(result)
        if violations:
            result.followed_persona = False
            result.persona_violations = violations


class SubagentTestSuite:
    """Collection of unit tests for a subagent."""

    def __init__(self, agent_type: str, tester: SubagentTester):
        self.agent_type = agent_type
        self.tester = tester
        self.tests: list[Callable[[], SubagentTestResult]] = []
        self.results: list[SubagentTestResult] = []

    def add_test(
        self,
        name: str,
        prompt: str,
        expected_schema: OutputSchema | None = None,
        expected_tools: list[str] | None = None,
    ) -> "SubagentTestSuite":
        """Add a test to the suite."""
        def test_fn():
            return self.tester.test_agent(
                agent_type=self.agent_type,
                test_name=name,
                prompt=prompt,
                expected_schema=expected_schema,
                expected_tools=expected_tools,
            )
        self.tests.append(test_fn)
        return self

    def run_all(self) -> list[SubagentTestResult]:
        """Run all tests in the suite."""
        self.results = [test() for test in self.tests]
        return self.results

    @property
    def all_passed(self) -> bool:
        """Check if all tests passed."""
        return all(r.passed for r in self.results)

    def summary(self) -> dict:
        """Get summary of test results."""
        return {
            "agent_type": self.agent_type,
            "total": len(self.results),
            "passed": sum(1 for r in self.results if r.passed),
            "failed": sum(1 for r in self.results if not r.passed),
            "tests": [r.to_dict() for r in self.results],
        }

"""Mock Context for Controlled Subagent Testing.

Provides mocked tool responses and VKG context for fast unit testing
without actual tool execution or API calls.

Use Cases:
- Fast unit tests (no network/subprocess overhead)
- Deterministic responses for reproducibility
- Edge case testing with injected failures
- Isolation from external dependencies
"""

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Callable


@dataclass
class InjectedResponse:
    """A pre-defined response for a tool call."""

    tool: str
    pattern: str  # Regex pattern to match against tool args
    response: str | dict | Exception
    use_count: int = 0
    max_uses: int | None = None  # None = unlimited

    def matches(self, tool: str, args: str) -> bool:
        """Check if this injection matches the tool call."""
        import re
        if self.tool != tool:
            return False
        if self.max_uses and self.use_count >= self.max_uses:
            return False
        return bool(re.search(self.pattern, args))

    def get_response(self) -> str | dict:
        """Get the response, raising if it's an exception."""
        self.use_count += 1
        if isinstance(self.response, Exception):
            raise self.response
        return self.response


class MockToolProvider:
    """Provides mocked tool responses for unit testing.

    Example:
        mock = MockToolProvider()

        # Inject specific response
        mock.inject(
            tool="Read",
            pattern=".*vulnerable.sol",
            response="contract Vulnerable { ... }"
        )

        # Inject failure
        mock.inject(
            tool="Bash",
            pattern=".*slither.*",
            response=TimeoutError("Tool timed out")
        )

        # Use in tests
        response = mock.call("Read", "contracts/vulnerable.sol")
    """

    def __init__(self):
        self.injections: list[InjectedResponse] = []
        self.call_history: list[tuple[str, str, Any]] = []
        self.default_responses: dict[str, Callable[[str], Any]] = {
            "Read": self._default_read,
            "Glob": self._default_glob,
            "Grep": self._default_grep,
            "Bash": self._default_bash,
        }

    def inject(
        self,
        tool: str,
        pattern: str,
        response: str | dict | Exception,
        max_uses: int | None = None
    ) -> "MockToolProvider":
        """Inject a response for matching tool calls."""
        self.injections.append(InjectedResponse(
            tool=tool,
            pattern=pattern,
            response=response,
            max_uses=max_uses,
        ))
        return self

    def call(self, tool: str, args: str) -> Any:
        """Simulate a tool call."""
        # Check for injected response
        for injection in self.injections:
            if injection.matches(tool, args):
                response = injection.get_response()
                self.call_history.append((tool, args, response))
                return response

        # Use default response
        if tool in self.default_responses:
            response = self.default_responses[tool](args)
            self.call_history.append((tool, args, response))
            return response

        # Unknown tool
        self.call_history.append((tool, args, None))
        return None

    def get_calls(self, tool: str | None = None) -> list[tuple[str, str, Any]]:
        """Get call history, optionally filtered by tool."""
        if tool:
            return [(t, a, r) for t, a, r in self.call_history if t == tool]
        return self.call_history

    def reset(self) -> None:
        """Reset call history and injection use counts."""
        self.call_history = []
        for inj in self.injections:
            inj.use_count = 0

    def _default_read(self, args: str) -> str:
        """Default Read response."""
        return f"// Mock content for {args}\ncontract Mock {{}}"

    def _default_glob(self, args: str) -> list[str]:
        """Default Glob response."""
        return [f"mock/path/{i}.sol" for i in range(3)]

    def _default_grep(self, args: str) -> list[dict]:
        """Default Grep response."""
        return [{"file": "mock.sol", "line": 10, "content": "// mock match"}]

    def _default_bash(self, args: str) -> dict:
        """Default Bash response."""
        return {"stdout": "Mock output", "stderr": "", "returncode": 0}

    def inject_into(self, mock: MockToolProvider) -> None:

"""Agent Contracts for Unit Testing.

Defines the expected behavior contracts for each subagent:
- Required output fields
- Expected tool usage patterns
- Persona constraints
- Quality thresholds

Contracts serve as specifications that subagents must satisfy.
"""

from dataclasses import dataclass, field
from typing import Any, Callable, TYPE_CHECKING

if TYPE_CHECKING:
    from true_vkg.testing.unit.subagent_tester import SubagentTestResult


@dataclass
class ExpectedOutput:
    """An expected output from a subagent."""

    field_path: str  # Dot-separated path like "findings.0.pattern"
    description: str
    validator: Callable[[Any], bool] | None = None  # Custom validation
    required: bool = True

    def check(self, output: dict | None) -> bool:
        """Check if output contains this expected field."""
        if output is None:
            return not self.required

        # Navigate to field
        value = output
        for key in self.field_path.split("."):
            if isinstance(value, dict):
                value = value.get(key)
            elif isinstance(value, list):
                try:
                    value = value[int(key)]
                except (ValueError, IndexError):
                    value = None
            else:
                value = None

            if value is None:
                return not self.required

        # Run custom validator
        if self.validator:
            return self.validator(value)

        return True


@dataclass
class OutputSchema:
    """Schema specification for subagent output.

    Validates that subagent output conforms to expected structure.
    """

    required_fields: list[str] = field(default_factory=list)
    optional_fields: list[str] = field(default_factory=list)
    field_types: dict[str, type] = field(default_factory=dict)
    nested_schemas: dict[str, "OutputSchema"] = field(default_factory=dict)

    def validate(self, output: dict | None) -> tuple[bool, list[str]]:
        """Validate output against schema.

        Returns:
            (is_valid, list_of_errors)
        """
        errors = []

        if output is None:
            if self.required_fields:
                errors.append("Output is None but has required fields")
            return len(errors) == 0, errors

        # Check required fields
        for field_name in self.required_fields:
            if field_name not in output:
                errors.append(f"Missing required field: {field_name}")
            elif field_name in self.field_types:
                expected_type = self.field_types[field_name]
                actual_type = type(output[field_name])
                if not isinstance(output[field_name], expected_type):
                    errors.append(
                        f"Field {field_name}: expected {expected_type.__name__}, "
                        f"got {actual_type.__name__}"
                    )

        # Check nested schemas
        for field_name, nested_schema in self.nested_schemas.items():
            if field_name in output:
                nested_valid, nested_errors = nested_schema.validate(output[field_name])
                errors.extend([f"{field_name}.{e}" for e in nested_errors])

        return len(errors) == 0, errors


@dataclass
class AgentContract:
    """Full behavioral contract for a subagent.

    Defines:
    - What the agent must output (schema)
    - What tools it should use
    - What persona it must maintain
    - Quality thresholds
    """

    agent_type: str
    description: str

    # Output requirements
    output_schema: OutputSchema
    expected_outputs: list[ExpectedOutput] = field(default_factory=list)

    # Tool usage
    required_tools: list[str] = field(default_factory=list)  # Must use
    forbidden_tools: list[str] = field(default_factory=list)  # Must not use

    # Persona constraints
    persona_requirements: list[str] = field(default_factory=list)

    # Quality thresholds
    min_confidence: float = 0.0
    max_response_tokens: int = 10000

    def check_compliance(self, result: "SubagentTestResult") -> list[str]:
        """Check if result complies with contract.

        Returns list of violations (empty if compliant).
        """
        violations = []

        # Check required tools
        for tool in self.required_tools:
            if not any(tool in called for called in result.tools_called):
                violations.append(f"Did not use required tool: {tool}")

        # Check forbidden tools
        for tool in self.forbidden_tools:
            if any(tool in called for called in result.tools_called):
                violations.append(f"Used forbidden tool: {tool}")

        # Check output schema
        if result.parsed_output:
            valid, errors = self.output_schema.validate(result.parsed_output)
            violations.extend(errors)

        return violations
