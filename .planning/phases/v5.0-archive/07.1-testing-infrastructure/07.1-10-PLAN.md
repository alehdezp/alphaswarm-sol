"""Pydantic schema for test scenario YAML validation.

Ensures all scenario configurations are valid and complete.
Used by ScenarioLoader to parse and validate YAML files.
"""

from pathlib import Path
from typing import Literal
import yaml

from pydantic import BaseModel, Field, field_validator


class GroundTruth(BaseModel):
    """Expected finding for a contract."""

    pattern: str = Field(description="Pattern ID that should be detected")
    severity: Literal["critical", "high", "medium", "low", "info"] = Field(
        description="Expected severity level"
    )
    location: str = Field(description="Code location (function:line or function name)")
    description: str | None = Field(
        default=None, description="Human description of the vulnerability"
    )
    confidence_min: float = Field(
        default=0.5, ge=0.0, le=1.0, description="Minimum confidence for valid match"
    )


class ContractCase(BaseModel):
    """A contract test case within a scenario."""

    path: str = Field(description="Path to contract file relative to corpus root")
    has_vulnerability: bool = Field(
        default=True, description="Whether contract has the target vulnerability"
    )
    expected_pattern: str | None = Field(
        default=None, description="Primary pattern expected to match"
    )
    expected_severity: str | None = Field(
        default=None, description="Expected severity if vulnerable"
    )
    ground_truth: list[GroundTruth] = Field(
        default_factory=list, description="All expected findings"
    )
    notes: str | None = Field(
        default=None, description="Notes about this test case (e.g., why it's safe)"
    )
    tags: list[str] = Field(
        default_factory=list, description="Tags for filtering (e.g., 'adversarial', 'edge-case')"
    )

    @field_validator("path")
    @classmethod
    def validate_path_format(cls, v: str) -> str:
        """Ensure path uses forward slashes."""
        return v.replace("\\", "/")


class IsolationConfig(BaseModel):
    """Configuration for blind/isolated testing."""

    exclude_vulndocs: bool = Field(
        default=False, description="Don't include vulnerability documentation in context"
    )
    blind_prompt: bool = Field(
        default=False, description="Don't hint at expected vulnerability type"
    )
    corpus_segment: str | None = Field(
        default=None, description="Use specific corpus segment (e.g., 'post-june-2025')"
    )
    shuffle_corpus: bool = Field(
        default=False, description="Randomize contract order"
    )
    anonymize_contracts: bool = Field(
        default=False, description="Strip identifying comments and names"
    )


class ChaosConfig(BaseModel):
    """Configuration for chaos/fault injection testing."""

    enabled: bool = Field(default=False, description="Enable chaos injection")
    timeout_probability: float = Field(
        default=0.0, ge=0.0, le=1.0, description="Probability of tool timeout"
    )
    malformed_probability: float = Field(
        default=0.0, ge=0.0, le=1.0, description="Probability of malformed response"
    )
    failure_probability: float = Field(
        default=0.0, ge=0.0, le=1.0, description="Probability of tool failure"
    )


class JsonSchemaConfig(BaseModel):
    """JSON schema for structured output."""

    type: str = "object"
    properties: dict = Field(default_factory=dict)
    required: list[str] = Field(default_factory=list)


class ScenarioConfig(BaseModel):
    """Complete test scenario configuration.

    Loaded from YAML files in scenarios/ directory.
    """

    name: str = Field(description="Human-readable scenario name")
    category: str = Field(description="Vulnerability category (e.g., 'reentrancy')")
    description: str = Field(default="", description="Detailed description of scenario")

    # Test cases
    contracts: list[ContractCase] = Field(
        description="Contracts to test in this scenario"
    )

    # Execution config
    prompt_template: str = Field(
        description="Prompt template with {contract_path} placeholder"
    )
    allowed_tools: list[str] = Field(
        default_factory=list, description="Tools to pre-approve for autonomous execution"
    )
    json_schema: JsonSchemaConfig = Field(
        default_factory=JsonSchemaConfig, description="Schema for structured output"
    )
    timeout_seconds: int = Field(default=120, ge=10, le=600)
    model: str | None = Field(default=None, description="Override default model")

    # Isolation for blind testing
    isolation: IsolationConfig = Field(default_factory=IsolationConfig)

    # Chaos injection
    chaos: ChaosConfig = Field(default_factory=ChaosConfig)

    # Metadata
    tags: list[str] = Field(default_factory=list)
    difficulty: Literal["easy", "medium", "hard", "expert"] = Field(default="medium")

    @classmethod
    def from_yaml(cls, path: Path) -> "ScenarioConfig":
        """Load scenario from YAML file."""
        with open(path) as f:
            data = yaml.safe_load(f)
        return cls(**data)

    @property
    def vulnerable_contracts(self) -> list[ContractCase]:
        """Get only contracts that should have vulnerabilities."""
        return [c for c in self.contracts if c.has_vulnerability]

    @property
    def safe_contracts(self) -> list[ContractCase]:
        """Get only contracts that should be safe (false positive tests)."""
        return [c for c in self.contracts if not c.has_vulnerability]

    @property
    def total_expected_findings(self) -> int:
        """Total number of expected findings across all contracts."""
        return sum(len(c.ground_truth) for c in self.contracts)

**ground_truth.py:**
```python
"""Ground Truth Management for accuracy measurement.

Compares agent findings against known vulnerabilities to calculate:
- Precision (what % of reported findings are real)
- Recall (what % of real vulnerabilities were found)
- F1 score (harmonic mean of precision and recall)

Supports fuzzy matching on location and severity weighting.
"""

from dataclasses import dataclass, field
from enum import Enum
from pathlib import Path
import json
import logging

from true_vkg.testing.scenarios import GroundTruth, ScenarioConfig

logger = logging.getLogger(__name__)


class Severity(Enum):
    """Vulnerability severity levels with weights."""

    CRITICAL = ("critical", 1.0)
    HIGH = ("high", 0.8)
    MEDIUM = ("medium", 0.5)
    LOW = ("low", 0.3)
    INFO = ("info", 0.1)

    def __init__(self, name: str, weight: float):
        self._name = name
        self._weight = weight

    @property
    def weight(self) -> float:
        return self._weight

    @classmethod
    def from_str(cls, s: str) -> "Severity":
        return cls[s.upper()]


@dataclass
class Finding:
    """A detected finding from analysis."""

    pattern: str
    severity: str
    location: str
    confidence: float = 0.5
    reasoning: str | None = None

    def matches_severity(self, expected: str, strict: bool = False) -> bool:
        """Check if severity matches.

        Args:
            expected: Expected severity level
            strict: If True, must match exactly. If False, >= expected is OK.
        """
        if strict:
            return self.severity.lower() == expected.lower()

        severity_order = ["info", "low", "medium", "high", "critical"]
        try:
            actual_idx = severity_order.index(self.severity.lower())
            expected_idx = severity_order.index(expected.lower())
            return actual_idx >= expected_idx
        except ValueError:
            return self.severity.lower() == expected.lower()


@dataclass
class MatchResult:
    """Result of matching a finding to ground truth."""

    finding: Finding
    ground_truth: GroundTruth | None
    is_match: bool
    match_quality: float = 0.0  # 0.0 to 1.0
    match_reason: str = ""


@dataclass
class CorpusStats:
    """Statistics about the test corpus."""

    total_contracts: int = 0
    vulnerable_contracts: int = 0
    safe_contracts: int = 0
    total_ground_truth: int = 0
    by_severity: dict[str, int] = field(default_factory=dict)
    by_pattern: dict[str, int] = field(default_factory=dict)


class GroundTruthManager:
    """Manage ground truth and calculate accuracy metrics.

    Example:
        manager = GroundTruthManager()
        manager.load_scenario(scenario)

        # Compare agent findings
        metrics = manager.evaluate(findings, contract_path)
        print(f"Precision: {metrics['precision']:.2%}")
        print(f"Recall: {metrics['recall']:.2%}")
    """

    def __init__(self, location_tolerance: int = 10):
        """Initialize manager.

        Args:
            location_tolerance: Line number tolerance for fuzzy matching
        """
        self.location_tolerance = location_tolerance
        self._ground_truth: dict[str, list[GroundTruth]] = {}  # contract_path -> truths
        self._scenarios: list[ScenarioConfig] = []

    def load_scenario(self, scenario: ScenarioConfig) -> None:
        """Load ground truth from a scenario."""
        self._scenarios.append(scenario)

        for contract in scenario.contracts:
            path = contract.path
            if path not in self._ground_truth:
                self._ground_truth[path] = []
            self._ground_truth[path].extend(contract.ground_truth)

    def load_from_json(self, json_path: Path) -> None:
        """Load ground truth from JSON file."""
        with open(json_path) as f:
            data = json.load(f)

        for contract_path, truths in data.items():
            self._ground_truth[contract_path] = [
                GroundTruth(**t) for t in truths
            ]

    def get_ground_truth(self, contract_path: str) -> list[GroundTruth]:
        """Get ground truth for a contract."""
        # Try exact match first
        if contract_path in self._ground_truth:
            return self._ground_truth[contract_path]

        # Try basename match
        basename = Path(contract_path).name
        for path, truths in self._ground_truth.items():
            if Path(path).name == basename:
                return truths

        return []

    def evaluate(
        self,
        findings: list[Finding],
        contract_path: str,
        severity_weighted: bool = True
    ) -> dict:
        """Evaluate findings against ground truth.

        Args:
            findings: Detected findings from analysis
            contract_path: Path to contract being evaluated
            severity_weighted: Weight by severity for recall calculation

        Returns:
            Dict with precision, recall, f1, and detailed match results
        """
        ground_truth = self.get_ground_truth(contract_path)

        if not ground_truth and not findings:
            # No ground truth and no findings - perfect for safe contract
            return {
                "precision": 1.0,
                "recall": 1.0,
                "f1_score": 1.0,
                "true_positives": 0,
                "false_positives": 0,
                "false_negatives": 0,
                "matches": [],
            }

        # Match findings to ground truth
        matches: list[MatchResult] = []
        matched_gt_indices: set[int] = set()
        true_positives = 0
        false_positives = 0

        for finding in findings:
            match_result = self._find_best_match(finding, ground_truth, matched_gt_indices)
            matches.append(match_result)

            if match_result.is_match:
                true_positives += 1
                if match_result.ground_truth:
                    gt_idx = ground_truth.index(match_result.ground_truth)
                    matched_gt_indices.add(gt_idx)
            else:
                false_positives += 1

        false_negatives = len(ground_truth) - len(matched_gt_indices)

        # Calculate metrics
        precision = true_positives / (true_positives + false_positives) if (true_positives + false_positives) > 0 else 0.0
        recall = true_positives / (true_positives + false_negatives) if (true_positives + false_negatives) > 0 else 0.0
        f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0.0

        # Severity-weighted recall
        if severity_weighted and ground_truth:
            weighted_found = sum(
                Severity.from_str(ground_truth[i].severity).weight
                for i in matched_gt_indices
            )
            weighted_total = sum(
                Severity.from_str(gt.severity).weight
                for gt in ground_truth
            )
            recall_weighted = weighted_found / weighted_total if weighted_total > 0 else 0.0
        else:
            recall_weighted = recall

        return {
            "precision": precision,
            "recall": recall,
            "recall_weighted": recall_weighted,
            "f1_score": f1,
            "true_positives": true_positives,
            "false_positives": false_positives,
            "false_negatives": false_negatives,
            "matches": matches,
            "missed_ground_truth": [
                ground_truth[i] for i in range(len(ground_truth))
                if i not in matched_gt_indices
            ],
        }

    def _find_best_match(
        self,
        finding: Finding,
        ground_truth: list[GroundTruth],
        already_matched: set[int]
    ) -> MatchResult:
        """Find best ground truth match for a finding."""
        best_match: GroundTruth | None = None
        best_quality = 0.0
        best_reason = ""

        for i, gt in enumerate(ground_truth):
            if i in already_matched:
                continue

            quality, reason = self._calculate_match_quality(finding, gt)

            if quality > best_quality:
                best_quality = quality
                best_match = gt
                best_reason = reason

        # Threshold for considering a match
        is_match = best_quality >= 0.5

        return MatchResult(
            finding=finding,
            ground_truth=best_match if is_match else None,
            is_match=is_match,
            match_quality=best_quality,
            match_reason=best_reason
        )

    def _calculate_match_quality(self, finding: Finding, gt: GroundTruth) -> tuple[float, str]:
        """Calculate match quality between finding and ground truth.

        Returns (quality_score, explanation).
        """
        scores: list[tuple[float, str]] = []

        # Pattern match (required)
        if finding.pattern.lower() == gt.pattern.lower():
            scores.append((0.4, "pattern exact match"))
        elif gt.pattern.lower() in finding.pattern.lower():
            scores.append((0.3, "pattern partial match"))
        else:
            return (0.0, "pattern mismatch")

        # Location match
        location_score = self._match_location(finding.location, gt.location)
        if location_score > 0:
            scores.append((location_score * 0.4, f"location score {location_score:.2f}"))

        # Severity match
        if finding.matches_severity(gt.severity, strict=True):
            scores.append((0.2, "severity exact match"))
        elif finding.matches_severity(gt.severity, strict=False):
            scores.append((0.1, "severity acceptable"))

        total_score = sum(s for s, _ in scores)
        reasons = [r for _, r in scores]

        return (total_score, "; ".join(reasons))

    def _match_location(self, finding_loc: str, gt_loc: str) -> float:
        """Match locations with fuzzy tolerance.

        Supports formats:
        - "function_name:line" -> "withdraw:45"
        - "function_name" -> "withdraw"
        - "contract.function:line" -> "Vault.withdraw:45"
        """
        # Normalize
        finding_loc = finding_loc.lower().strip()
        gt_loc = gt_loc.lower().strip()

        # Exact match
        if finding_loc == gt_loc:
            return 1.0

        # Extract function name and line
        def parse_location(loc: str) -> tuple[str, int | None]:
            if ":" in loc:
                parts = loc.rsplit(":", 1)
                try:
                    return parts[0], int(parts[1])
                except ValueError:
                    return loc, None
            return loc, None

        f_func, f_line = parse_location(finding_loc)
        gt_func, gt_line = parse_location(gt_loc)

        # Function name match
        func_match = 0.0
        if f_func == gt_func:
            func_match = 1.0
        elif f_func in gt_func or gt_func in f_func:
            func_match = 0.7

        if func_match == 0:
            return 0.0

        # Line match with tolerance
        line_match = 0.0
        if f_line is not None and gt_line is not None:
            diff = abs(f_line - gt_line)
            if diff == 0:
                line_match = 1.0
            elif diff <= self.location_tolerance:
                line_match = 1.0 - (diff / self.location_tolerance)
        elif f_line is None and gt_line is None:
            line_match = 1.0  # Neither has line, OK
        else:
            line_match = 0.5  # One has line, one doesn't

        return (func_match * 0.6) + (line_match * 0.4)

    def get_corpus_stats(self) -> CorpusStats:
        """Get statistics about loaded ground truth."""
        stats = CorpusStats()

        for scenario in self._scenarios:
            for contract in scenario.contracts:
                stats.total_contracts += 1
                if contract.has_vulnerability:
                    stats.vulnerable_contracts += 1
                else:
                    stats.safe_contracts += 1

                for gt in contract.ground_truth:
                    stats.total_ground_truth += 1

                    # By severity
                    sev = gt.severity
                    stats.by_severity[sev] = stats.by_severity.get(sev, 0) + 1

                    # By pattern
                    pat = gt.pattern
                    stats.by_pattern[pat] = stats.by_pattern.get(pat, 0) + 1

        return stats

# src/true_vkg/testing/benchmarks/__init__.py
"""Benchmark Suite Definitions.

Benchmarks define:
- Which scenarios to run
- Time budget
- Pass criteria
- Model settings
"""

from pathlib import Path
import yaml
from dataclasses import dataclass


@dataclass
class BenchmarkSuite:
    """Benchmark suite configuration."""
    name: str
    description: str
    time_budget_minutes: int
    scenarios: list[str]
    settings: dict
    metrics: list[str]
    pass_criteria: dict

    @classmethod
    def load(cls, name: str) -> "BenchmarkSuite":
        """Load benchmark by name (quick, standard, thorough)."""
        path = Path(__file__).parent / f"{name}.yaml"
        with open(path) as f:
            data = yaml.safe_load(f)
        return cls(**data)

    def passes(self, results: dict) -> tuple[bool, list[str]]:
        """Check if results pass criteria."""
        failures = []
        for metric, min_value in self.pass_criteria.items():
            actual = results.get(metric.replace("_min", ""), 0)
            if actual < min_value:
                failures.append(f"{metric}: {actual:.2%} < {min_value:.2%}")
        return len(failures) == 0, failures
```
