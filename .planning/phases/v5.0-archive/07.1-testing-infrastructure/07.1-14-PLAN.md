"""Chaos engineering layer for fault injection testing."""

from __future__ import annotations

import random
import time
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Callable, TypeVar
from functools import wraps

T = TypeVar("T")


class FailureType(Enum):
    """Types of failures that can be injected."""

    TIMEOUT = "timeout"
    CRASH = "crash"
    MALFORMED_RESPONSE = "malformed_response"
    PARTIAL_DATA = "partial_data"
    EMPTY_RESPONSE = "empty_response"
    RATE_LIMIT = "rate_limit"
    PERMISSION_DENIED = "permission_denied"
    NOT_FOUND = "not_found"
    CORRUPTION = "corruption"
    LATENCY = "latency"


@dataclass
class FailureConfig:
    """Configuration for a failure injection."""

    failure_type: FailureType
    probability: float  # 0.0 to 1.0
    target_pattern: str | None = None  # Regex pattern for target tools/commands
    delay_ms: int = 0  # For LATENCY type
    error_message: str = "Injected failure"

    def should_trigger(self) -> bool:
        """Determine if failure should trigger based on probability."""
        return random.random() < self.probability


@dataclass
class ChaosResult:
    """Result of a chaos-wrapped call."""

    success: bool
    result: Any = None
    injected_failure: FailureType | None = None
    failure_message: str | None = None
    latency_added_ms: int = 0


class ChaosLayer:
    """
    Wraps tool calls with configurable chaos injection.

    Based on ChaosAgent SDK patterns - injects realistic failures
    to test agent resilience and recovery capabilities.
    """

    def __init__(
        self,
        failures: list[FailureConfig] | None = None,
        seed: int | None = None,
        enabled: bool = True
    ):
        self.failures = failures or []
        self.enabled = enabled
        self._injection_log: list[ChaosResult] = []

        if seed is not None:
            random.seed(seed)

    @classmethod
    def from_config(cls, config_path: str) -> ChaosLayer:
        """Load chaos configuration from YAML file."""
        import yaml
        from pathlib import Path

        config = yaml.safe_load(Path(config_path).read_text())
        failures = [
            FailureConfig(
                failure_type=FailureType(f["type"]),
                probability=f.get("probability", 0.1),
                target_pattern=f.get("target"),
                delay_ms=f.get("delay_ms", 0),
                error_message=f.get("message", "Injected failure")
            )
            for f in config.get("failures", [])
        ]
        return cls(
            failures=failures,
            seed=config.get("seed"),
            enabled=config.get("enabled", True)
        )

    def add_failure(self, config: FailureConfig) -> None:
        """Add a failure configuration."""
        self.failures.append(config)

    def wrap(self, func: Callable[..., T]) -> Callable[..., ChaosResult]:
        """Wrap a function with chaos injection."""
        @wraps(func)
        def wrapped(*args: Any, **kwargs: Any) -> ChaosResult:
            if not self.enabled:
                return ChaosResult(success=True, result=func(*args, **kwargs))

            # Check each failure config
            for failure in self.failures:
                if failure.should_trigger():
                    return self._inject_failure(failure)

            # No failure triggered - execute normally
            try:
                result = func(*args, **kwargs)
                return ChaosResult(success=True, result=result)
            except Exception as e:
                return ChaosResult(success=False, failure_message=str(e))

        return wrapped

    def _inject_failure(self, config: FailureConfig) -> ChaosResult:
        """Inject a specific failure."""
        result = ChaosResult(
            success=False,
            injected_failure=config.failure_type,
            failure_message=config.error_message
        )

        if config.failure_type == FailureType.LATENCY:
            time.sleep(config.delay_ms / 1000)
            result.latency_added_ms = config.delay_ms
            result.success = True  # Latency doesn't cause failure

        elif config.failure_type == FailureType.TIMEOUT:
            result.failure_message = f"Operation timed out after {config.delay_ms}ms"

        elif config.failure_type == FailureType.MALFORMED_RESPONSE:
            result.result = "{malformed: json["

        elif config.failure_type == FailureType.PARTIAL_DATA:
            result.result = {"partial": True, "data": None}

        elif config.failure_type == FailureType.EMPTY_RESPONSE:
            result.result = None

        self._injection_log.append(result)
        return result

    def get_injection_log(self) -> list[ChaosResult]:
        """Get log of all injected failures."""
        return self._injection_log.copy()

    def clear_log(self) -> None:
        """Clear the injection log."""
        self._injection_log.clear()

    """Create a ChaosLayer configured for a specific scenario."""
    scenario = get_scenario(scenario_name)
    return ChaosLayer(failures=scenario.failures)


def create_combined_chaos_layer(
    scenario_names: list[str],
    probability_multiplier: float = 1.0
) -> ChaosLayer:
    """Create a ChaosLayer combining multiple scenarios."""
    all_failures = []
    for name in scenario_names:
        scenario = get_scenario(name)
        for failure in scenario.failures:
            # Adjust probability
            adjusted = FailureConfig(
                failure_type=failure.failure_type,
                probability=min(failure.probability * probability_multiplier, 1.0),
                target_pattern=failure.target_pattern,
                delay_ms=failure.delay_ms,
                error_message=failure.error_message
            )
            all_failures.append(adjusted)

    return ChaosLayer(failures=all_failures)

"""Trajectory recording for agent execution analysis."""

from __future__ import annotations

import time
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any


class StepType(Enum):
    """Types of steps in an agent trajectory."""

    REASONING = "reasoning"
    TOOL_CALL = "tool_call"
    TOOL_RESULT = "tool_result"
    DECISION = "decision"
    ERROR = "error"
    RECOVERY = "recovery"
    CHECKPOINT = "checkpoint"


@dataclass
class TrajectoryStep:
    """A single step in an agent trajectory."""

    step_number: int
    step_type: StepType
    timestamp: datetime
    content: str
    tool_name: str | None = None
    tool_args: dict[str, Any] | None = None
    duration_ms: int = 0
    success: bool = True
    metadata: dict[str, Any] = field(default_factory=dict)


@dataclass
class Trajectory:
    """Complete trajectory of an agent execution."""

    agent_id: str
    agent_type: str
    task_description: str
    start_time: datetime
    end_time: datetime | None = None
    steps: list[TrajectoryStep] = field(default_factory=list)
    final_result: Any = None
    success: bool = False
    metadata: dict[str, Any] = field(default_factory=dict)

    @property
    def total_duration_ms(self) -> int:
        """Total execution duration in milliseconds."""
        if not self.end_time:
            return 0
        delta = self.end_time - self.start_time
        return int(delta.total_seconds() * 1000)

    @property
    def tool_call_count(self) -> int:
        """Number of tool calls made."""
        return sum(1 for s in self.steps if s.step_type == StepType.TOOL_CALL)

    @property
    def error_count(self) -> int:
        """Number of errors encountered."""
        return sum(1 for s in self.steps if s.step_type == StepType.ERROR)

    @property
    def recovery_count(self) -> int:
        """Number of successful recoveries from errors."""
        return sum(1 for s in self.steps if s.step_type == StepType.RECOVERY)


class TrajectoryRecorder:
    """
    Records agent execution trajectories.

    Captures every step an agent takes for later analysis of
    reasoning quality, efficiency, and error handling.
    """

    def __init__(self, agent_id: str, agent_type: str, task_description: str):
        self.trajectory = Trajectory(
            agent_id=agent_id,
            agent_type=agent_type,
            task_description=task_description,
            start_time=datetime.now()
        )
        self._step_counter = 0
        self._current_step_start: float | None = None

    def record_reasoning(self, content: str, **metadata: Any) -> TrajectoryStep:
        """Record a reasoning step."""
        return self._add_step(
            step_type=StepType.REASONING,
            content=content,
            metadata=metadata
        )

    def record_tool_call(
        self,
        tool_name: str,
        tool_args: dict[str, Any],
        **metadata: Any
    ) -> TrajectoryStep:
        """Record a tool call (start timing)."""
        self._current_step_start = time.time()
        return self._add_step(
            step_type=StepType.TOOL_CALL,
            content=f"Calling {tool_name}",
            tool_name=tool_name,
            tool_args=tool_args,
            metadata=metadata
        )

    def record_tool_result(
        self,
        tool_name: str,
        result: Any,
        success: bool = True,
        **metadata: Any
    ) -> TrajectoryStep:
        """Record a tool result (end timing)."""
        duration_ms = 0
        if self._current_step_start:
            duration_ms = int((time.time() - self._current_step_start) * 1000)
            self._current_step_start = None

        return self._add_step(
            step_type=StepType.TOOL_RESULT,
            content=str(result)[:500],  # Truncate long results
            tool_name=tool_name,
            duration_ms=duration_ms,
            success=success,
            metadata=metadata
        )

    def record_decision(self, decision: str, **metadata: Any) -> TrajectoryStep:
        """Record a decision point."""
        return self._add_step(
            step_type=StepType.DECISION,
            content=decision,
            metadata=metadata
        )

    def record_error(self, error: str, **metadata: Any) -> TrajectoryStep:
        """Record an error."""
        return self._add_step(
            step_type=StepType.ERROR,
            content=error,
            success=False,
            metadata=metadata
        )

    def record_recovery(self, recovery_action: str, **metadata: Any) -> TrajectoryStep:
        """Record a recovery from an error."""
        return self._add_step(
            step_type=StepType.RECOVERY,
            content=recovery_action,
            metadata=metadata
        )

    def record_checkpoint(self, checkpoint: str, **metadata: Any) -> TrajectoryStep:
        """Record a checkpoint (significant progress marker)."""
        return self._add_step(
            step_type=StepType.CHECKPOINT,
            content=checkpoint,
            metadata=metadata
        )

    def finish(self, result: Any, success: bool) -> Trajectory:
        """Finish recording and return the complete trajectory."""
        self.trajectory.end_time = datetime.now()
        self.trajectory.final_result = result
        self.trajectory.success = success
        return self.trajectory

    def _add_step(
        self,
        step_type: StepType,
        content: str,
        tool_name: str | None = None,
        tool_args: dict[str, Any] | None = None,
        duration_ms: int = 0,
        success: bool = True,
        metadata: dict[str, Any] | None = None
    ) -> TrajectoryStep:
        """Add a step to the trajectory."""
        self._step_counter += 1
        step = TrajectoryStep(
            step_number=self._step_counter,
            step_type=step_type,
            timestamp=datetime.now(),
            content=content,
            tool_name=tool_name,
            tool_args=tool_args,
            duration_ms=duration_ms,
            success=success,
            metadata=metadata or {}
        )
        self.trajectory.steps.append(step)
        return step

"""Trajectory evaluation for agent quality assessment."""

from __future__ import annotations

from dataclasses import dataclass
from typing import Any

from .recorder import Trajectory, StepType, TrajectoryStep


@dataclass
class TrajectoryMetrics:
    """Metrics derived from trajectory evaluation."""

    # Efficiency metrics
    step_efficiency: float  # Ratio of productive steps to total
    tool_efficiency: float  # Ratio of successful tool calls
    time_efficiency: float  # Actual vs expected duration

    # Quality metrics
    tool_appropriateness: float  # Right tools for the task
    reasoning_coherence: float  # Logical flow score
    error_recovery_rate: float  # Recovery success rate

    # Problem metrics
    dead_end_ratio: float  # Unproductive exploration
    redundancy_ratio: float  # Repeated actions

    # Overall
    overall_score: float

    def to_dict(self) -> dict[str, float]:
        """Convert to dictionary."""
        return {
            "step_efficiency": self.step_efficiency,
            "tool_efficiency": self.tool_efficiency,
            "time_efficiency": self.time_efficiency,
            "tool_appropriateness": self.tool_appropriateness,
            "reasoning_coherence": self.reasoning_coherence,
            "error_recovery_rate": self.error_recovery_rate,
            "dead_end_ratio": self.dead_end_ratio,
            "redundancy_ratio": self.redundancy_ratio,
            "overall_score": self.overall_score
        }


class TrajectoryEvaluator:
    """
    Evaluates agent trajectories for quality assessment.

    Goes beyond final answers to analyze HOW agents reach conclusions.
    Based on research from arXiv 2510.02837.
    """

    def __init__(
        self,
        expected_tools: list[str] | None = None,
        expected_duration_ms: int | None = None,
        optimal_step_count: int | None = None
    ):
        self.expected_tools = expected_tools or []
        self.expected_duration_ms = expected_duration_ms
        self.optimal_step_count = optimal_step_count

    def evaluate(self, trajectory: Trajectory) -> TrajectoryMetrics:
        """Evaluate a trajectory and return metrics."""
        return TrajectoryMetrics(
            step_efficiency=self._calculate_step_efficiency(trajectory),
            tool_efficiency=self._calculate_tool_efficiency(trajectory),
            time_efficiency=self._calculate_time_efficiency(trajectory),
            tool_appropriateness=self._calculate_tool_appropriateness(trajectory),
            reasoning_coherence=self._calculate_reasoning_coherence(trajectory),
            error_recovery_rate=self._calculate_error_recovery_rate(trajectory),
            dead_end_ratio=self._calculate_dead_end_ratio(trajectory),
            redundancy_ratio=self._calculate_redundancy_ratio(trajectory),
            overall_score=self._calculate_overall_score(trajectory)
        )

    def _calculate_step_efficiency(self, trajectory: Trajectory) -> float:
        """Calculate ratio of productive steps to total steps."""
        if not trajectory.steps:
            return 0.0

        productive_types = {
            StepType.TOOL_CALL,
            StepType.DECISION,
            StepType.CHECKPOINT,
            StepType.RECOVERY
        }

        productive = sum(
            1 for s in trajectory.steps
            if s.step_type in productive_types and s.success
        )

        return productive / len(trajectory.steps)

    def _calculate_tool_efficiency(self, trajectory: Trajectory) -> float:
        """Calculate ratio of successful tool calls."""
        tool_calls = [s for s in trajectory.steps if s.step_type == StepType.TOOL_CALL]
        if not tool_calls:
            return 1.0

        successful = sum(1 for s in tool_calls if s.success)
        return successful / len(tool_calls)

    def _calculate_time_efficiency(self, trajectory: Trajectory) -> float:
        """Calculate time efficiency vs expected duration."""
        if not self.expected_duration_ms or not trajectory.total_duration_ms:
            return 1.0

        # Score decreases if took longer than expected
        ratio = self.expected_duration_ms / trajectory.total_duration_ms
        return min(ratio, 1.0)

    def _calculate_tool_appropriateness(self, trajectory: Trajectory) -> float:
        """Calculate how appropriate tool choices were."""
        if not self.expected_tools:
            return 1.0

        used_tools = {
            s.tool_name for s in trajectory.steps
            if s.step_type == StepType.TOOL_CALL and s.tool_name
        }

        expected_set = set(self.expected_tools)

        # Penalize missing expected tools and using unexpected tools
        missing = len(expected_set - used_tools)
        extra = len(used_tools - expected_set)

        total_expected = len(expected_set)
        if total_expected == 0:
            return 1.0

        score = max(0, total_expected - missing - (extra * 0.5)) / total_expected
        return score

    def _calculate_reasoning_coherence(self, trajectory: Trajectory) -> float:
        """Calculate logical coherence of reasoning steps."""
        reasoning_steps = [
            s for s in trajectory.steps
            if s.step_type == StepType.REASONING
        ]

        if len(reasoning_steps) < 2:
            return 1.0

        # Simple heuristic: check for logical progression
        # (In a real implementation, could use LLM to evaluate)
        coherent_count = 0
        for i in range(1, len(reasoning_steps)):
            prev = reasoning_steps[i - 1].content.lower()
            curr = reasoning_steps[i].content.lower()

            # Check for logical connectors or references
            connectors = ["therefore", "because", "since", "thus", "next", "then"]
            if any(c in curr for c in connectors):
                coherent_count += 1

        return coherent_count / (len(reasoning_steps) - 1) if len(reasoning_steps) > 1 else 1.0

    def _calculate_error_recovery_rate(self, trajectory: Trajectory) -> float:
        """Calculate rate of successful error recovery."""
        errors = sum(1 for s in trajectory.steps if s.step_type == StepType.ERROR)
        recoveries = sum(1 for s in trajectory.steps if s.step_type == StepType.RECOVERY)

        if errors == 0:
            return 1.0

        return min(recoveries / errors, 1.0)

    def _calculate_dead_end_ratio(self, trajectory: Trajectory) -> float:
        """Calculate ratio of dead-end explorations."""
        if not trajectory.steps:
            return 0.0

        # Detect dead ends: errors without recovery, or abandoned paths
        dead_ends = 0
        for i, step in enumerate(trajectory.steps):
            if step.step_type == StepType.ERROR:
                # Check if next step is not a recovery
                if i + 1 < len(trajectory.steps):
                    next_step = trajectory.steps[i + 1]
                    if next_step.step_type != StepType.RECOVERY:
                        dead_ends += 1
                else:
                    dead_ends += 1

        return dead_ends / len(trajectory.steps)

    def _calculate_redundancy_ratio(self, trajectory: Trajectory) -> float:
        """Calculate ratio of redundant/repeated actions."""
        if not trajectory.steps:
            return 0.0

        tool_calls = [
            (s.tool_name, str(s.tool_args))
            for s in trajectory.steps
            if s.step_type == StepType.TOOL_CALL
        ]

        if len(tool_calls) < 2:
            return 0.0

        unique_calls = len(set(tool_calls))
        redundant = len(tool_calls) - unique_calls

        return redundant / len(tool_calls)

    def _calculate_overall_score(self, trajectory: Trajectory) -> float:
        """Calculate weighted overall score."""
        weights = {
            "step_efficiency": 0.15,
            "tool_efficiency": 0.15,
            "time_efficiency": 0.10,
            "tool_appropriateness": 0.15,
            "reasoning_coherence": 0.15,
            "error_recovery_rate": 0.15,
            "dead_end_ratio": 0.075,  # Negative impact
            "redundancy_ratio": 0.075  # Negative impact
        }

        scores = {
            "step_efficiency": self._calculate_step_efficiency(trajectory),
            "tool_efficiency": self._calculate_tool_efficiency(trajectory),
            "time_efficiency": self._calculate_time_efficiency(trajectory),
            "tool_appropriateness": self._calculate_tool_appropriateness(trajectory),
            "reasoning_coherence": self._calculate_reasoning_coherence(trajectory),
            "error_recovery_rate": self._calculate_error_recovery_rate(trajectory),
            "dead_end_ratio": 1.0 - self._calculate_dead_end_ratio(trajectory),
            "redundancy_ratio": 1.0 - self._calculate_redundancy_ratio(trajectory)
        }

        return sum(scores[k] * weights[k] for k in weights)

"""Feedback collection and analysis for agent self-improvement."""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from typing import Any
from collections import defaultdict
import statistics


@dataclass
class PerformanceRecord:
    """Record of a single agent execution."""

    agent_id: str
    agent_type: str
    task_type: str
    timestamp: datetime
    success: bool
    precision: float
    recall: float
    f1_score: float
    latency_ms: int
    trajectory_score: float
    chaos_recovery_rate: float
    metadata: dict[str, Any] = field(default_factory=dict)


@dataclass
class PerformanceTrend:
    """Trend analysis for an agent's performance."""

    agent_type: str
    task_type: str
    period_start: datetime
    period_end: datetime
    sample_count: int

    # Trend metrics
    precision_trend: float  # Positive = improving
    recall_trend: float
    f1_trend: float
    latency_trend: float  # Negative = improving

    # Current averages
    avg_precision: float
    avg_recall: float
    avg_f1: float
    avg_latency_ms: float

    # Improvement rate
    improvement_rate: float


class FeedbackCollector:
    """
    Collects and analyzes agent performance feedback.

    Enables self-improvement by identifying patterns in agent performance
    and surfacing areas that need improvement.
    """

    def __init__(self, ema_alpha: float = 0.3):
        self.records: list[PerformanceRecord] = []
        self.ema_scores: dict[tuple[str, str], float] = {}  # (agent_type, task_type) -> EMA
        self.ema_alpha = ema_alpha

    def record(self, record: PerformanceRecord) -> None:
        """Record a performance result."""
        self.records.append(record)
        self._update_ema(record)

    def _update_ema(self, record: PerformanceRecord) -> None:
        """Update EMA score for agent/task combination."""
        key = (record.agent_type, record.task_type)
        current = self.ema_scores.get(key, record.f1_score)

        # EMA formula: new = alpha * value + (1 - alpha) * current
        new_ema = self.ema_alpha * record.f1_score + (1 - self.ema_alpha) * current
        self.ema_scores[key] = new_ema

    def get_ema(self, agent_type: str, task_type: str) -> float:
        """Get current EMA score for agent/task combination."""
        return self.ema_scores.get((agent_type, task_type), 0.0)

    def get_trend(
        self,
        agent_type: str,
        task_type: str,
        window_size: int = 10
    ) -> PerformanceTrend | None:
        """Analyze performance trend for an agent on a task type."""
        relevant = [
            r for r in self.records
            if r.agent_type == agent_type and r.task_type == task_type
        ]

        if len(relevant) < window_size:
            return None

        recent = relevant[-window_size:]
        older = relevant[-2 * window_size:-window_size] if len(relevant) >= 2 * window_size else relevant[:window_size]

        def avg(records: list[PerformanceRecord], attr: str) -> float:
            values = [getattr(r, attr) for r in records]
            return statistics.mean(values) if values else 0.0

        return PerformanceTrend(
            agent_type=agent_type,
            task_type=task_type,
            period_start=recent[0].timestamp,
            period_end=recent[-1].timestamp,
            sample_count=len(recent),
            precision_trend=avg(recent, "precision") - avg(older, "precision"),
            recall_trend=avg(recent, "recall") - avg(older, "recall"),
            f1_trend=avg(recent, "f1_score") - avg(older, "f1_score"),
            latency_trend=avg(older, "latency_ms") - avg(recent, "latency_ms"),  # Reversed
            avg_precision=avg(recent, "precision"),
            avg_recall=avg(recent, "recall"),
            avg_f1=avg(recent, "f1_score"),
            avg_latency_ms=avg(recent, "latency_ms"),
            improvement_rate=avg(recent, "f1_score") - avg(older, "f1_score")
        )

    def identify_weak_areas(
        self,
        agent_type: str,
        threshold: float = 0.7
    ) -> list[dict[str, Any]]:
        """Identify task types where agent performs below threshold."""
        by_task: dict[str, list[PerformanceRecord]] = defaultdict(list)

        for record in self.records:
            if record.agent_type == agent_type:
                by_task[record.task_type].append(record)

        weak_areas = []
        for task_type, records in by_task.items():
            if not records:
                continue

            avg_f1 = statistics.mean(r.f1_score for r in records)
            if avg_f1 < threshold:
                weak_areas.append({
                    "task_type": task_type,
                    "avg_f1": avg_f1,
                    "sample_count": len(records),
                    "worst_metrics": self._find_worst_metrics(records)
                })

        return sorted(weak_areas, key=lambda x: x["avg_f1"])

    def _find_worst_metrics(
        self,
        records: list[PerformanceRecord]
    ) -> dict[str, float]:
        """Find which metrics are weakest."""
        metrics = {}

        for metric in ["precision", "recall", "trajectory_score", "chaos_recovery_rate"]:
            values = [getattr(r, metric) for r in records]
            metrics[metric] = statistics.mean(values) if values else 0.0

        return metrics

    def generate_improvement_report(self, agent_type: str) -> dict[str, Any]:
        """Generate a comprehensive improvement report for an agent."""
        weak_areas = self.identify_weak_areas(agent_type)

        all_trends = {}
        for task_type in set(r.task_type for r in self.records if r.agent_type == agent_type):
            trend = self.get_trend(agent_type, task_type)
            if trend:
                all_trends[task_type] = {
                    "f1_trend": trend.f1_trend,
                    "improvement_rate": trend.improvement_rate,
                    "avg_f1": trend.avg_f1
                }

        return {
            "agent_type": agent_type,
            "total_records": sum(1 for r in self.records if r.agent_type == agent_type),
            "weak_areas": weak_areas,
            "trends": all_trends,
            "ema_scores": {
                k[1]: v for k, v in self.ema_scores.items()
                if k[0] == agent_type
            },
            "recommendations": self._generate_recommendations(agent_type, weak_areas)
        }

    def _generate_recommendations(
        self,
        agent_type: str,
        weak_areas: list[dict[str, Any]]
    ) -> list[str]:
        """Generate improvement recommendations based on analysis."""
        recommendations = []

        for area in weak_areas[:3]:  # Top 3 weak areas
            task = area["task_type"]
            worst = area["worst_metrics"]

            if worst.get("precision", 1.0) < 0.7:
                recommendations.append(
                    f"Improve precision on {task}: Consider stricter matching criteria"
                )

            if worst.get("recall", 1.0) < 0.7:
                recommendations.append(
                    f"Improve recall on {task}: Consider broader search patterns"
                )

            if worst.get("trajectory_score", 1.0) < 0.7:
                recommendations.append(
                    f"Improve reasoning efficiency on {task}: Reduce dead-end exploration"
                )

            if worst.get("chaos_recovery_rate", 1.0) < 0.7:
                recommendations.append(
                    f"Improve resilience on {task}: Add better error handling"
                )

        return recommendations

"""Agent evolution through mutation and selection."""

from __future__ import annotations

import random
from dataclasses import dataclass, field
from typing import Any, Callable
from enum import Enum

from .feedback import FeedbackCollector, PerformanceRecord
from .mutations import MutationType, AgentMutation, apply_mutation


class SelectionStrategy(Enum):
    """Strategies for selecting agents to evolve."""

    TOURNAMENT = "tournament"  # Tournament selection
    ROULETTE = "roulette"  # Fitness-proportionate
    RANK = "rank"  # Rank-based selection
    ELITIST = "elitist"  # Keep top performers


@dataclass
class AgentConfig:
    """Configuration for an agent that can be evolved."""

    agent_id: str
    agent_type: str
    system_prompt: str
    tool_whitelist: list[str]
    temperature: float = 0.7
    max_turns: int = 20
    metadata: dict[str, Any] = field(default_factory=dict)

    def fork(self, new_id: str) -> AgentConfig:
        """Create a copy with a new ID."""
        return AgentConfig(
            agent_id=new_id,
            agent_type=self.agent_type,
            system_prompt=self.system_prompt,
            tool_whitelist=self.tool_whitelist.copy(),
            temperature=self.temperature,
            max_turns=self.max_turns,
            metadata=self.metadata.copy()
        )


@dataclass
class EvolutionResult:
    """Result of an evolution cycle."""

    generation: int
    best_agent: AgentConfig
    best_fitness: float
    population_avg_fitness: float
    mutations_applied: list[tuple[str, MutationType]]
    improvement_from_baseline: float


class AgentEvolver:
    """
    Meta-agent that evolves VKG agents through mutation and selection.

    Based on aiXplain Evolver and evolutionary optimization research.
    Automatically improves agents without manual code changes.
    """

    def __init__(
        self,
        feedback_collector: FeedbackCollector,
        population_size: int = 10,
        mutation_rate: float = 0.3,
        selection_strategy: SelectionStrategy = SelectionStrategy.TOURNAMENT,
        elite_count: int = 2,
        seed: int | None = None
    ):
        self.feedback = feedback_collector
        self.population_size = population_size
        self.mutation_rate = mutation_rate
        self.selection_strategy = selection_strategy
        self.elite_count = elite_count

        if seed is not None:
            random.seed(seed)

    def evolve(
        self,
        base_agent: AgentConfig,
        fitness_fn: Callable[[AgentConfig], float],
        generations: int = 10,
        target_fitness: float | None = None
    ) -> EvolutionResult:
        """
        Evolve an agent over multiple generations.

        Args:
            base_agent: Starting agent configuration
            fitness_fn: Function to evaluate agent fitness (higher = better)
            generations: Number of evolution cycles
            target_fitness: Stop early if this fitness is reached

        Returns:
            EvolutionResult with best evolved agent
        """
        # Initialize population
        population = [
            base_agent.fork(f"{base_agent.agent_id}_gen0_{i}")
            for i in range(self.population_size)
        ]

        baseline_fitness = fitness_fn(base_agent)
        best_agent = base_agent
        best_fitness = baseline_fitness
        mutations_applied: list[tuple[str, MutationType]] = []

        for gen in range(generations):
            # Evaluate fitness
            fitness_scores = [(agent, fitness_fn(agent)) for agent in population]
            fitness_scores.sort(key=lambda x: x[1], reverse=True)

            # Check for new best
            if fitness_scores[0][1] > best_fitness:
                best_agent = fitness_scores[0][0]
                best_fitness = fitness_scores[0][1]

            # Check target
            if target_fitness and best_fitness >= target_fitness:
                break

            # Selection
            selected = self._select(fitness_scores)

            # Create next generation
            next_gen = []

            # Elitism - keep top performers
            for agent, _ in fitness_scores[:self.elite_count]:
                next_gen.append(agent.fork(f"{agent.agent_id}_elite"))

            # Fill rest with mutated offspring
            while len(next_gen) < self.population_size:
                parent = random.choice(selected)
                child = parent.fork(f"{base_agent.agent_id}_gen{gen+1}_{len(next_gen)}")

                if random.random() < self.mutation_rate:
                    mutation = self._select_mutation()
                    child = apply_mutation(child, mutation)
                    mutations_applied.append((child.agent_id, mutation.mutation_type))

                next_gen.append(child)

            population = next_gen

        # Final evaluation
        final_scores = [(a, fitness_fn(a)) for a in population]
        final_scores.sort(key=lambda x: x[1], reverse=True)

        if final_scores[0][1] > best_fitness:
            best_agent = final_scores[0][0]
            best_fitness = final_scores[0][1]

        avg_fitness = sum(s[1] for s in final_scores) / len(final_scores)

        return EvolutionResult(
            generation=generations,
            best_agent=best_agent,
            best_fitness=best_fitness,
            population_avg_fitness=avg_fitness,
            mutations_applied=mutations_applied,
            improvement_from_baseline=best_fitness - baseline_fitness
        )

    def _select(
        self,
        fitness_scores: list[tuple[AgentConfig, float]]
    ) -> list[AgentConfig]:
        """Select agents for reproduction based on strategy."""
        if self.selection_strategy == SelectionStrategy.TOURNAMENT:
            return self._tournament_select(fitness_scores)
        elif self.selection_strategy == SelectionStrategy.ROULETTE:
            return self._roulette_select(fitness_scores)
        elif self.selection_strategy == SelectionStrategy.RANK:
            return self._rank_select(fitness_scores)
        else:  # ELITIST
            return [a for a, _ in fitness_scores[:self.population_size // 2]]

    def _tournament_select(
        self,
        fitness_scores: list[tuple[AgentConfig, float]],
        tournament_size: int = 3
    ) -> list[AgentConfig]:
        """Tournament selection."""
        selected = []
        for _ in range(self.population_size // 2):
            tournament = random.sample(fitness_scores, min(tournament_size, len(fitness_scores)))
            winner = max(tournament, key=lambda x: x[1])
            selected.append(winner[0])
        return selected

    def _roulette_select(
        self,
        fitness_scores: list[tuple[AgentConfig, float]]
    ) -> list[AgentConfig]:
        """Fitness-proportionate selection."""
        total_fitness = sum(s[1] for s in fitness_scores)
        if total_fitness == 0:
            return [a for a, _ in fitness_scores[:self.population_size // 2]]

        selected = []
        for _ in range(self.population_size // 2):
            pick = random.uniform(0, total_fitness)
            current = 0
            for agent, fitness in fitness_scores:
                current += fitness
                if current >= pick:
                    selected.append(agent)
                    break
        return selected

    def _rank_select(
        self,
        fitness_scores: list[tuple[AgentConfig, float]]
    ) -> list[AgentConfig]:
        """Rank-based selection."""
        n = len(fitness_scores)
        weights = [n - i for i in range(n)]  # Higher rank = higher weight
        total = sum(weights)

        selected = []
        for _ in range(self.population_size // 2):
            pick = random.uniform(0, total)
            current = 0
            for i, (agent, _) in enumerate(fitness_scores):
                current += weights[i]
                if current >= pick:
                    selected.append(agent)
                    break
        return selected

    def _select_mutation(self) -> AgentMutation:
        """Select a random mutation to apply."""
        mutation_types = list(MutationType)
        selected_type = random.choice(mutation_types)
        return AgentMutation(mutation_type=selected_type)

from __future__ import annotations

import random
from dataclasses import dataclass
from enum import Enum
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from .evolver import AgentConfig


class MutationType(Enum):
    """Types of mutations that can be applied to agents."""

    PROMPT_REPHRASE = "prompt_rephrase"
    PROMPT_EXPAND = "prompt_expand"
    PROMPT_SIMPLIFY = "prompt_simplify"
    TOOL_ADD = "tool_add"
    TOOL_REMOVE = "tool_remove"
    TEMPERATURE_ADJUST = "temperature_adjust"
    MAX_TURNS_ADJUST = "max_turns_adjust"
    REASONING_STYLE = "reasoning_style"


@dataclass
class AgentMutation:
    """A mutation to apply to an agent."""

    mutation_type: MutationType
    parameters: dict = None

    def __post_init__(self):
        if self.parameters is None:
            self.parameters = {}


# Prompt mutation templates
PROMPT_EXPANSIONS = [
    "\n\nBe thorough in your analysis.",
    "\n\nConsider edge cases carefully.",
    "\n\nProvide evidence for each finding.",
    "\n\nExplain your reasoning step by step.",
    "\n\nDouble-check your conclusions."
]

PROMPT_SIMPLIFICATIONS = [
    lambda p: p.replace("thoroughly ", ""),
    lambda p: p.replace("carefully ", ""),
    lambda p: p.replace("step by step", ""),
    lambda p: p.split("\n\n")[0] if "\n\n" in p else p,
]

REASONING_STYLE_ADDITIONS = [
    "\n\nUse a systematic approach: 1) Identify, 2) Analyze, 3) Verify, 4) Report.",
    "\n\nThink like an attacker: What would you exploit?",
    "\n\nAssume the code is vulnerable until proven safe.",
    "\n\nFocus on high-impact vulnerabilities first."
]

# Available tools for mutation
AVAILABLE_TOOLS = [
    "Bash(uv run true-vkg build-kg*)",
    "Bash(uv run true-vkg query*)",
    "Bash(uv run true-vkg tools run*)",
    "Bash(slither*)",
    "Bash(aderyn*)",
    "Read",
    "Glob",
    "Grep",
    "WebSearch",
    "WebFetch"
]


def apply_mutation(agent: "AgentConfig", mutation: AgentMutation) -> "AgentConfig":
    """Apply a mutation to an agent configuration."""
    mutators = {
        MutationType.PROMPT_REPHRASE: _mutate_prompt_rephrase,
        MutationType.PROMPT_EXPAND: _mutate_prompt_expand,
        MutationType.PROMPT_SIMPLIFY: _mutate_prompt_simplify,
        MutationType.TOOL_ADD: _mutate_tool_add,
        MutationType.TOOL_REMOVE: _mutate_tool_remove,
        MutationType.TEMPERATURE_ADJUST: _mutate_temperature,
        MutationType.MAX_TURNS_ADJUST: _mutate_max_turns,
        MutationType.REASONING_STYLE: _mutate_reasoning_style,
    }

    mutator = mutators.get(mutation.mutation_type)
    if mutator:
        return mutator(agent, mutation.parameters)
    return agent


def _mutate_prompt_rephrase(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Rephrase parts of the system prompt."""
    # Simple word substitutions
    substitutions = [
        ("analyze", "examine"),
        ("find", "identify"),
        ("check", "verify"),
        ("look for", "search for"),
        ("vulnerability", "security issue"),
    ]

    new_prompt = agent.system_prompt
    for old, new in substitutions:
        if random.random() < 0.3:
            new_prompt = new_prompt.replace(old, new)

    agent.system_prompt = new_prompt
    return agent


def _mutate_prompt_expand(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Expand the system prompt with additional instructions."""
    expansion = random.choice(PROMPT_EXPANSIONS)
    agent.system_prompt = agent.system_prompt + expansion
    return agent


def _mutate_prompt_simplify(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Simplify the system prompt."""
    simplifier = random.choice(PROMPT_SIMPLIFICATIONS)
    agent.system_prompt = simplifier(agent.system_prompt)
    return agent


def _mutate_tool_add(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Add a tool to the whitelist."""
    available = set(AVAILABLE_TOOLS) - set(agent.tool_whitelist)
    if available:
        new_tool = random.choice(list(available))
        agent.tool_whitelist.append(new_tool)
    return agent


def _mutate_tool_remove(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Remove a tool from the whitelist (keep at least 2)."""
    if len(agent.tool_whitelist) > 2:
        to_remove = random.choice(agent.tool_whitelist)
        agent.tool_whitelist.remove(to_remove)
    return agent


def _mutate_temperature(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Adjust the temperature."""
    delta = random.uniform(-0.1, 0.1)
    agent.temperature = max(0.0, min(1.0, agent.temperature + delta))
    return agent


def _mutate_max_turns(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Adjust the maximum turns."""
    delta = random.choice([-5, -2, 2, 5])
    agent.max_turns = max(5, min(50, agent.max_turns + delta))
    return agent


def _mutate_reasoning_style(agent: "AgentConfig", params: dict) -> "AgentConfig":
    """Add reasoning style guidance."""
    style = random.choice(REASONING_STYLE_ADDITIONS)
    agent.system_prompt = agent.system_prompt + style
    return agent
