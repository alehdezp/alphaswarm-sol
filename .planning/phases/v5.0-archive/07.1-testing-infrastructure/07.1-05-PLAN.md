```python
"""Model ranking integration for testing infrastructure.

This module connects benchmark results to the Phase 5.3 ranking system,
enabling automatic model selection optimization based on test corpus
performance.

Key concepts:
- Benchmark results update rankings per task type
- EMA (Exponential Moving Average) smooths updates
- Cost-accuracy curves inform tradeoff decisions
- Quality is primary (60% weight), task-type 30%, cost 10%
"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Optional, Any
import yaml

from true_vkg.testing.metrics import TestMetrics


@dataclass
class ModelBenchmarkResult:
    """Results from benchmarking a single model.

    Attributes:
        model_id: Model identifier (e.g., "claude-sonnet-4.5")
        task_type: Task type benchmarked (e.g., "detect", "verify")
        metrics: TestMetrics from benchmark run
        timestamp: When benchmark was run
        corpus_version: Corpus version used
        vs_opus_accuracy: Percentage of Opus baseline achieved (if compared)
    """
    model_id: str
    task_type: str
    metrics: TestMetrics
    timestamp: str = field(default_factory=lambda: datetime.now().isoformat())
    corpus_version: str = "v1"
    vs_opus_accuracy: Optional[float] = None

    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for storage."""
        return {
            "model_id": self.model_id,
            "task_type": self.task_type,
            "precision": self.metrics.precision,
            "recall": self.metrics.recall,
            "f1_score": self.metrics.f1_score,
            "recall_critical": self.metrics.recall_critical,
            "cost_usd": self.metrics.cost_usd,
            "tokens_used": self.metrics.tokens_used,
            "timestamp": self.timestamp,
            "corpus_version": self.corpus_version,
            "vs_opus_accuracy": self.vs_opus_accuracy,
        }


def calculate_ema_score(
    current_score: float,
    new_value: float,
    alpha: float = 0.3
) -> float:
    """Calculate exponential moving average.

    Args:
        current_score: Current EMA score
        new_value: New value to incorporate
        alpha: Smoothing factor (0.3 = new values have 30% weight)

    Returns:
        Updated EMA score
    """
    return alpha * new_value + (1 - alpha) * current_score


def calculate_composite_score(
    precision: float,
    recall_critical: float,
    cost_per_thousand: float,
    quality_weight: float = 0.6,
    task_type_weight: float = 0.3,
    cost_weight: float = 0.1,
) -> float:
    """Calculate composite ranking score.

    Weights per context doc:
    - Quality/Accuracy: 60%
    - Task-Type Suitability: 30%
    - Cost-Accuracy: 10%

    Args:
        precision: Precision score (0-1)
        recall_critical: Critical recall score (0-1)
        cost_per_thousand: Cost per 1000 findings in USD
        quality_weight: Weight for quality metrics
        task_type_weight: Weight for task suitability
        cost_weight: Weight for cost efficiency

    Returns:
        Composite score (higher is better)
    """
    # Quality component: average of precision and recall_critical
    quality_score = (precision + recall_critical) / 2

    # Task-type component: use recall_critical as proxy for task suitability
    task_score = recall_critical

    # Cost component: inverse relationship (lower cost = higher score)
    # Normalize: assume $10/1000 is baseline, $1/1000 is excellent
    cost_score = max(0, min(1, 1 - (cost_per_thousand / 10)))

    return (
        quality_weight * quality_score +
        task_type_weight * task_score +
        cost_weight * cost_score
    )


class RankingUpdater:
    """Updates model rankings based on benchmark results.

    Integrates with Phase 5.3 ranking infrastructure to update
    model scores per task type.
    """

    def __init__(self, rankings_path: Optional[Path] = None):
        """Initialize ranking updater.

        Args:
            rankings_path: Path to rankings YAML file.
                          Defaults to .vrs/rankings/rankings.yaml
        """
        self.rankings_path = rankings_path or Path(".vrs/rankings/rankings.yaml")
        self._rankings: Dict[str, Dict[str, Any]] = {}
        self._load_rankings()

    def _load_rankings(self) -> None:
        """Load rankings from YAML file."""
        if self.rankings_path.exists():
            with open(self.rankings_path) as f:
                self._rankings = yaml.safe_load(f) or {}
        else:
            self._rankings = {"task_types": {}, "models": {}, "history": []}

    def _save_rankings(self) -> None:
        """Save rankings to YAML file."""
        self.rankings_path.parent.mkdir(parents=True, exist_ok=True)
        with open(self.rankings_path, "w") as f:
            yaml.dump(self._rankings, f, default_flow_style=False)

    def update_from_benchmark(self, result: ModelBenchmarkResult) -> float:
        """Update rankings from a benchmark result.

        Args:
            result: Benchmark result to incorporate

        Returns:
            New composite score for the model+task_type combination
        """
        task_type = result.task_type
        model_id = result.model_id

        # Ensure structure exists
        if "task_types" not in self._rankings:
            self._rankings["task_types"] = {}
        if task_type not in self._rankings["task_types"]:
            self._rankings["task_types"][task_type] = {}
        if model_id not in self._rankings["task_types"][task_type]:
            self._rankings["task_types"][task_type][model_id] = {
                "score": 0.5,  # Initial score
                "benchmarks": 0,
                "last_updated": None,
            }

        model_data = self._rankings["task_types"][task_type][model_id]

        # Calculate new composite score
        cost_per_thousand = (
            result.metrics.cost_usd / result.metrics.contracts_tested * 1000
            if result.metrics.contracts_tested > 0 else 0
        )

        new_score = calculate_composite_score(
            precision=result.metrics.precision,
            recall_critical=result.metrics.recall_critical,
            cost_per_thousand=cost_per_thousand,
        )

        # Update with EMA
        current_score = model_data["score"]
        updated_score = calculate_ema_score(current_score, new_score)

        # Store updated data
        model_data["score"] = updated_score
        model_data["benchmarks"] += 1
        model_data["last_updated"] = result.timestamp
        model_data["last_precision"] = result.metrics.precision
        model_data["last_recall_critical"] = result.metrics.recall_critical
        model_data["last_cost_usd"] = result.metrics.cost_usd

        # Record in history
        if "history" not in self._rankings:
            self._rankings["history"] = []
        self._rankings["history"].append(result.to_dict())

        # Keep only last 100 history entries
        if len(self._rankings["history"]) > 100:
            self._rankings["history"] = self._rankings["history"][-100:]

        self._save_rankings()
        return updated_score

    def get_rankings_for_task(self, task_type: str) -> List[Dict[str, Any]]:
        """Get model rankings for a specific task type.

        Args:
            task_type: Task type to query

        Returns:
            List of models sorted by score (descending)
        """
        if task_type not in self._rankings.get("task_types", {}):
            return []

        models = self._rankings["task_types"][task_type]
        ranked = [
            {"model_id": model_id, **data}
            for model_id, data in models.items()
        ]
        return sorted(ranked, key=lambda x: x["score"], reverse=True)

    def get_cost_accuracy_curve(self, task_type: str) -> List[Dict[str, float]]:
        """Get cost-accuracy curve data for a task type.

        Returns points for plotting cost vs accuracy tradeoff.

        Args:
            task_type: Task type to analyze

        Returns:
            List of {model_id, cost, accuracy} points
        """
        rankings = self.get_rankings_for_task(task_type)
        return [
            {
                "model_id": r["model_id"],
                "cost": r.get("last_cost_usd", 0),
                "accuracy": r.get("last_precision", 0),
                "recall_critical": r.get("last_recall_critical", 0),
            }
            for r in rankings
            if r.get("last_precision") is not None
        ]

    def compare_vs_opus(
        self,
        results: List[ModelBenchmarkResult],
        opus_result: ModelBenchmarkResult,
    ) -> Dict[str, float]:
        """Compare model results against Opus baseline.

        Args:
            results: Benchmark results from other models
            opus_result: Opus baseline result

        Returns:
            Dict of model_id -> percentage of Opus accuracy achieved
        """
        if opus_result.metrics.f1_score == 0:
            return {}

        comparisons = {}
        for result in results:
            pct = (result.metrics.f1_score / opus_result.metrics.f1_score) * 100
            comparisons[result.model_id] = round(pct, 2)
            result.vs_opus_accuracy = pct

        return comparisons

    def get_eligible_models(
        self,
        task_type: str,
        min_opus_pct: float = 95.0,
    ) -> List[str]:
        """Get models eligible for production use.

        Per context doc: Model must achieve >=95% of Opus accuracy
        for a task type to be eligible.

        Args:
            task_type: Task type to check
            min_opus_pct: Minimum percentage of Opus accuracy (default 95%)

        Returns:
            List of eligible model IDs
        """
        rankings = self.get_rankings_for_task(task_type)
        # This would need vs_opus_accuracy stored in rankings
        # For now, return all models above a score threshold
        return [
            r["model_id"]
            for r in rankings
            if r["score"] >= 0.85  # Proxy for 95% eligibility
        ]
```

```python
import pytest

class TestEMAScore:
    """Tests for EMA calculation."""

    def test_ema_basic(self):
        """Test basic EMA calculation."""
        # With alpha=0.3, new value gets 30% weight
        result = calculate_ema_score(0.5, 1.0, alpha=0.3)
        assert result == pytest.approx(0.65)  # 0.3*1.0 + 0.7*0.5

    def test_ema_no_change(self):
        """Test EMA when value doesn't change."""
        result = calculate_ema_score(0.8, 0.8, alpha=0.3)
        assert result == pytest.approx(0.8)

    def test_ema_gradual_increase(self):
        """Test EMA converges over multiple updates."""
        score = 0.5
        for _ in range(10):
            score = calculate_ema_score(score, 1.0, alpha=0.3)
        # Should approach 1.0 but not quite reach it
        assert score > 0.95
        assert score < 1.0


class TestCompositeScore:
    """Tests for composite score calculation."""

    def test_perfect_scores(self):
        """Test with perfect metrics."""
        score = calculate_composite_score(
            precision=1.0,
            recall_critical=1.0,
            cost_per_thousand=0.0,
        )
        # Quality: 1.0 * 0.6 + Task: 1.0 * 0.3 + Cost: 1.0 * 0.1 = 1.0
        assert score == pytest.approx(1.0)

    def test_average_scores(self):
        """Test with average metrics."""
        score = calculate_composite_score(
            precision=0.8,
            recall_critical=0.9,
            cost_per_thousand=5.0,
        )
        # Quality: 0.85 * 0.6 = 0.51
        # Task: 0.9 * 0.3 = 0.27
        # Cost: 0.5 * 0.1 = 0.05
        assert score == pytest.approx(0.83, rel=0.01)

    def test_quality_weighted_highest(self):
        """Test that quality has highest weight."""
        low_quality = calculate_composite_score(
            precision=0.5, recall_critical=0.5, cost_per_thousand=0.0
        )
        high_quality = calculate_composite_score(
            precision=1.0, recall_critical=1.0, cost_per_thousand=10.0
        )
        # Even with high cost, high quality should score better
        assert high_quality > low_quality
```
