"""Gap tracking for testing infrastructure.

This module provides infrastructure for documenting testing limitations,
failure modes, and improvement priorities.

Gap Categories:
- detection: Vulnerabilities we miss
- false_positive: What triggers incorrect findings
- technical: Builder/pattern/VQL limitations
- agent: Reasoning failures, context issues
- scalability: Performance bottlenecks
- ux: CLI/workflow problems

Gap Severity:
- critical: Blocks GA release
- high: Should be fixed before GA
- medium: Nice to have before GA
- low: Post-GA improvement
"""

from __future__ import annotations

import hashlib
from dataclasses import dataclass, field, asdict
from datetime import datetime
from enum import Enum
from pathlib import Path
from typing import List, Optional, Dict, Any
import yaml


class GapCategory(str, Enum):
    """Categories for gap classification."""
    DETECTION = "detection"
    FALSE_POSITIVE = "false_positive"
    TECHNICAL = "technical"
    AGENT = "agent"
    SCALABILITY = "scalability"
    UX = "ux"


class GapSeverity(str, Enum):
    """Severity levels for gaps."""
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"


class GapStatus(str, Enum):
    """Status of gap resolution."""
    OPEN = "open"
    INVESTIGATING = "investigating"
    FIXED = "fixed"
    WONTFIX = "wontfix"


@dataclass
class Gap:
    """A documented testing gap.

    Attributes:
        gap_id: Unique identifier (GAP-XXX format)
        category: Gap category
        severity: Gap severity
        title: Brief description
        description: Detailed explanation
        root_cause: Why this gap exists
        affected_component: Which module/component is affected
        test_case: Path to test that demonstrates gap (if exists)
        status: Current status
        roadmap_item: Link to BACKLOG item if actionable
        research_theme: Systemic research theme if applicable
        discovered_date: When gap was discovered
        discovered_by: Agent or human who found it
    """
    gap_id: str
    category: GapCategory
    severity: GapSeverity
    title: str
    description: str
    root_cause: str
    affected_component: str
    test_case: Optional[str] = None
    status: GapStatus = GapStatus.OPEN
    roadmap_item: Optional[str] = None
    research_theme: Optional[str] = None
    discovered_date: str = field(default_factory=lambda: datetime.now().isoformat()[:10])
    discovered_by: str = "manual"

    def to_yaml(self) -> str:
        """Serialize to YAML string."""
        data = {
            "gap_id": self.gap_id,
            "category": self.category.value,
            "severity": self.severity.value,
            "title": self.title,
            "description": self.description,
            "root_cause": self.root_cause,
            "affected_component": self.affected_component,
            "test_case": self.test_case,
            "status": self.status.value,
            "roadmap_item": self.roadmap_item,
            "research_theme": self.research_theme,
            "discovered_date": self.discovered_date,
            "discovered_by": self.discovered_by,
        }
        return yaml.dump(data, default_flow_style=False, sort_keys=False)

    @classmethod
    def from_yaml(cls, yaml_str: str) -> "Gap":
        """Deserialize from YAML string."""
        data = yaml.safe_load(yaml_str)
        return cls(
            gap_id=data["gap_id"],
            category=GapCategory(data["category"]),
            severity=GapSeverity(data["severity"]),
            title=data["title"],
            description=data["description"],
            root_cause=data["root_cause"],
            affected_component=data["affected_component"],
            test_case=data.get("test_case"),
            status=GapStatus(data.get("status", "open")),
            roadmap_item=data.get("roadmap_item"),
            research_theme=data.get("research_theme"),
            discovered_date=data.get("discovered_date", datetime.now().isoformat()[:10]),
            discovered_by=data.get("discovered_by", "manual"),
        )

    @classmethod
    def from_file(cls, path: Path) -> "Gap":
        """Load gap from YAML file."""
        return cls.from_yaml(path.read_text())


class GapTracker:
    """Manages gap tracking operations.

    Gaps are stored as individual YAML files in .vrs/testing/gaps/
    following the pattern GAP-XXX.yaml.
    """

    def __init__(self, gaps_dir: Optional[Path] = None):
        """Initialize gap tracker.

        Args:
            gaps_dir: Directory for gap storage. Defaults to .vrs/testing/gaps/
        """
        self.gaps_dir = gaps_dir or Path(".vrs/testing/gaps")
        self.gaps_dir.mkdir(parents=True, exist_ok=True)

    def _generate_gap_id(self) -> str:
        """Generate next gap ID."""
        existing = list(self.gaps_dir.glob("GAP-*.yaml"))
        if not existing:
            return "GAP-001"

        # Find highest number
        max_num = 0
        for path in existing:
            try:
                num = int(path.stem.split("-")[1])
                max_num = max(max_num, num)
            except (IndexError, ValueError):
                continue

        return f"GAP-{max_num + 1:03d}"

    def create_gap(
        self,
        title: str,
        category: GapCategory,
        severity: GapSeverity,
        description: str,
        root_cause: str,
        affected_component: str,
        test_case: Optional[str] = None,
        roadmap_item: Optional[str] = None,
        research_theme: Optional[str] = None,
        discovered_by: str = "manual",
    ) -> Gap:
        """Create and save a new gap.

        Args:
            title: Brief description
            category: Gap category
            severity: Gap severity
            description: Detailed explanation
            root_cause: Why this gap exists
            affected_component: Which component is affected
            test_case: Path to demonstrating test
            roadmap_item: Link to BACKLOG item
            research_theme: Systemic research theme
            discovered_by: Who/what found this

        Returns:
            Created Gap object
        """
        gap_id = self._generate_gap_id()
        gap = Gap(
            gap_id=gap_id,
            category=category,
            severity=severity,
            title=title,
            description=description,
            root_cause=root_cause,
            affected_component=affected_component,
            test_case=test_case,
            roadmap_item=roadmap_item,
            research_theme=research_theme,
            discovered_by=discovered_by,
        )

        # Save to file
        gap_path = self.gaps_dir / f"{gap_id}.yaml"
        gap_path.write_text(gap.to_yaml())

        return gap

    def get_gap(self, gap_id: str) -> Optional[Gap]:
        """Get gap by ID."""
        gap_path = self.gaps_dir / f"{gap_id}.yaml"
        if not gap_path.exists():
            return None
        return Gap.from_file(gap_path)

    def list_gaps(
        self,
        category: Optional[GapCategory] = None,
        severity: Optional[GapSeverity] = None,
        status: Optional[GapStatus] = None,
    ) -> List[Gap]:
        """List gaps with optional filtering.

        Args:
            category: Filter by category
            severity: Filter by severity
            status: Filter by status

        Returns:
            List of matching gaps
        """
        gaps = []
        for path in sorted(self.gaps_dir.glob("GAP-*.yaml")):
            gap = Gap.from_file(path)
            if category and gap.category != category:
                continue
            if severity and gap.severity != severity:
                continue
            if status and gap.status != status:
                continue
            gaps.append(gap)
        return gaps

    def update_status(self, gap_id: str, status: GapStatus) -> bool:
        """Update gap status.

        Args:
            gap_id: Gap identifier
            status: New status

        Returns:
            True if updated, False if gap not found
        """
        gap = self.get_gap(gap_id)
        if not gap:
            return False

        gap.status = status
        gap_path = self.gaps_dir / f"{gap_id}.yaml"
        gap_path.write_text(gap.to_yaml())
        return True

    def get_summary(self) -> Dict[str, Any]:
        """Get gap inventory summary.

        Returns:
            Dict with counts by category, severity, and status
        """
        gaps = self.list_gaps()
        summary = {
            "total": len(gaps),
            "by_category": {},
            "by_severity": {},
            "by_status": {},
            "blocking_ga": 0,
        }

        for gap in gaps:
            # Count by category
            cat = gap.category.value
            summary["by_category"][cat] = summary["by_category"].get(cat, 0) + 1

            # Count by severity
            sev = gap.severity.value
            summary["by_severity"][sev] = summary["by_severity"].get(sev, 0) + 1

            # Count by status
            stat = gap.status.value
            summary["by_status"][stat] = summary["by_status"].get(stat, 0) + 1

            # Count GA blockers
            if gap.severity == GapSeverity.CRITICAL and gap.status == GapStatus.OPEN:
                summary["blocking_ga"] += 1

        return summary

"""Tests for gap tracking infrastructure."""

import pytest
from pathlib import Path

from true_vkg.testing.gaps import (
    Gap,
    GapTracker,
    GapCategory,
    GapSeverity,
    GapStatus,
)


class TestGap:
    """Tests for Gap dataclass."""

    def test_to_yaml(self):
        """Test serialization to YAML."""
        gap = Gap(
            gap_id="GAP-001",
            category=GapCategory.DETECTION,
            severity=GapSeverity.HIGH,
            title="Cross-contract reentrancy",
            description="Detection fails for cross-contract callbacks",
            root_cause="CallTracker doesn't follow callback paths",
            affected_component="builder.calls",
            discovered_by="vrs-gap-finder",
        )
        yaml_str = gap.to_yaml()

        assert "gap_id: GAP-001" in yaml_str
        assert "category: detection" in yaml_str
        assert "severity: high" in yaml_str
        assert "Cross-contract reentrancy" in yaml_str

    def test_from_yaml(self):
        """Test deserialization from YAML."""
        yaml_str = """
gap_id: GAP-002
category: false_positive
severity: medium
title: Safe code flagged as vulnerable
description: Pattern too aggressive
root_cause: Missing context check
affected_component: patterns.reentrancy
status: open
discovered_date: "2026-01-22"
discovered_by: manual
"""
        gap = Gap.from_yaml(yaml_str)

        assert gap.gap_id == "GAP-002"
        assert gap.category == GapCategory.FALSE_POSITIVE
        assert gap.severity == GapSeverity.MEDIUM
        assert gap.status == GapStatus.OPEN

    def test_roundtrip(self):
        """Test YAML roundtrip preserves data."""
        original = Gap(
            gap_id="GAP-003",
            category=GapCategory.AGENT,
            severity=GapSeverity.LOW,
            title="Agent timeout on large contracts",
            description="Context-merge agent times out",
            root_cause="Token limit exceeded",
            affected_component="agents.context_merge",
            test_case="tests/test_large_contract.py",
            roadmap_item="BACKLOG-042",
            research_theme="context-optimization",
        )

        yaml_str = original.to_yaml()
        restored = Gap.from_yaml(yaml_str)

        assert restored.gap_id == original.gap_id
        assert restored.category == original.category
        assert restored.test_case == original.test_case
        assert restored.roadmap_item == original.roadmap_item
        assert restored.research_theme == original.research_theme


class TestGapTracker:

        tracker = GapTracker(gaps_dir=tmp_path / "gaps")

        gap = tracker.create_gap(
            title="Missing oracle check",

            severity=GapSeverity.HIGH,

            root_cause="Pattern incomplete",

        assert gap.gap_id == "GAP-001"
        assert (tmp_path / "gaps" / "GAP-001.yaml").exists()

    def test_sequential_ids(self, tmp_path):
        """Test gap IDs are sequential."""
        tracker = GapTracker(gaps_dir=tmp_path / "gaps")

        gap1 = tracker.create_gap(
            title="Gap 1", category=GapCategory.DETECTION,
            severity=GapSeverity.LOW, description="",
            root_cause="", affected_component=""
        )
        gap2 = tracker.create_gap(
            title="Gap 2", category=GapCategory.DETECTION,
            severity=GapSeverity.LOW, description="",

        assert gap1.gap_id == "GAP-001"

            severity=GapSeverity.MEDIUM, description="Test",
