"""CLI commands for protocol context pack management."""

import asyncio
from pathlib import Path
from typing import Optional, List

import structlog
import typer

from true_vkg.context import (
    ContextPackBuilder,
    BuildConfig,
    ContextPackStorage,
    ProtocolContextPack,
)

context_app = typer.Typer(help="Protocol context pack management")
logger = structlog.get_logger()


@context_app.command("generate")
def generate_context(
    path: str = typer.Argument(".", help="Path to Solidity project"),
    docs: Optional[str] = typer.Option(None, "--docs", "-d", help="Path to documentation directory"),
    output: Optional[str] = typer.Option(None, "--output", "-o", help="Output path for context pack"),
    sources: Optional[List[str]] = typer.Option(None, "--source", "-s", help="Additional doc sources (URLs or paths)"),
    name: Optional[str] = typer.Option(None, "--name", "-n", help="Protocol name override"),
    type: Optional[str] = typer.Option(None, "--type", "-t", help="Protocol type override (lending, dex, etc.)"),
    no_code: bool = typer.Option(False, "--no-code", help="Skip code analysis (docs only)"),
    no_docs: bool = typer.Option(False, "--no-docs", help="Skip doc parsing (code only)"),
    force: bool = typer.Option(False, "--force", "-f", help="Overwrite existing context pack"),
) -> None:
    """Generate protocol context pack from code and documentation.

    Examples:
        # Auto-discover docs and analyze code
        uv run alphaswarm context generate ./src

        # Specify docs directory
        uv run alphaswarm context generate ./src --docs ./docs

        # Add external sources
        uv run alphaswarm context generate ./src --source https://docs.protocol.io

        # Override protocol name
        uv run alphaswarm context generate ./src --name "MyProtocol" --type lending
    """
    try:
        project_path = Path(path).resolve()
        if not project_path.exists():
            typer.echo(f"Error: Path not found: {project_path}", err=True)
            raise typer.Exit(code=1)

        # Build config from options
        config = BuildConfig(
            include_code_analysis=not no_code,
            include_doc_parsing=not no_docs,
            additional_doc_sources=list(sources) if sources else [],
            protocol_name=name,
            protocol_type=type,
        )

        # Run async build
        result = asyncio.run(_generate_context(project_path, docs, config))

        # Determine output path
        if output:
            output_path = Path(output)
        else:
            output_path = project_path / ".vkg" / "context"

        # Save context pack
        storage = ContextPackStorage(output_path)
        pack_name = result.pack.protocol_name or "context"

        if storage.exists(pack_name) and not force:
            typer.echo(f"Context pack '{pack_name}' already exists. Use --force to overwrite.", err=True)
            raise typer.Exit(code=1)

        saved_path = storage.save(result.pack, pack_name)

        # Report results
        typer.echo(f"Context pack generated: {saved_path}")
        typer.echo(f"  Protocol: {result.pack.protocol_name or 'Unknown'}")
        typer.echo(f"  Type: {result.pack.protocol_type or 'Unknown'}")
        typer.echo(f"  Roles: {len(result.pack.roles)}")
        typer.echo(f"  Assumptions: {len(result.pack.assumptions)}")
        typer.echo(f"  Invariants: {len(result.pack.invariants)}")

        if result.conflicts:
            typer.echo(f"\n  Conflicts detected: {len(result.conflicts)}")
            for conflict in result.conflicts[:3]:
                typer.echo(f"    - {conflict}")

        if result.questions:
            typer.echo(f"\n  Gap questions: {len(result.questions)}")

    except Exception as exc:
        logger.error("context_generate_failed", error=str(exc))
        raise typer.Exit(code=1) from exc


async def _generate_context(
    project_path: Path,
    docs_path: Optional[str],
    config: BuildConfig,
) -> "BuildResult":
    """Async context generation helper."""
    from true_vkg.context import ContextPackBuilder
    from true_vkg.kg.builder import VKGBuilder

    # Build BSKG if code analysis enabled
    graph = None
    if config.include_code_analysis:
        builder = VKGBuilder()
        graph = builder.build(str(project_path))

    # Create context builder
    context_builder = ContextPackBuilder(
        graph=graph,
        project_path=project_path if config.include_doc_parsing else None,
        config=config,
    )

    return await context_builder.build()


@context_app.command("show")
def show_context(
    path: str = typer.Argument(".", help="Path to project with context pack"),
    name: str = typer.Option("context", "--name", "-n", help="Context pack name"),
    section: Optional[str] = typer.Option(None, "--section", "-s", help="Show specific section (roles, assumptions, invariants, etc.)"),
    json_output: bool = typer.Option(False, "--json", help="Output as JSON"),
) -> None:
    """Display context pack contents.

    Examples:
        # Show full context pack
        uv run alphaswarm context show ./src

        # Show specific section
        uv run alphaswarm context show ./src --section roles

        # Output as JSON
        uv run alphaswarm context show ./src --json
    """
    try:
        project_path = Path(path).resolve()
        storage_path = project_path / ".vkg" / "context"

        storage = ContextPackStorage(storage_path)
        pack = storage.load(name)

        if not pack:
            typer.echo(f"Context pack '{name}' not found at {storage_path}", err=True)
            raise typer.Exit(code=1)

        if json_output:
            import json
            if section:
                data = pack.get_section(section)
                typer.echo(json.dumps(data, indent=2))
            else:
                typer.echo(json.dumps(pack.to_dict(), indent=2))
        else:
            _display_context_pack(pack, section)

    except Exception as exc:
        logger.error("context_show_failed", error=str(exc))
        raise typer.Exit(code=1) from exc


def _display_context_pack(pack: ProtocolContextPack, section: Optional[str] = None) -> None:
    """Display context pack in human-readable format."""
    if section:
        data = pack.get_section(section)
        if data is None:
            typer.echo(f"Section '{section}' not found")
            return
        typer.echo(f"## {section.title()}\n")
        _display_section(section, data)
    else:
        typer.echo(f"# Protocol: {pack.protocol_name or 'Unknown'}")
        typer.echo(f"Type: {pack.protocol_type or 'Unknown'}")
        typer.echo(f"Generated: {pack.generated_at}")
        typer.echo(f"Auto-generated: {pack.auto_generated}")
        typer.echo(f"Reviewed: {pack.reviewed}\n")

        for section_name in ['roles', 'assumptions', 'invariants', 'offchain_inputs', 'accepted_risks']:
            data = pack.get_section(section_name)
            if data:
                typer.echo(f"## {section_name.title()} ({len(data)})\n")
                _display_section(section_name, data[:3])  # Show first 3
                if len(data) > 3:
                    typer.echo(f"  ... and {len(data) - 3} more\n")


def _display_section(section: str, data: List) -> None:
    """Display a section's data."""
    for item in data:
        if isinstance(item, dict):
            name = item.get('name', item.get('description', str(item)))
            typer.echo(f"  - {name}")
        else:
            typer.echo(f"  - {item}")


@context_app.command("update")
def update_context(
    path: str = typer.Argument(".", help="Path to project"),
    name: str = typer.Option("context", "--name", "-n", help="Context pack name to update"),
) -> None:
    """Update existing context pack with code changes.

    Examples:
        uv run alphaswarm context update ./src
    """
    try:
        project_path = Path(path).resolve()
        storage_path = project_path / ".vkg" / "context"

        storage = ContextPackStorage(storage_path)
        existing = storage.load(name)

        if not existing:
            typer.echo(f"No existing context pack '{name}' to update. Use 'generate' first.", err=True)
            raise typer.Exit(code=1)

        # Run update
        result = asyncio.run(_update_context(project_path, existing))

        # Save updated pack
        storage.save(result.pack, name)
        typer.echo(f"Context pack '{name}' updated.")

        if result.conflicts:
            typer.echo(f"New conflicts detected: {len(result.conflicts)}")

    except Exception as exc:
        logger.error("context_update_failed", error=str(exc))
        raise typer.Exit(code=1) from exc


async def _update_context(project_path: Path, existing: ProtocolContextPack) -> "BuildResult":
    """Async context update helper."""
    from true_vkg.context import ContextPackBuilder
    from true_vkg.kg.builder import VKGBuilder

    builder = VKGBuilder()
    graph = builder.build(str(project_path))

    context_builder = ContextPackBuilder(graph=graph, project_path=project_path)
    return await context_builder.update(existing)

"""Integration tests for Protocol Context Pack.

Tests the full workflow:
- Schema serialization/deserialization
- Storage save/load
- Code analysis extraction
- Builder orchestration
- Evidence/bead integration

import tempfile
from pathlib import Path

import pytest

from true_vkg.context import (
    ProtocolContextPack,
    ContextPackStorage,
    ContextPackBuilder,
    BuildConfig,
    Confidence,
    Role,
    Assumption,
    Invariant,
    OffchainInput,
)
from true_vkg.context.parser import CodeAnalyzer, AnalysisResult
from true_vkg.context.integrations import EvidenceContextProvider, BeadContextProvider


class TestContextPackSchema:
    """Test ProtocolContextPack schema."""

    def test_create_minimal_pack(self):
        """Create context pack with minimal fields."""
        pack = ProtocolContextPack(protocol_name="TestProtocol")
        assert pack.protocol_name == "TestProtocol"
        assert pack.version == "1.0"
        assert pack.auto_generated is True

    def test_create_full_pack(self):
        """Create context pack with all fields."""
        pack = ProtocolContextPack(
            protocol_name="FullProtocol",
            protocol_type="lending",
            roles=[
                Role(
                    name="admin",
                    capabilities=["pause", "upgrade"],
                    trust_assumptions=["trusted operator"],
                    confidence=Confidence.CERTAIN,
                )
            ],
            assumptions=[
                Assumption(
                    description="Oracle provides accurate prices",
                    category="price",
                    confidence=Confidence.INFERRED,
                    source="code:oracle_read",
                )
            ],
            invariants=[
                Invariant(
                    formal={"what": "totalSupply", "must": "lte", "value": "maxSupply"},
                    natural_language="Total supply never exceeds max supply",
                    confidence=Confidence.CERTAIN,
                    source="docs:whitepaper",
                )
            ],
        )
        assert len(pack.roles) == 1
        assert len(pack.assumptions) == 1
        assert len(pack.invariants) == 1

    def test_to_dict_from_dict_roundtrip(self):
        """Test serialization roundtrip."""
        pack = ProtocolContextPack(
            protocol_name="RoundtripTest",
            roles=[
                Role(
                    name="owner",
                    capabilities=["setFee"],
                    trust_assumptions=["benevolent"],
                    confidence=Confidence.CERTAIN,
                )
            ],
        )

        data = pack.to_dict()
        restored = ProtocolContextPack.from_dict(data)

        assert restored.protocol_name == pack.protocol_name
        assert len(restored.roles) == len(pack.roles)
        assert restored.roles[0].name == pack.roles[0].name

    def test_get_section(self):
        """Test targeted section retrieval."""
        pack = ProtocolContextPack(
            protocol_name="SectionTest",
            roles=[Role(name="admin", capabilities=[], trust_assumptions=[], confidence=Confidence.CERTAIN)],
            assumptions=[Assumption(description="test", category="trust", confidence=Confidence.INFERRED, source="test")],
        )

        roles = pack.get_section("roles")
        assert len(roles) == 1

        assumptions = pack.get_section("assumptions")
        assert len(assumptions) == 1

        missing = pack.get_section("nonexistent")
        assert missing is None

    def test_token_estimate(self):
        """Test token estimation for context budgeting."""
        pack = ProtocolContextPack(protocol_name="TokenTest")
        estimate = pack.token_estimate()
        assert estimate > 0
        assert isinstance(estimate, int)

                    source="code:oracle_read",

        # Get context for oracle-related finding
