```python
@dataclass
class BuildConfig:
    """Configuration for context pack generation."""
    # Source options
    include_code_analysis: bool = True
    include_doc_parsing: bool = True
    additional_doc_sources: List[str] = field(default_factory=list)

    # Generation options
    infer_unstated_assumptions: bool = True  # Per 03-CONTEXT.md
    cross_validate: bool = True  # Validate docs against code
    generate_questions: bool = True  # Generate gap questions

    # Output options
    protocol_name: Optional[str] = None
    protocol_type: Optional[str] = None

@dataclass
class BuildResult:
    """Result of context pack generation."""
    pack: ProtocolContextPack
    warnings: List[str]
    conflicts: List[str]  # Doc-code conflicts
    questions: List[str]  # Security-critical gaps
    sources_used: List[str]

class ContextPackBuilder:
    """Orchestrates protocol context pack generation.

    Combines:
    - Code analysis from BSKG (roles, critical functions, inferred assumptions)
    - Document parsing (roles, assumptions, invariants, economics)
    - Web research (additional sources)

    Per 03-CONTEXT.md:
    - LLM-generated, not human-authored
    - Review encouraged but not required
    - Cross-validate doc claims against code
    - Flag conflicts explicitly
    """

    def __init__(
        self,
        graph: Optional[KnowledgeGraph] = None,
        project_path: Optional[Path] = None,
        llm_client: Optional[LLMClient] = None,
        config: Optional[BuildConfig] = None,
    ):
        self.graph = graph
        self.project_path = Path(project_path) if project_path else None
        self.llm = llm_client or LLMClient()
        self.config = config or BuildConfig()

    async def build(self) -> BuildResult:
        """Build complete protocol context pack.

        Steps:
        1. Run code analysis (if graph provided)
        2. Discover and parse documentation
        3. Merge code and doc results
        4. Cross-validate and flag conflicts
        5. Generate security gap questions
        6. Assemble final context pack
        """

    async def _analyze_code(self) -> Optional[AnalysisResult]:
        """Run code analysis from VKG."""

    async def _parse_docs(self) -> List[DocParseResult]:
        """Discover and parse all documentation."""

    def _merge_roles(
        self,
        code_roles: List[Role],
        doc_roles: List[Role]
    ) -> List[Role]:
        """Merge roles from code and docs.

        - Match by name
        - Combine capabilities
        - Prefer doc-sourced trust assumptions (docs are often "the law")
        - Use higher confidence level when both sources agree
        """

    def _merge_assumptions(
        self,
        code_assumptions: List[Assumption],
        doc_assumptions: List[Assumption]
    ) -> List[Assumption]:
        """Merge assumptions from code and docs.

        - Deduplicate similar assumptions
        - Upgrade confidence when both sources agree
        - Keep source attribution
        """

    def _detect_conflicts(
        self,
        code_result: AnalysisResult,
        doc_results: List[DocParseResult]
    ) -> List[str]:
        """Detect conflicts between code and documentation.

        Per 03-CONTEXT.md: "Do NOT assume docs are outdated -
        docs/forums are often the law"

        Conflicts:
        - Role mentioned in docs but not found in code
        - Capability claimed in docs but not implemented
        - Invariant stated in docs but not enforced
        """

    def _infer_protocol_type(self, combined_data: Dict[str, Any]) -> str:
        """Infer protocol type from context.

        Types: lending, dex, nft, bridge, vault, staking, governance, other
        """

    def _assemble_pack(
        self,
        code_result: Optional[AnalysisResult],
        doc_results: List[DocParseResult],
        conflicts: List[str],
        questions: List[str],
    ) -> ProtocolContextPack:
        """Assemble final context pack from all sources."""
```

Implementation details:

1. **Build flow**:
   - Start with code analysis (synchronous, fast)
   - Fetch and parse docs in parallel (async)
   - Merge results with conflict detection
   - Generate questions for gaps
   - Mark as auto_generated=True, reviewed=False

2. **Role merging**:
   - Use role name as key for matching
   - Union capabilities from both sources
   - Doc trust assumptions take precedence (per 03-CONTEXT.md)
   - Upgrade confidence: if code AND docs agree, confidence=CERTAIN

3. **Conflict detection** (per 03-CONTEXT.md):
   - "Flag conflicts between docs and code explicitly"
   - "Treat doc-code conflicts as potential finding source"
   - Return list of conflict descriptions for human review

4. **Protocol type inference**:
   - Check for keywords in roles/functions (swap, lend, mint, stake, vote)
   - Map to standard types for context-driven pattern selection

```python
def _add_changelog_entry(
    self,
    pack: ProtocolContextPack,
    change_type: str,
    description: str,
) -> None:
    """Add changelog entry for version tracking.

    Per 03-CONTEXT.md: Explicit versioning with changelog support
    """
    if 'changelog' not in pack.notes:
        pack.notes = f"## Changelog\n\n{pack.notes}" if pack.notes else "## Changelog\n"

    entry = f"- {datetime.utcnow().isoformat()}: {change_type} - {description}\n"
    # Insert after changelog header
    pack.notes = pack.notes.replace("## Changelog\n", f"## Changelog\n{entry}")
```
